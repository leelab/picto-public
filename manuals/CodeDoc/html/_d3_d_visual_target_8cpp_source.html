237 void D3DVisualTarget::draw(QPoint location, QPoint compositingSurfaceOffset, QSharedPointer<CompositingSurface> compositingSurface)
238 {
239  location = location-compositingSurfaceOffset;
240  if(compositingSurface->getTypeName() == "Direct3D")
241  {
242  //retreive a pointer to the texture
243  LPDIRECT3DTEXTURE9 tex = compositingSurface.staticCast<D3DCompositingSurface>()->getTexture();
244 
245  //add texture pointer to the rendering list
246  textureList_.push_back(tex);
247 
248  D3DSURFACE_DESC texDesc;
249  tex->GetLevelDesc(0, &texDesc);
250 
251  //figure out the position for the sprite, and add it to the list
252  //NOTE: Picto uses the standard screen coodinates with (0,0) in the top left
253  //corner. However, the D3D setup we are using has (0,0) in the bottom left corner, so
254  //we need to find the "complement" of the y-coordinate.
255  //The value stored in the positionList_ is the bottom left corner (inclusive)
256  //of the sprite.
257  QPointF quadPoint;
258  quadPoint.setX(location.x());
259  quadPoint.setY((height_-location.y()-(int)texDesc.Height));
260  positionList_.push_back(quadPoint);
261  }
262  else
263  {
264  d3dFail("D3DVisualTarget: draw called with a compositing surface not of type \"Direct3D\"");
265  }

272 void D3DVisualTarget::present()
273 {
274  renderSuccess_ = false;
275  HRESULT coopLevel = pD3dDevice_->TestCooperativeLevel();
276  while(coopLevel == D3DERR_DEVICELOST || coopLevel == D3DERR_DEVICENOTRESET)
277  {
278  if(coopLevel == D3DERR_DEVICENOTRESET)
279  {
280  initializeD3DDevice();
281  }
282  else
283  {
284  //This code remaximizes the window-----------------------------
285  WINDOWPLACEMENT window_placement;
286  window_placement.length = sizeof(WINDOWPLACEMENT);
287  GetWindowPlacement( hWnd_, &window_placement );
288  window_placement.showCmd = SW_SHOWMAXIMIZED ;
289  SetWindowPlacement( hWnd_, &window_placement );
291  }
292  QCoreApplication::processEvents();
293  coopLevel = pD3dDevice_->TestCooperativeLevel();
294  }
295 
296  // initializeD3DDevice();
297  LPDIRECT3DTEXTURE9 tex;
298  QPointF quadPoint;
299  D3DXMATRIX matScale, matTrans;
300  D3DSURFACE_DESC texDesc;
301 
302  HRESULT hr;
303 
304 
305  //begin the scene
306  hr = pD3dDevice_->BeginScene();
307  if(hr)
308  d3dFail("D3DVisualTarget: BeginScene FAILED");
309 
310  //render each textured quad
311  while(!textureList_.isEmpty())
312  {
313  tex = textureList_.takeFirst();
314  quadPoint = positionList_.takeFirst();
315 
316  tex->GetLevelDesc(0, &texDesc);
317 
318  //The 0.5 offset on the translation matrix is done to ensure perfect texture mapping
319  D3DXMatrixScaling(&matScale, (float)texDesc.Width, (float)texDesc.Height, 0.0f);
320  D3DXMatrixTranslation(&matTrans, quadPoint.x()-0.5f, quadPoint.y()-0.5f, 0.0f);
321  pD3dDevice_->SetTransform( D3DTS_WORLD, (D3DMATRIX*)&(matScale*matTrans));
322 
323  pD3dDevice_->SetTexture( 0, tex );
324  pD3dDevice_->DrawPrimitive(D3DPT_TRIANGLELIST,
325  0, //Start Vertex
326  2); //Primitive Count
327  }
328 
329  //end the scene
330  hr = pD3dDevice_->EndScene();
331  if(hr)
332  d3dFail("D3DVisualTarget: EndScene FAILED");
333 
334  //Don't call present until we are in vertical blanking
335  //BYTE outBuf;
336  //DWORD bytesReturned;
337  //DeviceIoControl(hCRT,IOCTRL_WAIT_VBLANK_START,NULL,0,&outBuf,1,&bytesReturned,NULL);
338 
339  //Wait For Vertical Blank
340  D3DRASTER_STATUS rastStat;
341  HRESULT hRastRes = pD3dDevice_->GetRasterStatus(0,&rastStat);
342  while(!hRastRes && !rastStat.InVBlank)
343  {
344  hr = pD3dDevice_->GetRasterStatus(0,&rastStat);
345  }
346  if(hr)
347  d3dFail("D3DVisualTarget: Monitor does not support reading of the current scan line.");
348 
349  static int test = 0;
350 
351  //present the back buffer
352  hr = pD3dDevice_->Present( NULL, NULL, NULL, NULL );
353  if(hr)
354  {
355  d3dFail("D3DVisualTarget: Present FAILED");
356  }
357 
358  //Wait for rastering to begin
359  while(!hRastRes && rastStat.InVBlank)
360  {
361  hRastRes = pD3dDevice_->GetRasterStatus(0,&rastStat);
362  }
363  setFirstPhosphorTime();
364  renderSuccess_ = true;
365 
366  //clear the back buffer
367  pD3dDevice_->Clear( 0, NULL, D3DCLEAR_TARGET,0, 0.0f, 0 );

457 void D3DVisualTarget::initializeD3DDevice()
458 {
459  if(pVertexBuffer_)
460  {
461  pVertexBuffer_->Release();
462  }
463 
464  HRESULT hr;
465 
466  if(pD3dDevice_)
467  {
468  hr = pD3dDevice_->Reset(&d3dpp_);
469  if(hr)
470  return;
471  }
472  else
473  {
474  hr = pD3D_->CreateDevice(D3DADAPTER_DEFAULT, D3DDEVTYPE_HAL, hWnd_, D3DCREATE_SOFTWARE_VERTEXPROCESSING | D3DCREATE_FPU_PRESERVE,&d3dpp_, &pD3dDevice_);
475 
476  if(hr)
477  {
478  d3dFail("D3DVisualTarget: CreateDevice FAILED");
479  printf("D3DVisualTarget: CreateDevice FAILED %x\n", hr);
480  }
481  }
482 
483 
484  // Turn off D3D lighting
485  pD3dDevice_->SetRenderState( D3DRS_LIGHTING, FALSE );
486 
487  // Turn on alpha blending
488  //NOTE: at various stages during development, we had to use inverse alpha values
489  //so it's possible that if alpha blending isn't working, that may be a cause
490  pD3dDevice_->SetRenderState( D3DRS_ALPHABLENDENABLE, TRUE );
491  pD3dDevice_->SetRenderState(D3DRS_SRCBLEND,D3DBLEND_SRCALPHA);
492  pD3dDevice_->SetRenderState(D3DRS_DESTBLEND,D3DBLEND_INVSRCALPHA);
493 
494  //get device capabilities
495  D3DCAPS9 caps;
496  hr = pD3dDevice_->GetDeviceCaps(&caps);
497  if(hr)
498  d3dFail("D3DVisualTarget: GetDeviceCaps FAILED");
499 
500  //----------------------------------
501  //Set up the Vertex Stream
502  //----------------------------------
503  //Note that the texture y-coordinates are flipped. This is because Picto coordinates
504  //start in the top right corner, while D3D coordinates start in the bottom left.
505  CUSTOMVERTEX quadVertexData[] =
506  {
507  { 0.0f, 0.0f, 0.0f, D3DCOLOR_ARGB(0, 255, 255, 255), 0.0f, 1.0f },
508  { 0.0f, 1.0f, 0.0f, D3DCOLOR_ARGB(0, 255, 255, 255), 0.0f, 0.0f },
509  { 1.0f, 1.0f, 0.0f, D3DCOLOR_ARGB(0, 255, 255, 255), 1.0f, 0.0f },
510  { 0.0f, 0.0f, 0.0f, D3DCOLOR_ARGB(0, 255, 255, 255), 0.0f, 1.0f },
511  { 1.0f, 1.0f, 0.0f, D3DCOLOR_ARGB(0, 255, 255, 255), 1.0f, 0.0f },
512  { 1.0f, 0.0f, 0.0f, D3DCOLOR_ARGB(0, 255, 255, 255), 1.0f, 1.0f },
513  };
514 
515  // Create the vertex buffer.
516  hr = pD3dDevice_->CreateVertexBuffer(sizeof(quadVertexData), 0,
517  D3DFVF_XYZ | D3DFVF_DIFFUSE | D3DFVF_TEX1,
518  D3DPOOL_DEFAULT, &pVertexBuffer_, NULL);
519  if(hr)
520  d3dFail("D3DVisualTarget: CreateVertexBuffer FAILED");
521 
522  // Fill the vertex buffer.
523  void* pVertices;
524  hr = pVertexBuffer_->Lock(0, 0, &pVertices, 0);
525  if(hr)
526  d3dFail("D3DVisualTarget: pVertexBuffer_->Lock FAILED");
527 
528  memcpy(pVertices, quadVertexData, sizeof(quadVertexData));
529  pVertexBuffer_->Unlock();
530 
531  //Traditionally, the stream source would be set at the beginning of rendering each frame
532  //However to save CPU cycles (and since the rendering is kind of trivial), we'll
533  //set the Stream source here).
534  pD3dDevice_->SetStreamSource(0, pVertexBuffer_, 0, sizeof(CUSTOMVERTEX));
535  pD3dDevice_->SetFVF( D3DFVF_XYZ | D3DFVF_DIFFUSE | D3DFVF_TEX1 );
536 
537 
538  //----------------------------------
539  //Set up Matrices
540  //----------------------------------
541  // Set up our view matrix as an identity matrix
542  D3DXMATRIX matView;
543  D3DXMatrixIdentity( &matView );
544  pD3dDevice_->SetTransform( D3DTS_VIEW, (D3DMATRIX*)&matView);
545 
546  // For the projection matrix, we use a simple orthogonal matrix
547  // Basic Orthographic transform.
548  // Left = 0 Right = width_
549  // Bottom = 0 Top = height_
550  // Near = -1 Far = 1
551  D3DXMATRIX matProj;
552  D3DXMatrixOrthoOffCenterLH(&matProj,0.0f,(float)width_,0.0f,(float)height_,-1.0f,1.0f);
553  pD3dDevice_->SetTransform( D3DTS_PROJECTION, (D3DMATRIX*)&(matProj*matView));
554 
555  //The world matrix changes for each sprite, so we deal with that while rendering...
556
