35 QString FlowElement::run(QSharedPointer<Engine::PictoEngine> engine)
36 {
37  //The Condition list is sorted by the order values, so all we have to do is run
38  //through it evaluating conditions.
39  foreach(Condition c, conditions_)
40  {
41  if(c.isCompound)
42  {
43  if(c.compExpr->evaluate())
44  {
45  return c.name;
46  }
47  }
48  else
49  {
50  if(c.predExpr->evaluate())
51  {
52  return c.name;
53  }
54  }
55  }
56 
57  //if we made it this far, nothing evaluated true, so return the default result
58  return "default";

82  //Make sure that some variant of this condition doesn't already exist
83  foreach(Condition c, conditions_)
84  {
85  if(c.name == newCond.name)
86  return false;
87  if(c.order == newCond.order)
88  return false;

90  addChildScriptableContainer(newCond.predExpr);

92  conditions_.push_back(newCond);
93

96  //Conditions. Also, it's better to sort here, than when the FlowElement is run
97  qSort(conditions_.begin(), conditions_.end(), &conditionLessThan);

354 void FlowElement::postSerialize()
355 {
356  //clear out the existing conditions
357  conditions_.clear();
358 
359  QList<QSharedPointer<Asset>> newExprs = getGeneratedChildren("Result");
360  foreach(QSharedPointer<Asset> newExpr,newExprs)
361  {
362  if(newExpr.dynamicCast<PredicateExpression>())
363  {
364  addCondition(newExpr.staticCast<PredicateExpression>());
365  }
366  else if (newExpr.dynamicCast<CompoundExpression>())
367  {
368  addCondition(newExpr.staticCast<CompoundExpression>());
369  }
370  else
371  addRequiredResult(newExpr.staticCast<Result>());
372  }
373  //Sort the condition list
374  qSort(conditions_.begin(), conditions_.end(), &conditionLessThan);
375 
376  StateMachineElement::postSerialize();
377 }
