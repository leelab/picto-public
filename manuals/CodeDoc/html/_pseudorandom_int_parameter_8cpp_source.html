73 void PseudorandomIntParameter::randomize()
74 {
75  checkForPropertyChanges();
76 
77  currIndex_++;
78  if(currIndex_ >= randomArray_.size())
79  currIndex_ = 0;
80  //Swap the first value in the array with a randomized selection of the rest of the values in the array.
81  //The method below works. Really.
82  //See http://adrianquark.blogspot.com/2008/09/how-to-shuffle-array-correctly.html
83  int swapVal = randomArray_[currIndex_];
84  int maxSwapIndex = randomArray_.size()-1-currIndex_;
85  if(propertyContainer_->getPropertyValue("DontRepeatAnyValue").toBool() && maxSwapIndex)
86  { //We aren't allowed to repeat the last value in the array, so we subtract 1 from the swapRange
87  maxSwapIndex--;
88  }
89  int swapLoc = mtRand_.randInt(maxSwapIndex); //Returns a value between 0 and the input, including zero
90  randomArray_[currIndex_] = randomArray_[currIndex_+swapLoc];
91  randomArray_[currIndex_+swapLoc] = swapVal;
92 
93  propertyContainer_->setPropertyValue("Value",randomArray_[currIndex_]);
94 }

103 void PseudorandomIntParameter::reshuffleLastValue()
104 {
105  if(currIndex_ < 0)
106  {
107  //We haven't started yet. There's nothing to reshuffle.
108  return;
109  }
110  //Swap the last value in the array with the latest returned value and decrement the currIndex_
111  //We could just decrement the current index, but swapping with the last value assures us that
112  //we won't end up repeating a value if "DontRepeatAnyValue" is set (unless its the last value
113  //to be used in the list, in which case we should be repeating it anyway since there's nothing
114  //else to replace it).
115  int lastVal = randomArray_[currIndex_];
116  randomArray_[currIndex_] = randomArray_[randomArray_.size()-1];
117  randomArray_[randomArray_.size()-1] = lastVal;
118  if(currIndex_ == 0)
119  currIndex_ = randomArray_.size()-1;
120  else
121  currIndex_--;
122 }

142 void PseudorandomIntParameter::reset()
143 {

231 void PseudorandomIntParameter::checkForPropertyChanges()
232 {
233  bool useSeed = propertyContainer_->getPropertyValue("UseSeed").toBool();
234  if(useSeed)
235  {
236  int seed = propertyContainer_->getPropertyValue("Seed").toInt();
237  if(!useSeed_ || (seed_ != seed))
238  {
239  //We're using a seed and weren't before, or we were before but it just changed.
240  mtRand_.seed(seed);
241  }
242  seed_ = seed;
243  }
244  useSeed_ = useSeed;
245  int min = propertyContainer_->getPropertyValue("Min").toInt();
246  int max = propertyContainer_->getPropertyValue("Max").toInt();
247  if((randomArray_.size() == 0) || (min != min_) || (max != max_))
248  { //the max or min values changed. Rebuild the pseudorandom vector.
249  max_ = max;
250  min_ = min;
251  randomArray_.resize(max_-min_+1);
252  for(int i=min_;i<=max_;i++)
253  {
254  randomArray_[i-min_] = i;
255  }
256  //Set the current index to the last index in the array, so that we
257  //will start over next time.
258  currIndex_ = randomArray_.size()-1;
259  }
