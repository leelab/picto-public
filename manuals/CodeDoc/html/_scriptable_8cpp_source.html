15 bool Scriptable::hasContentsForScriptEngine()
16 {
17  //If the object has no Q_PROPERTIES OR local public slots, don't add it.
18  int numScriptableClassProps = Scriptable::staticMetaObject.propertyCount();
19  int numScriptableClassFuncs = Scriptable::staticMetaObject.methodCount();
20  const QMetaObject* metaObj = metaObject();
21  int props = metaObj->propertyCount();
22  int functions = metaObj->methodCount();
23  if(numScriptableClassProps == props)
24  { //No Q_PROPERTIES to add. Lets try public slots.
25  bool hasPubSlots = false;
26  for(int i = numScriptableClassFuncs;i<functions;i++)
27  {
28  QMetaMethod metaMeth = metaObj->method(i);
29  if( (metaMeth.access() == QMetaMethod::Public)
30  && (metaMeth.methodType() == QMetaMethod::Slot) )
31  {
32  hasPubSlots = true;
33  break;
34  }
35  }
36  if(!hasPubSlots)
37  return false;
38  }
39  return true;
40 }

45 bool Scriptable::bindToScriptEngine(QScriptEngine &engine)
46 {
47  if(!hasContentsForScriptEngine())
48  return true;
49  Q_ASSERT(!getName().isEmpty());
50  Q_ASSERT(!getName().contains(" "));
51 
52  QString myScriptValueName = getName().simplified().remove(' ');
53  QScriptValue myScriptValue = engine.globalObject().property(myScriptValueName);
54 
55  if(!myScriptValue.isValid())
56  { //This ScriptValue hasn't been created yet. Create it.
57  myScriptValue = engine.newQObject(this,QScriptEngine::QtOwnership);
58  engine.globalObject().setProperty(myScriptValueName,myScriptValue);
59  //Check for errors
60  if(engine.hasUncaughtException())
61  {
62  QString errorMsg = "Uncaught exception in Script Element " + getName() +"\n";
63  errorMsg += QString("Line %1: %2\n").arg(engine.uncaughtExceptionLineNumber())
64  .arg(engine.uncaughtException().toString());
65  errorMsg += QString("Backtrace: %1\n").arg(engine.uncaughtExceptionBacktrace().join(", "));
66  Q_ASSERT_X(!engine.hasUncaughtException(),"Scriptable::getScriptValue",errorMsg.toLatin1());
67  return false;
68  }
69  Q_ASSERT(myScriptValue.isValid());
70  }
71  return true;

104 QString Scriptable::getScriptingInfo()
105 {
106  QString returnVal;
107 
108  int numScriptableClassProps = Scriptable::staticMetaObject.propertyCount();
109  int numScriptableClassFuncs = Scriptable::staticMetaObject.methodCount();
110 
111  //Print out the scriptable's properties, ignoring those that are in scriptable and its superclasses.
112  const QMetaObject* metaObj = metaObject();
113  Q_ASSERT(metaObj);
114  int props = metaObj->propertyCount();
115  QMetaProperty metaProp;
116  for(int i=numScriptableClassProps;i<props;i++)
117  {
118  metaProp = metaObj->property(i);
119  returnVal.append(QString("<dd><span style=\"color:blue;\">%1</span> - <span style=\"color:grey;\">(type:%2)</span></dd>").arg(metaProp.name()).arg(metaProp.typeName()));
120  }
121 
122  //Now print out the functions that are available for scripting, ignoring those that are in
123  // scriptable and its superclasses. Only display functions defined as public slots.
124  int functions = metaObj->methodCount();
125  QMetaMethod metaMeth;
126  for(int i=numScriptableClassFuncs;i<functions;i++)
127  {
128  metaMeth = metaObj->method(i);
129  if(metaMeth.access() != QMetaMethod::Public)
130  continue;
131  if(metaMeth.methodType() == QMetaMethod::Slot)
132  {
133  QString signature = metaMeth.methodSignature();
134  QStringList commaSplitSig = signature.split(",");
135  QList<QByteArray> paramNames = metaMeth.parameterNames();
136  QStringList::iterator paramTypeIter = commaSplitSig.begin();
137  QList<QByteArray>::iterator paramNameIter = paramNames.begin();
138  //Color function name blue and the rest grey
139  int openParenthLoc = (*paramTypeIter).indexOf("(");
140  (*paramTypeIter).insert(openParenthLoc+1,"<span style=\"color:grey\">");
141  (*paramTypeIter).append("</span>");
142  while((paramTypeIter != commaSplitSig.end()) && (paramNameIter != paramNames.end()))
143  {
144  if (paramTypeIter != commaSplitSig.begin())
145  (*paramTypeIter) = QString("<span style=\"color:grey\">%1</span>").arg(*paramTypeIter);
146  //We don't want QString in our UI, so we replace it with "String".
147  (*paramTypeIter).replace("QString", "String");
148  (*paramTypeIter).append(QString(" <span style=\"color:blue\">%1</span>").arg(QString(*paramNameIter)));
149  paramTypeIter++;
150  paramNameIter++;
151  }
152  signature = commaSplitSig.join(",");
153  //Move the last name in front of the final paranthesis (right now it will be func(int a,int b,int) c
154  int parenthInd = signature.lastIndexOf(")");
155  signature.remove(parenthInd,1);
156  signature.append(")");
157 
158  returnVal.append(QString("<dd>%1</dd>").arg(signature));
159  }
160  }
161  if(!returnVal.isEmpty())
162  {
163  //There were scripting elements. Finish of the returnVal.
164  //Start the returnVal with this scriptables name.
165  returnVal.prepend(QString("<dl><dt>%1</dt>").arg(getName()));
166  //Close the definition list.
167  returnVal.append("</dl>");
168  }
169 
170  return returnVal;
171 }
