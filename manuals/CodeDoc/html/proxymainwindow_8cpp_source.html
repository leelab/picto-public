400 void ProxyMainWindow::updateState()
401 {
402  updatingState_ = true;
403  ProxyState nextState = currState_;
404  switch(currState_)
405  {
406  case InitState:
407  nextState = WaitForConnect;
408  break;
409  case WaitForConnect:
410  switch(stateTrigger_)
411  {
412  case Connected:
413  nextState = WaitForSession;
414  break;
415  }
416  break;
417  case WaitForSession:
418  switch(stateTrigger_)
419  {
420  case Disconnected:
421  nextState = WaitForConnect;
422  break;
423  case ConnectDeviceRequest:
424  nextState = WaitForDevice;
425  break;
426  }
427  break;
428  case WaitForDevice:
429  switch(stateTrigger_)
430  {
431  case SessionEnded:
432  nextState = WaitForSession;
433  break;
434  case DeviceStarted:
435  nextState = Running;
436  break;
437  }
438  break;
439  case Running:
440  switch(stateTrigger_)
441  {
442  case DeviceStopped:
443  nextState = WaitForDevice;
444  break;
445  case SessionEnded:
446  nextState = WaitForSession;
447  }
448  break;
449  }
450  stateTrigger_ = NoProxyTrigger;
451  if(nextState != currState_)
452  {
453  currState_ = nextState;
454  enterState();
455  }
456  runState();
457  if(stateTrigger_ != NoProxyTrigger)
458  endState();
459  //updateStatus();
460  updatingState_ = false;
461 }

485 void ProxyMainWindow::runState()
486 {
487  switch(currState_)
488  {
489  case WaitForConnect:
490  if(statusManager_)
491  statusManager_.staticCast<ProxyStatusManager>()->setName(name());
492  if(!dataCommandChannel_.isNull() && dataCommandChannel_->assureConnection())
493  stateTrigger_ = Connected;
494  break;
495  case WaitForSession:
496  setNeuralDataAcquisitionDevice(pluginCombo_->currentIndex());
497  if(statusManager_)
498  statusManager_.staticCast<ProxyStatusManager>()->setName(name());
499  if(dataCommandChannel_.isNull() || !dataCommandChannel_->assureConnection())
500  stateTrigger_ = Disconnected;
501  else if(isSessionActive())
502  stateTrigger_ = ConnectDeviceRequest;
503  break;
504  case WaitForDevice:
505  if(deviceIsRunning())
506  {
507  if(deviceHasData())
508  stateTrigger_ = DeviceStarted;
509  }
510  else
511  {
512  startDevice();
513  }
514  if(!isSessionActive())
515  {
516  stateTrigger_ = SessionEnded;
517  stopDevice();
518  }
519  break;
520  case Running:
521  if(!deviceHasData())
522  stateTrigger_ = DeviceStopped;
523  if(!isSessionActive())
524  {
525  stateTrigger_ = SessionEnded;
526  stopDevice();
527  }
528  break;
529  }
530  if(currState_ != WaitForConnect)
531  {
532  if(isServerConnected())
533  connectionStatus_->turnGreen();
534  else
535  connectionStatus_->turnRed();
536  }
537 }

562 void ProxyMainWindow::enterState()
563 {
564  switch(currState_)

566  case WaitForConnect:

568  systemNumber_->setEnabled(true);
569  runStatus_->turnRed();
570  sessionStatus_->turnRed();

573  break;
574  case WaitForSession:
575  lineEditName_->setEnabled(true);
576  systemNumber_->setEnabled(true);
577  runStatus_->turnRed();
578  sessionStatus_->turnRed();
579  pluginCombo_->setEnabled(true);
580  qDebug("Entering WaitForSession");
581  break;
582  case WaitForDevice:
583  lineEditName_->setEnabled(false);
584  systemNumber_->setEnabled(false);
585  runStatus_->turnRed();
586  sessionStatus_->turnGreen();
587  pluginCombo_->setEnabled(true);
588  qDebug("Entering WaitForDevice");
589  break;
590  case Running:
591  lineEditName_->setEnabled(false);
592  systemNumber_->setEnabled(false);
593  runStatus_->turnGreen();
594  sessionStatus_->turnGreen();
595  pluginCombo_->setEnabled(false);
596  qDebug("Entering Running");
597  break;
598  }
