395 void ProxyMainWindow::updateState()
396 {
397  updatingState_ = true;
398  ProxyState nextState = currState_;
399  switch(currState_)
400  {
401  case InitState:
402  nextState = WaitForConnect;
403  break;
404  case WaitForConnect:
405  switch(stateTrigger_)
406  {
407  case Connected:
408  nextState = WaitForSession;
409  break;
410  }
411  break;
412  case WaitForSession:
413  switch(stateTrigger_)
414  {
415  case Disconnected:
416  nextState = WaitForConnect;
417  break;
418  case ConnectDeviceRequest:
419  nextState = WaitForDevice;
420  break;
421  }
422  break;
423  case WaitForDevice:
424  switch(stateTrigger_)
425  {
426  case SessionEnded:
427  nextState = WaitForSession;
428  break;
429  case DeviceStarted:
430  nextState = Running;
431  break;
432  }
433  break;
434  case Running:
435  switch(stateTrigger_)
436  {
437  case DeviceStopped:
438  nextState = WaitForDevice;
439  break;
440  case SessionEnded:
441  nextState = WaitForSession;
442  }
443  break;
444  }
445  stateTrigger_ = NoProxyTrigger;
446  if(nextState != currState_)
447  {
448  currState_ = nextState;
449  enterState();
450  }
451  runState();
452  if(stateTrigger_ != NoProxyTrigger)
453  endState();
454  //updateStatus();
455  updatingState_ = false;
456 }

480 void ProxyMainWindow::runState()
481 {
482  switch(currState_)
483  {
484  case WaitForConnect:
485  if(statusManager_)
486  statusManager_.staticCast<ProxyStatusManager>()->setName(name());
487  if(!dataCommandChannel_.isNull() && dataCommandChannel_->assureConnection())
488  stateTrigger_ = Connected;
489  break;
490  case WaitForSession:
491  setNeuralDataAcquisitionDevice(pluginCombo_->currentIndex());
492  if(statusManager_)
493  statusManager_.staticCast<ProxyStatusManager>()->setName(name());
494  if(dataCommandChannel_.isNull() || !dataCommandChannel_->assureConnection())
495  stateTrigger_ = Disconnected;
496  else if(isSessionActive())
497  stateTrigger_ = ConnectDeviceRequest;
498  break;
499  case WaitForDevice:
500  if(deviceIsRunning())
501  {
502  if(deviceHasData())
503  stateTrigger_ = DeviceStarted;
504  }
505  else
506  {
507  startDevice();
508  }
509  if(!isSessionActive())
510  {
511  stateTrigger_ = SessionEnded;
512  stopDevice();
513  }
514  break;
515  case Running:
516  if(!deviceHasData())
517  stateTrigger_ = DeviceStopped;
518  if(!isSessionActive())
519  {
520  stateTrigger_ = SessionEnded;
521  stopDevice();
522  }
523  break;
524  }
525  if(currState_ != WaitForConnect)
526  {
527  if(isServerConnected())
528  connectionStatus_->turnGreen();
529  else
530  connectionStatus_->turnRed();
531  }
532 }
