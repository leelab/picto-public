96 void ChoiceController::upgradeVersion(QString deserializedVersion)
97 {
98  ControlElement::upgradeVersion(deserializedVersion);
99  if(deserializedVersion < "0.0.1")
100  { // In design syntax version "0.0.1", we added userOnTarget, userEnteredTarget, userExitedTarget functions.
101  // In an effort to keep the allowed script locations uniform, we are making TargetEntry
102  // and TargetExit scripts obsolete.
103  // We upgrade these older experiments by automatically adding their entry and exit scripts
104  // to the frame script and exit scripts in if(userEnteredTarget()) and if(userExitedTarget) blocks.
105 
106  //WARNING!!! userOnTarget(), userEnteredTarget() and userExitedTarget()
107  //functions cannot be used in AnalysisScripts of experiments that were saved with version below 0.0.1.
108  //This is because experiments with those versions used TargetEntry and TargetExit scripts and did not
109  //save OnTarget or OffTarget data in their session files. Attempting to use these functions in Analysis
110  //Scripts would cause return of invalid data.
111  QSharedPointer<Property> parentFrameScriptProp = getParentAsset().staticCast<DataStore>()->getGeneratedChildren("FrameScript").first().staticCast<Property>();
112  QSharedPointer<Property> parentExitScriptProp = getParentAsset().staticCast<DataStore>()->getGeneratedChildren("ExitScript").first().staticCast<Property>();
113  Q_ASSERT(parentFrameScriptProp);
114  QString parentFrameScript = parentFrameScriptProp->value().toString();
115  QString parentExitScript = parentExitScriptProp->value().toString();
116  QSharedPointer<ObsoleteAsset> obsScript;
117  if(getGeneratedChildren("TargetExitScript").size())
118  obsScript = getGeneratedChildren("TargetExitScript").first().staticCast<ObsoleteAsset>();
119  if(obsScript && obsScript->getValue().size())
120  {
121  //Put exit script into parent's frame script.
122  parentFrameScript.prepend(QString("if(%1.userExitedTarget()){\n%2\n}\n").arg(getName()).arg(obsScript->getValue()));
123  //Put exit script into parent's exit script.
124  parentExitScript.prepend(QString("if(%1.userExitedTarget()){\n%2\n}\n").arg(getName()).arg(obsScript->getValue()));
125  }
126  obsScript.clear();
127  if(getGeneratedChildren("TargetEntryScript").size())
128  obsScript = getGeneratedChildren("TargetEntryScript").first().staticCast<ObsoleteAsset>();
129  if(obsScript && obsScript->getValue().size())
130  {
131  //Put entry script into parent's frame script.
132  parentFrameScript.prepend(QString("if(%1.userEnteredTarget()){\n%2\n}\n").arg(getName()).arg(obsScript->getValue()));
133  //Put entry script into parent's exit script.
134  parentExitScript.prepend(QString("if(%1.userEnteredTarget()){\n%2\n}\n").arg(getName()).arg(obsScript->getValue()));
135  }
136  parentFrameScriptProp->setValue(parentFrameScript);
137  parentExitScriptProp->setValue(parentExitScript);
138  }
139 }

157 bool ChoiceController::isDonePrivate(QSharedPointer<Engine::PictoEngine> engine)
158 {
159  //This property tells us if the user entered or exited the target this frame. Initialize it to false.
160  propertyContainer_->getProperty("OnTargetChanged")->setValue(false);
161 
162  Controller::TimerUnits::TimerUnits timeUnits;
163  if(unitList_.value(timeUnits_,"") == "Sec")
164  timeUnits = Controller::TimerUnits::sec;
165  else if(unitList_.value(timeUnits_,"") == "Ms")
166  timeUnits = Controller::TimerUnits::ms;
167  else if(unitList_.value(timeUnits_,"") == "Us")
168  timeUnits = Controller::TimerUnits::us;
169  else
170  {
171  timeUnits = Controller::TimerUnits::ms;
172  Q_ASSERT(false);
173  }
174 
175 
176  //check to see if we've met or exceeded the total time
177  int totalTime = propertyContainer_->getPropertyValue("TotalTime").toInt();
178  int currTotalTime = cumulativeTimer_->elapsedTime(timeUnits);
179  int fixTime = propertyContainer_->getPropertyValue("FixationTime").toInt();
180  int currAcqTime = acquisitionTimer_->elapsedTime(timeUnits);
181  int remainingFixTime = targetAcquired_?fixTime-currAcqTime:fixTime;
182  if(currTotalTime+remainingFixTime > totalTime)
183  {
184  isDone_ = true;
185  result_ = "Total Time Exceeded";
186  return true;
187  }
188 
189  QString currTarget = insideTarget(engine);
190 
191  //just left a target
192  if(currTarget != lastTarget_ && targetAcquired_)
193  {
194  if(!engine->slaveMode())
195  {
196  propertyContainer_->getProperty("OnTarget")->setValue(false);
197  propertyContainer_->getProperty("OnTargetChanged")->setValue(true);
198  //if(propertyContainer_->getPropertyValue("TargetExitScript").toString() != "")
199  // runScript(getName().simplified().remove(' ').append("_TargetExit"));
200  }
201  //Are reentries allowed?
202  if(!propertyContainer_->getPropertyValue("AllowReentries").toBool())
203  {
204  isDone_ = true;
205  result_ = "Broke Fixation";
206  return true;
207  }
208  if(currTarget == "NotATarget")
209  targetAcquired_ = false;
210  else
211  {
212  acquisitionTimer_->start();
213  targetAcquired_ = true;
214  }
215 
216  //Recheck if fixation is possible in the timeframe since a target was left
217  if(currTotalTime+fixTime > totalTime)
218  {
219  isDone_ = true;
220  result_ = "Total Time Exceeded";
221  return true;
222  }
223  }
224  //just entered a target
225  else if(currTarget != "NotATarget" && !targetAcquired_)
226  {
227  if(!engine->slaveMode())
228  {
229  propertyContainer_->getProperty("OnTarget")->setValue(true);
230  propertyContainer_->getProperty("OnTargetChanged")->setValue(true);
231  //if(propertyContainer_->getPropertyValue("TargetEntryScript").toString() != "")
232  // runScript(getName().simplified().remove(' ').append("_TargetEntry"));
233  }
234  targetAcquired_ = true;
235  acquisitionTimer_->start();
236  //If fixation time is zero, we're done
237  if(fixTime <= 0)
238  {
239  isDone_ = true;
240  result_ = currTarget;
241  return true;
242  }
243  }
244  //staying inside a target
245  else if(currTarget == lastTarget_ && targetAcquired_)
246  {
247  if(currAcqTime > fixTime)
248  {
249  isDone_ = true;
250  result_ = currTarget;
251  return true;
252  }
253  }
254  //staying outside a target
255  else if(currTarget == "NotATarget" && !targetAcquired_)
256  {
257  //do nothing
258  }
259  lastTarget_ = currTarget;
260 
261  return false;
262 }
