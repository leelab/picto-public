50 PlaybackIndex AutoDataState::getNextIndex(double lookForwardTime)
51 {
52  if(dataIndex_ < -1)
53  {
54  querySessionForData(query_,lastQueryEndTime_,runEndTime_<0?1000000000:runEndTime_);
55  lastQueryEndTime_ = runEndTime_<0?-1:runEndTime_;
56  if(!query_->next() || query_->at() == QSql::AfterLastRow)
57  return PlaybackIndex();
58  currQueryData_ = convertQueryToDataVector(query_,runStartTime_);
59  if(!currQueryData_.size())
60  return PlaybackIndex();
61  if(query_->next() && query_->at() != QSql::AfterLastRow)
62  nextQueryData_ = convertQueryToDataVector(query_,runStartTime_);
63  dataIndex_ = -1;
64  }
65  if(dataIndex_ < currQueryData_.size()-1)
66  {
67  if(currQueryData_[dataIndex_+1]->index_.time() <= lookForwardTime)
68  return currQueryData_[dataIndex_+1]->index_;
69  return PlaybackIndex();
70  }
71  while(!nextQueryData_.size())
72  {
73  if(!query_->next() || query_->at() == QSql::AfterLastRow)
74  {
75  //Query Database starting after currentIndex!
76  if(lastQueryEndTime_ < 0)
77  lastQueryEndTime_ = getCurrentIndex().time()+runStartTime_;
78  if(lastQueryEndTime_ >= runEndTime_)
79  return PlaybackIndex();
80  querySessionForData(query_,lastQueryEndTime_,runEndTime_<0?1000000000:runEndTime_);
81  lastQueryEndTime_ = runEndTime_<0?-1:runEndTime_;
82  if(!query_->next() || query_->at() == QSql::AfterLastRow)
83  {
84  query_->previous();
85  return PlaybackIndex();
86  }
87  }
88  nextQueryData_ = convertQueryToDataVector(query_,runStartTime_);
89  }
90  if(nextQueryData_[0]->index_.time() <= lookForwardTime)
91  return nextQueryData_[0]->index_;
92  return PlaybackIndex();

95 void AutoDataState::moveToIndex(PlaybackIndex index)
96 {
97  PlaybackIndex curr = getCurrentIndex();
98  if(index == curr)
99  return;
100  Q_ASSERT(index >= curr);
101  PlaybackIndex next = getNextIndex(index.time());
102  while(next.isValid() && next <= index)
103  {
104  if(dataIndex_ < currQueryData_.size()-1)
105  {
106  dataIndex_++;
107  }
108  else
109  {
110  currQueryData_ = nextQueryData_;
111  nextQueryData_.clear();
112  dataIndex_ = 0;
113  }
114  curr = getCurrentIndex();
115  next = getNextIndex(index.time());
116  triggerDataChange(getCurrentValue());
117  }
118 }
