42 void PCMAuralTarget::present()
43 {
44  //Build up a new sounds lookup table and put it in a buffer
45  QSharedPointer<QHash<MixingSample*,QSharedPointer<MixingSample>>> newActiveLookup(new QHash<MixingSample*,QSharedPointer<MixingSample>>());
46  foreach(QSharedPointer<MixingSample> sound, soundsToMix_)
47  {
48  (*newActiveLookup)[sound.data()] = sound;
49  }
50 
51  //Find all sounds that were active before and are no longer active
52  //by removing all active sounds from the activeSoundLookup_ table
53  foreach(MixingSample* soundPtr,activeSoundLookup_->keys())
54  {
55  if(newActiveLookup->contains(soundPtr))
56  activeSoundLookup_->remove(soundPtr);
57  }
58  //We now retrieve all the sounds that were active but are no longer active by checking
59  //the values the just updated activeSoundLookup_ table. All of these sounds should
60  //be set to stop and added to the update list
61  QList<QSharedPointer<MixingSample>> soundsToDeactivate = activeSoundLookup_->values();
62  foreach(QSharedPointer<MixingSample> sound,soundsToDeactivate)
63  {
64  sound->scheduleVolumeChange(0);
65  sound->scheduleStop();
66  }
67  soundsToMix_ << soundsToDeactivate;
68 
69  //replace the old active sounds lookup table with the new one in the buffer
70  activeSoundLookup_ = newActiveLookup;
71 
72  //Go through each "sound to mix" and call update on it. This results in starting all
73  //sounds that weren't playing before and stopping all sounds that were just set to
74  //stop.
75  QSharedPointer<PCMMixingSample> pcmSound;
76  foreach(QSharedPointer<MixingSample> sound,soundsToMix_)
77  {
78  pcmSound = sound.dynamicCast<PCMMixingSample>();
79  Q_ASSERT(pcmSound);
80  pcmSound->update();
81  }
82  soundsToMix_.clear();
83 }
