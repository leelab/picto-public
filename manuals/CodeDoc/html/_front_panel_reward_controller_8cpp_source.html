183  void FrontPanelRewardController::startReward(unsigned int channel, int quantity)
184  {
185  if (!hasDevice_)
186  return;
187 
188  // For the older system Pictobox5 / System10 (series E), the reward enable line is on ao1, and the reward trigger is on a00.If we increase
189  //the NUMREWARDLINES, channel 2 will try to overwrite ao1, which won't work.
190  if (Picto::portNums->getSystemNumber() == 10)
191  NUMREWARDLINES = 1;
192 
193  if (channel > NUMREWARDLINES + 1 || channel < 1)
194  return;
195 
196  if (Picto::portNums->getSystemNumber() == 10)
197  {
198  DAQmxErrChk(DAQmxCreateTask(QString("RewardTask%1").arg(channel).toLatin1(), (TaskHandle*)&daqTaskHandle_[channel]));
199  DAQmxErrChk(DAQmxCreateAOVoltageChan(daqTaskHandle_[channel], QString("Dev1/ao%1").arg(channel - 1).toLatin1(), "", 0.0, 5.0, DAQmx_Val_Volts, NULL));
200  DAQmxErrChk(DAQmxCfgSampClkTiming(daqTaskHandle_[channel], "", 1000.0 / float64(quantity), DAQmx_Val_Rising, DAQmx_Val_FiniteSamps, 2));
201  int32 sampsPerChanWritten;
202  //(remember, we're using active low logic)
203  DAQmxErrChk(DAQmxWriteAnalogF64(daqTaskHandle_[channel], 2, 0, -1, DAQmx_Val_GroupByChannel, outputData, &sampsPerChanWritten, NULL));
204  }
205  else
206  {
207  DAQmxErrChk(DAQmxCreateTask(QString("RewardTaskDig%1").arg(channel).toLatin1(), (TaskHandle*)&daqTaskHandle_[channel]));
208  DAQmxErrChk(DAQmxCreateCOPulseChanTime(daqTaskHandle_[channel], QString("Dev1/ctr%1").arg(channel - 1).toLatin1(), "", DAQmx_Val_Seconds, DAQmx_Val_High, 0.00, float64(quantity) / 1000.0, 0.01));
209  }
210 
211 
212  if (!DAQmxStartTask(daqTaskHandle_[channel]))
213  {
214  //Since this code creates a task that raises the voltage for quantity ms and then lowers it for quantity ms, as it is it
215  //doesn't support inter reward intervals of less than quantity. Since we couldn't find a better way to do this, we just
216  //set our own timer and stop/clear the task anytime that it still exists and more than quantity ms have passed.
217  stopwatch_[channel].startWatch();
218  latestOnTime_[channel] = quantity;
219  }
220  else
221  {
222  DAQmxErrChk(DAQmxStopTask(daqTaskHandle_[channel]));
223  DAQmxErrChk(DAQmxClearTask(daqTaskHandle_[channel]));
224  daqTaskHandle_[channel] = 0;
225  }

232  bool FrontPanelRewardController::rewardWasSupplied(unsigned int channel)
233  {
234  // For the older system Pictobox5 / System10 (series E), the reward enable line is on ao1, and the reward trigger is on a00.If we increase
235  //the NUMREWARDLINES, channel 2 will try to overwrite ao1, which won't work.
236  if (Picto::portNums->getSystemNumber() == 10)
237  NUMREWARDLINES = 1;
238 
239  if (channel > NUMREWARDLINES + 1 || channel < 1)
240  return true;
241  if (!daqTaskHandle_[channel])
242  return true;
243  //The "+.5" below is just to avoid possible weird racing effects where the Nidaq doesn't turn the
244  //signal off. This hasn't ever actually happened, I'm just doing it to be safe.
245  if (stopwatch_[channel].elapsedMs() <= latestOnTime_[channel] + .5)
246  return false;
247  DAQmxErrChk(DAQmxStopTask(daqTaskHandle_[channel]));
248  DAQmxErrChk(DAQmxClearTask(daqTaskHandle_[channel]));
249  daqTaskHandle_[channel] = 0;
250  return true;
