181 void AutoSaver::initAutoSaveDir()
182 {
183  //Create a directory object and position it at the autosave root directory
184  Q_ASSERT(autoSaveDir_.isEmpty());
185  QDir autoSaveRoot(QCoreApplication::applicationDirPath()+"/../");
186  autoSaveRoot.mkdir(AUTOSAVEROOT);
187  if(!autoSaveRoot.cd(AUTOSAVEROOT))
188  return;
189  //Go through auto save directories
190  QStringList subDirectoryNames = autoSaveRoot.entryList(QDir::Dirs|QDir::NoDotAndDotDot|QDir::NoSymLinks,QDir::Time);
191  foreach(QString subDirName,subDirectoryNames)
192  {
193  //Get the files in the subdirectory
194  QDir subDir(autoSaveRoot.filePath(subDirName));
195  QStringList subDirFiles = subDir.entryList(QDir::Files);
196  //If the sub directory is empty, remove it
197  if(subDirFiles.isEmpty())
198  {
199  removeFilesThenDirectories(subDir);
200  subDir.rmdir(".");
201  continue;
202  }
203  //Try to remove any lockfile in the sub directory
204  subDir.remove(LOCKFILENAME);
205 
206  //If the sub directory doesn't contain contain a lock file then
207  //the workstation controlling is no longer using it, so it is
208  //an autosave directory that we should add to our list
209  if(!subDir.exists(LOCKFILENAME))
210  {
211  savedDesignDirs_.push_front(QDir(autoSaveRoot.filePath(subDirName)));
212  }
213  }
214  //We now have a list of all directores that contain auto saved files.
215 
216  //Create a directory that will contain auto saved files for
217  //this application.
218  QString newDirName = QUuid::createUuid().toString();
219  if(!autoSaveRoot.mkdir(newDirName))
220  return;
221  autoSaveDir_ = autoSaveRoot.filePath(newDirName);
222  //Create a lock file in the new directory so that other applications
223  //will know that we're still using this directory
224  QString lockFilePath = QDir(autoSaveDir_).filePath(LOCKFILENAME);
225  lockFile_ = QSharedPointer<QFile>(new QFile(lockFilePath));
226  bool rc = lockFile_->open(QIODevice::WriteOnly | QIODevice::Text);
227  IGNORED_PARAMETER(rc);
228  Q_ASSERT(rc);
229 }

235 void AutoSaver::removeFilesThenDirectories(QDir container)
236 {
237  QStringList files = container.entryList(QDir::Files);
238  foreach(QString fileName,files)
239  {
240  if(!container.remove(fileName))
241  {
242  return;
243  }
244  }
245  QStringList directories = container.entryList(QDir::Dirs|QDir::NoDotAndDotDot|QDir::NoSymLinks);
246  foreach(QString directory,directories)
247  {
248  if(!container.rmdir(directory))
249  {
250  QDir subDir = container;
251  subDir.cd(directory);
252  removeFilesThenDirectories(subDir);
253  container.rmdir(directory);
254  }
255  }

270 void AutoSaver::saveDesignToFile()
271 {
272  //Make sure that there is a designRoot_ to save to file
273  if(!designRoot_)
274  return;
275  //Get the latest design code;
276  QString newDesignCode;
277  if(designRoot_)
278  newDesignCode = designRoot_->getDesignRootText();
279  //If the design code hasn't changed, we're done
280  if(newDesignCode == latestDesignCode_)
281  return;
282  //Update latestDesignCode_
283  latestDesignCode_ = newDesignCode;
284  //If the latestDesignCode_ is empty, remove the designFile if it exists
285  //and we're done
286  if(latestDesignCode_.isEmpty())
287  {
288  if(designFile_)
289  {
290  designFile_->close();
291  designFile_->remove();
292  }
293  return;
294  }
295 
296  //Get the latest design name and build an updated file name from it
297  QString newDesignName;
298  if(designRoot_)
299  newDesignName = designRoot_->getDesignName();
300  QString newFileName = buildDesignFileName(newDesignName);
301  QString newDesignFilePath = QDir(autoSaveDir_).filePath(newFileName);
302 
303  //If there's no design file yet, make one
304  if(!designFile_)
305  {
306  designFile_ = QSharedPointer<QFile>(new QFile(newDesignFilePath));
307  bool rc = designFile_->open(QIODevice::ReadWrite | QIODevice::Text);
308  //If we didn't succesfully open the file, we can't autosave... we're done here.
309  if(!rc)
310  return;
311  }
312 
313  //Update contents of the design file with the latest design code
314  designFile_->resize(0);
315  designFile_->write(latestDesignCode_.toLatin1());
316  designFile_->flush();
317  //If the name of the design root has changed, update the name of
318  //the design file
319  if(designFile_->fileName() != newDesignFilePath)
320  {
321  designFile_->rename(newDesignFilePath); //This closes the file
322  bool rc = designFile_->open(QIODevice::ReadWrite | QIODevice::Text);
323  IGNORED_PARAMETER(rc);
324  Q_ASSERT(rc);
325  }
326 }
