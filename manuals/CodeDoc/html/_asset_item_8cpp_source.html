108 void AssetItem::setSvgIcon(QGraphicsSvgItem* svgIcon)
109 {
110  if(!svgIcon)
111  return;
112  if(lastSvgIcon_ && (lastSvgIcon_ != svgIcon))
113  {
114  lastSvgIcon_->setParent(NULL);
115  delete lastSvgIcon_;
116  }
117  lastSvgIcon_ = svgIcon;
118  svgIcon->setParentItem(this);
119  QRectF boundRect = svgIcon->boundingRect();
120  //Find the scale factor (always scale according to the smaller of width or height scale factors.
121  float widthScale = getIconRect().width()/boundRect.width();
122  float heightScale = getIconRect().height()/boundRect.height();
123  float smallerScale = (widthScale > heightScale)?heightScale:widthScale;
124  svgIcon->setTransform(svgIcon->transform().scale(smallerScale,smallerScale));

126  svgIcon->setPos(getIconRect().center()-QPointF(smallerScale*boundRect.width()/2.0,smallerScale*boundRect.height()/2.0));
127  svgIcon->setZValue(0);
