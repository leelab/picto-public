47 void RunSelectorWidget::addRun(bool saved,QString filePath,QString runName,int index,QString notes)
48 {
49  //Add new buttons to layout
50  if(filePath.isEmpty())
51  return;
52  if(notes.isEmpty())
53  notes = "No saved notes";
54  int lastBackslash = filePath.lastIndexOf("/");
55  int lastDot = filePath.lastIndexOf(".");
56  if(lastDot < 0)
57  lastDot = filePath.size();
58  int numChars = lastDot-(lastBackslash+1);
59  QString fileName = filePath.mid(lastBackslash+1,numChars);
60  if(!currGroupBox_ || currGroupBox_->title() != fileName)
61  {
62  currGroupBox_ = new QGroupBox(fileName);
63  currGroupBox_->setLayout(new QVBoxLayout());
64  qobject_cast<QVBoxLayout*>(mainWidget_->layout())->addWidget(currGroupBox_,1,Qt::AlignTop);
65  if(numSessions_ > 0)
66  qobject_cast<QVBoxLayout*>(mainWidget_->layout())->setStretch(numSessions_-1,0);
67  numSessions_++;
68  }
69  QCheckBox* taskCheckbox(new QCheckBox(runName));
70  taskCheckbox->setToolTip(notes);
71  taskCheckbox->setIcon(saved?QIcon(":/icons/filesave.png"):QIcon("://icons/delete.png"));
72  buttonIdRunLookup_[numRuns_] = QSharedPointer<RunInfo>(new RunInfo(filePath,index,saved,taskCheckbox,notes));
73  fileRunLookup_[filePath][index] = buttonIdRunLookup_[numRuns_];
74  buttonGroup_->addButton(taskCheckbox,numRuns_);
75  qobject_cast<QVBoxLayout*>(currGroupBox_->layout())->addWidget(taskCheckbox,0,Qt::AlignTop);
76  numRuns_++;
77  selectAll_->setVisible(true);
78  selectSaved_->setVisible(true);
79  clearSelection_->setVisible(true);
80  scrollArea_->setWidget(mainWidget_);
81 }

86 void RunSelectorWidget::clear()
87 {
88  //Remove all old buttons from layout. We do this in a roundabout but simple way. Moving a widget's
89  //layout to another widget and then destroying that widget destroys the layout and all the children that
90  //were attached to it.
91  if(mainWidget_->layout())
92  QWidget().setLayout(mainWidget_->layout());
93  mainWidget_->setLayout(new QVBoxLayout());
94  qobject_cast<QVBoxLayout*>(mainWidget_->layout())->setSizeConstraint(QLayout::SetMinAndMaxSize);
95  selectAll_->setVisible(false);
96  selectSaved_->setVisible(false);
97  clearSelection_->setVisible(false);
98  QList<QAbstractButton*> managedButtons = buttonGroup_->buttons();
99  foreach(QAbstractButton* button,managedButtons)
100  {
101  buttonGroup_->removeButton(button);
102  }
103  buttonIdRunLookup_.clear();
104  fileRunLookup_.clear();
105  numRuns_ = 0;
106  numSessions_ = 0;
107  currGroupBox_ = NULL;
108  scrollArea_->setWidget(mainWidget_);
109 }

167 void RunSelectorWidget::setRunStatus(QString fileName,int runIndex,RunStatus status,QString message)
168 {
169  if(!fileRunLookup_.contains(fileName) || !fileRunLookup_[fileName].contains(runIndex))
170  return;
171  fileRunLookup_[fileName][runIndex]->runStatus_ = status;
172  if(!message.isEmpty())
173  fileRunLookup_[fileName][runIndex]->button_->setToolTip(message);
174  switch(status)
175  {
176  case IDLE: //This is never used, but it shouldn't be setting the color to red, it should just clear it... fix this.
177  setRunColor(fileName,runIndex,QColor(255,100,100));
178  break;
179  case INPROGRESS:
180  setRunColor(fileName,runIndex,QColor(100,100,255));
181  break;
182  case COMPLETE:
183  setRunColor(fileName,runIndex,QColor(100,255,100));
184  break;
185  case ERROROCCURED:
186  setRunColor(fileName,runIndex,Qt::red);
187  break;
188  }
