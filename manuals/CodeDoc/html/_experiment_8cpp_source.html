156 bool Experiment::runTask(QString taskName)
157 {
158  if(engine_.isNull())
159  return false;
160  if(tasks_.isEmpty())
161  return false;
162 
163  Q_ASSERT(propTable_);
164 
165  //search through tasks_ for a matching task and run it!
166  // note that the taskname here may have had all of it's whitespace removed, so we need to check that possibility
167  do
168  {
169  QSharedPointer<Task> task = getTaskByName(taskName);
170  if(!task)
171  {
172  return false;
173  }
174  engine_->clearChangedPropertyPackages();
175  engine_->play();
176  //Start out by sending all starting property values to the server
177  engine_->sendAllPropertyValuesToServer();
178 
179  engine_->startAllSignalChannels();
180  //Initialize signal channel coefficients
181  updateSignalCoefficients(NULL,QVariant());
182  resetScriptableValues();
183  //Tell the design config that we are starting a new run
184  getDesignConfig()->markRunStart(taskName+"_"+QDateTime::currentDateTime().toString("yyyy_MM_dd__hh_mm_ss"));
185 
186  QString result = task->run(engine_);
187  engine_->stopAllSignalChannels();
188 
189  //Tell the design config that the run is over
190  getDesignConfig()->markRunEnd();
191 
192  //If the task changed somehow (ie. We are running as slaves)
193  //The prior task would end with a result that is a path within a different task, if that is the case, re-enter at
194  // that task.
195  QStringList pathElems = result.split("::",QString::SkipEmptyParts);
196  if(pathElems.size() <= 1)
197  taskName.clear();
198  else
199  taskName = pathElems[0];
200 
201  } while(!taskName.isEmpty());
202 
203  return true;
204 }

206 void Experiment::postDeserialize()
207 {
208  signalCoeffInitialized_ = false;
209  //designConfig_->disallowIdDuplication();
210  //Usually an object's parent adds itself to the objects scriptables list.
211  //Since an experiment is the highest level ScriptableContainer, we must do it manually.
212  addScriptable(selfPtr().staticCast<Scriptable>());
213  ScriptableContainer::postDeserialize();
214 
215  //Set the signal properties runtime editable
216  //We use the DataStore version of this function so that the actual properties, not the init properties, are the ones
217  // affected by changing the properties from the workstation. If we didn't do this, the init properties would change
218  // which would have no effect the experiment is not part of the state machine's run system and the values are never
219  // copied.
220  DataStore::setPropertyRuntimeEditable("XOffset");
221  DataStore::setPropertyRuntimeEditable("XGain");
222  DataStore::setPropertyRuntimeEditable("YOffset");
223  DataStore::setPropertyRuntimeEditable("YGain");
224  DataStore::setPropertyRuntimeEditable("XYSignalShear");
225 
226  //clear out the existing experiment
227  tasks_.clear();
228  QList<QSharedPointer<Asset>> newTasks = getGeneratedChildren("Task");
229  foreach(QSharedPointer<Asset> newTask,newTasks)
230  {
231  sortTasksIntoList(newTask);
232  }
233  //In case tasks are added after deserialization (ie. in the state machine editor). Attach the childAdded
234  //signal to the sortTasksIntoList slot.
235  connect(this,SIGNAL(childAddedAfterDeserialize(QSharedPointer<Asset>)),this,SLOT(sortTasksIntoList(QSharedPointer<Asset>)));
236 
237  propTable_ = QSharedPointer<PropertyTable>(new PropertyTable(getDesignConfig()));
238  QList<QSharedPointer<Property>> descendantProps = getDescendantsProperties();
239  foreach(QSharedPointer<Property> prop,descendantProps)
240  {
241  //This adds the property to the property table and gives it an index for use in transmission
242  propTable_->addProperty(prop);
243  }
244 
245  //Experiment objects are not part of the state machine. Init values should be bypassed and set immediately as
246  // run values in all properties
247  QHash<QString, QVector<QSharedPointer<Property>>> propMap = propertyContainer_->getProperties();
248  foreach(QVector<QSharedPointer<Property>> propVec, propMap)
249  {
250  foreach(QSharedPointer<Property> prop, propVec)
251  {
252  prop->enableInitRunValueSync(true);
253  }
254  }
255  QString hostIdStr = getHostId().toString();
256  ASSOCIATE_ROOT_HOST_POST_DESERIALIZE
257 }

270 void Experiment::updateSignalCoefficients(Property*,QVariant)
271 {
272  if(engine_.isNull())
273  return;
274  QSharedPointer<DoubletSignalChannel> posChannel = engine_->getSignalChannel("Position").dynamicCast<DoubletSignalChannel>();
275  QRect windowDims = engine_->getRenderingTargets().first()->getVisualTarget()->getDimensions();
276  int displayWidth = RESOLUTIONH;
277  int displayHeight = RESOLUTIONV;
278 
279  if(!signalCoeffInitialized_)
280  {
281  signalCoeffInitialized_ = true;
282  //If we're using a mouse channel, ignore the saved signal parameters since we know how to set them up automatically.
283  //We will still allow these values to be adjusted by the user later though for testing purposes.
284  bool isMouseChannel = posChannel->getPortType() == "mouse";
285  if(isMouseChannel)
286  { QRect windowDims = engine_->getRenderingTargets().first()->getVisualTarget()->getDimensions();
287  propertyContainer_->setPropertyValue("XOffset",-(displayWidth/2));
288  propertyContainer_->setPropertyValue("XGain",1.0);
289  propertyContainer_->setPropertyValue("YOffset",-(displayHeight/2));
290  propertyContainer_->setPropertyValue("YGain",1.0);
291  propertyContainer_->setPropertyValue("XYSignalShear",0.0);
292  }
293  //connect the signal properties to the updateSignalCoefficients function
294  connect(propertyContainer_->getProperty("XOffset").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
295  connect(propertyContainer_->getProperty("XGain").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
296  connect(propertyContainer_->getProperty("YOffset").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
297  connect(propertyContainer_->getProperty("YGain").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
298  connect(propertyContainer_->getProperty("XYSignalShear").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
299  }
300  double xZoom = propertyContainer_->getPropertyValue("XGain").toDouble();
301  double yZoom = propertyContainer_->getPropertyValue("YGain").toDouble();
302  double xGain = xZoom;
303  double yGain = yZoom;
304  int xOffset = displayWidth/2 + propertyContainer_->getPropertyValue("XOffset").toDouble()*xZoom;
305  int yOffset = displayHeight/2 + propertyContainer_->getPropertyValue("YOffset").toDouble()*yZoom;
306 
307  posChannel->setCalibrationCoefficients("x",xGain,xOffset,displayWidth/2);
308  posChannel->setCalibrationCoefficients("y",yGain,yOffset,displayHeight/2);
309  posChannel->setShear("x","y",propertyContainer_->getPropertyValue("XYSignalShear").toDouble());
310 
311 }
