302 void Experiment::updateSignalCoefficients(Property*,QVariant)
303 {
304  if(engine_.isNull())
305  return;
306  QSharedPointer<SignalChannel> posChannel = engine_->getSignalChannel("Position");
307  QRect windowDims = engine_->getRenderingTargets().first()->getVisualTarget()->getDimensions();
308  int displayWidth = 800;
309  int displayHeight = 600;
310 
311  if(!signalCoeffInitialized_)
312  {
313  signalCoeffInitialized_ = true;
314  //If we're using a mouse channel, ignore the saved signal parameters since we know how to set them up automatically.
315  //We will still allow these values to be adjusted by the user later though for testing purposes.
316  bool isMouseChannel = posChannel->getPortType() == "mouse";
317  if(isMouseChannel)
318  { QRect windowDims = engine_->getRenderingTargets().first()->getVisualTarget()->getDimensions();
319  propertyContainer_->setPropertyValue("XOffset",-(displayWidth/2));
320  propertyContainer_->setPropertyValue("XGain",1.0);
321  propertyContainer_->setPropertyValue("YOffset",-(displayHeight/2));
322  propertyContainer_->setPropertyValue("YGain",1.0);
323  propertyContainer_->setPropertyValue("XYSignalShear",0.0);
324  }
325  //connect the signal properties to the updateSignalCoefficients function
326  connect(propertyContainer_->getProperty("XOffset").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
327  connect(propertyContainer_->getProperty("XGain").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
328  connect(propertyContainer_->getProperty("YOffset").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
329  connect(propertyContainer_->getProperty("YGain").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
330  connect(propertyContainer_->getProperty("XYSignalShear").data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(updateSignalCoefficients(Property*,QVariant)));
331  }
332  double xZoom = propertyContainer_->getPropertyValue("XGain").toDouble();
333  double yZoom = propertyContainer_->getPropertyValue("YGain").toDouble();
334  double xGain = xZoom;
335  double yGain = yZoom;
336  int xOffset = displayWidth/2 + propertyContainer_->getPropertyValue("XOffset").toDouble()*xZoom;
337  int yOffset = displayHeight/2 + propertyContainer_->getPropertyValue("YOffset").toDouble()*yZoom;
338 
339  posChannel->setCalibrationCoefficients("x",xGain,xOffset,displayWidth/2);
340  posChannel->setCalibrationCoefficients("y",yGain,yOffset,displayHeight/2);
341  posChannel->setShear("x","y",propertyContainer_->getPropertyValue("XYSignalShear").toDouble());
342 
343 }
