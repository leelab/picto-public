93 void PortNums::updatePortNumbers()
94 {
95  QSettings settings((usedInSystemService_?QSettings::SystemScope:QSettings::UserScope),"Block Designs",appName_);
96  if(!settings.contains("SystemNum"))
97  {
98  settings.setValue("SystemNum",1);
99  }
100  else
101  systemNumber_ = settings.value("SystemNum").toInt();
102  if(systemNumber_ > 10 || systemNumber_ < 1)
103  systemNumber_ = 1;
104 
105  if(systemNumber_ > 1)
106  {
107  int referenceLoc = 52020+(100*systemNumber_);

109  minDiscoverPort_ = referenceLoc+5;

111  serverPort_ = referenceLoc+4;
