238 bool LfpState::loadData()
239 {
240  resetDataVariables();
241  emit lfpLoadProgress(0);
242  query_ = QSharedPointer<QSqlQuery>(new QSqlQuery(session_));
243 
244  //Get timestamp alignment constants
245  query_->prepare("SELECT value FROM sessioninfo WHERE key=\"AlignmentInfo\"");
246  if(!query_->exec() || !query_->next())
247  {
248  return true;
249  }
250  AlignmentInfo inf;
251  inf.fromXml(query_->value(0).toString());
252  double offsetTime = inf.getOffsetTime();
253  double temporalFactor = inf.getTemporalFactor();
254 
255 
256  //Set up lfp hash
257  query_->prepare("SELECT DISTINCT channel FROM lfp ORDER BY channel");
258  if(!query_->exec() || !query_->next())
259  {
260  return true;
261  }
262  int index = 0;
263  do
264  {
265  chanIndexMap_[query_->value(0).toInt()] = index++;
266  channels_.append(query_->value(0).toInt());
267  } while(query_->next());
268  numChannels_ = channels_.size();
269 
270  //Get lfp resolution
271  sampPeriod_ = 0;
272  query_->prepare("SELECT value FROM sessioninfo WHERE key='DataSource'");
273  if(!query_->exec())
274  {
275  return false;
276  }
277  while(query_->next())
278  {
279  DataSourceInfo info;
280  info.fromXml(query_->value(0).toString());
281  if(info.getName() == "lfp")
282  {
283  sampPeriod_ = temporalFactor*info.getResolution();
284  break;
285  }
286  }
287  if(sampPeriod_ <= 0)
288  {
289  return false;
290  }
291 
292  //Get minimum lfp time
293  query_->prepare("SELECT timestamp from lfp ORDER BY timestamp LIMIT 1");
294  if(!query_->exec() || !query_->next())
295  {
296  return true;
297  }
298  minLfpTime_ = offsetTime + temporalFactor*query_->value(0).toDouble();
299 
300  //Get maximum lfp time
301  //Start by getting highest timestamp entries for all channels. We could use distinct to make sure that we're
302  //actually getting the highest value from each channel. In practice; however, the DISTINCT keyword makes
303  //things take a lot longer in this call and unless the neural proxy is doing something really weird if we
304  //just select the last numberOfChannels timestamp entries whether or not the channels are distinct, we will get
305  //the highest time value in the experiment or certainly it will be close enough to the end of the experiment as
306  //to be past the end of the last run. The worst case here is that we miss one lfp data packet time length at
307  //the end of the session. Currently these packets are 500ms long and as mentioned before, this worst case will
308  //really never come up unless something really wierd happens with the proxy.
309  query_->prepare("SELECT timestamp,data FROM lfp ORDER BY timestamp DESC LIMIT :channels");
310  query_->bindValue(":channels",numChannels_);
311  if(!query_->exec())
312  {
313  return false;
314  }
315  double maxTime = minLfpTime_-1;
316  while(query_->next())
317  {
318  double currHighest = offsetTime+(temporalFactor*query_->value(0).toDouble()) + (sampPeriod_ * query_->value(1).toByteArray().size()/sizeof(float));
319  if(currHighest > maxTime)
320  maxTime = currHighest;
321  }
322  Q_ASSERT(maxTime >= minLfpTime_);
323 
324  emit lfpLoadProgress(10);
325 
326  //Create data list with an entry for every sample time from minTime to maxTime
327  numValues_ = arrayIndexFromGlobalTime(maxTime)+1;
328  try{
329  data_.reserve(numChannels_);
330  for(int i=0;i<numChannels_;i++)
331  {
332  data_.append(QList<float>());
333  data_[i].reserve(numValues_); //We use reserve and not resize so that we can be sure that we're allocating exactly the right amount of memory
334  //tests indicate that resize actually may reserve more than the requested amount
335  }
336  } catch(...)
337  {
338  resetDataVariables();
339  throw;
340  }
341  for(int i=0;i<numChannels_;i++)
342  {
343  QList<float> chanList;
344  for(int j=0;j<numValues_;j++)
345  chanList.append(0);
346  data_[i]= chanList;
347  }
348 
349  emit lfpLoadProgress(20);
350  //Go through each channel in database and fill data list
351  int channelNum = 0;
352  foreach(int channel,chanIndexMap_.keys())
353  {
354  channelNum++;
355  int valueIndex = chanIndexMap_[channel];
356  //Get data and load it into this channel's list from the lfp hash
357  query_->prepare("SELECT timestamp,data FROM lfp WHERE channel=:channel "
358  "ORDER BY timestamp");
359  query_->bindValue(":channel",channel);
360  if(!query_->exec())
361  {
362  Q_ASSERT(false);
363  return false;
364  }
365  int catchUpIndex = 0;
366  double startTime;
367  double currTime;
368  float latestValue = 0;
369  int indexForCurrTime = -1;
370  while(query_->next())
371  {
372  startTime = offsetTime + temporalFactor*query_->value(0).toDouble();
373  QByteArray dataArray = query_->value(1).toByteArray();
374  const float* dataArrayStart = reinterpret_cast<const float*>(dataArray.constData());
375  int dataArraySize = dataArray.size()/sizeof(float);
376  for(int i=0;i<dataArraySize;i++)
377  {
378  currTime = startTime + i*sampPeriod_;
379  indexForCurrTime = arrayIndexFromGlobalTime(currTime);
380  if(indexForCurrTime >= numValues_)
381  break; //Since we sped things up above by not necessarily checking all channels before making a highest time decision
382  //there is a very unlikely possiblity that there is more data available than will fit in the vector we made
383  //if this is the case we just throw the data out because in any case we will be very close to the end of the
384  //session and the data will almost certainly come from after the significant runs were finished.
385  while(catchUpIndex <= arrayIndexFromGlobalTime(currTime))
386  {
387  setDataValue(catchUpIndex,valueIndex,latestValue);
388  catchUpIndex++;
389  }
390  latestValue = *(dataArrayStart + i);
391  setDataValue(arrayIndexFromGlobalTime(currTime),valueIndex,latestValue);
392  }
393  }
394  emit lfpLoadProgress(20+(80.0*double(channelNum)/double(chanIndexMap_.keys().size())));
395  }
396  return true;
397 }
