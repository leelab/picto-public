38 bool AssociateRoot::LinkToAsset(QSharedPointer<Asset> asset, QString& feedback)
39 {
40  if(asset == linkedAsset_.toStrongRef())
41  return true;
42 
43  feedback = "";
44  AssociateElement* associateElement;
45  bool linkedAssetMatches = isLinkableAsset(asset);
46  QStringList pathMatchedAssociateRootAssets;
47  QStringList unmatchedAssociateRootAssets;
48  //Set this asset as our linked asset
49  setLinkedAsset(asset);
50  foreach(QString childTag,getValidChildTags())
51  {
52  foreach(QSharedPointer<Asset> child,getGeneratedChildren(childTag))
53  {
54  associateElement = dynamic_cast<AssociateElement*>(child.data());
55  if(!associateElement || child->isDeleted())
56  continue;
57 
58  int result = associateElement->attachToLinkedAsset(linkedAssetMatches); //Linked Asset needs to be stored before we call this, since associateElement gets it from us.
59  switch(result)
60  {
61  case AssociateElement::SUCCESS_BY_LINK:
62  break;
63  case AssociateElement::SUCCESS_BY_PATH:
64  pathMatchedAssociateRootAssets.append(associateElement->getInfoString());
65  break;
66  case AssociateElement::FAIL:
67  unmatchedAssociateRootAssets.append(associateElement->getInfoString());
68  child->setDeleted();
69  break;
70  }
71  }
72  }
73  if(pathMatchedAssociateRootAssets.size() || unmatchedAssociateRootAssets.size())
74  {
75  QString linkedAssetType = asset->identifier();
76  if(linkedAssetMatches)
77  feedback = QString("The %1 or %2 design appears to have been altered outside of the Picto Development Environment. ").arg(linkedAssetType).arg(friendlyTypeName());
78  else
79  feedback = QString("This %1 was built for a %2 design that does not appear in this experiment. ").arg(friendlyTypeName()).arg(linkedAssetType);
80  if(unmatchedAssociateRootAssets.isEmpty())
81  {
82  feedback += QString("All elements of the %1 were succesfully attached to this experiment's %2 %3.").arg(friendlyTypeName()).arg(asset->getName()).arg(linkedAssetType);
83  }
84  else
85  {
86  feedback += QString("Experimental element links could not be found for the following %1 elements which have been removed:\n").arg(friendlyTypeName())
87  + unmatchedAssociateRootAssets.join("\n") + "\n";
88  }
89  if(linkedAssetMatches && pathMatchedAssociateRootAssets.size())
90  {
91  feedback += QString("Imperfect matches were detected for the following %1 elements:\n").arg(friendlyTypeName()) + unmatchedAssociateRootAssets.join("\n") + "\n";
92  }
93 
94  }
95 
96  if(feedback.size())
97  return false;
98  return true;
99 }

140 void AssociateRoot::setLinkedAsset(QSharedPointer<Asset> asset)
141 {
142  Q_ASSERT(asset);
143  if(asset == linkedAsset_.toStrongRef())
144  return;
145  //If this AssociateRoot was already linked to an asset, disconnect that asset's Id and Name property signals
146  if(!linkedAsset_.isNull())
147  {
148  QSharedPointer<Property> linkedIdProp = getLinkedAssetHostIdProperty();
149  Q_ASSERT(linkedIdProp);
150  disconnect(linkedIdProp.data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(linkedAssetPropertyEdited(Property*,QVariant)));
151  QSharedPointer<Property> linkedNameProp = getLinkedHostNameProperty();
152  if(linkedNameProp)
153  disconnect(linkedNameProp.data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(linkedAssetPropertyEdited(Property*,QVariant)));
154  }
155  linkedAsset_ = asset;
156  AssociateRootHost* assocRootHost = dynamic_cast<AssociateRootHost*>(linkedAsset_.data());
157  Q_ASSERT(assocRootHost);
158  assocRootHost->setAssociateRoot(identifier(),selfPtr().staticCast<AssociateRoot>());
159  updateLinkedAssetProperties();
160 
161  //Connect to change signals from linked assets's Id and Name properties, so that we can update when
162  //they change
163  QSharedPointer<Property> linkedIdProp = getLinkedAssetHostIdProperty();
164  Q_ASSERT(linkedIdProp);
165  connect(linkedIdProp.data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(linkedAssetPropertyEdited(Property*,QVariant)));
166  QSharedPointer<Property> linkedNameProp = getLinkedHostNameProperty();
167  if(linkedNameProp)
168  connect(linkedNameProp.data(),SIGNAL(valueChanged(Property*,QVariant)),this,SLOT(linkedAssetPropertyEdited(Property*,QVariant)));
169 }
