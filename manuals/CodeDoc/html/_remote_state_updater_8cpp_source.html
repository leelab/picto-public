42 bool RemoteStateUpdater::updateState()
43 {
44  //Clear any old responses out of the response queue
45  while(serverChan_->incomingResponsesWaiting())
46  serverChan_->getResponse();
47  if(!serverChan_->assureConnection(100))
48  return false;
49 
50  //Collect the data from the server
51  QString commandStr = QString("GETDATA CurrentState:%1 PICTO/1.0").arg(lastTimeStateDataRequested_);
52  QSharedPointer<Picto::ProtocolCommand> command(new Picto::ProtocolCommand(commandStr));
53  QSharedPointer<Picto::ProtocolResponse> response;
54 
55  serverChan_->sendRegisteredCommand(command);
56  QString commandID = command->getFieldValue("Command-ID");
57  //qDebug(QString("Sent command: %1 at Time:%2").arg(commandID).arg(command->getFieldValue("Time-Sent")).toLatin1());
58 
59  do
60  {
61  QCoreApplication::processEvents();
62  if(!serverChan_->waitForResponse(1000))
63  {
64  return false;
65  }
66  response = serverChan_->getResponse();
67  }while(!response || response->getFieldValue("Command-ID") != commandID);
68 
69  //Response not 200:OK
70  if(response->getResponseCode() != Picto::ProtocolResponseType::OK)
71  return false;
72 
73  QByteArray xmlFragment = response->getContent();
74  QSharedPointer<QXmlStreamReader> xmlReader(new QXmlStreamReader(xmlFragment));
75 
76  while(!xmlReader->atEnd() && xmlReader->readNext() && xmlReader->name() != "Data");
77 
78  if(xmlReader->atEnd())
79  return false;
80 
81  xmlReader->readNext();
82 
83  while(!xmlReader->isEndElement() && xmlReader->name() != "Data" && !xmlReader->atEnd())
84  {
85  QString name = xmlReader->name().toString();
86  if (xmlHandlers_.contains(name))
87  {
88  xmlHandlers_[name]->handle(xmlReader);
89  }
90  else
91  {
92  qDebug() << "Unknown Data Packet of Name: " << name;
93  }
94  xmlReader->readNext();
95  }
96 
97  if (sendProcessQueueSignal_)
98  {
99  sendProcessQueueSignal_ = false;
100 
101  emit processQueue();
102  }
103 
104  return true;
105 }
