93 void Analysis::upgradeVersion(QString deserializedVersion)
94 {
95  AssociateRoot::upgradeVersion(deserializedVersion);
96  if(deserializedVersion < "0.0.2")
97  {
98  //Before 0.0.2, a copy paste error led to all data sources being saved with the AnalysisScriptContainer tag
99  //For this reason we changed the AnalysisScriptContainer name to AnalysisScriptHolder and in the upgradeVersion function we are recreating all
100  //AnalysisDataSource and AnalysisScriptHolder objects with the correct tag names.
101  QList<QSharedPointer<Asset>> childList = getGeneratedChildren("AnalysisScriptContainer");
102 
103  //Go through everything that was saved with an AnalysisScriptContainer tag.
104  QSharedPointer<ObsoleteNameAsset> obsNameChild;
105  foreach(QSharedPointer<Asset> child,childList)
106  {
107  obsNameChild = child.staticCast<ObsoleteNameAsset>();
108 
109  //If the type was empty, change the name to AnalysisScriptHolder
110  if(obsNameChild->getOriginalType().isEmpty())
111  obsNameChild->setNewTagName("AnalysisScriptHolder");
112  else //If there is a type, change the name to AnalysisDataSource
113  obsNameChild->setNewTagName("AnalysisDataSource");
114  }
115  }
116 }
