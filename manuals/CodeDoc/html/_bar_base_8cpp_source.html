58 void BarBase::replot()
59 {
60  QVector<QwtIntervalSample> qvSamples;
61  QVector<QwtIntervalSample> qvError;
62 
63  const bool bNormalized = getNormalized();
64  const double dBinSize = getBinSize();
65 
66  _handleErrorInitial(qvError);
67 
68  if (m_qhdCumulValue.isEmpty())
69  {
70  m_pHistoPlotItem->setSamples(new QwtIntervalSeriesData(qvSamples));
71 
72  if (bNormalized)
73  {
74  createNormalizedScale(1.0, 1.0);
75  }
76 
77  m_pPlot->replot();
78  return;
79  }
80 
81  QList<long> keyList = m_qhdCumulValue.keys();
82  std::sort(keyList.begin(), keyList.end());
83 
84  double total = 0.0;
85  double dRangeMax = 0.0;
86 
87  foreach(long key, keyList)
88  {
89  const double dSample = _getSampleValue(key);
90  const double dBinWidth = 0.5 / (getBinSpacing() + 1.0);
91  const double dMin = (key - dBinWidth) * dBinSize;
92  const double dMax = (key + dBinWidth) * dBinSize;
93  qvSamples.push_back(std::move(QwtIntervalSample(dSample, dMin, dMax)));
94 
95  if (bNormalized)
96  {
97  if (dSample > dRangeMax)
98  {
99  dRangeMax = dSample;
100  }
101 
102  total += dSample;
103  }
104 
105  _handleErrorValue(key, dRangeMax, qvError);
106  }
107 
108  m_pHistoPlotItem->setSamples(new QwtIntervalSeriesData(qvSamples));
109 
110  _handleErrorFinal(qvError);
111 
112  if (bNormalized)
113  {
114  createNormalizedScale(dRangeMax, total);
115  }
116 
117  if (m_bBinsModified)
118  {
119  m_bBinsModified = false;
120  m_pPlot->axisScaleEngine(QwtPlot::xBottom)->setMargins(0.5*dBinSize, 0.5*dBinSize);
121  handleXLabels(*keyList.begin(), *(keyList.end()-1));
122  }
123 
124  if (m_bUpdateBrush)
125  {
126  m_bUpdateBrush = false;
127  updateColors();
128  updateColumns();
129  }
130 
131  m_pPlot->replot();
132 }

156 void BarBase::createNormalizedScale(double dMaxValue, double dTotalValue)
157 {
158  double dRangeMax = dMaxValue / dTotalValue;
159 
160  double dAxisMax = 1.0;
161  double dTestRange = dAxisMax / 2.0;
162 
163  if (dRangeMax > 0.0)
164  {
165  while (dRangeMax <= dTestRange)
166  {
167  dAxisMax = dTestRange;
168  dTestRange /= 2.0;
169  }
170  }
171 
172  //Change the y-axis labeling to give the appearance of normalization.
173  QList<double> majorTicks;
174  QList<double> mediumTicks;
175  QList<double> minorTicks;
176 
177  int numOfDivs = 2;
178  for (int i = 0; i < numOfDivs + 1; i++)
179  {
180  majorTicks += (i / double(numOfDivs))*dTotalValue*dAxisMax;
181  }
182 
183  numOfDivs = 10;
184  for (int i = 0; i < numOfDivs + 1; i++)
185  {
186  mediumTicks += (i / double(numOfDivs))*dTotalValue*dAxisMax;
187  }
188 
189 
190  QwtScaleDiv scaleDiv(0, dAxisMax*dTotalValue, minorTicks, mediumTicks, majorTicks);
191 
192  QwtScaleDraw *scaleDraw = new NormalizedScaleDraw(dTotalValue);
193 
194  m_pPlot->setAxisScaleDiv(QwtPlot::yLeft, scaleDiv);
195  m_pPlot->setAxisScaleDraw(QwtPlot::yLeft, scaleDraw);
196 }

238 void BarBase::updateColumns()
239 {
240  const ColumnType::ColumnType eColumnType = getColumnType();
241  QPen pen(Qt::black, 0);
242  QwtColumnSymbol *symbol = nullptr;
243 
244  switch (eColumnType)
245  {
246  case ColumnType::COLUMN_OUTLINE:
247  m_pHistoPlotItem->setStyle(QwtPlotHistogram::Outline);
248  m_pHistoPlotItem->setSymbol(nullptr);
249  m_pHistoPlotItem->setPen(pen);
250  break;
251  case ColumnType::COLUMN_FLAT_COLUMN:
252  m_pHistoPlotItem->setStyle(QwtPlotHistogram::Columns);
253  m_pHistoPlotItem->setSymbol(nullptr);
254  m_pHistoPlotItem->setPen(pen);
255  break;
256  case ColumnType::COLUMN_RAISED_COLUMN:
257  m_pHistoPlotItem->setStyle(QwtPlotHistogram::Columns);
258 
259  symbol = new QwtColumnSymbol(QwtColumnSymbol::Box);
260  symbol->setFrameStyle(QwtColumnSymbol::Raised);
261  symbol->setLineWidth(2);
262  symbol->setPalette(QPalette(m_pHistoPlotItem->brush().color()));
263 
264  m_pHistoPlotItem->setSymbol(symbol);
265  break;
266  case ColumnType::COLUMN_LINES:
267  m_pHistoPlotItem->setStyle(QwtPlotHistogram::Lines);
268  m_pHistoPlotItem->setSymbol(nullptr);
269  pen.setBrush(m_pHistoPlotItem->brush());
270  pen.setWidth(2);
271  m_pHistoPlotItem->setPen(pen);
272  break;
273  default:
274  break;
275  }
