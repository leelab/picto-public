67 void BarBase::replot()
68 {
69  QVector<QwtIntervalSample> qvSamples;
70  QVector<QwtIntervalSample> qvError;
71 
72  const bool bNormalized = getNormalized();
73  const double dBinSize = getBinSize();
74 
75  _handleErrorInitial(qvError);
76 
77  if (m_qhdCumulValue.isEmpty())
78  {
79  m_pHistoPlotItem->setSamples(new QwtIntervalSeriesData(qvSamples));
80 
81  if (bNormalized)
82  {
83  createNormalizedScale(1.0, 1.0);
84  }
85 
86  m_pPlot->replot();
87  return;
88  }
89 
90  QList<long> keyList = m_qhdCumulValue.keys();
91  std::sort(keyList.begin(), keyList.end());
92 
93 
94  double total = 0.0;
95  double dRangeMax = 0.0;
96 
97  foreach(long key, keyList)
98  {
99  const double dSample = _getSampleValue(key);
100  const double dBinWidth = 0.5 / (getBinSpacing() + 1.0);
101  const double dMin = (key - dBinWidth) * dBinSize;
102  const double dMax = (key + dBinWidth) * dBinSize;
103  qvSamples.push_back(std::move(QwtIntervalSample(dSample, dMin, dMax)));
104 
105  if (bNormalized)
106  {
107  if (dSample > dRangeMax)
108  {
109  dRangeMax = dSample;
110  }
111 
112  total += dSample;
113  }
114 
115  _handleErrorValue(key, dRangeMax, qvError);
116  }
117 
118  m_pHistoPlotItem->setSamples(new QwtIntervalSeriesData(qvSamples));
119 
120  _handleErrorFinal(qvError);
121 
122  if (bNormalized)
123  {
124  createNormalizedScale(dRangeMax, total);
125  }
126 
127  if (m_bBinsModified)
128  {
129  m_bBinsModified = false;
130  m_pPlot->axisScaleEngine(QwtPlot::xBottom)->setMargins(0.5*dBinSize, 0.5*dBinSize);
131  handleXLabels(*keyList.begin(), *(keyList.end()-1));
132  }
133 
134  m_pPlot->replot();
135 }

145 void BarBase::createNormalizedScale(double dMaxValue, double dTotalValue)
146 {
147  double dRangeMax = dMaxValue / dTotalValue;
148 
149  double dAxisMax = 1.0;
150  double dTestRange = dAxisMax / 2.0;
151 
152  if (dRangeMax > 0.0)
153  {
154  while (dRangeMax <= dTestRange)
155  {
156  dAxisMax = dTestRange;
157  dTestRange /= 2.0;
158  }
159  }
160 
161  //Change the y-axis labeling to give the appearance of normalization.
162  QList<double> majorTicks;
163  QList<double> mediumTicks;
164  QList<double> minorTicks;
165 
166  int numOfDivs = 2;
167  for (int i = 0; i < numOfDivs + 1; i++)
168  {
169  majorTicks += (i / double(numOfDivs))*dTotalValue*dAxisMax;
170  }
171 
172  numOfDivs = 10;
173  for (int i = 0; i < numOfDivs + 1; i++)
174  {
175  mediumTicks += (i / double(numOfDivs))*dTotalValue*dAxisMax;
176  }
177 
178 
179  QwtScaleDiv scaleDiv(0, dAxisMax*dTotalValue, minorTicks, mediumTicks, majorTicks);
180 
181  QwtScaleDraw *scaleDraw = new NormalizedScaleDraw(dTotalValue);
182 
183  m_pPlot->setAxisScaleDiv(QwtPlot::yLeft, scaleDiv);
184  m_pPlot->setAxisScaleDraw(QwtPlot::yLeft, scaleDraw);
185 }
