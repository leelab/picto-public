58 {
59  const QString oldCurrentSet = currentSetName_;

62  QVector<QwtIntervalSample> qvSamples;

67  const double dBinSize = getBinSize();
68  const double dBinWidth = 0.5 / (getBinSpacing() + 1.0);
69 
70  _handleErrorInitial(qvError);
71 
72  if (m_qhCumulValue[setName].isEmpty())
73  {
74  //Branch point accumulating based on ColumnType
75  if (getColumnType() == ColumnType::COLUMN_CURVE ||

78  emit setPointsSig(setName, qvPoints);

93  currentSetName_ = oldCurrentSet;
94  return;
95  }
96 
97  QList<long> keyList = m_qhCumulValue[setName].keys();
98  std::sort(keyList.begin(), keyList.end());
99 
100  double total = 0.0;
101  double dRangeMax = 0.0;
102 
103  //Branch point accumulating based on ColumnType
104  if (getColumnType() == ColumnType::COLUMN_CURVE ||
105  getColumnType() == ColumnType::COLUMN_FILLED_CURVE)
106  {
107  foreach(long key, keyList)
108  {
109  const double dSample = _getSampleValue(key);
110  const double dValue = m_bBinOffset ? (key + dBinWidth) * dBinSize : key * dBinSize;
111  qvPoints.push_back(std::move(QPointF(dValue, dSample)));
112 
113  if (bNormalized)
114  {
115  if (dSample > dRangeMax)
116  {
117  dRangeMax = dSample;
118  }
119 
120  total += dSample;
121  }
122 
123  _handleErrorValue(key, dRangeMax, qvError);
124  }
125 
126  emit setPointsSig(setName, qvPoints);
127  }
128  else
129  {
130 
131  foreach(long key, keyList)
132  {
133  const double dSample = _getSampleValue(key);
134  const double dMin = m_bBinOffset ? key * dBinSize : (key - dBinWidth) * dBinSize;
135  const double dMax = m_bBinOffset ? (key + 2 * dBinWidth) * dBinSize : (key + dBinWidth) * dBinSize;
136  qvSamples.push_back(std::move(QwtIntervalSample(dSample, dMin, dMax)));
137

152  }

157  if (bNormalized)

159  createNormalizedScale(dRangeMax, total);
160  }
161

163  if (m_qhBinsModified[setName] || m_bTotalBinRecalc)
164  {
165  m_qhBinsModified[setName] = false;
166  if (*keyList.begin() < m_lCachedXMin)

175  }

390 void BarBase::_createSet(const QString &setName)
391 {
392  m_qhCumulValue[setName] = QHash<long, double>();
393  m_qhBinsModified[setName] = true;
394 
395  if (!m_qhColor.contains(setName))
396  {
397  char myColorNum = m_cUnasignedNum++;
398  // Scramble up the digits in a periodic way
399  myColorNum = ((myColorNum & 0x0F) << 4) | ((myColorNum & 0xF0) >> 4);
400  myColorNum = ((myColorNum & 0x33) << 2) | ((myColorNum & 0xCC) >> 2);
401  myColorNum = ((myColorNum & 0x55) << 1) | ((myColorNum & 0xAA) >> 1);
402 
403  int red = char(sin(0.0245 * myColorNum) * 127) + 128;
404  int green = char(sin(0.0245 * myColorNum + 2) * 127) + 128;
405  int blue = char(sin(0.0245 * myColorNum + 4) * 127) + 128;
406  int alpha = 255;
407 
408  switch (getColumnType())
409  {
410  case ColumnType::COLUMN_FLAT_COLUMN:
411  case ColumnType::COLUMN_RAISED_COLUMN:
412  case ColumnType::COLUMN_FILLED_CURVE:
413  alpha = 160;
414  break;
415  case ColumnType::COLUMN_OUTLINE:
416  case ColumnType::COLUMN_LINES:
417  case ColumnType::COLUMN_CURVE:
418  alpha = 255;
419  break;
420  default:
421  break;
422  }
423 
424  m_qhColor[setName] = QColor(red, green, blue, alpha);
425  }
426  m_qhUpdateBrush[currentSetName_] = true;
427 
428 }
