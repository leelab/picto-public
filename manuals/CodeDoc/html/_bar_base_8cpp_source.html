29 void BarBase::replot()
30 {
31  QVector<QwtIntervalSample> qvSamples;
32  QVector<QwtIntervalSample> qvError;
33 
34  const bool bNormalized = getNormalized();
35  const double dBinSize = getBinSize();
36 
37  _handleErrorInitial(qvError);
38 
39  if (m_qhdCumulValue.isEmpty())
40  {
41  emit setSamplesSig(qvSamples);
42  _handleErrorFinal(qvError);
43 
44  if (bNormalized)
45  {
46  createNormalizedScale(1.0, 1.0);
47  }
48 
49  emit callReplot();
50  return;
51  }
52 
53  QList<long> keyList = m_qhdCumulValue.keys();
54  std::sort(keyList.begin(), keyList.end());
55 
56  double total = 0.0;
57  double dRangeMax = 0.0;
58 
59  foreach(long key, keyList)
60  {
61  const double dSample = _getSampleValue(key);
62  const double dBinWidth = 0.5 / (getBinSpacing() + 1.0);
63  const double dMin = (key - dBinWidth) * dBinSize;
64  const double dMax = (key + dBinWidth) * dBinSize;
65  qvSamples.push_back(std::move(QwtIntervalSample(dSample, dMin, dMax)));
66 
67  if (bNormalized)
68  {
69  if (dSample > dRangeMax)
70  {
71  dRangeMax = dSample;
72  }
73 
74  total += dSample;
75  }
76 
77  _handleErrorValue(key, dRangeMax, qvError);
78  }
79 
80  emit setSamplesSig(qvSamples);
81 
82  _handleErrorFinal(qvError);
83 
84  if (bNormalized)
85  {
86  createNormalizedScale(dRangeMax, total);
87  }
88 
89  if (m_bBinsModified)
90  {
91  m_bBinsModified = false;
92  emit scaleAxisSig(dBinSize);
93  emit handleXLabelsSig(*keyList.begin(), *(keyList.end() - 1));
94  }
95 
96  if (m_bUpdateBrush)
97  {
98  m_bUpdateBrush = false;
99  emit updateColumnsSig(getColor(), getColumnType());
100  }
101 
102  emit callReplot();
103 }

127 void BarBase::createNormalizedScale(double dMaxValue, double dTotalValue)
128 {
129  double dRangeMax = dMaxValue / dTotalValue;
130 
131  double dAxisMax = 1.0;
132  double dTestRange = dAxisMax / 2.0;
133 
134  if (dRangeMax > 0.0)
135  {
136  while (dRangeMax <= dTestRange)
137  {
138  dAxisMax = dTestRange;
139  dTestRange /= 2.0;
140  }
141  }
142 
143  //Change the y-axis labeling to give the appearance of normalization.
144  QList<double> majorTicks;
145  QList<double> mediumTicks;
146  QList<double> minorTicks;
147 
148  int numOfDivs = 2;
149  for (int i = 0; i < numOfDivs + 1; i++)
150  {
151  majorTicks += (i / double(numOfDivs))*dTotalValue*dAxisMax;
152  }
153 
154  numOfDivs = 10;
155  for (int i = 0; i < numOfDivs + 1; i++)
156  {
157  mediumTicks += (i / double(numOfDivs))*dTotalValue*dAxisMax;
158  }
159 
160  emit normalizeScaleSig(dAxisMax, dTotalValue, mediumTicks, majorTicks);
161 }
