225  void RasterBase::createNormalizedScale(double dMaxValue, double dTotalValue)
226  {
227  double dRangeMax = dMaxValue / dTotalValue;
228 
229  double dAxisMax = 1.0;
230  double dTestRange = dAxisMax / 2.0;
231 
232  if (dRangeMax > 0.0)
233  {
234  while (dRangeMax <= dTestRange)
235  {
236  dAxisMax = dTestRange;
237  dTestRange /= 2.0;
238  }
239  }
240 
241  //Change the y-axis labeling to give the appearance of normalization.
242  QList<double> majorTicks;
243  QList<double> mediumTicks;
244  QList<double> minorTicks;
245 
246  int numOfDivs = 2;
247  for (int i = 0; i < numOfDivs + 1; i++)
248  {
249  majorTicks += (i / double(numOfDivs))*dTotalValue*dAxisMax;
250  }
251 
252  numOfDivs = 10;
253  for (int i = 0; i < numOfDivs + 1; i++)
254  {
255  mediumTicks += (i / double(numOfDivs))*dTotalValue*dAxisMax;
256  }
257  }
