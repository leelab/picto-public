44 bool FileSessionLoader::setFile(QString path)
45 {
46  sessionDataLoaded_ = false;
47  Q_ASSERT(connectionName_.isEmpty()); //Currently, we just create a new file session loader whenever we need to load a new file
48  //
49  //Load Sqlite Database
50  path = path.replace("\\","/");
51  Q_ASSERT(path.lastIndexOf("/") >=0);
52  Q_ASSERT(path.lastIndexOf(".") >=0);
53  QString connectionName = path.mid(path.lastIndexOf("/")+1);
54  connectionName = connectionName.left(connectionName.lastIndexOf("."));
55  Q_ASSERT(QSqlDatabase::contains(connectionName) == connectionUsers_.contains(connectionName));
56  if(!QSqlDatabase::contains(connectionName))
57  {
58  QSqlDatabase newSession = QSqlDatabase::addDatabase("QSQLITE",connectionName);
59  newSession.setDatabaseName(path);
60  newSession.open();
61  if(!newSession.isOpen())
62  {
63  newSession.close();
64  QSqlDatabase::removeDatabase(connectionName);
65  qDebug("Error: Could not open session file.");
66  return false;
67  }
68  connectionUsers_[connectionName] = 0;
69  }
70  connectionUsers_[connectionName]++;
71  connectionName_ = connectionName;
72 
73  //Intialize Object Data
74  if(!loadRunData())
75  return false;
76  if(!getSignalInfo())
77  return false;
78  if(!loadDesignDefinition())
79  return false;
80  runIndex_ = -1;
81  return true;
82 }

196 bool FileSessionLoader::loadRunData()
197 {
198  runs_.clear();
199  QSqlQuery query(getDatabase());
200  query.setForwardOnly(true);
201 
202  //Get runs list.
203  query.prepare("SELECT r.runid,r.name,r.notes,r.saved,r.firstframe,r.lastframe,s.time,e.time "
204  "FROM runs r, frames s, frames e WHERE s.dataid=r.firstframe AND ((r.lastframe>0 AND e.dataid=r.lastframe) "
205  "OR (r.lastframe=0 AND e.dataid=(SELECT dataid FROM frames ORDER BY dataid DESC LIMIT 1))) ORDER BY runid");
206  bool success = query.exec();
207  if(!success)
208  {
209  qDebug("Failed to select data from runs table with error: " + query.lastError().text().toLatin1());
210  return false;
211  }
212 
213  RunData run;
214  while(query.next()){
216  //if(!query.value(3).toBool())
217  // continue;
218  //Build up the run object
219  run.dataId_ = query.value(0).toLongLong();
220  run.startFrame_ = query.value(4).toLongLong();
221  run.endFrame_ = query.value(5).toLongLong();
222  run.name_ = query.value(1).toString();
223  run.notes_ = query.value(2).toString();
224  run.saved_ = query.value(3).toBool();
225  run.startTime_ = query.value(6).toDouble();
226  run.endTime_ = query.value(7).toDouble();
227  //Add RunData to the runs_ list.
228  runs_.append(run);
229  }
230  query.finish();
231  if(runs_.size() == 0)
232  return false;
233  return true;
234 }

303 bool FileSessionLoader::loadDesignDefinition()
304 {
306  //Load design definition
308  QSqlQuery query(getDatabase());
309  query.setForwardOnly(true);
310  //ExperimentXML name changed to DesignXML when we started putting UI Data and Analysis Data together
311  //with it on server. They never appeared at the same time so we can just look for either one.
312  query.prepare("SELECT value FROM sessioninfo WHERE key='ExperimentXML' OR key='DesignXML'");
313  bool success = query.exec();
314  if(!success)
315  {
316  qDebug("Query Failed: " + query.lastError().text().toLatin1());
317  return false;
318  }
319  if(!query.next())
320  {
321  return false;
322  }
323  QString designDef = query.value(0).toString();
324 
326  //Create the design root
328  designRoot_ = QSharedPointer<DesignRoot>(new DesignRoot());
329  if(!designRoot_->resetDesignRoot(designDef) || !designRoot_->compiles())
330  {
331  Q_ASSERT(false); //This would mean that somehow the session had a bad design in it.
332  return false;
333  }
334 
336  //Since asset ids can change for unsaved (default) asset values when the same experiment
337  //is loaded in different versions of picto, us a SessionVersionInterfacer to correct the
338  //asset ids in the new version of the experiment.
340  QSharedPointer<Picto::Experiment> experiment = designRoot_->getExperiment().staticCast<Experiment>();
341  if(!experiment)
342  {
343  Q_ASSERT(false); //This would mean that somehow the session had no experiment in it.
344  return false;
345  }
346  SessionVersionInterfacer updater(experiment);
348  //Add all Session asset data to SessionVersionInterfacer
350  query.prepare("SELECT assetid,name,parent FROM propertylookup WHERE parent <> 0");
351  success = query.exec();
352  if(!success)
353  {
354  qDebug("Query Failed: " + query.lastError().text().toLatin1());
355  return false;
356  }
357  while(query.next())
358  {
359  updater.addSessionProperty(query.value(1).toString(),query.value(0).toInt(),query.value(2).toInt());
360  }
361 
362  query.prepare("SELECT source,sourceresult,assetid,parent FROM transitionlookup");
363  success = query.exec();
364  if(!success)
365  {
366  qDebug("Query Failed: " + query.lastError().text().toLatin1());
367  return false;
368  }
369  while(query.next())
370  {
371  updater.addSessionTransition(query.value(0).toString(),query.value(1).toString(),query.value(2).toInt(),query.value(3).toInt());
372  }
373 
374  query.prepare("SELECT assetid,path FROM elementlookup");
375  success = query.exec();
376  if(!success)
377  {
378  qDebug("Query Failed: " + query.lastError().text().toLatin1());
379  return false;
380  }
381  while(query.next())
382  {
383  updater.addSessionElement(query.value(0).toInt(),query.value(1).toString());
384  }
385 
387  //Update Experiment's DesignConfig
389  updater.updateSessionConfigFromSessionAssets();
390 
392  //Get obsolete Asset Id table
394  obsoleteAssetIds_ = updater.getObsoleteAssets();
395  return true;
396 }
