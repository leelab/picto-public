77 void UIEnabled::upgradeVersion(QString deserializedVersion)
78 {
79  DataStore::upgradeVersion(deserializedVersion);
80  if(deserializedVersion < "0.0.1")
81  {
82  //Before Design syntax version 0.0.1, all UI position data was stored in UIInfo objects that were sub-elements
83  //of UIEnabled objects. At version 0.0.1, we moved UI data out of the Experimental design into a separate UIElements of a UIData design.
84 
85  //Get the old position from the obsolete UIInfo asset
86  QList<QSharedPointer<Asset>> oldUIInfoList = getGeneratedChildren("UIInfo");
87  Q_ASSERT(oldUIInfoList.size() <= 1);
88  if(oldUIInfoList.isEmpty())
89  return; //Looks like there's nothing to do here.
90  QSharedPointer<ObsoleteAsset> oldPosAsset = oldUIInfoList.first().staticCast<ObsoleteAsset>()->getObsoleteChildAsset("Pos").first().staticCast<ObsoleteAsset>();
91  //Check if the position data is stored as attributes or a tag value (this has been updated to always use tag values in new versions)
92  //and set it to a position variable.
93  QPoint pos;
94  if(oldPosAsset->numAttributes()<=1)
95  {
96  QStringList values = oldPosAsset->getValue().split(",");
97  pos = QPoint(values[0].toInt(),values[1].toInt());
98  }
99  else
100  {
101  pos = QPoint(oldPosAsset->getAttributeValue("x").toInt(),oldPosAsset->getAttributeValue("y").toInt());
102  }
103  //Set the position to a UIElement in the newly created UIData object
104  setPos(pos);
105  }
106 }

163 QSharedPointer<Asset> UIEnabled::getUIElement()
164 {
165  //If we've already done this before, use the UIData associate ID that we got last time to find the UIElement
166  QList<QSharedPointer<Asset>> assocChildren;
167  if(!uiAssociateId_.isNull())
168  {
169  assocChildren = getAssociateChildren(uiAssociateId_,"UIElement");
170  if(assocChildren.size())
171  {
172  return assocChildren.first();
173  }
174  }
175 
176  //If we got here, we don't have a stored UIData associate ID that has a UIElement in it. Find one and get the element.
177  foreach(QUuid associateId,getAttachedAssociateIds())
178  {
179  assocChildren = getAssociateChildren(associateId,"UIElement");
180  if(assocChildren.size())
181  {
182  uiAssociateId_ = associateId;
183  return assocChildren.first();
184  }
185  }
186 
187  //If we got here, there is no attached UIElement, create one in the UIData attached to the root of this tree
188 
189  //Get the root of this tree, beneath the PictoData object. It will be an AssociateRootHost.
190  QSharedPointer<Asset> ancestor = selfPtr();
191  while(ancestor->getParentAsset() && ancestor->getParentAsset()->getParentAsset())
192  {
193  ancestor = ancestor->getParentAsset();
194  }
195  //Get the AssociateRoot attached to it. This will be a UIData object.
196  AssociateRootHost* assocRootHost = dynamic_cast<AssociateRootHost*>(ancestor.data());
197  Q_ASSERT(assocRootHost);
198  QSharedPointer<AssociateRoot> uiData = assocRootHost->getAssociateRoot("UIData");
199  //Store the uiAssociateId of the UIData object
200  uiAssociateId_ = uiData->getAssociateId();
201  //Add a new UIElement to the data object and link it to me.
202  QSharedPointer<Asset> uiElement = uiData->createChildAsset("UIElement",QString(),QString());
203  AddAssociateChild(uiAssociateId_,"UIElement",uiElement);
204 
205  //Return the newly generated element
206  return uiElement;
207 }
