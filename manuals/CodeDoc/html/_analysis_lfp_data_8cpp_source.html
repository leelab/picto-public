248 QVariantList AnalysisLfpData::getPrevValues(double secsPreceding)
249 {
250  QVariantList channels = getChannels();
251  QList<QVariantList> resultsByName;
252  foreach(QVariant channel,channels)
253  {
254  resultsByName.append(getPrevValues(channel.toInt(),secsPreceding));
255  }
256  //Make sure all lists are the same size
257  for(int i=1;i<resultsByName.size();i++)
258  {
259  if(resultsByName[i].size() != resultsByName[i-1].size())
260  {
261  Q_ASSERT(resultsByName[i].size() == resultsByName[i-1].size());
262  return QVariantList();
263  }
264  }
265  //Put together result list
266  QVariantList result;
267  QVariantList singleRow;
268  if(resultsByName.size())
269  {
270  result.reserve(resultsByName[0].size());
271  for(int i=0;i<resultsByName[0].size();i++)
272  {
273  singleRow.clear();
274  for(int j=0;j<resultsByName.size();j++)
275  {
276  singleRow.append(resultsByName[j][i]);
277  }
278  result.append(QVariant(singleRow));
279  }
280  }
281  return result;
282 }
