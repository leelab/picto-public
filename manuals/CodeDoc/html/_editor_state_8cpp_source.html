55 bool EditorState::setCurrentAnalysis(QSharedPointer<Analysis> currAnalysis)
56 {
57  if(!topAsset_)
58  return false;
59  //Run through this function even if the current analysis hasn't changed, because
60  //the current analysis in the designConfig may have been changed and this will
61  //reset it.
62 
63  //If there is a current analysis
64  if(currAnalysis)
65  {
66  //Verify that its linked into the experiment.
67  QSharedPointer<Experiment> exp = topAsset_.staticCast<Experiment>();
68  Q_ASSERT(exp);
69  //Link the analysis to the experiment
70  QString feedback;
71  bool linkResult = currAnalysis->LinkToAsset(exp,feedback);
72  if(!linkResult)
73  { //If anything didn't go perfectly in the link (had to link by paths or failed) tell the user.
74  QMessageBox msg;
75  msg.setText("The Analysis was not fully compatible with this Experiment.");
76  msg.setDetailedText(feedback);
77  msg.setIcon(QMessageBox::Warning);
78  msg.exec();
79  }
80  }
81  currAnalysis_ = currAnalysis;
82  QList<QUuid> activeAnalysisList;
83  if(currAnalysis_)
84  activeAnalysisList.append(currAnalysis_->getAssociateId());
85  topAsset_->getDesignConfig()->setActiveAnalysisIds(activeAnalysisList);
86  emit currentAnalysisChanged(currAnalysis_);
87  return true;
88 }

146 void EditorState::setWindowAsset(QSharedPointer<Asset> asset,bool undoable)
147 {
148  if(asset.isNull())
149  return;
150  Q_ASSERT(!asset.dynamicCast<DataStore>().isNull());
151  Q_ASSERT(!asset.dynamicCast<UIEnabled>().isNull());
152  QSharedPointer<UIEnabled> uiEnabled = asset.staticCast<UIEnabled>();
153  //If this asset has no editable descendants, it should not be openable.
154  if(!uiEnabled->hasEditableDescendants())
155  {
156  return;
157  }
158 
159  //Store the current window asset
160  windowAsset_ = asset;
161 
162  //Clear any selected items to insert into the scene
163  setInsertionItem();
164 
165  //Set the currently selected item to null
166  setSelectedItem(NULL);
167 
168  //Tell everyone that the window asset changed.
169  emit windowAssetChanged(windowAsset_);
170 
171  //Set the window asset as the currently selected asset.
172  setSelectedAsset(asset);
173 
174  if(undoable)
175  {
176  //Set this action as undoable so that when people undo the next change they make, they
177  //will still be on the window where they made that change, rather than moving back to
178  //the last place where they made an undoable change.
179  setLastActionUndoable();
180  }
181 }
