65  return;

67  uInt8 data[] = { 1 };

70  for (int channel = 1; channel<NUMREWARDLINES + 1; channel++)
71  {
72  DAQmxErrChk(DAQmxCreateTask("RewardTaskDig", (TaskHandle*)&daqTaskHandleDig_[channel]));
73  DAQmxErrChk(DAQmxCreateDOChan(daqTaskHandleDig_[channel], QString("Dev1/port2/line%1").arg(channel + 3).toLatin1(), "", DAQmx_Val_ChanPerLine));
74  //If the function fails on StartTask, don't trigger assertions because it might just be that another process
75  //is using the pin we want to control. If there's an error here, just skip the task.
76  if (!DAQmxStartTask(daqTaskHandleDig_[channel]))
77  DAQmxErrChk(DAQmxWriteDigitalLines(daqTaskHandleDig_[channel], 1, true, 1, DAQmx_Val_GroupByChannel, data, &sampsPerChanWritten, NULL));
78  DAQmxErrChk(DAQmxStopTask(daqTaskHandleDig_[channel]));

159  return false;
160  DAQmxErrChk(DAQmxStopTask(daqTaskHandleDig_[channel]));
