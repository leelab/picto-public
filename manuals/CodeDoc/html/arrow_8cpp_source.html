144  bool startItemIsStartBar = myStartItem->getHeight() > 1000;
145  if(startItemIsStartBar) //Its a start bar
146  startPos.setY(0); //Do calculations as if we're starting from top corner. Later we'll update startPos for a horizontal line
147  //Find the end point with the nearest Y value.
148  QPointF endTop = mapFromItem(myEndItem,0,0);
149  QPointF endBottom = mapFromItem(myEndItem,0,myEndItem->boundingRect().height());
150  QPointF endPos;
151  if((startPos.y() >= endTop.y()) && (startPos.y() <= endBottom.y()) )
152  {
153  endPos = endTop;
154  endPos.setY(startPos.y());
155  }
156  else if(startPos.y() < endTop.y())
157  {
158  endPos = endTop;
159  }
160  else

162  endPos = endBottom;
163  }
164  //endPos = mapFromItem(myEndItem,0,0)+QPointF(0,myStartItem->myEndItem()/2.0);
165  if(startPos == endPos)
166  {
167  updatePosition();
168  return;
169  }
170 
171  if(startItemIsStartBar) //If start item is a start bar, make the arrow start at the same height as it stops
172  startPos.setY(endPos.y());
173 
174  QPen myPen = pen();
175  QColor finalColor(myColor);
176  //If there's an active analysis, grey out the arrow
177  qobject_cast<QGraphicsOpacityEffect*>(graphicsEffect())->setOpacity(1.0);
178  if(!editorState_->getCurrentAnalysis().isNull())
179  qobject_cast<QGraphicsOpacityEffect*>(graphicsEffect())->setOpacity(100.0/255.0);
180  myPen.setColor(finalColor);
181  qreal arrowSize = 20;
182  painter->setPen(myPen);
183  painter->setBrush(finalColor);
184 
185 
186  QLineF centerLine(startPos, endPos);
187  QPolygonF endPolygon = myEndItem->polygon();
188  QPointF p1 = endPolygon.first() + endPos;
189  QPointF p2;
190  QPointF intersectPoint;
191  QLineF polyLine;
192  for (int i = 1; i < endPolygon.count(); ++i) {
193  p2 = endPolygon.at(i) + endPos;
194  polyLine = QLineF(p1, p2);
195  QLineF::IntersectType intersectType =
196  polyLine.intersect(centerLine, &intersectPoint);
197  if (intersectType == QLineF::BoundedIntersection)
198  break;
199  p1 = p2;
200  }
201  if(p1 == p2)
202  {
203  updatePosition();
204  return;
205  }
206

208
