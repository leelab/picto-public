207 bool CommandChannel::processResponses(int timeoutMs)
208 {
209  QTime timeoutTimer;
210  // A negative input means loop forever (until the status manager's ComponentStatusManager::exitTriggered() returns true)
211  bool loopForever = (timeoutMs < 0);
212  timeoutTimer.start(); //Use this timer to keep track of when we have passed the input timeout time
213  QSharedPointer<Picto::ProtocolResponse> response;
214  QSharedPointer<ProtocolResponseHandler> responseHandler;
215  //This value will store the next time at which we may need to resend registered commands
216  //whose replies were never recived. The fact that its initialized to the current time is confusing, but it just
217  //makes sure that this value is ignored until it gets some meaningful data.
218  QDateTime nextPendingMessageTime = QDateTime::currentDateTime();
219  bool keepLooping = true;
220  do
221  { //This loop keeps looping so long as:
222  // - either there are still registered commands that haven't recieved matching reponses,
223  // and the next time that a registered command will have to be resent is before the input timeout time.
224  // - or this function is set to loop forever
225  // if the statusManager's exitTriggered() function returns true it will always stop.
226  do
227  { //This loop keeps looping until there are no more incoming responses available to be processed
228  //by their respective ProtocolResponseHander objects.
229 
230  if(!waitForResponse(0))
231  break;
232  //Take the latest response off of the queue.
233  response = getResponse();
234  //Get the appropriate ProtocolResponseHander to process the response, and call its acceptResponse() function.
235  if(!response.isNull())
236  {
237  QString directive = response->getDecodedContent();
238  int directiveEnd = directive.indexOf(' ');
239  int altDirectiveEnd = directive.indexOf('\n');
240  if((altDirectiveEnd > 0) && (altDirectiveEnd < directiveEnd))directiveEnd = altDirectiveEnd;
241  QString statusDirective = directive.left(directiveEnd);
242  //qDebug("Directive Received: " + statusDirective.toLatin1());
243  responseHandler = responseHandlerMap_.value(statusDirective,QSharedPointer<ProtocolResponseHandler>());
244  if(!responseHandler.isNull())
245  {
246  responseHandler->acceptResponse(response);
247  }
248  else
249  {
250  Q_ASSERT(!responseHandler.isNull());
251  }
252  }
253  }while(numIncomingResponses() > 0);
254 
255  int remainingMs = timeoutMs - timeoutTimer.elapsed();
256  // This will verify connection and attempt to reconnect if there's time left before the input timeout time
257  if(assureConnection((timeoutMs == 0)?0:(loopForever)?100:(remainingMs<0)?0:remainingMs))
258  //If there is time left before the input timeout time, send any RegisteredCommands for whom their timeout time has passed without any matching response received.
259  if(timeoutTimer.elapsed() > timeoutMs)
260  nextPendingMessageTime = resendPendingCommands().addSecs(resendPendingInterval_);
261 
262  // Update Status Manager
263  //We do this here because, well, someone has to do it and this function gets called at least once per frame.
264  //The StatusManager does things like sending the periodic Event code, so we need to make sure it has the opportunity to do that frequently.
265  if(!statusManager_.isNull())
266  {
267  remainingMs = timeoutMs - timeoutTimer.elapsed();
268  statusManager_.toStrongRef()->update((timeoutMs<=0)?timeoutMs:(remainingMs<0)?0:remainingMs);
269  }
270  remainingMs = timeoutMs - timeoutTimer.elapsed();
271  //If the next time at which a registered command will need to be resent is beyond the timeout time input into this function, don't
272  //continue unless the function was set to loop forever
273  if(!loopForever && QDateTime::currentDateTime().addMSecs(remainingMs) < nextPendingMessageTime)
274  keepLooping = false;
275  }while((loopForever || ((pendingResponses() > 0) && (timeoutTimer.elapsed() < timeoutMs) && keepLooping)) && !(statusManager_.toStrongRef() && statusManager_.toStrongRef()->exitTriggered()) );
276  //If all Registered commands have been matched with responses, return true
277  if(pendingResponses() == 0)
278  return true;
279  //If there are still registered commands with no matching responses, return false.
280  return false;
281 }

435 bool CommandChannel::assureConnection(int acceptableTimeoutMs)
436 {
437  if(consumerSocket_->state() == QAbstractSocket::ConnectedState)
438  {
439  //This is an incredibly dumb call.
440  //
441  //If the CommandChannel is being used in a situation with no event loop, the
442  //sockets behave oddly. When the socket loses its connection, it doesn't realize it.
443  //This means that we can't reliably check for the connection by calling socket::state().
444  //A call to socket->flush also fails to work.
445  //Instead, I call waitForReadyRead, which has the side effect of realizing that the connection
446  //is dead, emmitting the disconnected signal, and then calling the disconnectHandler slot.
447  //
448  //Surely there is a better way to do this, but I can't find one and I've already spent
449  //way too long dealing with this issue.
450  //Someday we need to just rewrite this whole thing properly by using the Qt Event Loop (see processResponses())
451  consumerSocket_->waitForReadyRead(0);
452  }
453  //Now we can really check if we're connected and if not,
454  //then if there is an acceptable timeout lets try reconnecting.
455  //
456  if(consumerSocket_->state() != QAbstractSocket::ConnectedState)
457  {
458  status_ = disconnected;
459  if(!statusManager_.isNull() && statusManager_.toStrongRef()->getStatus() == idle)
460  statusManager_.toStrongRef()->setStatus(ComponentStatus::disconnected);
461  if(acceptableTimeoutMs > 0)
462  {
463  QTime timer;
464  timer.start();
465  if((acceptableTimeoutMs > 50) && !discoverServer(acceptableTimeoutMs))
466  return false;
467  if(timer.elapsed() >= acceptableTimeoutMs)
468  return false;
469  consumerSocket_->connectToHost(serverAddr_, serverPort_, QIODevice::ReadWrite);
470  int remainingTime = acceptableTimeoutMs - timer.elapsed();
471  if(remainingTime <= 0.0)
472  return false;
473  consumerSocket_->waitForConnected(remainingTime);
474  }
475  }
476  if(consumerSocket_->state() == QAbstractSocket::ConnectedState)
477  {
478  if(status_ != connected)
479  lastReconnectTime_ = QDateTime::currentDateTime();
480  status_ = connected;
481  if(!statusManager_.isNull() && statusManager_.toStrongRef()->getStatus() < idle)
482  statusManager_.toStrongRef()->setStatus(idle);
483  return true;
484  }
485  return false;
486 }

498 bool CommandChannel::discoverServer(int timeoutMs)
499 {
500  if(timeoutMs == 0)
501  return false;
502  if(timeoutMs < 0)
503  timeoutMs = 10000;
504  QTime timer;
505  timer.start();
506  //For some reason, the discovery socket seems to periodically forget to be bound to
507  //the port that we set. If this happens, we close and rebind it.
508  if(discoverySocket_->state() != QAbstractSocket::BoundState)
509  {
510  discoverySocket_->close();
511  QHostAddress serverAddress(QHostAddress::Any);
512  discoveryPort_ = Picto::portNums->getMinDiscoverPort();
513  while((!discoverySocket_->bind(serverAddress, discoveryPort_)) && (discoveryPort_ < Picto::portNums->getMaxDiscoverPort()))
514  {
515  if(timer.elapsed() >= timeoutMs)
516  {
517  return false;
518  }
519  discoveryPort_++;
520  }
521  }
522  if(timer.elapsed() >= timeoutMs)
523  return false;
524 
525  //If a long time has passed since we sent our last DISCOVER message, send a new one.
526  if(discoverMsgSentTime_.secsTo(QDateTime::currentDateTime()) >= 10)
527  {
528  QByteArray datagram = QString("DISCOVER %1 PICTO/1.0").arg(discoveryPort_).toLatin1();
529  QUdpSocket sendSocket;
530  sendSocket.writeDatagram(datagram.data(), datagram.size(), QHostAddress::Broadcast, Picto::portNums->getServerPort());
531  discoverMsgSentTime_ = QDateTime::currentDateTime();
532  }
533  int msLeft = timeoutMs-timer.elapsed();
534  if(msLeft <= 0)
535  return false;
536 
537  //Check to see if we got any new discover messages back.
538  discoverySocket_->waitForReadyRead(msLeft);
539  while ((discoverySocket_->state() == QAbstractSocket::BoundState) && discoverySocket_->hasPendingDatagrams())
540  {
541  if(timer.elapsed() >= timeoutMs)
542  return false;
543  QByteArray datagram;
544  datagram.resize(discoverySocket_->pendingDatagramSize());
545  discoverySocket_->readDatagram(datagram.data(), datagram.size());
546  QString request = datagram.data();
547  QString method, targetAddress, targetPort, protocolName, protocolVersion;
548  QStringList tokens = request.split(QRegExp("[ ][ ]*"));
549  if(tokens.count() == 3)
550  {
551  method = tokens[0];
552 
553  int targetPortPosition = tokens[1].indexOf(':');
554  if(targetPortPosition != -1)
555  {
556  targetAddress = tokens[1].left(targetPortPosition);
557  serverAddr_.setAddress(targetAddress);
558  targetPort = tokens[1].mid(targetPortPosition+1);
559  serverPort_ = targetPort.toInt();
560  }
561 
562  int protocolVersionPosition = tokens[2].indexOf('/');
563  if(protocolVersionPosition != -1)
564  {
565  protocolName = tokens[2].left(protocolVersionPosition);
566  protocolVersion = tokens[2].mid(protocolVersionPosition+1);
567  }
568 
569  if(method == "ANNOUNCE" &&
570  !serverAddr_.isNull() &&
571  serverPort_ &&
572  protocolName == "PICTO")
573  {
574  qDebug("Time when discovered: " + QString::number(timer.elapsed()).toLatin1());
575  discoverySocket_->close();
576  return true;
577  }
578  }
579  }
580  //qDebug("Time for discover func: " + QString::number(timer.elapsed()).toLatin1() + " of max: " + QString::number(timeoutMs).toLatin1());
581  return false;
582 }

594 bool CommandChannel::sendCommand(QSharedPointer<Picto::ProtocolCommand> command)
595 {
596  // We need to call assureConnection every time we send a command. It fails if there is no
597  // connection and starts the process of reconnecting without blocking.
598  if(!assureConnection())
599  return false;
600 
601  //We also always add a source-ID field so that we can be uniquely identified by the recipient.
602  command->setFieldValue("Source-ID",sourceId_.toString());
603 
604  //We also always add a source-Type field so that we our component type can be identified by the recipient
605  command->setFieldValue("Source-Type",sourceType_);
606 
607  //For messages that have a status element, update it to the latest status.
608  if(!statusManager_.isNull() && ((command->getMethod() == "PUTDATA") || (command->getMethod() == "COMPONENTUPDATE")))
609  {
610  command->setTarget(statusManager_.toStrongRef()->getName()+":"+statusManager_.toStrongRef()->getStatusAsString());

617

701 QDateTime CommandChannel::resendPendingCommands()
702 {
703  QDateTime currTime = QDateTime::currentDateTime();
704  if(!resendEnabled_)
705  return earliestPendingCommand_;
706  if(!assureConnection(0))
707  return earliestPendingCommand_;
708  if(earliestPendingCommand_.secsTo(currTime) < resendPendingInterval_)
709  return earliestPendingCommand_;
710  QList<QSharedPointer<ProtocolCommand> > commandsToResend;
711  QDateTime timeSent;
712  QDateTime nextEarliestCommand = QDateTime::currentDateTime();
713  foreach(QSharedPointer<ProtocolCommand> command, pendingCommands_)
714  {
715  // If the command was sent more than resendInterval secs ago, send it again.
716  timeSent = QDateTime::fromString(command->getFieldValue("Time-Sent"));
717  if(timeSent.secsTo(currTime) > resendPendingInterval_)
718  {
719  commandsToResend.push_back(command);
720  }
721  else if(timeSent < nextEarliestCommand)
722  {
723  nextEarliestCommand = timeSent;
724  }
725  }
726 
727  foreach(QSharedPointer<ProtocolCommand> command, commandsToResend)
728  {
729  command->setFieldValue("Time-Sent",QDateTime::currentDateTime().toString());
730  sendCommand(command);
731  //qDebug((QString("Sent message with ID: ") + command->getFieldValue("Command-ID")).toLatin1());
732  }
733  earliestPendingCommand_ = nextEarliestCommand;
734  return earliestPendingCommand_;
