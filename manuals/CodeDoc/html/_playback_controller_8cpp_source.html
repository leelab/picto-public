413 void PlaybackController::update()
414 {
415  PlaybackControllerData::Status status = data_.getStatus();
416  PlaybackCommand cmd = data_.getNextCommand();
417  QString errorMsg;
418  switch(status)
419  {
420  case PlaybackControllerData::None:
421  data_.setNextStatus(PlaybackControllerData::Idle);
422  break;
423  case PlaybackControllerData::Idle:
424  switch(cmd.commandType)
425  {
426  case PlaybackCommand::PreLoad:
427  data_.setNextStatus(PlaybackControllerData::PreLoading);
428  data_.pushCommand(cmd,false);
429  break;
430  case PlaybackCommand::ChangeSpeed:
431  playbackUpdater_->setPlaybackSpeed(cmd.commandData.toDouble());
432  break;
433  case PlaybackCommand::ChangeUserType:
434  engine_->setOperatorAsUser(cmd.commandData.toInt() == PlaybackCommand::Operator);
435  break;
436  case PlaybackCommand::EnableLFP:
437  Q_ASSERT(false);
438  break;
439  case PlaybackCommand::Stop:
440  data_.setNextStatus(PlaybackControllerData::Stopped);
441  break;
442  case PlaybackCommand::Jump:
443  data_.setNextStatus(status);
444  break;
445  }
446  break;
447  case PlaybackControllerData::PreLoading:
448  switch(cmd.commandType)
449  {
450  case PlaybackCommand::PreLoad:
451  {
452  //Preload the first file in the list and send the designRoot to signal
453  //listeners. Then remove the file from the list and re-enter the command
454  //until the list is empty. At that point we can move on to stop.
455 
456  //Take first file path from cmd list
457  QStringList filePaths = cmd.commandData.toStringList();
458  QString filePath = filePaths.takeFirst();
459 
460  FileSessionLoader loader;
461  if(!loader.setFile(filePath) || !loader.getDesignRoot())
462  {
463  emit sessionPreloadFailed("The session in file: " + filePath.toLatin1() + " could not be loaded.");
464  }
465  else
466  {
467  PreloadedSessionData preloadData;
468  preloadData.fileName_ = filePath;
469  preloadData.runs_ = loader.getRunNames();
470  preloadData.notes_ = loader.getRunNotes();
471  preloadData.savedRuns_ = loader.getSavedRunNames();
472  QSharedPointer<DesignRoot> designRoot = loader.getDesignRoot();
473  for(int i=0;i<designRoot->getNumAnalyses();i++)
474  {
475  preloadData.analysisIds_.append(designRoot->getAnalysis(i).staticCast<Analysis>()->getAssociateId());
476  preloadData.analysisNames_.append(designRoot->getAnalysis(i)->getName());
477  }
478  emit sessionPreloaded(preloadData);
479  }
480 
481  //If we're done with all the preloading, move to the stopped state.
482  if(filePaths.isEmpty())
483  {
484  data_.setNextStatus(PlaybackControllerData::Stopped);
485  break;
486  }
487  //If there are more files to preload, attach the updated list
488  //to the command and put it back on the command queue.
489  cmd.commandData = filePaths;
490  data_.pushCommand(cmd);
491  break;
492  }
493  case PlaybackCommand::ChangeSpeed:
494  playbackUpdater_->setPlaybackSpeed(cmd.commandData.toDouble());
495  break;
496  case PlaybackCommand::ChangeUserType:
497  engine_->setOperatorAsUser(cmd.commandData.toInt() == PlaybackCommand::Operator);
498  break;
499  case PlaybackCommand::EnableLFP:
500  Q_ASSERT(false);
501  break;
502  case PlaybackCommand::Stop:
503  data_.setNextStatus(PlaybackControllerData::Idle);
504  break;
505  case PlaybackCommand::Jump:
506  data_.setNextStatus(status);
507  break;
508  }
509  break;
510  case PlaybackControllerData::Stopped:
511  switch(cmd.commandType)
512  {
513  case PlaybackCommand::Play:
514  case PlaybackCommand::Pause:
515  data_.setNextStatus(PlaybackControllerData::Loading);
516  data_.pushCommand(cmd,false);
517  break;
518  case PlaybackCommand::PreLoad:
519  data_.setNextStatus(PlaybackControllerData::Idle);
520  data_.pushCommand(cmd,false);
521  break;
522  case PlaybackCommand::ChangeFile:
523  data_.setNextFilePath(cmd.commandData.toString());
524  setCurrTime(0.0);
525  break;
526  case PlaybackCommand::ChangeRun:
527  currRun_ = cmd.commandData.toInt();
528  setCurrTime(0.0);
529  break;
530  case PlaybackCommand::ChangeSpeed:
531  playbackUpdater_->setPlaybackSpeed(cmd.commandData.toDouble());
532  break;
533  case PlaybackCommand::ChangeUserType:
534  engine_->setOperatorAsUser(cmd.commandData.toInt() == PlaybackCommand::Operator);
535  break;
536  case PlaybackCommand::EnableLFP:
537  //If the session for the upcoming run is already loaded, set it to the playabck state updater.
538  //otherwise this happens automatically during the setup() call used when the session is loaded
539  if(data_.getLoadedFilePath() == data_.getNextFilePath())
540  {
541  if(playbackUpdater_)
542  playbackUpdater_->enableLfp(cmd.commandData.toBool());
543  }
544  data_.enableLfp(cmd.commandData.toBool());
545  break;
546  case PlaybackCommand::Jump:
547  data_.setNextStatus(status);
548  break;
549  }
550  break;
551  case PlaybackControllerData::Loading:
552  switch(cmd.commandType)
553  {
554  case PlaybackCommand::Play:
555  case PlaybackCommand::Pause:
556  {
557  QString nextFilePath = data_.getNextFilePath();
558  if(nextFilePath != data_.getLoadedFilePath())
559  {
560  setup();
561  slaveExpDriver_.clear(); //Both SlaveExperimentDriver and Playback Updater contain pointers to the SessionState which is where
562  //all the session data is stored in RAM. By clearing the slaveExpDriver_ here, we can be sure that
563  //RAM that was allocated for a previous playback will be freed up for loading the new file. If
564  //we didn't do this, we would effectively be cutting our maximum RAM capacity in half because when
565  //switching from one session to the next, only half of the maximum RAM available to the application
566  //is available for the new session.
567  experiment_.clear(); //Experiment also uses up a lot of RAM.
568  designRoot_.clear(); //Design root point to the experiment too.
569 
570 
571 #if defined WIN32 || defined WINCE
572  //Since we're creating lots of giant data structures on the heap, we found that there were some heap fragmentation issues. ie. 'new' commands
573  //would start to fail wbile creating analysis data structures after we analyzed multiple other sessions, presumably since there weren't
574  //large enough holes in memory for the new data structures. For this reason, we are defragmenting the heap each time we start loading a new
575  //session. This code for heap deframentation is from: http://www.cplusplus.com/forum/windows/8010/
576  {
577  const int MaxNumHeaps= 5000;
578  HANDLE HeapHandles[MaxNumHeaps];
579  int NumUsedHeaps= GetProcessHeaps(MaxNumHeaps,HeapHandles);
580  for (int i=0;i<NumUsedHeaps;i++)
581  {
582  HeapCompact(HeapHandles[i],0);
583  }
584  }
586 #endif
587 
588  if(!playbackUpdater_->setFile(nextFilePath))
589  {
590  emit loadError("Failed to load session design. This often means that the session contains no experimental data.");
591  data_.setNextStatus(PlaybackControllerData::Stopped);
592  break;
593  }
594  QSharedPointer<DesignRoot> newDesignRoot = QSharedPointer<DesignRoot>(playbackUpdater_->getDesignRoot());
595  if(!newDesignRoot)
596  {
597  emit loadError("Failed to load session design.");
598  data_.setNextStatus(PlaybackControllerData::Stopped);
599  break;
600  }
601  if(!newDesignRoot->compiles())
602  {
603  emit loadError("Session's experiment does not compile.");
604  data_.setNextStatus(PlaybackControllerData::Stopped);
605  break;
606  }
607  QSharedPointer<Picto::Experiment> currExp = newDesignRoot->getExperiment().staticCast<Experiment>();
608  if(!currExp || !currExp->getTaskNames().size())
609  {
610  emit loadError("Failed to load experiment from session.");
611  data_.setNextStatus(PlaybackControllerData::Stopped);
612  break;
613  }
614 
615  if(!playbackUpdater_->loadRun(currRun_))
616  {
617  emit loadError("Failed to load data from this session run. This is usually due to RAM issues when running a batch analysis. Try analyzing this run by itself.");
618  data_.setNextStatus(PlaybackControllerData::Stopped);
619  break;
620  }
621 
622  designRoot_ = newDesignRoot;
623  experiment_ = currExp;
624  experiment_->setEngine(engine_);
625 
626  numImportedAnalyses_ = 0;
627  data_.clearEnabledBuiltInAnalyses();
628  data_.clearEnabledImportedAnalyses();
629 
630  data_.setRunLength(playbackUpdater_->getRunLength());
631 
632  data_.setLoadedFilePath(nextFilePath);
633 
634  //Set up SlaveExperimentDriver to connect StateUpdater and Experiment
635  slaveExpDriver_ = QSharedPointer<SlaveExperimentDriver>(new SlaveExperimentDriver(experiment_,playbackUpdater_));
636  emit designRootChanged();
637  }
638 
639  //If we got here, the session is loaded
640  //Update to use the current set run
641  playbackUpdater_->loadRun(currRun_);
642  data_.setRunLength(playbackUpdater_->getRunLength());
643 
644  //Activate the necessary analyses
645  errorMsg = activateAnalyses(cmd.commandData.toStringList());
646  if(!errorMsg.isEmpty())
647  {
648  emit loadError(errorMsg);
649  data_.setNextStatus(PlaybackControllerData::Stopped);
650  break;
651  }
652 
653  //Now that the analyses have been added, we need to reinitialize scripting for the experiment
654  //Since a whole lot of memory is allocated at this point, the script engine allocation occuring
655  //during the initScripting may fail, so we put this in a try catch block.
656  try{
657  if(!designRoot_->getExperiment().staticCast<Experiment>()->initScripting(false))
658  {
659  emit loadError("Failed to initialize the loaded session. This may result from RAM issues when running batch analysis. Try analyzing this run by itself.");
660  data_.setNextStatus(PlaybackControllerData::Stopped);
661  break;
662  }
663  }
664  catch(...)
665  {
666  emit loadError("Failed to initialize the loaded session. This may have resulted from RAM issues when running batch analysis. Try analyzing this run by itself.");
667  data_.setNextStatus(PlaybackControllerData::Stopped);
668  break;
669  }
670  designRoot_->getExperiment()->getDesignConfig()->setActiveAnalysisIds(data_.getEnabledAnalyses());
671 
672  designRoot_->enableRunMode(true);//We do this here so that we're sure all the analyses are attached first
673  playbackUpdater_->play();
674  data_.setNextStatus(PlaybackControllerData::Running);
675  if(cmd.commandType == PlaybackCommand::Pause)
676  data_.pushCommand(cmd,false);
677  }
678  break;
679  }
680  break;
681  case PlaybackControllerData::Running:
682  case PlaybackControllerData::Paused:
683  switch(cmd.commandType)
684  {
685  case PlaybackCommand::Play:
686  if(status == PlaybackControllerData::Paused)
687  {
688  playbackUpdater_->play();
689  data_.setNextStatus(PlaybackControllerData::Running);
690  }
691  break;
692  case PlaybackCommand::Pause:
693  if(status == PlaybackControllerData::Running)
694  {
695  playbackUpdater_->pause();
696  data_.setNextStatus(PlaybackControllerData::Paused);
697  }
698  break;
699  case PlaybackCommand::Stop:
700  playbackUpdater_->stop();
701  data_.setCurrTime(0.0);
702  data_.setNextStatus(PlaybackControllerData::Stopped);
703  break;
704  case PlaybackCommand::Jump:
705  playbackUpdater_->jumpToTime(cmd.commandData.toDouble());
706  data_.setNextStatus(status);
707  break;
708  case PlaybackCommand::PreLoad:
709  playbackUpdater_->stop();
710  data_.setCurrTime(0.0);
711  data_.setNextStatus(PlaybackControllerData::Idle);
712  data_.pushCommand(cmd,false);
713  break;
714  case PlaybackCommand::ChangeFile:
715  case PlaybackCommand::ChangeRun:
716  playbackUpdater_->stop();
717  data_.setCurrTime(0.0);
718  data_.setNextStatus(PlaybackControllerData::Stopped);
719  data_.pushCommand(cmd,false);
720  break;
721  case PlaybackCommand::ChangeSpeed:
722  playbackUpdater_->setPlaybackSpeed(cmd.commandData.toDouble());
723  break;
724  case PlaybackCommand::ChangeUserType:
725  engine_->setOperatorAsUser(cmd.commandData.toInt() == PlaybackCommand::Operator);
726  break;
727  case PlaybackCommand::EnableLFP:
728  Q_ASSERT(false);
729  break;
730  }
731  playbackUpdater_->updateState();
732  break;
733  }
734  PlaybackControllerData::Status nextStatus = data_.getNextStatus();
735  if(status != nextStatus)
736  {
737  data_.setStatus(data_.getNextStatus());
738  emit statusChanged(nextStatus);
739  }
740 }
