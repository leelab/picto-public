54  else
55  b = QPoint(headLength*cos(lineAngle), headLength*sin(lineAngle));

58  -b.x()*sin(headAngle) + b.y()*cos(headAngle));
59  QPoint negHeadEndPoint(b.x()*cos(-headAngle) + b.y()*sin(-headAngle),
60  -b.x()*sin(-headAngle) + b.y()*cos(-headAngle));
61  posHeadEndPoint.setX(posHeadEndPoint.x()+end.x());
62  posHeadEndPoint.setY(posHeadEndPoint.y()+end.y());
