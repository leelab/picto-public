369 int MediaMuxer::encodeAudioFrame(unsigned pts, double spikeTime, bool reward)
370 {
371  printf("encodeAudioFrame: %d\n", pts);
372 
373  if (firstSpike_ == 0)
374  {
375  firstSpike_ = spikeTime;
376  firstPts_ = pts;
377  }
378 
379  pAudioCodecCtx_->coded_frame->pts = pts; // Set the time stamp
380 
381  unsigned currentSpikePts = (unsigned)pts; //ms: in theory *1000/FPS but here FPS=1000 (fps)
382 
383  // simple sound encoding
384  double Fs = 8000.; //8 samples per ms - sound is 1ms
385  uint8_t samples[5 * N * 8] = { 0 }; // large buffer
386  //int n; // buffer index
387 
388  double duration = 1.0; //sound duration in ms
389  double silenceDuration = 50; //ms
390  double amplitude = 100.; //amplitude
391 
392  int num_silent_samples = 0;
393 
394  // add silence
395  double tmp = (double)(currentSpikePts - lastSpikePts_);
396  if (tmp > 0)
397  {
398  double FPSd = (double)(FPS);
399  silenceDuration = (double)((tmp * 1000 / FPS) - 1); //time difference in number of frames converted to number of ms
400  if (silenceDuration > 0)
401  {
402  num_silent_samples = round(silenceDuration * (Fs / 1000.0));
403  num_silent_samples = 2 * num_silent_samples;
404 
405  for (int x = 0; x < num_silent_samples; ++x)
406  {
407  samples[x] = 0.0;
408  }
409  }
410  }
411 
412  if (reward) //1 ms
413  {
414  duration = 1.;
415  amplitude = 90.;
416  double freq = 2000.;
417 
418  // add sound
419  int num_samples = round(duration * (Fs / 1000.0));
420  num_samples = 2 * num_samples;
421 
422  for (int x = num_silent_samples; x < num_silent_samples + num_samples; ++x)
423  {
424  samples[x] = amplitude * sin(2 * M_PI * freq * (x / Fs));
425  }
426 
427  audio_outbuf_size = 1 * (num_silent_samples + num_samples);
428 
429  }
430  else //1ms
431  {
432  // add sound
433  int num_samples = round(duration * (Fs / 1000.0));
434  num_samples = 2 * num_samples;
435 
436  for (int x = num_silent_samples; x < num_silent_samples + num_samples; ++x)
437  {
438  samples[x] = amplitude;
439  }
440  //set buffer
441  audio_outbuf_size = 1 * (num_silent_samples + num_samples);
442  }
443 
444 
445  lastSpikePts_ = currentSpikePts;
446 
447  audio_outbuf = new uint8_t[audio_outbuf_size];
448 
449  if (audio_outbuf == 0)

451 
452  //int frame_size = pAudioCodecCtx_->frame_size;
453  int out_size = ffmpeg::avcodec_encode_audio(pAudioCodecCtx_, audio_outbuf, audio_outbuf_size, (const short*)samples);
454  pAudioCodecCtx_->coded_frame->pts = pts; // Set the time stamp
455 
456  if (out_size>0)
457  {
458  // Packet
459  ffmpeg::AVPacket pkt = { 0 };
460  av_init_packet(&pkt);
461  pkt.data = NULL; // packet data will be allocated by the encoder
462  pkt.size = 0;
463  if (pAudioCodecCtx_->coded_frame->pts != (0x8000000000000000LL))
464  pkt.pts = av_rescale_q(pAudioCodecCtx_->coded_frame->pts, pAudioCodecCtx_->time_base, pAudioStream_->time_base);
465 
466  if (pAudioCodecCtx_->coded_frame->key_frame)
467  pkt.flags |= AV_PKT_FLAG_KEY;
468 
469  pkt.stream_index = pAudioStream_->index;
470  pkt.data = audio_outbuf;
471 
472  pkt.size = out_size;
473  //pkt.duration = 100;
474  //pkt.dts = pkt.pts;
475 
476  int ret = av_interleaved_write_frame(pOutputFormatCtx_, &pkt);

586 bool MediaMuxer::convertImage(const QImage &img)
587 {
588  // Check if the image matches the size
589  if (img.width() != getWidth() || img.height() != getHeight())
590  {
591  printf("Wrong image size!\n");
592  return false;
593  }
594  if (img.format() != QImage::Format_RGB32 && img.format() != QImage::Format_ARGB32)
595  {
596  printf("Wrong image format\n");
597  return false;
598  }
599 
600  // RGB32 to YUV420
601 
602  int size = getWidth()*getHeight();
603  // Y
604  for (unsigned y = 0; y<getHeight(); y++)
605  {
606 
607  unsigned char *s = (unsigned char*)img.scanLine(y);
608  unsigned char *d = (unsigned char*)&picture_buf[y*getWidth()];
609  //printf("Line %d. d: %p. picture_buf: %p\n",y,d,picture_buf);
610 
611  for (unsigned x = 0; x<getWidth(); x++)
612  {
613  unsigned int r = s[2];
614  unsigned int g = s[1];
615  unsigned int b = s[0];
616 
617  unsigned Y = (r * 2104 + g * 4130 + b * 802 + 4096 + 131072) >> 13;
618  if (Y>235) Y = 235;
619 
620  *d = Y;
621 
622  d += 1;
623  s += 4;
624  }
625  }
626 
627  // U,V
628  for (unsigned y = 0; y<getHeight(); y += 2)
629  {
630  unsigned char *s = (unsigned char*)img.scanLine(y);
631  unsigned int ss = img.bytesPerLine();
632  unsigned char *d = (unsigned char*)&picture_buf[size + y / 2 * getWidth() / 2];
633 
634  //printf("Line %d. d: %p. picture_buf: %p\n",y,d,picture_buf);
635 
636  for (unsigned x = 0; x<getWidth(); x += 2)
637  {
638  // Cr = 128 + 1/256 * ( 112.439 * R'd - 94.154 * G'd - 18.285 * B'd)
639  // Cb = 128 + 1/256 * (- 37.945 * R'd - 74.494 * G'd + 112.439 * B'd)
640 
641  // Get the average RGB in a 2x2 block
642  int r = (s[2] + s[6] + s[ss + 2] + s[ss + 6] + 2) >> 2;
643  int g = (s[1] + s[5] + s[ss + 1] + s[ss + 5] + 2) >> 2;
644  int b = (s[0] + s[4] + s[ss + 0] + s[ss + 4] + 2) >> 2;
645 
646  int Cb = (-1214 * r - 2384 * g + 3598 * b + 4096 + 1048576) >> 13;
647  if (Cb<16)
648  Cb = 16;
649  if (Cb>240)
650  Cb = 240;
651 
652  int Cr = (3598 * r - 3013 * g - 585 * b + 4096 + 1048576) >> 13;
653  if (Cr<16)
654  Cr = 16;
655  if (Cr>240)
656  Cr = 240;
657 
658  *d = Cb;
659  *(d + size / 4) = Cr;
660 
661  d += 1;
662  s += 8;
663  }
664  }
665  return true;
666 }
