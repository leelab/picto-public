133 void DirectorStatusManager::doServerUpdate()
134 {
135  //We keep on setting status/user info here just in case the director got minimized. These actions will cause the splash
136  //screen to re-present which will cause the director to remaximize and take focus.
137  if(getStatus() <= idle)
138  setStatus(getStatus());
139  else if(getStatus() == stopped)
140  setUserInfo("Engine Stopped");
141 
142  QSharedPointer<CommandChannel> dataChannel = getEngine()->getDataCommandChannel();
143  if(dataChannel.isNull())
144  return;
145  if((getStatus() > stopped) && (lastAlignTime_.secsTo(QDateTime::currentDateTime()) > SECS_PRE_ALIGN))
146  {
147  alignmentCode_ = (alignmentCode_ == 0x7F)? 1 : alignmentCode_+1;
148  alignmentID_++;
149  Timestamper timestamper;
150 
151  double clockBack = getEngine()->generateEvent(alignmentCode_); //It is crucial that we put this line before
152  //In order to get the Event Timestamp right.
153  double timestamp = timestamper.stampSec()-clockBack;
154 
155  //Create an alignment command
156  QSharedPointer<ProtocolCommand> command(new ProtocolCommand());
157  command->setMethod("PUTDATA");
158  command->setProtocolName("PICTO");
159  command->setProtocolVersion("1.0");
160 
161  //Create the content of the alignment command
162  QByteArray alignDataXml;
163  QSharedPointer<QXmlStreamWriter> xmlWriter(new QXmlStreamWriter(&alignDataXml));
164 
165  AlignmentDataUnit alignData;
166  alignData.setAlignCode(alignmentCode_);
167  alignData.setAlignNumber(alignmentID_);
168  alignData.setTimestamp(timestamp);
169 
170  xmlWriter->writeStartElement("Data");
171  alignData.toXml(xmlWriter);
172  xmlWriter->writeEndElement();
173 
174  command->setContent(alignDataXml);
175  command->setFieldValue("Content-Length",QString::number(alignDataXml.length()));
176 
177  dataChannel->sendRegisteredCommand(command);
178  lastAlignTime_ = QDateTime::currentDateTime();
179  }
180  else
181  {
182  QSharedPointer<Picto::ProtocolCommand> updateCommand(new Picto::ProtocolCommand());
183  updateCommand->setMethod("COMPONENTUPDATE");
184  updateCommand->setProtocolName("PICTO");
185  updateCommand->setProtocolVersion("1.0");
186  Q_ASSERT(directorData_);
187  QByteArray directorDataXml(directorData_->toXml().toLatin1());
188  updateCommand->setContent(directorDataXml);
189  updateCommand->setFieldValue("Content-Length",QString::number(directorDataXml.length()));
190  dataChannel->sendCommand(updateCommand);
191  }
192 }
