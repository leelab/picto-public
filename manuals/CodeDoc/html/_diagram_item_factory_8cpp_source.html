129 DiagramItem* DiagramItemFactory::create(QSharedPointer<Asset> asset)
130 {
131  AssetItem* returnVal = NULL;
132  if(asset.isNull())
133  returnVal = NULL;
134  //Note that the order here is very important since, for example, StateMachineElements inherit Asset.
135  if(asset->inherits("Picto::Property"))
136  returnVal = NULL;
137  else if(asset->inherits("Picto::Transition"))
138  returnVal = NULL;
139  else if(asset->inherits("Picto::ControlLink"))
140  returnVal = NULL;
141  else if(asset->inherits("Picto::AnalysisScriptHolder"))
142  returnVal = NULL;
143  else if(asset->inherits("Picto::LogicResult"))
144  returnVal = new WireableResultItem(editorState_, contextMenu_,asset);
145  else if(asset->inherits("Picto::Result"))
146  returnVal = new WireableResultItem(editorState_, contextMenu_,asset);
147  else if(asset->inherits("Picto::StateMachineElement"))
148  returnVal = new StateMachineElementItem(editorState_, contextMenu_, asset);
149  else if (asset->inherits("Picto::DataViewElement"))
150  returnVal = new AnalysisItem(editorState_, contextMenu_, asset);
151  else if(asset->inherits("Picto::ControlElement"))
152  returnVal = new ControlElementItem(editorState_, contextMenu_,asset);
153  else if(asset->inherits("Picto::AnalysisVariable"))
154  returnVal = new AnalysisItem(editorState_, contextMenu_,asset);
155  else if(asset->inherits("Picto::AnalysisFunction"))
156  returnVal = new AnalysisItem(editorState_, contextMenu_,asset);
157  else if(asset->inherits("Picto::UIEnabled"))
158  returnVal = new WireableItem(editorState_, contextMenu_,asset);
159  else if(asset->inherits("Picto::Asset"))
160  returnVal = new AssetItem(editorState_, contextMenu_,asset);
161  else
162  Q_ASSERT_X(false,"DiagramItemFactory::create",QString("Unknown Item Type: %1").arg(asset->metaObject()->className()).toLatin1());
163 
164  if(returnVal)
165  {
166  IconDef iconDef = getIconDef(asset);
167  QRectF iconRect(-iconDef.width_/2.0,-iconDef.height_/2.0,iconDef.width_,iconDef.height_);
168  returnVal->setRect(iconRect);
169  returnVal->setSvgIcon(iconDef.fileName_);
170  QSharedPointer<UIEnabled> uiEnabled = asset.dynamicCast<UIEnabled>();
171  if(uiEnabled)
172  {
173  QPointF pos = uiEnabled->getPos();
174  if(!scene_->sceneRect().contains(pos))
175  pos = scene_->sceneRect().center();
176  returnVal->setPos(pos);
177  }
178  }
179  return returnVal;
180 }
