406 void TokenFactoryGraphic::draw()
407 {
408  int numTokens = getNumTokens();
409  QRect trayDimensions = QRect(QPoint(),propertyContainer_->getPropertyValue("Size").toSize());
410  QRect tokenDimensions = QRect(QPoint(),propertyContainer_->getPropertyValue("TokenSize").toSize());
411  //Make sure size of sizes,colors,shapes lists matches numTokens before drawing
412  QStringList sizes = propertyContainer_->getPropertyValue("TokenSizes").toString().split(",",QString::SkipEmptyParts);
413  QStringList colors = propertyContainer_->getPropertyValue("TokenColors").toString().split(",",QString::SkipEmptyParts);
414  QStringList shapes = propertyContainer_->getPropertyValue("TokenShapes").toString().split(",",QString::SkipEmptyParts);
415  QStringList outlines = propertyContainer_->getPropertyValue("TokenOutlines").toString().split(",",QString::SkipEmptyParts);
416  QStringList outlineWidths = propertyContainer_->getPropertyValue("TokenOutlineWidths").toString().split(",",QString::SkipEmptyParts);
417  QStringList xs = propertyContainer_->getPropertyValue("TokenXs").toString().split(",",QString::SkipEmptyParts);
418  QStringList ys = propertyContainer_->getPropertyValue("TokenYs").toString().split(",",QString::SkipEmptyParts);
419  if ( (sizes.size() != getNumTokens())
420  || (colors.size() != getNumTokens())
421  || (shapes.size() != getNumTokens())
422  || (outlines.size() != getNumTokens())
423  || (outlineWidths.size() != getNumTokens())
424  || (xs.size() != getNumTokens())
425  || (ys.size() != getNumTokens())
426  )
427  {
428  //Stuff is in the process of being updated.
429  return;
430  }
431  //Whenever a list's value is "_" it should juist used the default value
432  for(int i=0;i<numTokens;i++)
433  {
434  if(sizes[i] == "_")
435  sizes[i] = "1.0";
436  if(colors[i] == "_")
437  colors[i] = getColor().name();
438  if(shapes[i] == "_")
439  shapes[i] = getTokenShape();
440  if(outlines[i] == "_")
441  outlines[i] = QString::number(int(getOutline()));
442  if(outlineWidths[i] == "_")
443  outlineWidths[i] = QString::number(getOutlineWidth());
444  if(xs[i] == "_")
445  xs[i] = QString("0");
446  if(ys[i] == "_")
447  ys[i] = QString("0");
448  }
449  QRect imageDimensions = getImageDims();
450  QImage image(imageDimensions.width(),imageDimensions.height(),QImage::Format_ARGB32);
451  image.fill(0);
452  posOffset_ = QPoint(-imageDimensions.left(),-imageDimensions.top());
453 
454  for(int i=0;i<numTokens;i++)
455  {
456  QColor color(colors[i]);
457  double size = sizes[i].toDouble();
458  QRect dimensions(tokenDimensions.x(),tokenDimensions.y(),tokenDimensions.width()*size,tokenDimensions.height()*size);
459  QPoint tokenPosOffset = QPoint(dimensions.width()/2.0,dimensions.height()/2.0);
460  QPoint offset = posOffset_+QPoint(xs[i].toInt(),ys[i].toInt())-tokenPosOffset;
461  QPainter p(&image);
462  QPen pen(color);
463  p.setBrush(color);
464  p.translate(offset);
465  if(outlines[i].toInt())
466  {
467  p.setBrush(QColor(0,0,0,0));
468  pen.setWidth(outlineWidths[i].toInt());
469  }
470  p.setPen(pen);
471  p.setRenderHint(QPainter::Antialiasing, true);
472 
473  if(shapes[i] == "Ellipse")
474  {
475  p.drawEllipse(dimensions);
476  }
477  else if(shapes[i] == "Rectangle")
478  {
479  p.drawRect(dimensions);
480  }
481  else if(shapes[i] == "Diamond")
482  {
483  QPolygon diamond;
484  int maxWPoint = dimensions.width()-1;
485  int maxHPoint = dimensions.height()-1;
486  diamond << QPoint(maxWPoint/2,0)
487  << QPoint(maxWPoint,maxHPoint/2)
488  << QPoint(maxWPoint/2,maxHPoint)
489  << QPoint(0,maxHPoint/2)
490  << QPoint(maxWPoint/2,0);
491  p.drawPolygon(diamond);
492  }
493 
494  p.end();
495  }
496  image_ = image;
497  //updateCompositingSurfaces();
498 
499  shouldUpdateCompositingSurfaces_ = true;
500 }

579 QRect TokenFactoryGraphic::getImageDims()
580 {
581  int defaultOutlineWidth = getOutlineWidth();
582  int defaultWidth = getTokenWidth();
583  int defaultHeight = getTokenHeight();
584  QStringList sizes = propertyContainer_->getPropertyValue("TokenSizes").toString().split(",",QString::SkipEmptyParts);
585  QStringList outlineWidths = propertyContainer_->getPropertyValue("TokenOutlineWidths").toString().split(",",QString::SkipEmptyParts);
586  QStringList xs = propertyContainer_->getPropertyValue("TokenXs").toString().split(",",QString::SkipEmptyParts);
587  QStringList ys = propertyContainer_->getPropertyValue("TokenYs").toString().split(",",QString::SkipEmptyParts);
588 
589  if ( (sizes.size() < getNumTokens())
590  || (outlineWidths.size() < getNumTokens())
591  || (xs.size() < getNumTokens())
592  || (ys.size() < getNumTokens())
593  )
594  return QRect(); //If this is the case, somethings being updated and calling draw and the system won't really need this data yet.
595 
596  QRect returnVal;
597  for(int i=0;i<getNumTokens();i++)
598  {
599  QPoint currPoint;
600  if(xs[i] != "_")
601  currPoint.setX(xs[i].toInt());
602  if(ys[i] != "_")
603  currPoint.setY(ys[i].toInt());
604  int oWidth = defaultOutlineWidth;
605  int height = defaultHeight;
606  int width = defaultWidth;
607  if(sizes[i] != "_")
608  {
609  height *= sizes[i].toDouble();
610  width *= sizes[i].toDouble();
611  }
612  if(outlineWidths[i] != "_")
613  oWidth = outlineWidths[i].toInt();
614  int halfWidth = (width+oWidth)/2;
615  int halfHeight = (height+oWidth)/2;
616  int minX = currPoint.x()-halfWidth;
617  int minY = currPoint.y()-halfHeight;
618  int maxX = currPoint.x()+halfWidth;
619  int maxY = currPoint.y()+halfHeight;
620  if(returnVal.isEmpty())
621  {
622  returnVal = QRect(QPoint(minX,minY),QPoint(maxX,maxY));
623  continue;
624  }
625  if(minX < returnVal.left())
626  returnVal.setLeft(minX);
627  if(minY < returnVal.top())
628  returnVal.setTop(minY);
629  if(maxX > returnVal.right())
630  returnVal.setRight(maxX);
631  if(maxY > returnVal.bottom())
632  returnVal.setBottom(maxY);
633  }
634  return returnVal;
635 }
