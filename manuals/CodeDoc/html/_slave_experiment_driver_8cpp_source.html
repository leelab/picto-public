52 void SlaveExperimentDriver::renderFrame()
53 {
54  if(!currElement_)
55  return;
56 
57  if(frameTimer_.elapsed() < FRAMEMS || !renderingEnabled_)
58  {

81 void SlaveExperimentDriver::handleEvent(SlaveEvent& event)
82 {
83  switch(event.type)
84  {
85  case SlaveEvent::PROP_VAL_CHANGE:
86  {
87  QSharedPointer<Asset> asset = designConfig_->getAsset(event.id);
88  Q_ASSERT(asset && asset->inherits("Picto::Property"));
89  asset.staticCast<Property>()->valFromUserString(event.value);
90  }
91  break;
92  case SlaveEvent::INIT_PROP_VAL_CHANGE:
93  {
94  QSharedPointer<Asset> asset = designConfig_->getAsset(event.id);
95  Q_ASSERT(asset && asset->inherits("Picto::Property"));
96  asset.staticCast<Property>()->initValFromUserString(event.value);
97  }
98  break;
99  case SlaveEvent::TRANS_ACTIVATED:
100  {
101  QSharedPointer<Asset> asset = designConfig_->getAsset(event.id);
102  if(!asset || !asset->inherits("Picto::Transition"))
103  {
104  Q_ASSERT(!asset);
105  return;
106  }
107  QSharedPointer<Transition> trans = asset.staticCast<Transition>();
108 
109  //As soon as a transition comes in and it has a source result,
110  //we set that transition's source's latest result to the transition's
111  //source result. Then, if the source is a state machine element, we
112  //call it's AnalysisExitScripts
113  QSharedPointer<Asset> sourceResult = trans->getSourceResultAsset();
114  QSharedPointer<Asset> sourceAsset = trans->getSourceAsset();
115  if(sourceResult)
116  {
117  sourceAsset.staticCast<ResultContainer>()->setLatestResult(sourceResult->getName());
118  }
119  if(sourceAsset && sourceAsset->inherits("Picto::StateMachineElement"))
120  {
121  sourceAsset.staticCast<StateMachineElement>()->runAnalysisExitScripts();
122  }
123 
124  QString newResult = trans->getDestination();
125  if(newResult == "EngineAbort")
126  {
127  return;
128  }
129 
130  QSharedPointer<Asset> destAsset = trans->getDestinationAsset();
131  QString destName = destAsset?destAsset->getName():"";
132  if(!destAsset || !destAsset->inherits("Picto::StateMachineElement"))
133  {
134  Q_ASSERT(!destAsset || destAsset->inherits("Picto::Result"));
135  return;
136  }
137  //If we got here, we transitioned to a StateMachineElement.
138  //Tell the world if the task has changed
139  QString currTask = destAsset->getPath().split("::").first();
140  if(currTask != currTask_)
141  {
142  currTask_ = currTask;
143  emit taskChanged(currTask_);
144  }
145  //Set it as the current element
146  //Reset its values to initial conditions
147  //run its AnalysisEntryScripts
148  //start it up
149  currElement_ = destAsset.staticCast<StateMachineElement>();
150  currElement_->resetScriptableValues();
151  currElement_->resetScriptableAnalysisValues();
152  currElement_->runAnalysisEntryScripts();
153  currElement_->slaveRun(experiment_->getEngine());
154  }
155  break;
156  }
157 }

216 void SlaveExperimentDriver::masterFramePresented(double time)
217 {
218  //Report the time of the frame following all the property and transition updates that are
219  //about to occur.
220  experiment_->getDesignConfig()->getFrameTimerFactory()->setNextFrameTime(time);
221  //Perform all queued events that occured up to this frame since the last one
222  //(We do this after setting next frame time so that when reading absolute time values during the course
223  //of state machine execution, we will read the absolute time value of the frame that follows
224  //the current position in the state machine. This is necessary since the timing
225  //paradigm in Picto is that all property changes and transitions take effect at the time
226  //of the frame that follows them. Note that this matters in analysis because times
227  //of behavioral events can be compared to times of separately timed signals (eye position,
228  //neural data, etc). See FrameResolutionTimer for more info.
229  SlaveEvent event;
230  while((event = eventQueue_.takeFirstEvent()).type != SlaveEvent::INVALID)
231  {
232  handleEvent(event);
233  }
234  //Render frame
235  renderFrame();
236 
237  //Tell all timers that work on single frame resolution what the latest frame time is
238  experiment_->getDesignConfig()->getFrameTimerFactory()->setLastFrameTime(time);
239 
240  //---------- Run the analysis frame scripts ---------------------------------------
241  QSharedPointer<State> currState = currElement_.dynamicCast<State>();
242  if(currState)
243  {
244  currState->runAnalysisFrameScripts();
245  }
246 }
