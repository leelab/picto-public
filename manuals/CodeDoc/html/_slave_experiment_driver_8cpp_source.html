75 void SlaveExperimentDriver::renderFrame()
76 {
77  if(!currElement_)
78  return;
79 
80  if(frameTimer_.elapsed() < FRAMEMS || !renderingEnabled_)
81  {
82  //Even if we don't render the frame, we need to make sure that the signal channels' data is getting cleared out
83  // each frame. Otherwise they'll become huge and use up all our memory. In general, slaveRenderFrame() takes

103 void SlaveExperimentDriver::handleEvent(SlaveEvent& event)
104 {
105  //These conditions should only fulfilled when joining an experiment mid-session
106  if (!initialized_ && !(event.id < 0 && event.type == SlaveEvent::TRANS_ACTIVATED))
107  {
108  QSharedPointer<Asset> asset = designConfig_->getAsset(event.id);
109  while (asset && !asset->inherits("Picto::Task"))
110  {
111  asset = asset->getParentAsset();
112  }
113 
114  if (asset)
115  {
116  remoteRunStarting(asset->getName());
117  }
118  }
119 
120  switch(event.type)
121  {
122  case SlaveEvent::PROP_VAL_CHANGE:
123  {
124  QSharedPointer<Asset> asset = designConfig_->getAsset(event.id);
125  Q_ASSERT(asset && asset->inherits("Picto::Property"));
126  asset.staticCast<Property>()->valFromUserString(event.value);
127  }
128  break;
129  case SlaveEvent::INIT_PROP_VAL_CHANGE:
130  {
131  QSharedPointer<Asset> asset = designConfig_->getAsset(event.id);
132  Q_ASSERT(asset && asset->inherits("Picto::Property"));
133  asset.staticCast<Property>()->initValFromUserString(event.value);
134  }
135  break;
136  case SlaveEvent::TRANS_ACTIVATED:
137  {
138  QSharedPointer<Asset> asset = designConfig_->getAsset(event.id);
139  if(!asset || !asset->inherits("Picto::Transition"))
140  {
141  Q_ASSERT(!asset);
142  if (!event.value.isEmpty())
143  {
144  remoteRunEnding();
145  }
146  return;
147  }
148 
149  QSharedPointer<Transition> trans = asset.staticCast<Transition>();
150 
151  //As soon as a transition comes in and it has a source result, we set that transition's source's latest
152  // result to the transition's source result. Then, if the source is a state machine element, we call its
153  // AnalysisExitScripts
154  QSharedPointer<Asset> sourceResult = trans->getSourceResultAsset();
155  QSharedPointer<Asset> sourceAsset = trans->getSourceAsset();
156  if(sourceResult)
157  {
158  sourceAsset.staticCast<ResultContainer>()->setLatestResult(sourceResult->getName());
159  }
160  if(sourceAsset && sourceAsset->inherits("Picto::StateMachineElement"))
161  {
162  sourceAsset.staticCast<StateMachineElement>()->runAnalysisExitScripts();
163  }
164 
165  QString newResult = trans->getDestination();
166  if(newResult == "EngineAbort")
167  {
168  return;
169  }
170 
171  QSharedPointer<Asset> destAsset = trans->getDestinationAsset();
172  QString destName = destAsset?destAsset->getName():"";
173  if(!destAsset || !destAsset->inherits("Picto::StateMachineElement"))
174  {
175  Q_ASSERT(!destAsset || destAsset->inherits("Picto::Result"));
176  if (event.id < 0 && !event.value.isEmpty())
177  {
178  QString currTask = "";
179  if (trans->getParentAsset())
180  currTask = trans->getParentAsset()->getName();
181  remoteRunStarting(currTask);
182  }
183  return;
184  }
185  //If we got here, we transitioned to a StateMachineElement.
186  //Tell the world if the task has changed
187  QString currTask = destAsset->getPath().split("::").first();
188  if(currTask != currTask_)
189  {
190  currTask_ = currTask;
191  emit taskChanged(currTask_);
192  }
193  //Set it as the current element
194  //Reset its values to initial conditions
195  //run its AnalysisEntryScripts
196  //start it up
197  currElement_ = destAsset.staticCast<StateMachineElement>();
198  currElement_->resetScriptableValues();
199  currElement_->resetScriptableAnalysisValues();
200  currElement_->runAnalysisEntryScripts();
201  currElement_->slaveRun(experiment_->getEngine());
202  }
203  break;
204  }
205 }

336 void SlaveExperimentDriver::masterFramePresented(double time)
337 {
338  //Report the time of the frame following all the property and transition updates that are about to occur.
339  experiment_->getDesignConfig()->getFrameTimerFactory()->setNextFrameTime(time);
340 
341  if (experiment_->getDesignConfig()->getFrameReader().objectCast<LiveFrameReader>())
342  {
343  experiment_->getDesignConfig()->getFrameReader().objectCast<LiveFrameReader>()->setLatestFrameTime(time);
344  }
345 
346  //Perform all queued events that occured up to this frame since the last one
347  // (We do this after setting next frame time so that when reading absolute time values during the course of state
348  // machine execution, we will read the absolute time value of the frame that follows the current position in the
349  // state machine. This is necessary since the timing paradigm in Picto is that all property changes and transitions
350  // take effect at the time of the frame that follows them. Note that this matters in analysis because times of
351  // behavioral events can be compared to times of separately timed signals (eye position, neural data, etc).
352  // See FrameResolutionTimer for more info.
353  handleEvents();
354  //Render frame
355  renderFrame();
356 
357  //Tell all timers that work on single frame resolution what the latest frame time is
358  experiment_->getDesignConfig()->getFrameTimerFactory()->setLastFrameTime(time);
359 
360  //---------- Run the analysis frame scripts ---------------------------------------
361  QSharedPointer<State> currState = currElement_.dynamicCast<State>();
362  if (currState && !paused_)
363  {
364  currState->runAnalysisFrameScripts();
365  }
366 }
