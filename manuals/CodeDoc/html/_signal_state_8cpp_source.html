24 void SignalState::setDatabase(QSqlDatabase session)
25 {
26  runStart_ = runEnd_ = curr_ = -1;
27  session_ = session;
28  QSqlQuery query(session_);
29  query.exec(QString("SELECT COUNT(*) FROM %1").arg(tableName_));
30  if(!query.exec() || !query.next())
31  {
32  Q_ASSERT(false);
33  return;
34  }
35  data_.clear();
36  data_.reserve(query.value(0).toInt());
37  for(int i=0;i<query.value(0).toInt();i++)
38  {
39  data_.append(PlaybackSignalData());
40  }
41 
42  query.exec(QString("SELECT f.time,s.offsettime,s.data,f.dataid "
43  "FROM %1 s,frames f WHERE f.dataid=s.frameid ORDER BY f.dataid").arg(tableName_));
44 
45  if(!query.exec())
46  {
47  Q_ASSERT(false);
48  return;
49  }
50  PlaybackIndex index;
51  QString val;
52  int arrayIndex = 0;
53  QByteArray byteArray;
54  int numEntries;
55  qulonglong lastFrameId = 0; //Before 6/2013 data ids of signals in session file were not being transferred
56  //to the session file correctly. This means that if their signal data got
57  //sent more than once from the diretor due to a server slowdown, it could
58  //appear in the session file more than once. For this reason, we store the
59  //last frameId and throw out any data that arrives with a frame id that
60  //duplicates the last one.
61  qulonglong currFrameId;
62  while(query.next())
63  {
64  Q_ASSERT(arrayIndex < data_.size());
65  currFrameId = query.value(3).toLongLong();
66  if(lastFrameId == currFrameId)
67  continue; //See big comment a few lines up
68  lastFrameId = currFrameId;
69  //Note: With signals, the definition is such that offsetTime after the frameTime of frameId is when the first signal data was read.
70  data_[arrayIndex] = PlaybackSignalData(query.value(0).toDouble()+query.value(1).toDouble());
71  byteArray = query.value(2).toByteArray();
72  numEntries = byteArray.size()/sizeof(float);
73  const float* floatArray = reinterpret_cast<const float*>(byteArray.constData());
74  data_[arrayIndex].vals_.resize(numEntries);
75  for(int i=0;i<numEntries;i++)
76  {
77  data_[arrayIndex].vals_[i] = floatArray[i];
78  }
79  arrayIndex++;
80  }
81  //In cases where there was duplicated data, we bring the data_ array size down to where it should bec (see comments above)
82  if(data_.size() > arrayIndex)
83  data_.erase(data_.begin()+arrayIndex,data_.end());
84 }

119 void SignalState::moveToIndex(PlaybackIndex index)
120 {
121  Q_ASSERT(runStart_ >= 0);
122  Q_ASSERT(index >= getCurrentIndex());
123  PlaybackIndex nextIndex = getNextIndex();
124  QVector<float> sigVec(numSubChans_);
125  while(nextIndex.isValid() && nextIndex <= index)
126  {
127  goToNext();
128  for(int i=0;i<numSubChans_;i++)
129  {
130  sigVec[i]=data_[curr_].vals_[currSub_*numSubChans_+i];
131  }
132  emit signalChanged(name_,subChanNames_,sigVec);
133  nextIndex = getNextIndex();
134  }

185 QVariantList SignalState::getValuesSince(QString channel,double time)
186 {
187  Q_ASSERT(runStart_ >= 0);
188  Q_ASSERT(curr_ >= 0);
189  if(!subChanIndexLookup_.contains(channel))
190  return QVariantList();
191  int chanIndex = subChanIndexLookup_[channel];
192  double afterTime = time;
193  double latestTime = getLatestTime();
194  if(afterTime >= latestTime)
195  return QVariantList();
196  double upToGlobalTime = data_[curr_].time_+double(currSub_)*sampPeriod_;
197  PlaybackSignalData beyondVal = PlaybackSignalData(afterTime+runStart_);
198  QList<PlaybackSignalData>::iterator iter = qLowerBound<QList<PlaybackSignalData>::iterator,PlaybackSignalData>(data_.begin(),curr_>=0?data_.begin()+curr_:data_.begin(),beyondVal);
199  //The above line puts the iterator at the data_ entry at or one after the one containing the beyondVal,
200  //so we move it backward by one to assure that its low enough for us to capture all relevant data
201  if(iter != data_.begin())
202  iter--;
203  QVariantList returnVal;
204  double sampleTime;
205  for (;((iter != data_.end()) && (iter->time_ <= upToGlobalTime)); iter++)
206  {
207  for(int i=0;i<iter->vals_.size();i+=numSubChans_)
208  {
209  sampleTime = iter->time_ + (i/numSubChans_)*sampPeriod_;
210  if(sampleTime <= beyondVal.time_)
211  continue;
212  if(sampleTime > upToGlobalTime)
213  break;
214  returnVal.append(iter->vals_[i+chanIndex]);
215  }
216  }
217  //Q_ASSERT(iter != data_.end());
218  return returnVal;
219 }

334 bool SignalState::moveIndecesToNextTime(int& outerIndex, int& innerIndex)
335 {
336  Q_ASSERT(runStart_ >= 0);
337  int nextOuterIndex = outerIndex;
338  int nextInnerIndex = innerIndex;
339  if(nextOuterIndex < 0 && data_.size())
340  {
341  nextOuterIndex = 0;
342  nextInnerIndex = -1;
343  }
344  if(nextInnerIndex >= data_[nextOuterIndex].vals_.size()/numSubChans_-1)
345  {
346  if(nextOuterIndex >= data_.size()-1)
347  return false;
348  nextOuterIndex++;
349  nextInnerIndex = -1;
350  }
351  nextInnerIndex++;
352  outerIndex = nextOuterIndex;
353  innerIndex = nextInnerIndex;
354  return true;
355 }
