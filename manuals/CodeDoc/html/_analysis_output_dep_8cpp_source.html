41 //Will create an empty AnalysisOutputDep directory in the runpath once
42 //per executable run, clearing out any previous contents if there
43 //were any.
44 void AnalysisOutputDep::initTempOutputDir()
45 {
46  //We want to make sure that even if multiple Picto Workstations are open at the
47  //same time, no one will remove the data that we are currently using. We also
48  //want to erase the data if we are just opening for the first time and no one else
49  //is using the data in the temporary folder. For this reason, we attempt to remove
50  //all files at the top level of the temporary directory, then if successful, remove
51  //all other contained directories as well. We then create a top level file in the
52  //temporary directory and leave it open until the application closes.
53  //If another workstation is open, we won't succeed in removing its open file so we
54  //wont delete the temporary directory contents, otherwise, we will.
55  if(!outputDir_.isEmpty())
56  return;
57  QDir dir(QCoreApplication::applicationDirPath()+"/../");
58  dir.mkdir("analysisoutput");
59  dir.cd("analysisoutput");
60  removeFilesThenDirectories(dir);
61 
62  outputDir_ = dir.absolutePath();
63 
64  QString fileName = outputDir_+"/"+QUuid::createUuid().toString();
65  lockFile_ = QSharedPointer<QFile>(new QFile(fileName));
66  bool rc = lockFile_->open(QIODevice::WriteOnly | QIODevice::Text);
67  Q_ASSERT(rc);
68 
69 }

71 void AnalysisOutputDep::removeFilesThenDirectories(QDir container)
72 {
73  QStringList files = container.entryList(QDir::Files);
74  foreach(QString fileName,files)
75  {
76  if(!container.remove(fileName))
77  {
78  return;
79  }
80  }
81  QStringList directories = container.entryList(QDir::Dirs|QDir::NoDotAndDotDot|QDir::NoSymLinks);
82  foreach(QString directory,directories)
83  {
84  if(!container.rmdir(directory))
85  {
86  QDir subDir = container;
87  subDir.cd(directory);
88  removeFilesThenDirectories(subDir);
89  container.rmdir(directory);
90  }
91  }
92 }
