40 void D3DCompositingSurface::convertImage(QImage image)
41 {
42  HRESULT hr;
43 
44  //Figure out the dimensions of the texture.
45  //NOTE: This assumes that the textures must be 2^n x 2^m
46  int texWidth = 1;
47  int texHeight = 1;
48 
49  while(texWidth < image.width())
50  texWidth *= 2;
51  while(texHeight < image.height())
52  texHeight *= 2;
53 
54  //If we don't have a valid texture, or is the incorrect size,
55  //we will need to generate a new texture interface
56  bool generateNewTexture = false;
57  if(!texture_)
58  {
59  generateNewTexture = true;
60  }
61  else
62  {
63  D3DSURFACE_DESC texDesc;
64  texture_->GetLevelDesc(0,&texDesc);
65  if(texDesc.Height != texHeight)
66  generateNewTexture = true;
67  else if(texDesc.Width != texWidth)
68  generateNewTexture = true;
69  }
70 
71  if(generateNewTexture)
72  {
73  if(texture_)
74  texture_->Release();
75 
76  // Determine if the device can create textures in video memory
77  // by testing the device caps bits.
78  D3DCAPS9 caps;
79  hr = pD3dDevice_->GetDeviceCaps(&caps);
80  if(hr)
81  d3dFail("D3DCompositingSurface: GetDeviceCaps FAILED");
82 
83  //Create a texture with dimensions large enough for the image
84  hr = pD3dDevice_->CreateTexture(texWidth,texHeight,1,0,D3DFMT_A8R8G8B8,D3DPOOL_MANAGED,&texture_, NULL);
85  if(hr)
86  d3dFail("D3DCompositingSurface: CreateTexture FAILED");
87  }
88 
89  //store the image height (since this information will be lost when we
90  //place the image on a texture). We need this info to accurately determine
91  //the coordinates
92  imageHeight_ = image.height();
93 
94  //if needed, convert the image format
95  if(image.format() != QImage::Format_ARGB32)
96  image = image.convertToFormat(QImage::Format_ARGB32);
97 
98 
99  //lock the entire texture
100  D3DLOCKED_RECT lockedRect;
101  hr = texture_->LockRect(0,&lockedRect,0,0);
102  if(hr)
103  d3dFail("D3DCompositingSurface: texture_->LockRect FAILED");
104 
105  DWORD *pTexel;
106  pTexel = (DWORD*)lockedRect.pBits;
107 
108  //clear the existing texture
109  memset(pTexel,0,texHeight*lockedRect.Pitch);
110 
111 
112  //NOTE: The following code assumes that the pitch is a multiple of 4
113  //so far this has always been true. If there are rendering problems,
114  //this may be a cause...
115  Q_ASSERT_X(lockedRect.Pitch%4 == 0, "D3DCompositingSurface::convertImage", "pitch not a multiple of 4");
116 
117  //be sure to use a const pointer since we won't be modifying the image data and want to avoid an expensive and unnecessary deep copy
118  const uchar * imageData = image.bits();
119  const unsigned int * imageDataUint32 = (const unsigned int *) imageData;
120 
121  unsigned int upperBound = (unsigned int) pTexel + image.height() * lockedRect.Pitch;
122  unsigned int incrementAmount = lockedRect.Pitch;
123  unsigned int widthBound = image.width();
124 
125  unsigned int rowDataPointer, columnDataOffset;
126 
127  for(rowDataPointer=(unsigned int) pTexel;rowDataPointer<upperBound; rowDataPointer+=incrementAmount)
128  {
129  for(columnDataOffset=0; columnDataOffset < widthBound; columnDataOffset++)
130  {
131  *((unsigned int *) (rowDataPointer+columnDataOffset*sizeof(int))) = *imageDataUint32++;
132  }
133 
134  }
135  texture_->UnlockRect(0);
136 
137 }
