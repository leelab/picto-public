331 void RecordingVisualTargetHost::updateRecordingTime()
332 {
333  //Get the last recording time
334  double lastFrameTime = getRecordingTime();
335  //If it was less than zero, the new time will be zero
336  //Otherwise, new time equals last time plus the time on the frame timer clock
337  double newFrameTime = 0;
338  if (lastFrameTime >= 0)
339  {
340  newFrameTime = lastFrameTime + frameTimer_.elapsed();
341  }
342  //Restart the frame timer clock
343  frameTimer_.restart();
344 
345  //Store the new recording time
346  setRecordingTime(newFrameTime);
347 }

363 void RecordingVisualTargetHost::setRecordingTime(int currMs)
364 {
365  //If time didn't change, do nothing
366  if (currFrameMs_ == currMs)
367  return;
368  //Update the stored current frame time value
369  currFrameMs_ = currMs;
370  //If the new current frame time value is less than zero, it is an
371  //initialization value, emit a time of zero to listeners
372  if (currFrameMs_ < 0)
373  {
374  emit updateRecordingTime(0);
375  return;
376  }
377  //Otherwise, emit the current time in seconds to listeners
378  emit updateRecordingTime(double(currFrameMs_) / 1000.0);
