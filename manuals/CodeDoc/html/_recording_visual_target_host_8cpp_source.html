220 void RecordingVisualTargetHost::updateRecordingTime()
221 {
222  //Get the last recording time
223  double lastFrameTime = getRecordingTime();
224  //If it was less than zero, the new time will be zero
225  //Otherwise, new time equals last time plus the time on the frame timer clock
226  double newFrameTime = 0;
227  if(lastFrameTime >= 0)
228  {
229  newFrameTime = lastFrameTime + frameTimer_.elapsed();
230  }
231  //Restart the frame timer clock
232  frameTimer_.restart();
233 
234  //Store the new recording time
235  setRecordingTime(newFrameTime);
236 }

253 void RecordingVisualTargetHost::setRecordingTime(int currMs)
254 {
255  //If time didn't change, do nothing
256  if(currFrameMs_ == currMs)
257  return;
258  //Update the stored current frame time value
259  currFrameMs_ = currMs;

263  {
264  emit updateRecordingTime(0);
265  return;
266  }
267  //Otherwise, emit the current time in seconds to listeners
