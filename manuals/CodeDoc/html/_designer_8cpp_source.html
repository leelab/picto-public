483 void Designer::updateEnabledActions()
484 {
485  QSharedPointer<Asset> asset = editorState_->getSelectedAsset();
486  if(editorState_->getCurrentAnalysis())
487  { //An Analysis is active
488  analysisExportAction->setVisible(true);
489  analysisImportAction->setVisible(true);
490  if(!asset)
491  {
492  deleteAction->setEnabled(false);
493  copyAction->setEnabled(false);
494  analysisExportAction->setEnabled(false);
495  analysisImportAction->setEnabled(false);
496  pasteAction->setEnabled(false);
497  }
498  else if(dynamic_cast<AssociateElement*>(asset.data()))
499  { //The asset is an analysis element
500  deleteAction->setEnabled(true);
501  copyAction->setEnabled(true);
502  analysisExportAction->setEnabled(false);
503  analysisImportAction->setEnabled(false);
504  pasteAction->setEnabled(false);
505  }
506  else //The asset is an experiment element
507  {
508  deleteAction->setEnabled(false);
509  copyAction->setEnabled(false);
510  analysisExportAction->setEnabled(asset->inherits("Picto::StateMachineElement") || asset->inherits("Picto::Task"));
511  analysisImportAction->setEnabled((asset->inherits("Picto::StateMachineElement") || asset->inherits("Picto::Task")) && (Copier::availablePasteType() == Copier::ANALYSIS_IMPORT));
512  pasteAction->setEnabled(false);
513  }
514  }
515  else //No Analysis is active
516  {
517  analysisExportAction->setVisible(false);
518  analysisImportAction->setVisible(false);
519 
520  if(!asset)
521  {
522  deleteAction->setEnabled(false);
523  copyAction->setEnabled(false);
524  analysisExportAction->setEnabled(false);
525  analysisImportAction->setEnabled(false);
526  pasteAction->setEnabled(false);
527  }
528  else
529  {
530  deleteAction->setEnabled(true);
531  copyAction->setEnabled(true);
532  analysisExportAction->setEnabled(false);
533  analysisImportAction->setEnabled(false);
534  pasteAction->setEnabled((Copier::availablePasteType() == Copier::EXPERIMENT_PASTE));
535  }
536  }
537  //If the window is the selected asset, copy should be disabled,
538  //paste depends on whether we're in analysis mode
539  if(asset && (asset == editorState_->getWindowAsset()))
540  {
541  copyAction->setEnabled(false);
542  analysisExportAction->setEnabled(false);
543  analysisImportAction->setEnabled(false);
544  bool pasteEnabled = (editorState_->getCurrentAnalysis().isNull() && (Copier::availablePasteType() == Copier::EXPERIMENT_PASTE))
545  || (!editorState_->getCurrentAnalysis().isNull() && (Copier::availablePasteType() == Copier::ANALYSIS_PASTE));
546  pasteAction->setEnabled(pasteEnabled);
547  }
548  //If the window is the selected asset, and the selected item isn't a transition delete should be disabled
549  if ( asset && (editorState_->getSelectedAsset() == editorState_->getWindowAsset())
550  && (dynamic_cast<Arrow*>(editorState_->getSelectedItem()) == NULL)
551  )
552  {
553  deleteAction->setEnabled(false);
554  }
555 }
