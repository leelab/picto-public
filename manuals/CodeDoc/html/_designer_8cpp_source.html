472 void Designer::updateEnabledActions()
473 {
474  QSharedPointer<Asset> asset = editorState_->getSelectedAsset();
475  if(editorState_->getCurrentAnalysis())
476  { //An Analysis is active
477  analysisExportAction->setVisible(true);
478  analysisImportAction->setVisible(true);
479  if(!asset)
480  {
481  deleteAction->setEnabled(false);
482  copyAction->setEnabled(false);
483  analysisExportAction->setEnabled(false);
484  analysisImportAction->setEnabled(false);
485  pasteAction->setEnabled(false);
486  }
487  else if(dynamic_cast<AssociateElement*>(asset.data()))
488  { //The asset is an analysis element
489  deleteAction->setEnabled(true);
490  copyAction->setEnabled(true);
491  analysisExportAction->setEnabled(false);
492  analysisImportAction->setEnabled(false);
493  pasteAction->setEnabled(false);
494  }
495  else //The asset is an experiment element
496  {
497  deleteAction->setEnabled(false);
498  copyAction->setEnabled(false);
499  analysisExportAction->setEnabled(asset->inherits("Picto::StateMachineElement") || asset->inherits("Picto::Task"));
500  analysisImportAction->setEnabled((asset->inherits("Picto::StateMachineElement") || asset->inherits("Picto::Task")) && (Copier::availablePasteType() == Copier::ANALYSIS_IMPORT));
501  pasteAction->setEnabled(false);
502  }
503  }
504  else //No Analysis is active
505  {
506  analysisExportAction->setVisible(false);
507  analysisImportAction->setVisible(false);
508 
509  if(!asset)
510  {
511  deleteAction->setEnabled(false);
512  copyAction->setEnabled(false);
513  analysisExportAction->setEnabled(false);
514  analysisImportAction->setEnabled(false);
515  pasteAction->setEnabled(false);
516  }
517  else
518  {
519  deleteAction->setEnabled(true);
520  copyAction->setEnabled(true);
521  analysisExportAction->setEnabled(false);
522  analysisImportAction->setEnabled(false);
523  pasteAction->setEnabled((Copier::availablePasteType() == Copier::EXPERIMENT_PASTE));
524  }
525  }
526  //If the window is the selected asset, copy should be disabled,
527  //paste depends on whether we're in analysis mode
528  if(asset && (asset == editorState_->getWindowAsset()))
529  {
530  copyAction->setEnabled(false);
531  analysisExportAction->setEnabled(false);
532  analysisImportAction->setEnabled(false);
533  bool pasteEnabled = (editorState_->getCurrentAnalysis().isNull() && (Copier::availablePasteType() == Copier::EXPERIMENT_PASTE))
534  || (!editorState_->getCurrentAnalysis().isNull() && (Copier::availablePasteType() == Copier::ANALYSIS_PASTE));
535  pasteAction->setEnabled(pasteEnabled);
536  }
537  //If the window is the selected asset, and the selected item isn't a transition delete should be disabled
538  if ( asset && (editorState_->getSelectedAsset() == editorState_->getWindowAsset())
539  && (dynamic_cast<Arrow*>(editorState_->getSelectedItem()) == NULL)
540  )
541  {
542  deleteAction->setEnabled(false);
543  }
544 }
