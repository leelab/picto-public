158 bool FixationController::isDonePrivate(QSharedPointer<Engine::PictoEngine> engine)
159 {
160  Controller::TimerUnits::TimerUnits timeUnits;
161  if(unitList_.value(propertyContainer_->getPropertyValue("TimeUnits").toInt(),"") == "Sec")
162  timeUnits = Controller::TimerUnits::sec;
163  else if(unitList_.value(propertyContainer_->getPropertyValue("TimeUnits").toInt(),"") == "Ms")
164  timeUnits = Controller::TimerUnits::ms;
165  else if(unitList_.value(propertyContainer_->getPropertyValue("TimeUnits").toInt(),"") == "Us")
166  timeUnits = Controller::TimerUnits::us;
167  else
168  {
169  timeUnits = Controller::TimerUnits::ms;
170  Q_ASSERT(false);
171  }
172 
173  int fixTime = getFixationTime();
174 
175  int currAcqTime = acquisitionTimer_->elapsedTime(timeUnits);
176  int currReAcqTime = reacquisitionTimer_->elapsedTime(timeUnits);
177 
178  bool isInsideTarget = insideTarget(engine);
179 
180  //just entered target
181  if(!targetAcquired_ && isInsideTarget)
182  {
183  //if we're past the Min Initial Acquisition Time mark the target as acquired and start running the timer
184  if (initialAcquisitionOccurred_ || currAcqTime >= getMinAcqTime())
185  {
186  if (!initialAcquisitionOccurred_)
187  {
188  acquisitionTimer_->start();
189  currAcqTime = 0;
190  }
191  initialAcquisitionOccurred_ = true;
192 
193  targetAcquired_ = true;
194 
195  if(!engine->slaveMode())
196  {
197  propertyContainer_->getProperty("OnTarget")->setValue(true);
198  propertyContainer_->getProperty("OnTargetChanged")->setValue(true);
199  }
200  //If fixation time is zero, we're done
201  if(fixTime <= 0)
202  {
203  isDone_ = true;
204  result_ = "Success";
205  return true;
206  }
207  }
208  }
209  //just left target
210  else if(targetAcquired_ && !isInsideTarget)
211  {
212  targetAcquired_ = false;
213 
214  if(!engine->slaveMode())
215  {
216  propertyContainer_->getProperty("OnTarget")->setValue(false);
217  propertyContainer_->getProperty("OnTargetChanged")->setValue(true);
218  }
219 
220  //If reacquisition isn't allowed, then we're done with a failure value.
221  if (getMaxReacqTime() <= 0)
222  {
223  isDone_ = true;
224  result_ = "Broke Fixation";
225  return true;
226  }
227  else
228  {
229  waitingForReacquisition_ = true;
230  reacquisitionTimer_->start();
231  }
232  }
233  //staying inside target
234  else if(targetAcquired_ && isInsideTarget)
235  {
236  //We have no particular tasks to perform when staying in the target.
237  }
238  //staying outside target
239  else if(!targetAcquired_ && !isInsideTarget)
240  {
241  //check to see if we've met or exceeded the max acquisition time (if relevant)
242  if(!initialAcquisitionOccurred_)
243  {
244  if (currAcqTime >= getMaxAcqTime())
245  {
246  isDone_ = true;
247  result_ = "Initial Acquisition Time Exceeded";
248  return true;
249  }
250  }
251 
252  //check to see if we've met or exceeded the reacquisition time
253  if(waitingForReacquisition_)
254  {
255  if (currReAcqTime >= getMaxReacqTime())
256  {
257  isDone_ = true;
258  result_ = "Broke Fixation";
259  return true;
260  }
261  }
262 
263  }
264 
265  //If the user has initially fixated, and they've made it to the end of the fixation window without failure, then
266  //they have, by definition, succeeded.
267  if (initialAcquisitionOccurred_ && currAcqTime >= fixTime)
268  {
269  isDone_ = true;
270  result_ = "Success";
271  return true;
272  }
273 
274  return false;
275 }

279 bool FixationController::insideTarget(QSharedPointer<Engine::PictoEngine> engine)
280 {
281  if(engine.isNull())
282  return false;
283  if(signal_.isNull())
284  {
285  //grab the signal channel
286  signal_ = engine->getSignalChannel(propertyContainer_->getPropertyValue("SignalChannel").toString());
287  if(signal_.isNull())
288  return false;
289 
290  //confirm that the signal channel contains "x" and "y" subchannels
291  Q_ASSERT(signal_->getSubchannels().contains("x"));
292  Q_ASSERT(signal_->getSubchannels().contains("y"));
293  }
294 
295  int x = signal_->peekValue("x");

297  if(controlTarget_ && controlTarget_.toStrongRef()->contains(x,y))
298  return true;
