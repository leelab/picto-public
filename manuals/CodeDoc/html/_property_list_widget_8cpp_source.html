162 void PropertyListWidget::propertyWasEdited(QSharedPointer<Property> prop,QVariant val)
163 {
164  //We use QMetaObject::invokeMethod() below instead of calling the methods directly
165  //so that we won't encounter any multithreaded write issues. By using this method,
166  //if the properties are in another thread, the function call will be scheduled to
167  //run in that thread's event loop automatically. If the property is in the current
168  //thread, the function will be called immediately. This is one of the few places
169  //where the UI thread and an experiment that might possibly in a separate thread need
170  //to communicate.
171  if(trackInitVals_)
172  QMetaObject::invokeMethod(prop.data(),"setInitValue",Q_ARG(QVariant,val));
173  else
174  QMetaObject::invokeMethod(prop.data(),"setValue",Q_ARG(QVariant,val));
175  emit propertyEdited(prop,val);
176 }
