154 void PropertyListWidget::propertyWasEdited(QSharedPointer<Property> prop,QVariant val)
155 {
156  //We use QMetaObject::invokeMethod() below instead of calling the methods directly
157  //so that we won't encounter any multithreaded write issues. By using this method,
158  //if the properties are in another thread, the function call will be scheduled to
159  //run in that thread's event loop automatically. If the property is in the current
160  //thread, the function will be called immediately. This is one of the few places
161  //where the UI thread and an experiment that might possibly in a separate thread need
162  //to communicate.
163  if(trackInitVals_)
164  QMetaObject::invokeMethod(prop.data(),"setInitValue",Q_ARG(QVariant,val));
165  else
166  QMetaObject::invokeMethod(prop.data(),"setValue",Q_ARG(QVariant,val));
167  emit propertyEdited(prop,val);
168 }
