303 void SessionState::updatePercentLoaded()
304 {
305  //Handle lfp separately because its so big and so that
306  //a meaningful value can appear here when it is being loaded
307  //on its own
308  double currLoadedStates = loadedStates_;
309  if(lfpPercent_ < 100)
310  {
311  if(loadedStates_ == totalSubStates_)
312  currLoadedStates = loadedStates_-1;
313  currLoadedStates += double(lfpPercent_)/100.0;
314  }
315  if(totalSubStates_ > 0)
316  emit percentLoaded(100.0*currLoadedStates/double(totalSubStates_));
317 }
