323 void SessionState::updatePercentLoaded()
324 {
325  //Handle lfp separately because its so big and so that
326  //a meaningful value can appear here when it is being loaded
327  //on its own
328  double currLoadedStates = loadedStates_;
329  if(lfpPercent_ < 100)
330  {
331  if(loadedStates_ == totalSubStates_)
332  currLoadedStates = loadedStates_-1;
333  currLoadedStates += double(lfpPercent_)/100.0;
334  }
335  if(totalSubStates_ > 0)
336  emit percentLoaded(100.0*currLoadedStates/double(totalSubStates_));
337 }
