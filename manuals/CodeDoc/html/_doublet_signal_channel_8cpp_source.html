99 double DoubletSignalChannel::peekValue(QString subchannel)
100 {
101  getDataFromPort();
102 
103  if(!rawDataBuffer_.contains(subchannel))
104  return 0.0;
105  QString shearAsFuncOf = scaleFactorsMap_.value(subchannel).shearAsFuncOf;
106  Q_ASSERT(shearAsFuncOf.isEmpty() || rawDataBuffer_.contains(shearAsFuncOf));
107  //Update the rawDataLastValue_ map with this subchannel and a shearAsFuncOf channel
108  if(!rawDataBuffer_.value(subchannel).isEmpty())
109  {
110  rawDataLastValue_[subchannel] = rawDataBuffer_.value(subchannel).last();
111  }
112  if(!shearAsFuncOf.isEmpty() && !rawDataBuffer_.value(shearAsFuncOf).isEmpty())
113  {
114  rawDataLastValue_[shearAsFuncOf] = rawDataBuffer_.value(shearAsFuncOf).last();
115  }
116  Q_ASSERT(rawDataLastValue_.contains(subchannel));
117  Q_ASSERT(shearAsFuncOf.isEmpty() || rawDataLastValue_.contains(subchannel));
118 
119  //Get the lastRawDataValue, scale and shear it.
120  double rawValue = rawDataLastValue_.value(subchannel);
121  double scaledValue = scaleFactorsMap_.value(subchannel).scaleA +
122  scaleFactorsMap_.value(subchannel).scaleB * rawValue;
123  if(!shearAsFuncOf.isEmpty())
124  {
125  double otherRawValue = rawDataLastValue_.value(shearAsFuncOf);
126  double otherScaledValue = scaleFactorsMap_.value(shearAsFuncOf).scaleA +
127  scaleFactorsMap_.value(shearAsFuncOf).scaleB * otherRawValue;
128  scaledValue = scaledValue + scaleFactorsMap_.value(subchannel).shearFactor * (otherScaledValue - scaleFactorsMap_.value(shearAsFuncOf).centerVal);
129  }
130  //qDebug(QString("%1: Last: %2 Scaled: %3").arg(rawDataLastValue_[subchannel]).arg(subchannel).arg(scaledValue).toLatin1());
131  return scaledValue;
132 }

139 QMap<QString, QVector<double> > DoubletSignalChannel::getValues()
140 {
141  QMap<QString, QVector<double> > dataBuffer = getRawValues();
142 
143  //scale the values
144  QMap<QString, QVector<double> >::iterator x = dataBuffer.begin();
145 
146  while(x != dataBuffer.end())
147  {
148  for(QVector<double>::iterator subChanValIter = x.value().begin();
149  subChanValIter != x.value().end();
150  subChanValIter++)
151  {
152  double scaledValue = scaleFactorsMap_.value(x.key()).scaleA +
153  scaleFactorsMap_.value(x.key()).scaleB * (*subChanValIter);
154  (*subChanValIter) = scaledValue;
155  }
156  x++;
157  }
158 
159  //Add shear
160  x = dataBuffer.begin();
161  while(x != dataBuffer.end())
162  {
163  QString shearAsFuncOf = scaleFactorsMap_.value(x.key()).shearAsFuncOf;
164  if(shearAsFuncOf.isEmpty())
165  {
166  x++;
167  continue;
168  }
169 
170  QVector<double>::iterator funcOfIterator = dataBuffer.find(shearAsFuncOf).value().begin();
171  for(QVector<double>::iterator subChanValIter = x.value().begin();
172  subChanValIter != x.value().end();
173  subChanValIter++,funcOfIterator++)
174  {
175  double shearedVal = (*subChanValIter) +
176  scaleFactorsMap_.value(x.key()).shearFactor * ((*funcOfIterator)-scaleFactorsMap_.value(shearAsFuncOf).centerVal);
177  (*subChanValIter) = shearedVal;
178  }
179  x++;
180  }
181 
182  return dataBuffer;
183 }
