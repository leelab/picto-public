371  || (colors.size() != getNumLabels())
372  || (Contents.size() != getNumLabels())
373  || (xs.size() != getNumLabels())
374  || (ys.size() != getNumLabels())

401

403  {

448  QRect LabelFactoryGraphic::adjustSizeToFont(QString str, QFont txtFont, QRect dimensions)
449  {
450  QRect returnVal(dimensions.left(), dimensions.top(), dimensions.width(), dimensions.height());
451 
452  QFontMetrics fm(txtFont);
453  int strWidth = fm.boundingRect(str).width();
454  int strHeight = fm.boundingRect(str).height();
455 
456  int rectWidth = 0;
457  int rectHeight = 0;
458 
459  if (dimensions.width() != 0 && dimensions.height() != 0 && dimensions.isValid())
460  {
461  //default is defined size (percent of the default size)
462  rectWidth = dimensions.width();
463  rectHeight = dimensions.height();
464  }
465 
466  QRect adjDimensions; //dimension of the longest subString in the text
467 
468  //find all the return carriage \n in the text and keep the width of the longest substring
469  QStringList strList = str.split('\n');
470  if (strList.size() > 0)
471  {
472  int longestIndex = 0;
473  strWidth = 0;
474  for (int j = 0; j < strList.size(); ++j)
475  {
476  if (fm.width(strList.at(j)) > strWidth)
477  {
478  strWidth = fm.width(strList.at(j));
479  longestIndex = j;
480  }
481  }
482  adjDimensions = fm.boundingRect(strList.at(longestIndex));
483  }
484  else
485  adjDimensions = fm.boundingRect(str);
486 
487  if (rectWidth < strWidth || rectHeight < strHeight)
488  {
489  //auto-adjust size to text if size was not defined or if it is null or too small for the text
490  if (adjDimensions.width() == 0 || adjDimensions.height() == 0 || !adjDimensions.isValid())
491  {
492  if (dimensions.width() == 0 || dimensions.height() == 0 || !dimensions.isValid())
493  {
494  //use original font size only
495  returnVal.setWidth(fm.boundingRect(str).width());
496  returnVal.setHeight(fm.boundingRect(str).height());
497  }
498 
499  if (strWidth > 0)
500  {
501  //auto-adjust font-size to specified size with a limit of 60% to 125%
502  double factor = dimensions.width() / strWidth;
503  if ((factor < 1) || factor > 1.25)
504  {
505  if (factor < 0.6)
506  factor = 0.6;
507  else if (factor > 1.5)
508  factor = 1.5;
509 
510  txtFont.setPointSizeF(txtFont.pointSizeF()*factor);
511  }
512  }
513  }
514  else
515  {
516  //number of lines in the text
517  int nbLines = 1;
518  nbLines = nbLines + str.count('\n');
519 
520  returnVal.setWidth(strWidth);
521  returnVal.setHeight(strHeight*nbLines);
522  }
523  }
524  return returnVal;
525  }

596  QRect LabelFactoryGraphic::getImageDims()
597  {
598  int defaultWidth = getLabelWidth();
599  int defaultHeight = getLabelHeight();
600  QStringList sizes = propertyContainer_->getPropertyValue("LabelSizes").toString().split(",", QString::SkipEmptyParts);
601  QStringList xs = propertyContainer_->getPropertyValue("LabelXs").toString().split(",", QString::SkipEmptyParts);
602  QStringList ys = propertyContainer_->getPropertyValue("LabelYs").toString().split(",", QString::SkipEmptyParts);
603  QStringList fontSizes = propertyContainer_->getPropertyValue("LabelFontSizes").toString().split(",", QString::SkipEmptyParts);
604  QStringList Contents = propertyContainer_->getPropertyValue("LabelContents").toString().split(",", QString::SkipEmptyParts);
605 
606  if ((sizes.size() < getNumLabels() && fontSizes.size() < getNumLabels())
607  || (xs.size() < getNumLabels())
608  || (ys.size() < getNumLabels())
609  || (Contents.size() < getNumLabels())
610  )
611  return QRect(); //If this is the case, somethings being updated and calling draw and the system won't really need this data yet.
612 
613  QRect returnVal;
614  for (int i = 0; i<getNumLabels(); i++)
615  {
616  QPoint currPoint;
617  if (xs[i] != "_")
618  currPoint.setX(xs[i].toInt());
619  if (ys[i] != "_")
620  currPoint.setY(ys[i].toInt());
621  int height = defaultHeight;
622  int width = defaultWidth;
623  if (sizes.size() > i && sizes[i] != "_")
624  {
625  height *= sizes[i].toDouble();
626  width *= sizes[i].toDouble();
627  }
628 
629  int fontsize = getLabelFontSize().toInt();
630  QString str = Contents[i];
631  if (fontSizes.size() > i && fontSizes[i] != "_" && fontSizes[i] != "0")
632  fontsize = fontSizes[i].toInt();
633 
634  QFont txtFont = QFont("Times", fontsize);
635 
636  QRect Tmpdimensions(0., 0., width, height);
637  QRect dimensions = adjustSizeToFont(str, txtFont, Tmpdimensions);
638 
639  width = dimensions.width();
640  height = dimensions.height();
641 
642  int halfWidth = width / 2;
643  int halfHeight = height / 2;
644 
645  int minX = currPoint.x() - halfWidth;
646  int minY = currPoint.y() - halfHeight;
647  int maxX = currPoint.x() + halfWidth;
648  int maxY = currPoint.y() + halfHeight;
649 
650  if (returnVal.isEmpty())
651  {
652  returnVal = QRect(QPoint(minX, minY), QPoint(maxX, maxY));
653  continue;
654  }
655  if (minX < returnVal.left())
656  returnVal.setLeft(minX);
657  if (minY < returnVal.top())
658  returnVal.setTop(minY);
659  if (maxX > returnVal.right())
660  returnVal.setRight(maxX);
661  if (maxY > returnVal.bottom())
662  returnVal.setBottom(maxY);
663 
664  }
665  return returnVal;
666  }
