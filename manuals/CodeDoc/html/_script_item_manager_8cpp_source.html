76 void ScriptItemManager::updateScriptItemDimensions()
77 {
78  float currPos = 0;
79  float totalHeight = horizontal_?scriptBoundingRect_.width():scriptBoundingRect_.height();
80  float elementHeight = 0;
81  for(int i = 0; i < TYPE_NUM; i++)
82  {
83  if(scriptItems_.contains(i))
84  {
85  currPos = scriptRegionStartPos_[i/2]*totalHeight+(horizontal_?scriptBoundingRect_.topLeft().x():scriptBoundingRect_.topLeft().y());
86  if( !(i%2) && scriptItems_.contains(i+1) )
87  {
88  elementHeight = minScriptItemHeight_*totalHeight;
89  }
90  else if((i%2 && scriptItems_.contains(i-1)))
91  {
92  elementHeight = minScriptItemHeight_*totalHeight;
93  currPos += elementHeight;
94  }
95  else
96  {
97  elementHeight = minScriptItemHeight_*2.0*totalHeight;
98  }
99 
100  ScriptItem* scriptItem = scriptItems_.value(i);
101  if(horizontal_)
102  {
103  scriptItem->setWidth(elementHeight);
104  scriptItem->setPos(QPointF(currPos,0));
105  scriptItem->setHeight(scriptBoundingRect_.height());
106  }
107  else
108  { //Vertical
109  scriptItem->setHeight(elementHeight);
110  scriptItem->setPos(QPointF(0,currPos));

124 void ScriptItemManager::updateScriptItems()
125 {
126  QSharedPointer<PropertyContainer> propContainer;
127  QSharedPointer<Property> prop;
128  for(int i = 0; i < TYPE_NUM; i++)
129  {
130  bool isAnalysisScript = orderedScriptNames_[i].left(1).toLower() == "a";
131  //If its a run Script
132  if(!isAnalysisScript)
133  { //Set the property container directly from the asset attached to this manager
134  propContainer = asset_.staticCast<DataStore>()->getPropertyContainer();
135  }
136  else
137  { //if its an analysis script
138  //If the object does not have an AnalysisScriptHolder, skip this script
139  QSharedPointer<Analysis> activeAnalysis = editorState_->getCurrentAnalysis();
140  if(!activeAnalysis || asset_.staticCast<DataStore>()->getAssociateChildren(activeAnalysis->getAssociateId(),"AnalysisScriptHolder").isEmpty())
141  continue;
142 
143  //Set the property container from the scriptContainer
144  QSharedPointer<AnalysisScriptHolder> scriptContainer = asset_.staticCast<DataStore>()->getAssociateChildren(activeAnalysis->getAssociateId(),"AnalysisScriptHolder").first().staticCast<AnalysisScriptHolder>();
145  propContainer = scriptContainer.staticCast<DataStore>()->getPropertyContainer();
146  }
147  //Check the property container for the prop
148  prop = propContainer->getProperty(orderedScriptNames_[i]);
149  if(!scriptItems_.contains(i))
150  {
151  if(prop && !prop->value().toString().isEmpty())
152  {
153  scriptItems_[i] = new ScriptItem(prop,scriptTypeColors_[i],editorState_,scriptsParent_);
154  scriptItems_[i]->setZValue(scriptsParent_->zValue()+1);
155  }
156  }
157  else
158  {
159  Q_ASSERT(prop);
160  if(prop->value().toString().isEmpty())
161  {
162  delete scriptItems_[i];
163  scriptItems_.remove(i);
164  }
165  }
166 
167  }
168  updateScriptItemDimensions();
169 }
