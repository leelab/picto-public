138 double SignalChannel::peekValue(QString subchannel)
139 {
140  getDataFromPort();
141 
142  if(!rawDataBuffer_.contains(subchannel))
143  return 0.0;
144  QString shearAsFuncOf = scaleFactorsMap_.value(subchannel).shearAsFuncOf;
145  Q_ASSERT(shearAsFuncOf.isEmpty() || rawDataBuffer_.contains(shearAsFuncOf));
146  //Update the rawDataLastValue_ map with this subchannel and a shearAsFuncOf channel
147  if(!rawDataBuffer_.value(subchannel).isEmpty())
148  {
149  rawDataLastValue_[subchannel] = rawDataBuffer_.value(subchannel).last();
150  }
151  if(!shearAsFuncOf.isEmpty() && !rawDataBuffer_.value(shearAsFuncOf).isEmpty())
152  {
153  rawDataLastValue_[shearAsFuncOf] = rawDataBuffer_.value(shearAsFuncOf).last();
154  }
155  Q_ASSERT(rawDataLastValue_.contains(subchannel));
156  Q_ASSERT(shearAsFuncOf.isEmpty() || rawDataLastValue_.contains(subchannel));
157 
158  //Get the lastRawDataValue, scale and shear it.
159  double rawValue = rawDataLastValue_.value(subchannel);
160  double scaledValue = scaleFactorsMap_.value(subchannel).scaleA +
161  scaleFactorsMap_.value(subchannel).scaleB * rawValue;
162  if(!shearAsFuncOf.isEmpty())
163  {
164  double otherRawValue = rawDataLastValue_.value(shearAsFuncOf);
165  double otherScaledValue = scaleFactorsMap_.value(shearAsFuncOf).scaleA +
166  scaleFactorsMap_.value(shearAsFuncOf).scaleB * otherRawValue;
167  scaledValue = scaledValue + scaleFactorsMap_.value(subchannel).shearFactor * (otherScaledValue - scaleFactorsMap_.value(shearAsFuncOf).centerVal);
168  }
169  //qDebug(QString("%1: Last: %2 Scaled: %3").arg(rawDataLastValue_[subchannel]).arg(subchannel).arg(scaledValue).toLatin1());
170  return scaledValue;
171 }

216 QMap<QString, QVector<double> > SignalChannel::getValues()
217 {
218  QMap<QString, QVector<double> > dataBuffer = getRawValues();
219 
220  //scale the values
221  QMap<QString, QVector<double> >::iterator x = dataBuffer.begin();
222 
223  while(x != dataBuffer.end())
224  {
225  for(QVector<double>::iterator subChanValIter = x.value().begin();
226  subChanValIter != x.value().end();
227  subChanValIter++)
228  {
229  double scaledValue = scaleFactorsMap_.value(x.key()).scaleA +
230  scaleFactorsMap_.value(x.key()).scaleB * (*subChanValIter);
231  (*subChanValIter) = scaledValue;
232  }
233  x++;
234  }
235 
236  //Add shear
237  x = dataBuffer.begin();
238  while(x != dataBuffer.end())
239  {
240  QString shearAsFuncOf = scaleFactorsMap_.value(x.key()).shearAsFuncOf;
241  if(shearAsFuncOf.isEmpty())
242  {
243  x++;
244  continue;
245  }
246 
247  QVector<double>::iterator funcOfIterator = dataBuffer.find(shearAsFuncOf).value().begin();
248  for(QVector<double>::iterator subChanValIter = x.value().begin();
249  subChanValIter != x.value().end();
250  subChanValIter++,funcOfIterator++)
251  {
252  double shearedVal = (*subChanValIter) +
253  scaleFactorsMap_.value(x.key()).shearFactor * ((*funcOfIterator)-scaleFactorsMap_.value(shearAsFuncOf).centerVal);
254  (*subChanValIter) = shearedVal;
255  }
256  x++;
257  }
258 
259  return dataBuffer;
260 }
