28 {
29  //Monitor for changes in any Experimental properties during the running of the scripts below.
30  //If any experimental properties change, Analysis changed them and a runtime error should be issued.
31  Property::startMonitoringForExperimentalValueChange();
32  switch(type)
33  {
34  case ENTRY:
35  if(!propertyContainer_->getPropertyValue("AnalysisEntryScript").toString().isEmpty())
36  {
37  QString scriptNamePrefix = getScriptNamePrefix();
38  ScriptableContainer::runScript(scriptNamePrefix+"Entry");
39  }
40  break;
41  case FRAME:
42  if(propertyContainer_->getProperty("AnalysisFrameScript")
43  && !propertyContainer_->getProperty("AnalysisFrameScript")->isDeleted()
44  && (!propertyContainer_->getPropertyValue("AnalysisFrameScript").toString().isEmpty()))
45  {
46  QString scriptNamePrefix = getScriptNamePrefix();
47  ScriptableContainer::runScript(scriptNamePrefix+"Frame");
48  }
49  break;
50  case EXIT:
51  if(propertyContainer_->getProperty("AnalysisExitScript") && (!propertyContainer_->getPropertyValue("AnalysisExitScript").toString().isEmpty()))
52  {
53  QString scriptNamePrefix = getScriptNamePrefix();
54  ScriptableContainer::runScript(scriptNamePrefix+"Exit");
55  }
56  break;
57  }
