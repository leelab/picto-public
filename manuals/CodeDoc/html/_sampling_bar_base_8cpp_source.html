52 void SamplingBarBase::_handleErrorValue(long bin, double &rdRangeMax, QVector<QwtIntervalSample> &qvError)
53 {
54  if (m_bLastErrorBarState)
55  {
56  const double dAvg = m_qhdCumulValue[bin] / double(m_qhlCumulNum[bin]);
57  const double dValue = bin * getBinSize();
58  const double stdErr = (sqrt(m_qhdCumulValSq[bin] - dAvg*m_qhdCumulValue[bin])) / double(m_qhlCumulNum[bin]);
59  const double dMax = dAvg + stdErr;
60 
61  qvError.push_back(std::move(QwtIntervalSample(dValue, dAvg - stdErr, dMax)));
62 
63  if (getNormalized())
64  {
65  if (dMax > rdRangeMax)
66  {
67  rdRangeMax = dMax;
68  }
69  }
70  }
