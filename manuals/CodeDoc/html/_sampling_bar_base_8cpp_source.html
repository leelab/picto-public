44 void SamplingBarBase::_handleErrorValue(long bin, double &rdRangeMax, QVector<QwtIntervalSample> &qvError)
45 {
46  if (m_bLastErrorBarState)
47  {
48  const double dAvg = m_qhCumulValue[currentSetName_][bin] / double(m_qhCumulNum[currentSetName_][bin]);
49  const double dValue = bin * getBinSize();
50  const double stdErr = (sqrt(m_qhCumulValSq[currentSetName_][bin] - dAvg*m_qhCumulValue[currentSetName_][bin])) / double(m_qhCumulNum[currentSetName_][bin]);
51  const double dMax = dAvg + stdErr;
52 
53  qvError.push_back(std::move(QwtIntervalSample(dValue, dAvg - stdErr, dMax)));
54 
55  if (getNormalized())
56  {
57  if (dMax > rdRangeMax)
58  {
59  rdRangeMax = dMax;
60  }
61  }
62  }
