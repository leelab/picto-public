38 void SamplingBarBase::_handleErrorValue(long bin, double &rdRangeMax, QVector<QwtIntervalSample> &qvError)
39 {
40  if (m_bLastErrorBarState)
41  {
42  const double dAvg = m_qhdCumulValue[bin] / double(m_qhlCumulNum[bin]);
43  const double dValue = bin * getBinSize();
44  const double stdErr = (sqrt(m_qhdCumulValSq[bin] - dAvg*m_qhdCumulValue[bin])) / double(m_qhlCumulNum[bin]);
45  const double dMax = dAvg + stdErr;
46 
47  qvError.push_back(std::move(QwtIntervalSample(dValue, dAvg - stdErr, dMax)));
48 
49  if (getNormalized())
50  {
51  if (dMax > rdRangeMax)
52  {
53  rdRangeMax = dMax;
54  }
55  }
56  }
