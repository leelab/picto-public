334 void DesignConfig::fixDuplicatedAssetIds()
335 {
336  if(allowIdDuplication_)
337  return;
338  QSharedPointer<Asset> sAsset;
339  int id;
340  QList<QSharedPointer<Asset>> fixList;
341 
342  while(unsortedIdAssets_.size())
343  {
344  QWeakPointer<Asset> wAsset = unsortedIdAssets_.takeFirst();
345  sAsset = QSharedPointer<Asset>(wAsset);
346  if(!sAsset)
347  continue; //This asset was already deleted
348  id = sAsset->getAssetId();
349  if(id == 0 || assetsById_.contains(id))
350  {
351  fixList.append(sAsset);
352  continue;
353  }
354  assetsById_[id] = wAsset; //Add the asset to the id map with its id.
355  }
356  while(fixList.size())
357  {
358  sAsset = fixList.takeFirst();
359  id = getNewDataId(); //Get a new asset id and set it to the asset.
360  sAsset->setAssetId(id);
361  assetsById_[id] = QWeakPointer<Asset>(sAsset); //Add the asset to the id map with its id.
362  }
363 }
