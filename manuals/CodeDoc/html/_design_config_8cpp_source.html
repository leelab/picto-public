339 void DesignConfig::fixDuplicatedAssetIds()
340 {
341  if(allowIdDuplication_)
342  return;
343  QSharedPointer<Asset> sAsset;
344  int id;
345  QList<QSharedPointer<Asset>> fixList;
346 
347  while(unsortedIdAssets_.size())
348  {
349  QWeakPointer<Asset> wAsset = unsortedIdAssets_.takeFirst();
350  sAsset = QSharedPointer<Asset>(wAsset);
351  if(!sAsset)
352  continue; //This asset was already deleted
353  id = sAsset->getAssetId();
354  if(id == 0 || assetsById_.contains(id))
355  {
356  fixList.append(sAsset);
357  continue;
358  }
359  assetsById_[id] = wAsset; //Add the asset to the id map with its id.
360  }
361  while(fixList.size())
362  {
363  sAsset = fixList.takeFirst();
364  id = getNewDataId(); //Get a new asset id and set it to the asset.
365  sAsset->setAssetId(id);
366  assetsById_[id] = QWeakPointer<Asset>(sAsset); //Add the asset to the id map with its id.
367  }
368 }
