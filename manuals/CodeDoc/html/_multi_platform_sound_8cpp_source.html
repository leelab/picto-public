123 bool MultiplatformSound::isFileSupported(QString path)
124 {
125  QFile waveFile;
126  QAudioFormat fileFormat;
127  waveFile.setFileName(path);
128  waveFile.open(QIODevice::ReadOnly);
129  waveFile.seek(0);
130 
131 
132  CombinedHeader header;
133  bool result = waveFile.read(reinterpret_cast<char *>(&header), sizeof(CombinedHeader)) == sizeof(CombinedHeader);
134  if (result) {
135  if ((memcmp(&header.riff.descriptor.id, "RIFF", 4) == 0
136  || memcmp(&header.riff.descriptor.id, "RIFX", 4) == 0)
137  && memcmp(&header.riff.type, "WAVE", 4) == 0
138  && memcmp(&header.wave.descriptor.id, "fmt ", 4) == 0
139  && (header.wave.audioFormat == 1 || header.wave.audioFormat == 0)) {
140 
141  // Read off remaining header information
142  DATAHeader dataHeader;
143 
144  if (qFromLittleEndian<quint32>(header.wave.descriptor.size) > sizeof(WAVEHeader)) {
145  // Extended data available
146  quint16 extraFormatBytes;
147  if (waveFile.peek((char*)&extraFormatBytes, sizeof(quint16)) != sizeof(quint16))
148  return false;
149  const qint64 throwAwayBytes = sizeof(quint16) + qFromLittleEndian<quint16>(extraFormatBytes);
150  if (waveFile.read(throwAwayBytes).size() != throwAwayBytes)
151  return false;
152  }
153 
154  if (waveFile.read((char*)&dataHeader, sizeof(DATAHeader)) != sizeof(DATAHeader))
155  return false;
156 
157  // Establish format
158  if (memcmp(&header.riff.descriptor.id, "RIFF", 4) == 0)
159  fileFormat.setByteOrder(QAudioFormat::LittleEndian);
160  else
161  fileFormat.setByteOrder(QAudioFormat::BigEndian);
162 
163  int bps = qFromLittleEndian<quint16>(header.wave.bitsPerSample);
164  fileFormat.setChannelCount(qFromLittleEndian<quint16>(header.wave.numChannels));
165  fileFormat.setCodec("audio/pcm");
166  fileFormat.setSampleRate(qFromLittleEndian<quint32>(header.wave.sampleRate));
167  fileFormat.setSampleSize(qFromLittleEndian<quint16>(header.wave.bitsPerSample));
168  fileFormat.setSampleType(bps == 8 ? QAudioFormat::UnSignedInt : QAudioFormat::SignedInt);
169  } else {
170  return false;
171  }
172  }
173  else
174  {
175  return false;
176  }
177 
178  QAudioDeviceInfo info(QAudioDeviceInfo::defaultOutputDevice());
179  if (!info.isFormatSupported(fileFormat))
180  return false;
181  return true;
182 }
