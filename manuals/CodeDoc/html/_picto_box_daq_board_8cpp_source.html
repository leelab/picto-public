69 bool PictoBoxDaqBoard::initRewardController(int _numControllers, unsigned int _rewardDurationsMs[])
70 {
71  if(hDAQ_ == NULL)
72  {
73  OutputDebugString(L"DaqBoard.initRewardController: Null DAQ handle\n");
74  return false;
75  }
76  if(_numControllers > MAX_NUM_REWARD_CONTROLLERS || _numControllers < 0)
77  {
78  OutputDebugString(L"DaqBoard.initRewardController: Invalid number of reward controllers specified\n");
79  return false;
80  }
81  if(rewardControllersInitialized_)
82  {
83  OutputDebugString(L"DaqBoard.initRewardController: Reward controllers alread initialized\n");
84  return false;
85  }
86 
87  activeRewardControllers_ = _numControllers;
88  for(int i=0; i<activeRewardControllers_; i++)
89  {
90  rewardDurations_[i] = _rewardDurationsMs[i];
91  rewardBitmasks_[i] = 1 << i;
92  }
93 
94  //Set the DIO pins to output. I'm setting them all to output because this
95  //makes life easier. If at some point we want to use the other pins for
96  //something else, we'll have to change this.
97  DWORD data = 0xFFFFFFFF;
98  DWORD bytesWritten;
99 
100  SetFilePointer(hDAQ_, REG_DIO_DIRECTION,NULL, FILE_BEGIN);
101  WriteFile(hDAQ_,&data, FOUR_BYTES, &bytesWritten, NULL);
102 
103  if(bytesWritten != FOUR_BYTES)
104  {
105  OutputDebugString(L"DaqBoard.initRewardController: Failed writing to DIO_DIRECTION register\n");
106  return false;
107  }
108 
109  //set the reward controller values to 1 (off)
110  data = 0x0000000F;
111  SetFilePointer(hDAQ_,REG_STATIC_DIGITAL_OUTPUT,NULL,FILE_BEGIN);
112  WriteFile(hDAQ_,&data,FOUR_BYTES,&bytesWritten,NULL);
113 
114  if(bytesWritten != FOUR_BYTES)
115  {
116  OutputDebugString(L"DaqBoard.initRewardController: Failed writing to DIO_DIRECTION register\n");
117  return false;
118  }
119 
120 
121  rewardControllersInitialized_ = true;
122 
123  return true;
124

201 bool PictoBoxDaqBoard::giveReward(int _controller)
202 {
203  DWORD buffer;
204  DWORD bytesWritten,bytesRead;
205 
206  if(_controller > activeRewardControllers_)
207  {
208  OutputDebugString(L"DaqBoard.giveReward: Invalid reward controller number\n");
209  return false;
210  }
211  if(hDAQ_ == NULL)
212  {
213  OutputDebugString(L"DaqBoard.giveReward: Null DAQ handle\n");
214  return false;
215  }
216  if(!rewardControllersInitialized_)
217  {
218  OutputDebugString(L"DaqBoard.giveReward: Reward controllers not initialized\n");
219  return false;
220  }
221 
222 
223  //Turn on the reward controller (set the bit to 0)
224  SetFilePointer(hDAQ_,REG_STATIC_DIGITAL_OUTPUT,NULL,FILE_BEGIN);
225  ReadFile(hDAQ_,&buffer,FOUR_BYTES,&bytesRead,NULL);
226  buffer &= ~rewardBitmasks_[_controller-1];
227  SetFilePointer(hDAQ_,REG_STATIC_DIGITAL_OUTPUT,NULL,FILE_BEGIN);
228  WriteFile(hDAQ_,&buffer, FOUR_BYTES,&bytesWritten, NULL);
229 
230 
231 
232  //Delay for appropriate duration
233  //code stolen/modified from MSDN:
234  //http://msdn.microsoft.com/en-us/library/ms836797.aspx
235  LARGE_INTEGER ticksPerMs;
236  LARGE_INTEGER currTicks,endTicks;
237  // Query number of ticks per second
238  if (QueryPerformanceFrequency(&ticksPerMs))
239  {
240  //get ticks per ms
241  ticksPerMs.QuadPart /= 1000;
242 
243  // Get current ticks
244  if (QueryPerformanceCounter(&currTicks))
245  {
246  // Create timeout value
247  endTicks.QuadPart = currTicks.QuadPart + ticksPerMs.QuadPart*rewardDurations_[_controller-1];
248  do
249  {
250  // Get current ticks
251  QueryPerformanceCounter(&currTicks);
252  // Delay until timeout
253  } while (currTicks.QuadPart<endTicks.QuadPart);
254  }
255  }
256 
257  //turn off the reward controller (set the bit to 1)
258  SetFilePointer(hDAQ_,REG_STATIC_DIGITAL_OUTPUT,NULL,FILE_BEGIN);
259  ReadFile(hDAQ_,&buffer,FOUR_BYTES,&bytesRead,NULL);
260  buffer |= rewardBitmasks_[_controller-1];
261  SetFilePointer(hDAQ_,REG_STATIC_DIGITAL_OUTPUT,NULL,FILE_BEGIN);
262  WriteFile(hDAQ_,&buffer, FOUR_BYTES,&bytesWritten, NULL);
263 
264  return true;
265 
266 
267 }

343 bool PictoBoxDaqBoard::initEventLines()
344 {
345  //set PFI[0:7] as output
346  //Each register contains 3 5-bit values, which are right justified
347  //For example, bits 4:0 of register 1 control PFI0, bits 9:4
348  //of register 1 control PFI1, etc.
349 
350  //There are a lot of possible settings (see the M-series register map),
351  //but the only one we care about is 16, which sets the line as PFI_DO.
352  //We're going to set PFI[7:0] to PFI_DO (16), and the remaining lines
353  //to default (0).
354 
355  if(hDAQ_ == NULL)
356  {
357  OutputDebugString(L"DaqBoard.initEventLines: Null DAQ handle\n");
358  return false;
359  }
360  if(eventLinesInitialized_)
361  {
362  OutputDebugString(L"DaqBoard.initEventLines: Event lines alread initialized\n");
363  return false;
364  }
365 
366  DWORD data;
367  DWORD bytesWritten;
368 
369  //REG_IO_BIDIRECTION_PIN: 0x00FF (1=output, 0=input)
370  data = 0x00FF;
371  SetFilePointer(hDAQ_, REG_IO_BIDIRECTION_PIN, NULL, FILE_BEGIN);
372  WriteFile(hDAQ_,&data, TWO_BYTES, &bytesWritten, NULL);
373  if(bytesWritten != TWO_BYTES)
374  {
375  OutputDebugString(L"DaqBoard.initEventLines: Failed writing to REG_IO_BIDIRECTION_PIN register\n");
376  return false;
377  }
378 
379  // REG_PFI_OUTPUT_SELECT_1: 0x4210
380  data = 0x4210;
381  SetFilePointer(hDAQ_, REG_PFI_OUTPUT_SELECT_1,NULL, FILE_BEGIN);
382  WriteFile(hDAQ_,&data, TWO_BYTES, &bytesWritten, NULL);
383  if(bytesWritten != TWO_BYTES)
384  {
385  OutputDebugString(L"DaqBoard.initEventLines: Failed writing to REG_PFI_OUTPUT_SELECT_1 register\n");

387  }

463 bool PictoBoxDaqBoard::sendEvent(unsigned char _eventCode)
464 {
465 
466  if(!eventLinesInitialized_)
467  return false;
468 
469  SHORT data;
470  DWORD bytesWritten;
471 
472  //set event code on PFI lines
473  data = _eventCode & 0x7F;
474  SetFilePointer(hDAQ_, PFI_DO,NULL, FILE_BEGIN);
475  WriteFile(hDAQ_,&data, TWO_BYTES, &bytesWritten, NULL);
476 
477  //set strobe high
478  data = (_eventCode & 0x7F) | 0x80;
479  SetFilePointer(hDAQ_, PFI_DO,NULL, FILE_BEGIN);
480  WriteFile(hDAQ_,&data, TWO_BYTES, &bytesWritten, NULL);
481 
482  //wait 250 us
483  LARGE_INTEGER ticksPerSec;
484  LARGE_INTEGER tick, tock;
485  double elapsedTime;
486  double delayTime = 0.000250; //250 microseconds
487 
488  QueryPerformanceFrequency(&ticksPerSec);
489  QueryPerformanceCounter(&tick);
490  do
491  {
492  QueryPerformanceCounter(&tock);
493  elapsedTime = (double)(tock.LowPart-tick.LowPart)/(double)(ticksPerSec.LowPart);
494  }
495  while(elapsedTime < delayTime);
496 
497  //set strobe low
498  data = _eventCode & 0x7F;
499  SetFilePointer(hDAQ_, PFI_DO,NULL, FILE_BEGIN);
500  WriteFile(hDAQ_,(LPCVOID)&data, TWO_BYTES, &bytesWritten, NULL);
501 
502  //clear PFI lines
503  data = 0x0;
504  SetFilePointer(hDAQ_, PFI_DO,NULL, FILE_BEGIN);
505  WriteFile(hDAQ_,&data, TWO_BYTES, &bytesWritten, NULL);
506 
507 
508  return true;
509 }

693 int PictoBoxDaqBoard::CollectAIData(unsigned int _numSamples)
694 {
695  if(!hDAQ_)
696  {
697  OutputDebugString(L"DaqBoard.ReadAIData: DAQ1 file NULL. D\n");
698  return 0;
699  }
700  unsigned int numChannels = AIChannelList_.size();
701  unsigned int chan, sample;
702 
703  ULONG *dataBuffer;
704  dataBuffer = new ULONG[_numSamples*numChannels];
705 
706  ULONG fifoBytes=0;
707 
708  //dump the data fifo
709  DeviceIoControl(hDAQ_,IOCTRL_DATA_FIFO_DUMP,NULL,0, dataBuffer, _numSamples*numChannels*sizeof(ULONG), &fifoBytes, NULL);
710  //sort the data fifo into the channels
711 
712  //run through each channel
713  for(chan=0; chan<numChannels; chan++)
714  {
715  //clear out the data currently stored for each channel
716  AIChannelList_[chan].rawData.clear();
717  AIChannelList_[chan].voltages.clear();
718 
719 
720  //for each channel run through the data buffer, storing the raw values
721  //and converting them to voltages
722  for(sample=0; sample<_numSamples && sample<fifoBytes/sizeof(ULONG)/numChannels; sample++)
723  {
724  AIChannelList_[chan].rawData.push_back(dataBuffer[sample*numChannels+chan]);
725  AIChannelList_[chan].voltages.push_back(_aiPolynomialScaler(dataBuffer[sample*numChannels+chan],&scale_));
726  }
727  }
728 
729  delete dataBuffer;
730 
731  return sample;
732 
733 }

796  SetFilePointer(hDAQ_,REG_AI_STATUS_1,NULL,FILE_BEGIN);

798  FIFOEmpty = data & 0x0001000;
799 
800  while (!FIFOEmpty)
801  {
802  //value = board->AI_FIFO_Data.readRegister ();
803  SetFilePointer(hDAQ_,REG_AI_FIFO_DATA,NULL,FILE_BEGIN);
804  ReadFile(hDAQ_,&data, FOUR_BYTES, &bytesRead,NULL);
805

807  SetFilePointer(hDAQ_,REG_AI_STATUS_1,NULL,FILE_BEGIN);
808  ReadFile(hDAQ_,&data, TWO_BYTES, &bytesRead,NULL);
809  FIFOEmpty = data & 0x0001000;
810 
811  numReads++;
812 
813  //If, for some reason, the fifo is being read at a slower rate than
814  //samples are being collected, then the FIFO will never be empty.
815  //If this happens, break out of the loop
816  if(numReads>4096)
817  break;
818  }
819  return numReads;
