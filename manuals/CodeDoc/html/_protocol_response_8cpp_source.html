58 QString ProtocolResponse::generateHeadersString()
59 {
60  //NOTE: The approach to generate headers changed sometime around 12/1/09. If
61  //you're experiencing problems with protocolResponses this might be the cause...
62  QString headers;
63 
64  headers = QString("%1/%2 %3 %4\r\n").arg(protocol)
65  .arg(version)
66  .arg(protocolResponseCode)
67  .arg(protocolResponseTypeStrings[protocolResponseCode]);
68 
69  //Print the content length and encoding, but only if this isn't a multipart response
70  if(!fields.value("Content-Type").contains("multipart", Qt::CaseInsensitive))
71  {
72  if(fields.contains("Content-Encoding"))
73  headers+= QString("Content-Encoding: %1\r\n").arg(fields.value("Content-Encoding"));
74  headers+= QString("Content-Length: %1\r\n").arg(fields.value("Content-Length","0"));
75  }
76 
77  //update (or add) the date field
78  fields["Date"] = QString("%1 GMT").arg(QDateTime::currentDateTime().toUTC().toString("ddd, dd MMM yyyy hh:mm:ss"));
79 
80  //print everything else that we haven't dealt with yet
81  QMap<QString,QString>::const_iterator fieldsIter = fields.begin();
82  while(fieldsIter != fields.end())
83  {
84  if(fieldsIter.key() != "Content-Encoding" &&
85  fieldsIter.key() != "Content-Length")
86  {
87  headers += QString("%1: %2\r\n")
88  .arg(fieldsIter.key())
89  .arg(fieldsIter.value());
90  }
91  fieldsIter++;
92  }
93  return headers;
94 }

403 int ProtocolResponse::write(QAbstractSocket *socket)
404 {
405  QByteArray response;
406 
407  //before doing anything, make sure that Content-Length is accurate
408  if(fields.value("Content-Length").toInt() != content.size())
409  return -1;
410 
411  if(multiPartResponseState <= MultiPartResponseType::MultiPartInitial)
412  {
413  QString headers = generateHeadersString();
414  headers += "\r\n";
415  response.append(headers);
416  }
417 
418  if(multiPartResponseState == MultiPartResponseType::MultiPartPart)
419  {
420  response.append("\r\n");
421  }
422 
423  if(multiPartResponseState >= Picto::MultiPartResponseType::MultiPartInitial)
424  {
425  QString multiPartHeaders = getMultiPartHeaders();
426  multiPartHeaders += "\r\n";
427  response.append(multiPartHeaders);
428  }
429 
430  if(hasContent())
431  {
432  response.append(getContent());
433  }
434 
435  int bytesWritten = socket->write(response);
436 
437  socket->flush();
438 
439  return bytesWritten;
440 }

462 int ProtocolResponse::read(QAbstractSocket *socket, int timeoutMs)
463 {
464  QString currentLine;
465  int contentLength=0;
466 
467  if(socket->bytesAvailable() <= 0)
468  {
469  if(!socket->waitForReadyRead(timeoutMs))
470  {
471  return -1;
472  }
473  }
474 
475  //read the first line, it might be a status line (e.g. ACQ/1.0 200 OK)
476  //or it might be a CRLF followed by a multipart boundary string. This
477  //tells us what type of response we're dealing with. Adittionally, if the
478  //first line is null, there's no data on the socket, so we should give up.
479 
480  currentLine = socket->readLine();
481 
482  //there is no response.
483  if(currentLine.isEmpty())
484  {
485  return -1;
486  }
487  //it's a multipart part
488  //NOTE: if it's the first multipart part, it will start with the boundary --....
489  //Otherwise, it will start with \r\n bollowed by the boundary
490  //I'm trying to leave the flexibilty for the user to not have to define the
491  //boundary string
492 
493  //It's a multi part part, but not the first one
494  else if(currentLine == "\r\n")
495  {
496  multiPartResponseState = MultiPartResponseType::MultiPartPart;
497 
498  //grab the boundary string, and check it
499  currentLine = socket->readLine();
500  currentLine.remove(QRegExp("[\r\n]"));
501 
502  //If the user never set the boundary string, we're just going to assume that
503  //this is a correct boundary. However, if they did set the string, we
504  //need to confirm that it is correct
505  if(!multiPartBoundary.isEmpty() &&
506  currentLine != multiPartBoundary)
507  return -3;
508  }
509  //Its' the first part of a multipary part (meaning that it starts with the boundary
510  //rather than LFCR)
511  else if ((!multiPartBoundary.isEmpty() && currentLine.remove("\r\n") == multiPartBoundary) ||
512  (multiPartBoundary.isEmpty() && currentLine.startsWith("--")))
513  {
514  multiPartResponseState = MultiPartResponseType::MultiPartPart;
515  }
516  //it's a multipart initial or, a non multipart, or something else entirely
517  //(in which case we return -2
518  else
519  {
520  //if it turns out to be multiPartInitial, this gets changed when the
521  //Content-type field is read.
522  multiPartResponseState = MultiPartResponseType::NotMultiPart;
523 
524  QStringList tokens = currentLine.split(" ");
525 
526  if(tokens.count() < 2)
527  {
528  return -2;
529  }
530 
531  protocolResponseCode = (ProtocolResponseType::ProtocolResponseType)(tokens[1].toInt());
532 
533  int protocolVersionPosition = tokens[0].indexOf('/');
534  if(protocolVersionPosition == -1)
535  {
536  return -2;
537  }
538  else
539  {
540  protocol = tokens[0].left(protocolVersionPosition);
541  version = tokens[0].mid(protocolVersionPosition+1);
542  }
543  }
544 
545  //read headers
546  QRegExp newLineRegExp("[\r\n]+");
547  currentLine = socket->readLine();
548  while(!newLineRegExp.exactMatch(currentLine))
549  {
550  currentLine.remove(QRegExp("[\r\n]"));
551  QString fieldKey,fieldValue;
552 
553  int keyEndPosition = currentLine.indexOf(':');
554  if(keyEndPosition < 0)
555  return -4;
556 
557  fieldKey = currentLine.left(keyEndPosition);
558  fieldValue = currentLine.mid(keyEndPosition+2);
559  fields[fieldKey] = fieldValue;
560 
561  currentLine = socket->readLine();
562  }
563 
564  //Interpret the headers
565  contentLength = fields.value("Content-Length","0").toInt();
566 
567  if(fields.value("Content-Type").contains("multipart", Qt::CaseInsensitive))
568  {
569  multiPartResponseState = MultiPartResponseType::MultiPartInitial;
570  int boundaryIndex = fields.value("Content-Type").indexOf("boundary=");
571  if(boundaryIndex >= 0)
572  multiPartBoundary = fields.value("Content-Type").mid(boundaryIndex+9);
573  }
574 
575  //read content
576  encodedContent = socket->read(contentLength);
577  while(encodedContent.size() < contentLength &&
578  (socket->bytesAvailable() > 0 || socket->waitForReadyRead(timeout)))
579  {
580  encodedContent.append(socket->read(contentLength-encodedContent.size()));
581  }
582 
583  //decode content
584  decodeContent();
585 
586  //return length of content read
587  if(encodedContent.size() == contentLength)
588  return encodedContent.size();
589  else
590  return 0-encodedContent.size();
591 }
