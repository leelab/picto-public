179  QObject::connect(engine_.data(),SIGNAL(nameChangeRequest(QString)),this,SLOT(changeName(QString)));
180  QObject::connect(engine_.data(),SIGNAL(rewardDurationChanged(int,int)),this,SLOT(changeRewardDuration(int,int)));
181  QObject::connect(engine_.data(),SIGNAL(flushDurationChangeRequest(int,int)),this,SLOT(changeFlushDuration(int,int)));
182 
183  HardwareSetup hwSetup(engine_);
184 
185  //If there is a command argument of "-pixmap", we should use a pixmpa rendering
186  //otherwise use d3d.
187  if(!hwSetup.setupRenderingTargets(visualTarget_,sigChannel_ != HardwareSetup::Mouse))
188  return 1;
189  if(!hwSetup.setXYChannelNums(xChannel_,yChannel_))
190  return 1;
191  if(!hwSetup.setXYDiamChannelNums(xDiamChannel_,yDiamChannel_))
192  return 1;
193  if(!hwSetup.setSamplePeriods(posSampPer_,diamSampPer_))
194  return 1;
195  if(!hwSetup.setupSignalChannel(sigChannel_))
196  return 1;
197  if(!hwSetup.setupRewardController(rewardController_))
198  return 1;
199  if(!hwSetup.setupOutputSignalController(outSigController_,eventCodeGenerator_))
200  return 1;
201  if(!hwSetup.setupEventCodeGenerator(eventCodeGenerator_))
202  return 1;
203  if(!hwSetup.isSetup())
204  return 1;
205 
206  //If running on PictoBox create front panel
207  if(useFrontPanel_)
208  {
209  frontPanelProcess_ = QSharedPointer<QProcess>(new QProcess());
210  frontPanelProcess_->start(QCoreApplication::applicationDirPath() + "/" + FRONTPANELEXECUTABLE,QStringList() << "-systemNumber" << QString::number(Picto::portNums->getSystemNumber()));
211  if(!frontPanelProcess_->waitForStarted())
212  return 1;
213  //Assure that the update downloader will kill the front panel process if it needs to restart the application.
214  connect(UpdateDownloader::getInstance().data(),SIGNAL(closingProcess()),frontPanelProcess_.data(),SLOT(kill()));
215  connect(engine_.data(),SIGNAL(escapePressed()),frontPanelProcess_.data(),SLOT(kill()));
216  fpInterface_ = QSharedPointer<FPInterface>(new FPInterface());
217  engine_->addControlPanel(fpInterface_);
218  }
