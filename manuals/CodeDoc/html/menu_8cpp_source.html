70 void Menu::userInputSlot(int type)
71 {
72  if((type == PanelInfo::rewardButton) || (type == PanelInfo::flushButton))
73  { //reward and flush should always be honored regardless of the current menu state
74  if(type == PanelInfo::rewardButton)
75  {
76  directorIf_->startReward(panelInfo->getRewardController());
77  }
78  else
79  {
80  directorIf_->flush(panelInfo->getRewardController());
81  }
82  return;
83  }
84  //Stop the activity timer, since something happened.
85  activityTimer->stop();
86  QSharedPointer<DisplayMode> newMode;
87  //Route the user input into the current DisplayMode, get the next mode that should
88  //be activated and initialize it if it is not the same as the current one.
89  newMode = panelModes_[currMode_->userInputSlot(PanelInfo::InputType(type))];
90  if(newMode != currMode_)
91  newMode->initMode();
92  currMode_ = newMode;
93  //Restart the activity timer that brings control back to the StatusMode after a set time
94  //has passed without user interaction. Since user interaction just happened, this is the
95  //time to start.
96  if(currMode_ != panelModes_[StatusModeType])
97  activityTimer->start(currMode_->activityTimeoutMs());
98 }
