181 void AnalysisOutput::initTempOutputDir()
182 {
183  //We want to make sure that even if multiple Picto Workstations are open at the same time, no one will remove the data
184  // that we are currently using. We also want to erase the data if we are just opening for the first time and no one
185  // else is using the data in the temporary folder. For this reason, we attempt to remove all files at the top level
186  // of the temporary directory, then if successful, remove all other contained directories as well. We then create a
187  // top level file in the temporary directory and leave it open until the application closes. If another workstation
188  // is open, we won't succeed in removing its open file so we won't delete the temporary directory contents, otherwise
189  // we will.
190  if(!outputDir_.isEmpty())
191  return;
192  QDir dir(QCoreApplication::applicationDirPath()+"/../");
193  dir.mkdir("analysisoutput");
194  dir.cd("analysisoutput");
195  removeFilesThenDirectories(dir);
196 
197  outputDir_ = dir.absolutePath();
198 
199  QString fileName = outputDir_+"/"+QUuid::createUuid().toString();
200  lockFile_ = QSharedPointer<QFile>(new QFile(fileName));
201  bool rc = lockFile_->open(QIODevice::WriteOnly | QIODevice::Text);
202  IGNORED_PARAMETER(rc);
203  Q_ASSERT(rc);
204 
205 }

212 void AnalysisOutput::removeFilesThenDirectories(QDir container)
213 {
214  QStringList files = container.entryList(QDir::Files);
215  foreach(QString fileName,files)
216  {
217  if(!container.remove(fileName))
218  {
219  return;
220  }
221  }
222  QStringList directories = container.entryList(QDir::Dirs|QDir::NoDotAndDotDot|QDir::NoSymLinks);
223  foreach(QString directory,directories)
224  {
225  if(!container.rmdir(directory))
226  {
227  QDir subDir = container;
228  subDir.cd(directory);
229  removeFilesThenDirectories(subDir);
230  container.rmdir(directory);
231  }
232  }
233 }
