115  //waveform data
116  //The preamp is typically configured to provide a gain of 1000 and the SIG gain is specified by the

118  //the digital spike samples of the waveform back to the original analog voltage (mV):
119  //For file versions <103:

220 bool PlexonPlugin::acqDataAfterNow()
221 {
222  int NumMAPEvents = MAX_MAP_EVENTS_PER_READ;
223  PL_Event* pServerSkipDataBuffer = (PL_Event*)malloc(sizeof(PL_Event)*MAX_MAP_EVENTS_PER_READ);
224  //Every time we call PL_GetTimeStampArrays it moves the plexon's read pointer forward
225  //we just have to call it until it stops returning data to move the read pointer to
226  //the current time.
227  while(NumMAPEvents == MAX_MAP_EVENTS_PER_READ)
228  {
229  NumMAPEvents = MAX_MAP_EVENTS_PER_READ;
230  PL_GetTimeStampStructures(&NumMAPEvents,pServerSkipDataBuffer);
231  }
232  free(pServerSkipDataBuffer);
233 
234  return true;
235 }
