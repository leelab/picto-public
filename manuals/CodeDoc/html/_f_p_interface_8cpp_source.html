127 QSharedPointer<Picto::ProtocolCommand> FPInterface::readCommand()
128 {
129  //grab the command
130  QString pendingCommand("");
131  QString currentLine("");
132  QRegExp newLineRegExp("[\r\n]+");
133 
134  //read in the command and headers
135  while(commSocket->canReadLine() && !newLineRegExp.exactMatch(currentLine))
136  {
137  currentLine = commSocket->readLine();
138  pendingCommand += currentLine;
139  }
140 
141  //create the command
142  QSharedPointer<Picto::ProtocolCommand> command(new Picto::ProtocolCommand(pendingCommand));
143 
144  //read in the content. Since this has to be done without dropping frames,
145  //and we know that incoming commands will have extremely small amounts of
146  //content (and definitely won't be streaming), we're going to simply assume
147  //that the content is ready and waiting, and if it isn't, the command will
148  //be sent on with insufficient content
149  int remainingContentLength = command->getFieldValue("Content-Length").toInt();
150 
151  if(commSocket->bytesAvailable() >= remainingContentLength)
152  command->addToContent(commSocket->read(remainingContentLength));
153 
154  return command;
155 
156 }

161 QSharedPointer<Picto::ProtocolResponse> FPInterface::processCommand(QSharedPointer<Picto::ProtocolCommand> command)
162 {

166  //we only deal with Picto/1.0 commands

174 
175  return unknownProtocolResponse;
176 
177  }
178  //The command hopefully got all of its content, if not, we can't reply..
179  else if(command->isPendingContent())
180  {
181  QSharedPointer<Picto::ProtocolResponse> incompleteRequestResponse(
182  new Picto::ProtocolResponse(Picto::Names->directorAppName,
183  "PICTO",
184  "1.0",

186 
187  return incompleteRequestResponse;
188  }
189  //the command is good!
190  else
191  {

193  handler = commandHandlers.find(command->getMethod());
194

202  return notFoundResponse;
