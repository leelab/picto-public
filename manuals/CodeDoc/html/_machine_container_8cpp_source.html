96 bool MachineContainer::addTransition(QSharedPointer<Transition> transition)
97 {
98  QSharedPointer<ResultContainer> source,sourceResult,destination;
99  if(!getTransitionAssets(transition,source,sourceResult,destination))
100  return false;
101 
102  addingTransition_ = true;
103  //It looks like this is a new, valid transition.
104  //Set the pointers to source, sourceResult, and Destination into the transition so that it will be updated
105  // if their names change.
106  //(Note that a name change would mess up our transitions_ list since it is indexed by name, but that only
107  // matters in an experiment. In a real experiment the name won't ever change, so it is irrelevant.
108  //We should never need to reset the asset pointers in a transition. If the transition already has these
109  // pointers, we're probably just resetting the same pointers to it, and if we must be in a copy, paste
110  // situation where two assets temporarily have the same name and asset id. In this case, we don't want
111  // the transition to move from one asset source/result/dest to another so we should skip the setting as well.
112  if(transition->getDestinationAsset().isNull())
113  {
114  transition->setSource(source.staticCast<Asset>());
115  transition->setSourceResult(sourceResult.staticCast<Asset>());
116  transition->setDestination(destination.staticCast<Asset>());
117  }
118 
119  //Check if the transition is already in our child map and if not add it.
120  QList<QSharedPointer<Asset>> transChildren = getGeneratedChildren("Transition");
121  bool found = false;
122  foreach(QSharedPointer<Asset> transChild,transChildren)
123  {
124  if(transChild == transition)
125  {
126  found = true;
127  break;
128  }
129  }
130  if(!found)
131  AddChild("Transition",transition);
132 
133  //Add transition to transitions list
134  transitions_.insert(transition->getSource(), transition);
135  if(transition->getSource().isEmpty() && transition->getSourceResult().isEmpty())
136  initTransition_ = transition;
137 
138  addingTransition_ = false;
139  return true;
140 }

169 bool MachineContainer::validateObject(QSharedPointer<QXmlStreamReader> xmlStreamReader)
170 {
171  if(!OutputElementContainer::validateObject(xmlStreamReader))
172  return false;
173 
174  //Confirm that all transitions are legal
175  QList<QSharedPointer<Asset>> transitionList = getGeneratedChildren(transitionTag_);
176  foreach(QSharedPointer<Asset> transition, transitionList)
177  {
178  QSharedPointer<Transition> tran = transition.staticCast<Transition>();
179  QSharedPointer<ResultContainer> s,r,d;
180  if(!getTransitionAssets(tran,s,r,d))
181  {
182  if(s.isNull() && !r.isNull())
183  {
184  QString errMsg = QString("Contains an illegal transition. "
185  "Source: %1 is not a element.").arg(tran->getSource());
186  addError(errMsg);
187  return false;
188  }
189  if(r.isNull() && !s.isNull())
190  {
191  QString errMsg = QString("Contains an illegal transition. "
192  "Result: %1 is not a result generated by source: %2")
193  .arg(tran->getSourceResult()).arg(tran->getSource());
194  addError(errMsg);
195  return false;
196  }
197  if(d.isNull())
198  {
199  QString errMsg = QString("Contains an illegal transition. "
200  "Destination: %1 is not a element.").arg(tran->getDestination());
201  addError(errMsg);
202  return false;
203  }
204  //If we got here and there is no source and no source result, there is more than on initial state transition.
205  if(s.isNull() && r.isNull())
206  {
207  QString errMsg = QString(
208  "Contains multiple initial state transitions. ");
209  addError(errMsg);
210  return false;
211  }
212  //If we got to here, they're all valid, but the function failed. This means that something
213  //else is already transitioning from this source result.
214  QString errMsg = QString(
215  "Contains multiple transitions originating from the same result of %1 on %2"
216  ).arg(tran->getSourceResult()).arg(tran->getSource());
217  addError(errMsg);
218  return false;
219  }
220  }
221 
222  //Confirm that every element has all of its results connected
223  QList<QSharedPointer<Asset>> elementList = getGeneratedChildren(elementTag_);
224  foreach(QSharedPointer<Asset> elem, elementList)
225  {
226  QSharedPointer<ResultContainer> element = elem.staticCast<ResultContainer>();
227  QList<QSharedPointer<Asset>> elemResults = element->getGeneratedChildren("Result");
228  foreach(QSharedPointer<Asset> result, elemResults)
229  {
230  bool found = false;
231  foreach(QSharedPointer<Asset> transition, transitionList)
232  {
233  if((transition.staticCast<Transition>()->getSourceAsset() == element)
234  && (transition.staticCast<Transition>()->getSourceResultAsset() == result))
235  {
236  found = true;
237  break;
238  }
239  }
240  if(!found)
241  {
242  QString elementName = element->getName();
243  QString unconnectedResult = result->getName();
244  QString errMsg = QString("%1 has an "
245  "unconnected result of %2").arg(elementName).arg(unconnectedResult);
246  addError(errMsg);
247  return false;
248  }
249  }
250  }
251 
252  //Confirm that all of this MachineContainer's results are internally linked to something
253  QList<QSharedPointer<Asset>> results = getGeneratedChildren("Result");
254  foreach(QSharedPointer<Asset> result, results)
255  {
256  bool found = false;
257  foreach(QSharedPointer<Asset> transition, transitionList)
258  {
259  if(transition.staticCast<Transition>()->getDestinationAsset() == result)
260  {
261  found = true;
262  continue;
263  }
264  }
265  if(!found)
266  {
267  QString errMsg = QString("%1 has an unconnected result of %2").arg(getName()).arg(result->getName());
268  addError(errMsg);
269  return false;
270  }
271  }
272 
273  //Confirm that all element, and result names are unique (otherwise transition destinations can be ambiguous)
274  QStringList names;
275  foreach(QSharedPointer<Asset> asset,elementList)
276  {
277  names.push_back(asset->getName());
278  }
279  foreach(QSharedPointer<Asset> asset,results)
280  {
281  names.push_back(asset->getName());
282  }
283  names.sort();
284  if(names.size()>0)
285  {
286  for(QStringList::iterator it = names.begin();it != (names.end()-1);it++)
287  {
288  if((*it) == (*(it+1)))
289  {
290  QString errMsg =
291  QString("MachineContainer: Two components of a MachineContainer can not have the same name: \"%1\".")
292  .arg(*it);
293  addError(errMsg);
294  return false;
295  }
296  }
297  }
298 
299  //If we made it this far, all the transitions are "legal"
300  return true;

309 bool MachineContainer::getTransitionAssets(QSharedPointer<Transition> transition, QSharedPointer<ResultContainer>& source,
310  QSharedPointer<ResultContainer>& sourceResult, QSharedPointer<ResultContainer>& destination)
311 {
312  source = sourceResult = destination = QSharedPointer<ResultContainer>();
313  //Get source, sourceResult, and Destination pointers. Use child lists instead of elements_ / transitions_ lists
314  //in case something happened post serialization (ie. in the state machine editor ) and the elements_, transitions_
315  //lists weren't updated (there is no need to update them except just before running an experiment, in which case we
316  //regenerate the experiment from xml).
317  //Search for source and destination elements.
318  //First try to find the asset based on asset id.
319  QList<QSharedPointer<Asset>> elementsList = getGeneratedChildren(elementTag_);
320  foreach(QSharedPointer<Asset> element, elementsList)
321  {
322  if(transition->getSourceId())
323  {
324  if(element->getAssetId() == transition->getSourceId())
325  {
326  if(element->getName() == transition->getSource())
327  {
328  source = element.staticCast<ResultContainer>();
329  if(!destination.isNull())
330  break;
331  }
332  }
333  }
334 
335  if(transition->getDestinationId())
336  {
337  if(element->getAssetId() == transition->getDestinationId())
338  {
339  if(element->getName() == transition->getDestination())
340  {
341  destination = element.staticCast<ResultContainer>();
342  if(!source.isNull())
343  break;
344  }
345  }
346  }
347  }
348 
349  if(destination.isNull() && transition->getDestinationId())
350  { //Check for result destination by assetid
351  QList<QSharedPointer<Asset>> resultList = getGeneratedChildren("Result");
352  foreach(QSharedPointer<Asset> res, resultList)
353  {
354  if(res->getAssetId() == transition->getDestinationId())
355  {
356  destination = res.staticCast<ResultContainer>();
357  break;
358  }
359  }
360  }
361 
362  //If we don't have both source and destination, try to get them by name
363  if(source.isNull() || destination.isNull())
364  {
365  foreach(QSharedPointer<Asset> element, elementsList)
366  {
367  if(source.isNull() && (element->getName() == transition->getSource()))
368  {
369  source = element.staticCast<ResultContainer>();
370  if(!destination.isNull())
371  break;
372  }
373  if(destination.isNull() && (element->getName() == transition->getDestination()))
374  {
375  destination = element.staticCast<ResultContainer>();
376  if(!source.isNull())
377  break;
378  }
379  }
380 
381  //Destination must be a result
382  if(destination.isNull())
383  {
384  QList<QSharedPointer<Asset>> resultList = getGeneratedChildren("Result");
385  foreach(QSharedPointer<Asset> res, resultList)
386  {
387  if(res->getName() == transition->getDestination())
388  {
389  destination = res.staticCast<ResultContainer>();
390  break;
391  }
392  }
393  }
394  }
395 
396  //If there's no source but there is a source result, something is wrong. If both are empty
397  //its a transition to the initial state.
398  if(source.isNull() && !transition->getSourceResult().isEmpty())
399  return false;
400  if(!transition->getSourceResult().isEmpty())
401  {
402  //Search for sourceResult in source.
403  //First search by Id
404  if(transition->getSourceResultId())
405  {
406  QList<QSharedPointer<Asset>> resultList = source->getGeneratedChildren("Result");
407  foreach(QSharedPointer<Asset> res, resultList)
408  {
409  if(res->getAssetId() == transition->getSourceResultId())
410  {
411  sourceResult = res.staticCast<ResultContainer>();
412  break;
413  }
414  }
415  }
416 
417  //Now try searching by name
418  if(sourceResult.isNull())
419  {
420  QList<QSharedPointer<Asset>> resultList = source->getGeneratedChildren("Result");
421  foreach(QSharedPointer<Asset> res, resultList)
422  {
423  if(res->getName() == transition->getSourceResult())
424  {
425  sourceResult = res.staticCast<ResultContainer>();
426  break;
427  }
428  }
429  }
430  if(sourceResult.isNull())
431  return false;
432  }
433  if(destination.isNull())
434  return false;
435 
436  //If we made it this far, we found source, sourceResult and Destination.
437  //Make sure nothing else is transitioning from this transitions sourceResult
438  foreach(QSharedPointer<Transition> trans, transitions_)
439  {
440  QString transitionSrcRealName;
441  QString transSrcRealName;
442  if(transition->getSourceAsset())
443  transitionSrcRealName = transition->getSourceAsset()->getName();
444  if(trans->getSourceAsset())
445  transSrcRealName = trans->getSourceAsset()->getName();
446  QString src = transition->getSource();
447  QString dest = transition->getDestination();
448  QString result = transition->getSourceResult();
449  if( (trans != transition) && (trans->getSourceAsset() == source)
450  && (trans->getSourceResultAsset() == sourceResult)
451  )
452  return false;
453  }
454  return true;
455 }
