120 void AlignmentTool::updateCoefficients(double bAlignTimestamp,
121  double nAlignTimestamp)
122 
123 {
124  //update the sums
125  n_++;
126  sumXX_ += nAlignTimestamp*nAlignTimestamp;
127  sumYY_ += bAlignTimestamp*bAlignTimestamp;
128  sumXY_ += nAlignTimestamp*bAlignTimestamp;
129  sumX_ += nAlignTimestamp;
130  sumY_ += bAlignTimestamp;
131 
132  //calulate the new coefficients
133  //SSxx = sum(x[i]^2) - n*mean(x)^2
134  //SSyy = sum(y[i]^2) - n*mean(y)^2
135  //SSxy = sum(x[i]y[i]) - n*mean(x)*mean(y)
136 
137  //B = SSxy/SSxx
138  //A = mean(y) - B*mean(x)
139  //corr = r^2 = SSxy^2/(SSxx*SSyy)
140 
141  double meanX = sumX_/n_;
142  double meanY = sumY_/n_;
143 
144  double SSxx = sumXX_ - n_*meanX*meanX;
145  double SSyy = sumYY_ - n_*meanY*meanY;
146  double SSxy = sumXY_ - n_*meanX*meanY;
147  if(SSxx == 0)
148  { // This means the variance of the neural timestamps is zero (ie. only one value was ever received.
149  // If this happens, its either the first trial, or something has gone horribly wrong. If its the
150  // first trial SSyy and SSxy will also be zero and Coeff_.B should be 1 (in the limit), with the resulting value
151  // for coeff_.A. coeff_.corr should be 1 (in the limit), so we just set all the SS values to 1 to make these things
152  // happen.
153  Q_ASSERT_X(n_==1,"AlignmentTool::updateCoefficients()","A single neural timestamp value is being used repeatedly in alignment calculations.");
154  SSxx = SSyy = SSxy = 1;
155  }
156 
157  QMutexLocker locker(alignmentMutex_.data());
158  coeff_.B = SSxy/SSxx;
159  coeff_.A = meanY - coeff_.B*meanX;
160  // Only update correlation coefficient once n_ >= 2. This is because n==1 doesn't provide any slope information for our linear fit, only constant offset info
161  // we want that fact reflected in the output.
162  if(n_ < 2)
163  coeff_.corr = 0;
164  else
165  coeff_.corr = (SSxy*SSxy)/(SSxx*SSyy);
166 }
