556 bool MainWindow::loadFile(const QString filename)
557 {
558  QFile file(filename);
559  QSqlDatabase newSession;
560  QString designXML = "";
561  bool isDesignFile = filename.right(4).toLower() == ".xml";
562  if(isDesignFile)
563  { //Its an xml design file
564  if(!file.open(QIODevice::ReadOnly))
565  {
566  QMessageBox::critical(this, Picto::Names->workstationAppName,
567  "Failed to open file: " + filename);
568  return false;
569  }
570 
571  designXML = file.readAll();
572 #ifdef _WIN32
573  //Adds the file to Windows' list of recent documents
574  SHAddToRecentDocs(SHARD_PATHA, filename.toLocal8Bit());
575 #endif
576  file.close();
577  }
578  else
579  { //Its a sqlite session file
580  QString path = filename;
581  path = path.replace("\\","/");
582  Q_ASSERT(path.lastIndexOf("/") >=0);
583  Q_ASSERT(path.lastIndexOf(".") >=0);
584  QString connectionName = path.mid(path.lastIndexOf("/")+1);
585  connectionName = connectionName.left(connectionName.lastIndexOf("."));
586  newSession = QSqlDatabase::addDatabase("QSQLITE",connectionName);
587  newSession.setDatabaseName(path);
588  newSession.open();
589  if(!newSession.isOpen())
590  {
591  QSqlError error = newSession.lastError();
592  QMessageBox::critical(0,"Could not open session",error.text());
593  return false;
594  }
595  QSqlQuery query(newSession);
596 
597  //Get runs list.
598  //ExperimentXML name changed to DesignXML when we started putting UI Data and Analysis Data together
599  //with it on server. They never appeared at the same time so we can just look for either one.
600  query.prepare("SELECT value FROM sessioninfo WHERE key='ExperimentXML' OR key='DesignXML' LIMIT 1");
601  bool success = query.exec();
602  if(!success || !query.next())
603  {
604  QMessageBox::critical(0,"Failed to select Design XML from Session",success?"":query.lastError().text().toLatin1());
605  return false;
606  }
607  designXML = query.value(0).toString();
608 #ifdef _WIN32
609  //Adds the file to Windows' list of recent documents
610  SHAddToRecentDocs(SHARD_PATHA, filename.toLocal8Bit());
611 #endif
612  newSession.close();
613  }
614  if(designXML.isEmpty())
615  {
616  QMessageBox::critical(0,"Experiment design was empty.","Please try a different file");
617  return false;
618  }
619  QSharedPointer<DesignRoot> newDesignRoot = QSharedPointer<DesignRoot>(new DesignRoot());
620  if(!newDesignRoot->resetDesignRoot(designXML))
621  {
622 
623  DesignMessage errorMsg = newDesignRoot->getLastError();
624  QMessageBox box;
625  box.setTextFormat(Qt::RichText);
626  box.setText(errorMsg.name);
627  box.setDetailedText(errorMsg.details);
628  box.setIconPixmap(QPixmap(":/icons/x.png"));
629  box.exec();
630  return false;
631  }
632  if(newDesignRoot->hasWarning())
633  {
634  DesignMessage warnMsg = newDesignRoot->getLastWarning();
635  QMessageBox::warning(0,warnMsg.name,warnMsg.details);
636  }
637  designRoot_ = newDesignRoot;
638  //Load the design root into the autosaver. Don't auto save the current version
639  //because we just opened it from file, we already have a saved version.
640  AutoSaver::getSingleton()->setDesignRoot(designRoot_,false);
641 
642  if(filename == DEFAULT_FILE)
643  setCurrentFile("");
644  else
645  setCurrentFile(filename);
646 
647  return true;
648 }

680 void MainWindow::updateRecentFileActions()
681 {
682  //Clear out any non-existent files
683  foreach(QString filename, recentFiles_)
684  {
685  if(!QFile::exists(filename))
686  recentFiles_.removeAll(filename);
687  }
688 
689  //truncate the list so that only the 5 most recent files are included
690  recentFiles_ = recentFiles_.mid(0,5);
691 
692  //Set up an action for each recent file
693  int actionIdx = 0;
694 
695  foreach(QString filename, recentFiles_)
696  {
697  QString actionText = QString("%1 %2").arg(actionIdx+1)
698  .arg(QFileInfo(filename).fileName());
699  recentExperimentsActions_[actionIdx]->setText(actionText);
700  recentExperimentsActions_[actionIdx]->setVisible(true);
701  recentExperimentsActions_[actionIdx]->setData(filename);
702  actionIdx++;
703  }
704 
705  //hide the leftover actions
706  while(actionIdx++<MaxRecentFiles)
707  recentExperimentsActions_[actionIdx]->setVisible(false);
708 
709  //Show/hide the seperator
710  seperatorAction_->setVisible(!recentFiles_.isEmpty());
711 }
