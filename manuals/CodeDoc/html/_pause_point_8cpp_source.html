61 QString PausePoint::run(QSharedPointer<Engine::PictoEngine> engine)
62 {
63  resetScriptableValues();
64  int command = engine->getEngineCommand();
65  //Pause the engine if force pause is enabled
66  if( (command != Engine::PictoEngine::StopEngine)
67  && propertyContainer_->getPropertyValue("ForcePause").toBool())
68  {
69  //We don't directly pause the engine because then whoever is controlling the engine (componentStatusManager,
70  // TestViewer, etc) won't know about it. Instead we use requestPause() to ask the master to pause the engine
71  // for us.
72  engine->requestPause();
73 
74  command = engine->getEngineCommand();
75  }
76  //If engine is paused, suspend execution until play is pressed
77  runEntryScript();
78  while(command == Engine::PictoEngine::PauseEngine)
79  {
80  //Show frame
81  scene_->render(engine,getAssetId());
82  QCoreApplication::processEvents();
83  command = engine->getEngineCommand();
84  }
85  //If it was a stop that got us out of the pause, return an EngineAbort.
86  if(command == Engine::PictoEngine::StopEngine)
87  {
88  setLatestResult("EngineAbort");
89  return "EngineAbort";
90  }
91  setLatestResult("done");
92  getResult("done")->runResultScript();
93  runExitScript();
94  return "done";
95 }

188 void PausePoint::rebuildScene()
189 {
190  //Since the scene needs access to visual elements stored above it in the tree, we get our output elements from the
191  // output element list.
192  scene_ = Scene::createScene();
193  hasCursor_ = false;
194  QList<QSharedPointer<ContainerElement>> outputs = getElementList();
195  QList<QUuid> activeAnalyses = getDesignConfig()->getActiveAnalysisIds();
196  QHash<QUuid,bool> activeAnalysisHash;
197  foreach(QUuid analysisId,activeAnalyses)
198  {
199  activeAnalysisHash[analysisId] = true;
200  }
201  foreach(QSharedPointer<ContainerElement> output, outputs)
202  {
203  if(output.isNull())
204  continue;
205  if(output.dynamicCast<VisualElement>())
206  {
207  //If this is an analysis element and its analysis id isn't active, don't add it to the scene.
208  AssociateElement* analysisVisElem = dynamic_cast<AssociateElement*>(output.data());
209  if(analysisVisElem && !activeAnalysisHash.contains(analysisVisElem->getAssociateId()))
210  continue;
211  scene_->addVisualElement(output.staticCast<VisualElement>());
212  }
213  else if (output.dynamicCast<OutputSignal>())
214  {
215  scene_->addOutputSignal(output.staticCast<OutputSignal>());
216  }
217  else if(output.dynamicCast<AudioElement>())
218  {
219  scene_->addAudioElement(output.staticCast<AudioElement>());
220  }
221  else if (output.dynamicCast<DataViewElement>())
222  {
223  scene_->addDataViewElement(output.staticCast<DataViewElement>());
224  }
225  }
226  QColor backgroundColor;
227  backgroundColor.setNamedColor(propertyContainer_->getPropertyValue("BackgroundColor").toString());
228  scene_->setBackgroundColor(QColor(propertyContainer_->getPropertyValue("BackgroundColor").toString()));
229 }

237  //Even though this happens as a result of a change in the UI. Since its connected on a signal->slot basis, the
238  // function should run in the Experiment thread. Since the experiment thread never runs during the course of scene
239  // rendering (if they are in two different threads, the experiment thread pauses and waits), there should be no issue
240  // of accidentally changing scene values while they are being used in the UI thread.
241  rebuildScene();
