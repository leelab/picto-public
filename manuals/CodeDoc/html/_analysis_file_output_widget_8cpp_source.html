154  //Disable previous and next buttons
155  prev_->setEnabled(false);
156  next_->setEnabled(false);
157 
158  if(currPage_ < 0)
159  currPage_ = 0;
160 
161  int lastPageEnd = 0;
162  int newPageEnd = -1;
163 
164  //If we don't yet have all the page limits defined up the current page selected by the user
165  while(currPage_ >= pages_.size() && !outputFileStream_->atEnd())
166  { //Expand pages_ definitions until we reach the desired page
167  bool finishingOldPage = false;
168  if(pages_.size())
169  {
170  if(pages_.last().completePage)
171  {
172  lastPageEnd = pages_.last().endPos;
173  }
174  else
175  {
176  finishingOldPage = true;
177  lastPageEnd = pages_.last().startPos;
178  }
179  }
180  patchQtTextStreamBug();
181  if(!outputFileStream_->seek(lastPageEnd)
182  || outputFileStream_->atEnd())
183  break;
184  bool pageComplete = true;
185  for(int i=0;i<linesPerPage_;i++)
186  {
187  outputFileStream_->readLine();
188  if(outputFileStream_->atEnd())
189  {
190  break;
191  }
192  }
193  int currPos = outputFileStream_->pos(); //Note: putting pos() inside the above loop makes things really slow!!
194  if(currPos < 0 || outputFileStream_->atEnd())
195  pageComplete = false;
196  newPageEnd = currPos;
197  if(finishingOldPage)
198  pages_[pages_.size()-1] = PageDef(lastPageEnd,newPageEnd,pageComplete);
199  else
200  pages_.append(PageDef(lastPageEnd,newPageEnd,pageComplete));
201  }
202  //If pages_ still isn't big enough, we must have hit the end of the file. Correct currPage_
203  if(currPage_ >= pages_.size())
204  { //We must have run out of data. Show the last page with data.
205  currPage_ = pages_.size() - 1;
206  }
207 
208  patchQtTextStreamBug();
209  //If there is no data in the file or we cant seek the beginning of the current page.
210  //Clear output and go back to the beginning of the file.
211  if(pages_.empty() || !outputFileStream_->seek(pages_[currPage_].startPos))
212  {
213  //Something went wrong. Start from the beginning.
214  lastPage_ = -1;
215  currPage_ = -1;
216  textEdit_->setText("");
217  return;
218  }

220  //I would have liked to use the QTextStream::read() function to read the text for this page in all in
221  //one shot. The problem was that for some reason, when specifying the number of characters to read in
222  //with the read function, you have to skip "\n" characters. This means that if we simply used the
223  //pages_[currPage_].endPos-pages_[currPage_].startPos, the value would be too high. We could simply
224  //delete the number of "\n" characters from the read() input, but that would mean recording the number
225  //of lines read on the last page. It also isn't clear if there are other characters that aren't
226  //counted by read() as well (whitespace for example). By simply reading in linesPerPage_ lines, we
