195  void RasterBasePlotHandler::exportClicked()
196  {
197  QString selectedFilter;
198 
199  QString fileName = QFileDialog::getSaveFileName(
200  m_pDataSelectionWidget,
201  "Export Plot",
202  m_tmpTitle,
203  "Small (*.png *.pdf *.bmp);;Medium (*.png *.pdf *.bmp);;Large (*.png *.pdf *.bmp);;Huge (*.png *.pdf *.bmp)",
204  &selectedFilter);
205 
206  //Return if canceled
207  if (fileName.isEmpty())
208  {
209  return;
210  }
211 
212  ExportType::ExportType exportType = ExportType::EXPORT_PNG;
213  DataViewSize::ViewSize exportSize = DataViewSize::VIEW_SIZE_2x2;
214 
215  if (selectedFilter == "Small (*.png *.pdf *.bmp)")
216  {
217  exportSize = DataViewSize::VIEW_SIZE_1x1;
218  }
219  else if (selectedFilter == "Medium (*.png *.pdf *.bmp)")
220  {
221  exportSize = DataViewSize::VIEW_SIZE_2x2;
222  }
223  else if (selectedFilter == "Large (*.png *.pdf *.bmp)")
224  {
225  exportSize = DataViewSize::VIEW_SIZE_3x3;
226  }
227  else if (selectedFilter == "Huge (*.png *.pdf *.bmp)")
228  {
229  exportSize = DataViewSize::VIEW_SIZE_4x4;
230  }
231 
232  if (fileName.endsWith(".png", Qt::CaseInsensitive))
233  {
234  exportType = ExportType::EXPORT_PNG;
235  }
236  else if (fileName.endsWith(".bmp", Qt::CaseInsensitive))
237  {
238  exportType = ExportType::EXPORT_BMP;
239  }
240  else if (fileName.endsWith(".pdf", Qt::CaseInsensitive))
241  {
242  exportType = ExportType::EXPORT_PDF;
243  }
244  else
245  {
246  exportType = ExportType::EXPORT_PNG;
247  fileName += ".png";
248  qDebug() << "Unrecognized export extension: defaulting to png";
249  }
250 
251  exportPlot(exportType, exportSize, fileName);
252  }
