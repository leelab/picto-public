52 void Scene::render(QSharedPointer<Engine::PictoEngine> engine,int callerId)
53 {
54  //Setup rendering routine.
55  QMutexLocker locker(mutex_.data());
56  engine_ = engine;
57  readyToRender_ = true;
58 
59  //Tell the render routine to run the next time the main UI thread has the processor.
60  emit readyForRender(callerId);
61 
62  //Wait for the render routine to finish.
63  bool endLoop = false;
64  if(readyToRender_)
65  { //If we're in here, we're running with more the experiment in a non-ui thread.
66  do
67  {
68  //Allow UI thread to do the render. Unlock, yield this thread and relock.
69  locker.unlock();
70  QThread::currentThread()->yieldCurrentThread();
71 
72  staticMutex_.lock();
73  endLoop = closeRenderLoops_;
74  staticMutex_.unlock();
75  locker.relock();
76  //Check if the render occured, if not, loop back to "Allow UI thread to do the render"
77  }while(readyToRender_ && !endLoop);
78  }
79 }

213  }

218  if(!unaddedOutputSignals_.isEmpty())
219  {
220  //add the appropriate compositing surfaces to the element
221  for(int i=0; i<unaddedOutputSignals_.length(); i++)
222  {
223  //Do Output Signal setup here
224  outputSignals_.append(unaddedOutputSignals_[i]);
225  }
