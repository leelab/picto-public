36 void TaskRunSelector::loadSessions(QList<QSqlDatabase> sessions)
37 {
38  selectArea_->clear();
39  runs_.clear();
40  sessions_.clear();
41  foreach(QSqlDatabase session,sessions)
42  {
43  QSqlQuery query(session);
44  query.setForwardOnly(true);
45 
46  //Get frame value list. If there is no last frame listed, the run might still be in progress or
47  //their was a failure somewhere and the session ended while the Run was still in progress.
48  query.prepare("SELECT r.runid,r.name,r.notes,r.saved,r.firstframe,r.lastframe,s.time,e.time "
49  "FROM runs r, frames s, frames e WHERE s.dataid=r.firstframe AND ((r.lastframe>0 AND e.dataid=r.lastframe) "
50  "OR (r.lastframe=0 AND e.dataid=(SELECT dataid FROM frames ORDER BY dataid DESC LIMIT 1))) ORDER BY runid");
51  bool success = query.exec();
52  if(!success)
53  {
54  qDebug("Failed to select data from runs table with error: " + query.lastError().text().toLatin1());
55  return;
56  }
57  while(query.next()){
58  //continue if the run wasn't saved.
59  if(!query.value(3).toBool())
60  continue;
61  //Build up the run object
62  QSharedPointer<Picto::TaskRunDataUnit> run(
63  new Picto::TaskRunDataUnit
64  (
65  query.value(4).toLongLong(),
66  query.value(5).toLongLong(),
67  query.value(1).toString(),
68  query.value(2).toString(),
69  query.value(3).toBool()
70  )
71  );
72  run->setDataID(query.value(0).toLongLong());
73  //Add RunData to the runs_ list.
74  runs_.append(RunData(run,query.value(6).toDouble(),query.value(7).toDouble()));
75  sessions_.append(session);
76  //Add run name to the selection area.
77  QListWidgetItem* runItem(new QListWidgetItem(run->name_));
78  runItem->setData(32,selectArea_->count());
79  runItem->setToolTip(run->notes_);
80  selectArea_->addItem(runItem);
81  }
82  query.finish();
83  }
84  if(!selectArea_->count())
85  {
86  QMessageBox box;
87  box.setText("No Saved Runs ");
88  box.setDetailedText("No saved runs were found in these sessions.");
89  box.setIconPixmap(QPixmap(":/icons/x.png"));
90  box.exec();
91  }
92  else
93  selectAll();
94 }
