206 void ViewSelectionWidget::updateSizeAndPositionOptions()
207 {
208  //Update size combobox to reflect current widget's size
209  DataViewWidget *pCurrentView = nullptr;
210  int iNewSizeIndex = 0;
211 
212  if (widgetIndexMap_.contains(plotSelection_->currentData().toInt()))
213  {
214  pCurrentView = widgetIndexMap_[plotSelection_->currentData().toInt()];
215  iNewSizeIndex = sizeSelection_->findData(pCurrentView->getCurrentSize());
216  }
217 
218  sizeSelection_->setCurrentIndex(iNewSizeIndex);
219 
220  //Initialize the state of the checkboxes.
221  for (int x = 0; x < VIEWGRIDWIDTH; x++)
222  {
223  for (int y = 0; y < VIEWGRIDHEIGHT; y++)
224  {
225  positionCheckBoxes_[x][y]->setDisabled(false);
226  positionCheckBoxes_[x][y]->setCheckState(Qt::Unchecked);
227  }
228  }
229 
230  //Update position checkboxes to reflect current options
231  foreach(DataViewWidget *pWidget, dataViewWidgets_)
232  {
233  if (pWidget->getDisplayed())
234  {
235  int xPos, yPos;
236  pWidget->getPosition(xPos, yPos);
237 
238  if (pWidget != pCurrentView)
239  {
240  for (int x = xPos; x < xPos + pWidget->getCurrentSize(); x++)
241  {
242  for (int y = yPos; y < yPos + pWidget->getCurrentSize(); y++)
243  {
244  Q_ASSERT(x < VIEWGRIDWIDTH);
245  Q_ASSERT(y < VIEWGRIDHEIGHT);
246  positionCheckBoxes_[x][y]->setDisabled(true);
247  positionCheckBoxes_[x][y]->setCheckState(Qt::Unchecked);
248  }
249  }
250  }
251  else
252  {
253  for (int x = xPos; x < xPos + pWidget->getCurrentSize(); x++)
254  {
255  for (int y = yPos; y < yPos + pWidget->getCurrentSize(); y++)
256  {
257  Q_ASSERT(x < VIEWGRIDWIDTH);
258  Q_ASSERT(y < VIEWGRIDHEIGHT);
259 
260  if (x == xPos && y == yPos)
261  {
262  positionCheckBoxes_[x][y]->setCheckState(Qt::Checked);
263  }
264  else
265  {
266  positionCheckBoxes_[x][y]->setCheckState(Qt::PartiallyChecked);
267  }
268  }
269  }
270  }
271  }
272  }
273 
274  if (pCurrentView)
275  {
276  //Disable impossible positions
277  for (int x = VIEWGRIDWIDTH - pCurrentView->getCurrentSize() + 1; x < VIEWGRIDWIDTH; x++)
278  {
279  for (int y = 0; y < VIEWGRIDHEIGHT; y++)
280  {
281  positionCheckBoxes_[x][y]->setDisabled(true);
282  }
283  }
284 
285  for (int x = 0; x < VIEWGRIDWIDTH; x++)
286  {
287  for (int y = VIEWGRIDHEIGHT - pCurrentView->getCurrentSize() + 1; y < VIEWGRIDHEIGHT; y++)
288  {
289  positionCheckBoxes_[x][y]->setDisabled(true);
290  }
291  }
