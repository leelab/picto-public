62 bool DataStore::serializeAsXml(QSharedPointer<QXmlStreamWriter> xmlStreamWriter)
63 {
64  Q_ASSERT_X(tagText_ != "","DataStore::serializeAsXml","This datastore has no tag text, it was either not serialized in or not initialized to default values.");
65  QSharedPointer<QXmlStreamReader> xmlReader(new QXmlStreamReader(tagText_));
66 
67  //Move read pointer to the first token
68  while(!xmlReader->isStartElement())
69  xmlReader->readNext();
70 
71  //Write out first token (ie. My tag name).
72 
73  //Check if the token includes an id and if that id has changed
74  if(xmlReader->attributes().hasAttribute("id"))
75  {
76  //Check if the id value has changed
77  if(xmlReader->attributes().value("id").toString().toInt()!=assetId_)
78  {
79  xmlStreamWriter->writeStartElement(xmlReader->name().toString());
80  foreach(QXmlStreamAttribute attribute,xmlReader->attributes())
81  {
82  if(attribute.name() == "id")
83  continue;
84  xmlStreamWriter->writeAttribute(attribute.name().toString(),attribute.value().toString());
85  }
86  //Write the new Id.
87  xmlStreamWriter->writeAttribute("id",QString::number(assetId_));
88  }
89  else
90  {
91  //Nothing changed. Just write out old value.
92  xmlStreamWriter->writeCurrentToken(*xmlReader);
93  }
94  }
95  else
96  {
97  //There is no id value. Add it.
98  xmlStreamWriter->writeCurrentToken(*xmlReader);
99  xmlStreamWriter->writeAttribute("id",QString::number(assetId_));
100  }
101 
102  //Get this DataStore's tag name from the stored xml.
103  myTagName_ = xmlReader->name().toString();
104 
105  bool returnVal = true;
106  //Make a list of iterators to use for iterating through the children_ data.
107  //Make an "end iterator" list as well to find out when you're done.
108  QMap<QString,QList<QSharedPointer<Asset>>::iterator> readIterators;
109  QMap<QString,QList<QSharedPointer<Asset>>::iterator> readEndIterators;
110  for(QMap<QString,QList<QSharedPointer<Asset>>>::iterator iter=children_.begin();iter!=children_.end();iter++)
111  {
112  readIterators[iter.key()] = (*iter).begin();
113  readEndIterators[iter.key()] = (*iter).end();
114  }
115  QMap<QString,QList<QSharedPointer<Asset>>::iterator>::iterator dataIter, endIter;
116 
117  //DO ALL THE FANCY READING AND WRITING OF XML TO KEEP ORDERING INTACT HERE
118  while(xmlReader->readNext() && !(xmlReader->isEndElement() && (xmlReader->name().toString() == myTagName_)) && !xmlReader->atEnd())
119  {
120  //Continue loop if we're not at a start element
121  if(!xmlReader->isStartElement())
122  {
123  xmlStreamWriter->writeCurrentToken(*xmlReader);// Serialize out whatever we read
124  continue;
125  }
126  //Check to see if this token corresponds to a child.
127  //Get Tag name
128  QString name = xmlReader->name().toString();
129  //See if there's a read iterator for this tag.
130  if(!readIterators.contains(name))
131  {
132  //It must be a comment. Start the process of writing it out.
133  xmlStreamWriter->writeCurrentToken(*xmlReader);
134  continue;
135  }
136  //Get the Asset for the tag
137  Q_ASSERT_X(readIterators[name] != readEndIterators[name],"DataStore::serializeAsXml","The original deserialized xml string contains a tag for which there are no remaining children to deserialize.");
138  QSharedPointer<Asset> child = *(readIterators[name]++);
139  Q_ASSERT_X(!child.isNull(),"DataStore::serializeAsXml","A child asset pointer in this DataStore's list is null.");
140  //Deserialize all new children until you get to one that isn't new
141  while(child->isNew())
142  {
143  child->toXml(xmlStreamWriter);
144  Q_ASSERT_X(readIterators[name] != readEndIterators[name],"DataStore::serializeAsXml","The original deserialized xml string contains a tag for which there are no remaining children to deserialize.");
145  child = *(readIterators[name]++);
146  }
147  //This is the child represented by this tag. Deserialize it.
148  child->toXml(xmlStreamWriter);
149  //Move read pointer forward. The next thing in it should be an end tag, and we don't want
150  //to write that out because the child Asset already wrote out its end tag.
151  xmlReader->readNext();
152  }
153 
154  //Write out everything that wasn't yet written that needs to be
155  dataIter = readIterators.begin();
156  endIter = readEndIterators.begin();
157  QSharedPointer<Asset> childAsset;
158  while(dataIter!=readIterators.end())
159  {
160  while((*dataIter) != (*endIter))
161  {
162  childAsset = (*(*dataIter));
163  // If the childAsset is not New, the only explanation is that the asset was serialized in under one tag name,
164  // then copied to a list of child assets under another tag (as in the case of a choice controller's targets
165  // that are also used as results. In these types of cases, the asset was already serialized under the correct
166  // tag name when we were going through our tagText_, so we skip childAssets that are not new here.
167  //
168  // If the childAsset IS new but was not edited, it must have been an unedited default
169  // value and should not be serialized out. Otherwise it must have been a
170  // default value that was altered by the user or a new value altogether. It must now be serialized out.
171  if(childAsset->isNew() && childAsset->wasEdited())
172  childAsset->toXml(xmlStreamWriter);
173  (*dataIter)++;
174  }
175  dataIter++;
176  endIter++;
177  }
178  //Write out this Datastore's closing tag.
179  xmlStreamWriter->writeCurrentToken(*xmlReader);
180  return returnVal;
181 }

224  tagText_ = "";
225  QSharedPointer<QXmlStreamWriter> xmlWriter(new QXmlStreamWriter(&tagText_));
226  xmlWriter->writeCurrentToken(*xmlStreamReader);// Lets write the start tag to our tagText.

230  children_.clear();
231  propertyContainer_->clear();

233  {
234  if(factory.isNull())
235  continue; //It is a placeholder tag
236  factory->startSourcing();
237  }
238 
239  //Get Start tag name (depending on who the parent is, they may have given us different names.
240  myTagName_ = xmlStreamReader->name().toString();

244  addError(QString("Incorrect tag, expected <%1>").arg(myTagName_).toLatin1());
245  returnVal = false;
246  return returnVal;
247  }
248 
249 
250  foreach(QXmlStreamAttribute attribute,xmlStreamReader->attributes())
251  {
252  if(attribute.name() == "id")

254  assetId_ = attribute.value().toString().toInt();
255  }
256  else if(attribute.name() == "type")
257  {
258  }
259  else
260  {
261  //Make sure that there are no attributes besides "type" and "id"
262  addError(QString("Incorrect attribute:\"%1\", only \"type\" and \"id\" attributes are allowed").arg(attribute.name().toString()));
263  returnVal = false;
264  return returnVal;
265  }
266  }
267 
269  //int numAttributes = xmlStreamReader->attributes().size();
270  //if((numAttributes > 1) || ((numAttributes == 1) && !xmlStreamReader->attributes().hasAttribute("type")))
271  //{
272  // addError(myTagName_.toLatin1(),QString("Incorrect attribute(s), only \"type\" attributes are allowed"),xmlStreamReader);
273  // returnVal = false;
274  // return returnVal;
275  //}
276 
277  //Loop until we're done with the tag or we reach the end of the XMLStream
278  bool childWithSameTag = false; //If this is true, we just deserialized a child with the same tag, so we should ignore its end tag.
279  while(!(xmlStreamReader->isEndElement() && (xmlStreamReader->name().toString() == myTagName_) && !childWithSameTag) && !xmlStreamReader->atEnd())
280  {
281  childWithSameTag = false;
282  //Move forward in XMLStream
283  if(xmlStreamReader->readNext() == QXmlStreamReader::Invalid)
284  {
285  addError(QString("XML syntax has been violated.").toLatin1());
286  return false;
287  }
288  xmlWriter->writeCurrentToken(*xmlStreamReader);// Lets add the current tag to our tagText.
289  //Continue loop if we're not at a start element
290  if(!xmlStreamReader->isStartElement())
291  continue;
292  //Get Tag name
293  QString name = xmlStreamReader->name().toString();
294  //See if there's a Asset factory for this tag.
295  if(!factories_.contains(name))
296  {
297  //There's no entry in our factory list for this tag. Syntax Error.
298  //Report error and attempt to move XMLStream pointer to the end of this tag
299  returnVal = false;
300  addError(QString("The tag: <%1> is invalid in the %2 object definition").arg(name).arg(myTagName_).toLatin1());
301  while(!(xmlStreamReader->isEndElement() && (xmlStreamReader->name().toString() == name)) && !xmlStreamReader->atEnd())
302  {
303  xmlStreamReader->readNext();
304  }
305  xmlWriter->writeCurrentToken(*xmlStreamReader);// Lets add the closing tag from the child to our tagText.
306  continue;
307  }
308  //Check if its a child with the same tag as us. If so, we'll need to ignore the end tag when we read it on the next loop.
309  if(name == myTagName_)
310  childWithSameTag = true;
311 
312  //Get the factory for the tag
313  QSharedPointer<AssetFactory> factory = factories_.value(name);
314  if(factory.isNull())
315  {
316  //If factory is null, this is a place holder tag to be ignored.
317  Q_ASSERT_X(!childWithSameTag,"DataStore::deserializeFromXml",QString("The tag: <%1> has the same name as its parent and cannot be used as a placeholder").arg(name).toLatin1());
318  continue;
319  }
320  //Get the new Asset
321  //
322  //The command below will put an empty string into type if the "type" attribute
323  //isn't found. This should map to the default value in the factory.
324  QString type = xmlStreamReader->attributes().value("type").toString();
325  QString error = "";
326  QSharedPointer<Asset> newChild = factory->getAsset(error,type);
327  //If the new child is Null then that factory must not support the input type.
328  //Report an error and attempt to continue deserialization to catch any other errors.
329  if(newChild.isNull())
330  {
331  returnVal = false;
332  addError(QString(error));
333  //Attempt to loop until the end tag of the bad tag.
334  while(!(xmlStreamReader->isEndElement() && (xmlStreamReader->name().toString() == name)) && !xmlStreamReader->atEnd())
335  {
336  xmlStreamReader->readNext();
337  }
338  xmlWriter->writeCurrentToken(*xmlStreamReader);// Lets add the closing tag from the child to our tagText.
339  continue;
340  }
341  //We created the new child.
342  //Lets add it to our list of children
343  AddChild(name,newChild);
344  //Lets deserialize it.
345  if(!newChild->fromXml(xmlStreamReader))
346  {
347  //Child deserialization failed. For now, just exit.
348  //We may want to attempt to continue deserialization in the futures in order
349  //to present other errors.
350  returnVal = false;
351  return false;
352  }
353  xmlWriter->writeCurrentToken(*xmlStreamReader); // Lets add the closing tag from the child to our tagText.
354  //continue the loop
355  }
356  //Make sure we didn't finish document
357  if(xmlStreamReader->atEnd())
358  {
359  returnVal = false;
360  addError("Unexpected end of document");
361  }
362  //Generate any default Assets that were not in the XML
363  for(QMap<QString,QSharedPointer<AssetFactory>>::iterator iter = factories_.begin();iter!=factories_.end();iter++)
364  {
365  if(iter.value().isNull())
366  continue; //Its a placeholder tag
367  QSharedPointer<Asset> newChild;
368  forever{
369  QString returnedType = "";
370  newChild = iter.value()->getRequiredAsset(returnedType);
371  if(newChild.isNull())
372  break;
373  //We created the new child.
374  //Lets add it to our list of children
375  AddChild(iter.key(),newChild);
376  //Initialize the child into a default state.
377  newChild->initializeToDefault(iter.key(),returnedType);
378  //continue the loop
379  }
380  }
381  xmlWriter->writeCurrentToken(*xmlStreamReader);// Lets write the end tag to our tagText.
382 
383  return returnVal;
384 }

464 {

466  QString name;
467  //If necessary check to see if children have unique names

492  QMap<QString,bool> associateNameMap;
493  foreach(QList<QSharedPointer<Asset>> assetList,associateChildren.values())
494  {
495  foreach(QSharedPointer<Asset> child,assetList)
496  {
497  if(child->needsUniqueName())
498  {
499  name = child->getName();
500  if(nameMap.contains(name))
501  {
502  QString errMsg = QString("%1 contains experimental and analysis elements with the same name: %2").arg(getName()).arg(name);
503  addError(errMsg);
504  return false;
505  }
506  if(associateNameMap.contains(name))
507  {
508  QString errMsg = QString("%1 contains more than one analysis element child with the same name: %2").arg(getName()).arg(name);
509  addError( errMsg);
510  return false;

512  associateNameMap[name] = true;
513  }
514  }
515  }
516  }
517  }

928 {

939  return QSharedPointer<Asset>();

942  type = xmlStreamReader->attributes().value("type").toString();
943

945  QSharedPointer<AssetFactory> factory = getAssetFactory(tagName);

947  return QSharedPointer<Asset>();

949  if(newAsset.isNull())

953  AddChild(tagName,newAsset);
954

957  if(!success)
