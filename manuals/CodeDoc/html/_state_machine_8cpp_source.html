66 void StateMachine::upgradeVersion(QString deserializedVersion)
67 {
68  MachineContainer::upgradeVersion(deserializedVersion);
69  if(deserializedVersion < "0.0.1")
70  { // In design syntax version "0.0.1", we removed "ScriptElements" and replaced them with "SwitchElements"
71  // "ScriptElement" were essentially "SwitchElements" that contained only two results, "true" and "false"
72  // that were triggered with a boolean value.
73  // To upgrade, we serialize "ScriptElements" into ObsoleteAssets and create new equivalent "SwitchElements".
74  QList<QSharedPointer<Asset>> stateMachineElements = getGeneratedChildren("StateMachineElement");
75  QList<QSharedPointer<ObsoleteAsset>> scriptElements;
76  foreach(QSharedPointer<Asset> element, stateMachineElements)
77  {
78  if(element->inherits("Picto::ObsoleteAsset")
79  && element.staticCast<ObsoleteAsset>()->getAttributeValue("type") == "ScriptElement")
80  {
81  scriptElements.append(element.staticCast<ObsoleteAsset>());
82  }
83  }
84 
85  //For each script element, create a switch element and script data into it.
86  foreach(QSharedPointer<ObsoleteAsset> scriptElement,scriptElements)
87  {
88  QString error;
89  QSharedPointer<Asset> switchElementAsset = createChildAsset("StateMachineElement","Switch",error);
90  QSharedPointer<SwitchElement> switchElement = switchElementAsset.staticCast<SwitchElement>();
91  QList<QSharedPointer<ObsoleteAsset>> obsAssetList;
92  //Copy name to Switch Element
93  obsAssetList = scriptElement->getObsoleteChildAsset("Name");
94  switchElement->setName(obsAssetList.size()?obsAssetList.first()->getValue():"");
95  //Copy Script to Switch Element
96  obsAssetList = scriptElement->getObsoleteChildAsset("Script");
97  switchElement->getPropertyContainer()->setPropertyValue("Script",obsAssetList.size()?obsAssetList.first()->getValue():"");
98 
99  //Get UI position info and set it to the switch element
100  obsAssetList = scriptElement->getObsoleteChildAsset("UIInfo"); //Get UIINfo sub asset
101  if(!obsAssetList.isEmpty())
102  {
103  obsAssetList = obsAssetList.first()->getObsoleteChildAsset("Pos"); //Get Pos sub tag
104  if(!obsAssetList.isEmpty())
105  {
106  QStringList xy = obsAssetList.first()->getValue().split(","); //Convery ?,? of tag value to string list
107  if(xy.size() == 2) //If string list has correct size, set it to the switch element position
108  switchElement->setPos(QPoint(xy.first().toFloat(),xy.last().toFloat()));
109  else //It must be stored in attributes (we did that at one point)
110  switchElement->setPos(QPoint(obsAssetList.first()->getAttributeValue("x").toInt(),obsAssetList.first()->getAttributeValue("y").toInt()));
111  }
112  }
113 
114  //Create true, false results in Switch Element
115  QSharedPointer<Asset> result = switchElement->createChildAsset("Result","",error);
116  result.staticCast<UIEnabled>()->setName("true");
117  result = switchElement->createChildAsset("Result","",error);
118  result.staticCast<UIEnabled>()->setName("false");\
119  }
120  }
121 }
