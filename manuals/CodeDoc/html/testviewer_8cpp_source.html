51 void TestViewer::init()
52 {
54  //Sometimes while working on an experiment, if we change something and don't reopen the experiment
55  //the experimental run doesn't work right. To prevent this, we are simply reseting the the experiment
56  //from xml whenever we open this viewer
57  designRoot_->refreshFromXml();
59 
60  deiniting_ = false;
61  designRoot_->enableRunMode(true);
62  QSharedPointer<DesignRoot> myDesignRoot = designRoot_;
63  if(!myDesignRoot->compiles())
64  {
65  QMessageBox msg;
66  msg.setText("Experiment does not compile.");
67  msg.setIconPixmap(QPixmap(":/icons/triangle.png"));
68  msg.exec();
69  return;
70  }
71  experiment_ = myDesignRoot->getExperiment().staticCast<Experiment>();
72  if(!experiment_)
73  {
74  QMessageBox msg;
75  msg.setText("Failed to load current experiment.");
76  msg.setIconPixmap(QPixmap(":/icons/triangle.png"));
77  msg.exec();
78  return;
79  }
80  if(experiment_)
81  {
82  experiment_->setEngine(engine_);
83  static_cast<PropertyFrame*>(propertyFrame_)->setTopLevelDataStore(experiment_.staticCast<DataStore>());
84  loadPropsAction_->setEnabled(true);
85  testController_ = QSharedPointer<TestPlaybackController>(new TestPlaybackController(experiment_));
86  connect(playAction_,SIGNAL(triggered()),this, SLOT(playTriggered()));
87  connect(playAction_,SIGNAL(triggered()),testController_.data(), SLOT(play()));
88  connect(pauseAction_,SIGNAL(triggered()),testController_.data(), SLOT(pause()));
89  connect(engine_.data(),SIGNAL(pauseRequested()),testController_.data(),SLOT(pause()));
90  connect(stopAction_,SIGNAL(triggered()),testController_.data(), SLOT(stop()));

94  }

118 void TestViewer::deinit()
119 {
120  deiniting_ = true;
121  if(testController_ && testController_->isRunning())
122  {
123  //If the experiment is still running, tell it to stop.
124  //After it stops, the testController will trigger its stopped() signal which
125  //will trigger our stopped() slot. This will handle emitting the deinitCompleted()
126  //signal. It is important that we do it this way in case the event loop is
127  //currently running from within the experimental run. In that case, we need to actually leave
128  //this deinit() function after telling the experiment to stop in order to give it a
129  //chance to actually follow the stop command. Only once testController emits stopped()
130  //can we be sure that we're really finished with the current experiment. Only then
131  //is it safe to give control to another viewer.
132  testController_->stop();
133  }
134  else
135  stopped();
136  Q_ASSERT(propertyFrame_);
137  //Clear the property frame by setting its data store to null. If we don't do this then an unexpected
138  //path leading from property value changes through the property frame back to
139  //a different part of the propery value would be active... this gets rid of that path
140  static_cast<PropertyFrame*>(propertyFrame_)->setTopLevelDataStore(QSharedPointer<DataStore>());
141 }

346 void TestViewer::playTriggered()
347 {
348  if(!testController_->isRunning())
349  {
350  //Find out which analyses should be enabled and enable them
351  QList<QUuid> analysisIds = analysisSelector_->getSelectedAnalysisIds();
352  QSharedPointer<DesignConfig> designConfig = experiment_->getDesignConfig();
353  //If there are active analysis ids, add the various data readers (frameReader, etc) to the
354  //designConfig. Then add them to the Picto Engine so that it can update them with data.
355  //Generally, the SlaveExperimentDriver is responsible for creating the readers and the StateUpdater
356  //is responsible for making sure they have data. Eventually, the Test Viewer should contain a
357  //running experiment and an interface to a SlaveExperimentViewer that handles the playback and analysis.
358  //Until that happens, we create the DataReaders here and update them in the PictoEngine.
359  if(analysisIds.size())
360  {
361  if(!liveFrameReader_)
362  {
363  //Setup Test Data Readers for Analysis
364  liveFrameReader_ = QSharedPointer<LiveFrameReader>(new LiveFrameReader());
365  liveLfpReader_ = QSharedPointer<LiveLfpReader>(new LiveLfpReader(QList<int>() << 1 << 3 << 4 << 5 << 8,0.001));
366  liveRewardReader_ = QSharedPointer<LiveRewardReader>(new LiveRewardReader());
367  liveRunNotesReader_ = QSharedPointer<LiveRunNotesReader>(new LiveRunNotesReader());
368  liveSpikeReader_ = QSharedPointer<LiveSpikeReader>(new LiveSpikeReader(5,2,32));
369  signalReaders_.append(QSharedPointer<LiveSignalReader>(new LiveSignalReader("Position",QStringList()<<"x"<<"y",0.002)));
370  signalReaders_.append(QSharedPointer<LiveSignalReader>(new LiveSignalReader("Diameter", QStringList() << "x" << "y", 0.004)));
371  signalReaders_.append(QSharedPointer<LiveSignalReader>(new LiveSignalReader("GenericInputs", QStringList() << "input0" << "input1" << "input2" << "input3" << "input4" << "input5" << "input6" << "input7", 0.004)));
372  }
373  designConfig->setFrameReader(liveFrameReader_.staticCast<FrameReader>());
374  designConfig->setLfpReader(liveLfpReader_.staticCast<LfpReader>());
375  designConfig->setRewardReader(liveRewardReader_.staticCast<RewardReader>());
376  designConfig->setRunNotesReader(liveRunNotesReader_.staticCast<RunNotesReader>());
377  designConfig->setSpikeReader(liveSpikeReader_.staticCast<SpikeReader>());
378  foreach(QSharedPointer<LiveSignalReader> signalReader,signalReaders_)
379  {
380  designConfig->setSignalReader(signalReader->getName(),signalReader.staticCast<SignalReader>());
381  }
382  engine_->setFrameReader(liveFrameReader_);
383  engine_->setLfpReader(liveLfpReader_);
384  engine_->setRewardReader(liveRewardReader_);
385  engine_->setSpikeReader(liveSpikeReader_);
386  foreach(QSharedPointer<LiveSignalReader> signalReader,signalReaders_)
387  {
388  engine_->setSignalReader(signalReader->getName(),signalReader);
389  }
390  }
391  else
392  {
393  //If there's no analysis. Set frame readers to null so that we don't waste memory on them.
394  //in case they were set last time.
395  designConfig->setFrameReader(QSharedPointer<FrameReader>());
396  designConfig->setLfpReader(QSharedPointer<LfpReader>());
397  designConfig->setRewardReader(QSharedPointer<RewardReader>());
398  designConfig->setSpikeReader(QSharedPointer<SpikeReader>());
399  designConfig->clearSignalReaders();
400  engine_->setFrameReader(QSharedPointer<LiveFrameReader>());
401  engine_->setLfpReader(QSharedPointer<LiveLfpReader>());
402  engine_->setRewardReader(QSharedPointer<LiveRewardReader>());
403  engine_->setSpikeReader(QSharedPointer<LiveSpikeReader>());
404  engine_->clearSignalReaders();
405  }
406 
407 
408  designConfig->setActiveAnalysisIds(analysisIds);
409  }
410 
411  analysisSelector_->setEnabled(false);
412  //QMetaObject::invokeMethod(testController_.data(),"play");
413 }
