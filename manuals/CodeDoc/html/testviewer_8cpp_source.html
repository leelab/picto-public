126 void TestViewer::deinit()
127 {
128  deiniting_ = true;
129  if(testController_ && testController_->isRunning())
130  {
131  //If the experiment is still running, tell it to stop.
132  //After it stops, the testController will trigger its stopped() signal which
133  //will trigger our stopped() slot. This will handle emitting the deinitCompleted()
134  //signal. It is important that we do it this way in case the event loop is
135  //currently running from within the experimental run. In that case, we need to actually leave
136  //this deinit() function after telling the experiment to stop in order to give it a
137  //chance to actually follow the stop command. Only once testController emits stopped()
138  //can we be sure that we're really finished with the current experiment. Only then
139  //is it safe to give control to another viewer.
140  testController_->stop();
141  }
142  else
143  stopped();
144  Q_ASSERT(propertyFrame_);
145  //Clear the property frame by setting its data store to null. If we don't do this then an unexpected
146  //path leading from property value changes through the property frame back to
147  //a different part of the propery value would be active... this gets rid of that path
148  static_cast<PropertyFrame*>(propertyFrame_)->setTopLevelDataStore(QSharedPointer<DataStore>());
149 }

416 void TestViewer::playTriggered()
417 {
418  if(!testController_->isRunning())
419  {
420  //Find out which analyses should be enabled and enable them
421  QList<QUuid> analysisIds = analysisSelector_->getSelectedAnalysisIds();
422  QSharedPointer<DesignConfig> designConfig = experiment_->getDesignConfig();
423  //If there are active analysis ids, add the various data readers (frameReader, etc) to the
424  //designConfig. Then add them to the Picto Engine so that it can update them with data.
425  //Generally, the SlaveExperimentDriver is responsible for creating the readers and the StateUpdater
426  //is responsible for making sure they have data. Eventually, the Test Viewer should contain a
427  //running experiment and an interface to a SlaveExperimentViewer that handles the playback and analysis.
428  //Until that happens, we create the DataReaders here and update them in the PictoEngine.
429  if(analysisIds.size())
430  {
431  if(!liveFrameReader_)
432  {
433  //Setup Test Data Readers for Analysis
434  liveFrameReader_ = QSharedPointer<LiveFrameReader>(new LiveFrameReader());
435  liveLfpReader_ = QSharedPointer<LiveLfpReader>(new LiveLfpReader(QList<int>() << 1 << 3 << 4 << 5 << 8,0.001));
436  liveRewardReader_ = QSharedPointer<LiveRewardReader>(new LiveRewardReader());
437  liveRunNotesReader_ = QSharedPointer<LiveRunNotesReader>(new LiveRunNotesReader());
438  liveSpikeReader_ = QSharedPointer<LiveSpikeReader>(new LiveSpikeReader(5,2,32));
439  signalReaders_.append(QSharedPointer<LiveSignalReader>(new LiveSignalReader("Position",QStringList()<<"x"<<"y",0.002)));
440  signalReaders_.append(QSharedPointer<LiveSignalReader>(new LiveSignalReader("Diameter", QStringList() << "x" << "y", 0.004)));
441  signalReaders_.append(QSharedPointer<LiveSignalReader>(new LiveSignalReader("GenericInputs", QStringList() << "input0" << "input1" << "input2" << "input3" << "input4" << "input5" << "input6" << "input7", 0.004)));
442  }
443  designConfig->setFrameReader(liveFrameReader_.staticCast<FrameReader>());
444  designConfig->setLfpReader(liveLfpReader_.staticCast<LfpReader>());
445  designConfig->setRewardReader(liveRewardReader_.staticCast<RewardReader>());
446  designConfig->setRunNotesReader(liveRunNotesReader_.staticCast<RunNotesReader>());
447  designConfig->setSpikeReader(liveSpikeReader_.staticCast<SpikeReader>());
448  foreach(QSharedPointer<LiveSignalReader> signalReader,signalReaders_)
449  {
450  designConfig->setSignalReader(signalReader->getName(),signalReader.staticCast<SignalReader>());
451  }
452  engine_->setFrameReader(liveFrameReader_);
453  engine_->setLfpReader(liveLfpReader_);
454  engine_->setRewardReader(liveRewardReader_);
455  engine_->setSpikeReader(liveSpikeReader_);
456  connect(liveSpikeReader_.data(), SIGNAL(spikeEvent(double, int, int, QVariantList)), this, SLOT(spikeEvent(double, int, int, QVariantList)));
457  foreach(QSharedPointer<LiveSignalReader> signalReader,signalReaders_)
458  {
459  engine_->setSignalReader(signalReader->getName(),signalReader);
460  }
461  }
462  else
463  {
464  //If there's no analysis. Set frame readers to null so that we don't waste memory on them.
465  //in case they were set last time.
466  designConfig->setFrameReader(QSharedPointer<FrameReader>());
467  designConfig->setLfpReader(QSharedPointer<LfpReader>());
468  designConfig->setRewardReader(QSharedPointer<RewardReader>());
469  designConfig->setSpikeReader(QSharedPointer<SpikeReader>());
470  designConfig->clearSignalReaders();
471  engine_->setFrameReader(QSharedPointer<LiveFrameReader>());
472  engine_->setLfpReader(QSharedPointer<LiveLfpReader>());
473  engine_->setRewardReader(QSharedPointer<LiveRewardReader>());
474  engine_->setSpikeReader(QSharedPointer<LiveSpikeReader>());
475  engine_->clearSignalReaders();
476  }
477 
478 
479  designConfig->setActiveAnalysisIds(analysisIds);
480  }
481 
482  analysisSelector_->setEnabled(false);
483  //QMetaObject::invokeMethod(testController_.data(),"play");
484 }

641 void TestViewer::updateNeuralUI()
642 {
643  noCallBack_ = true;
644  int currCh = channelBox_->itemData(channelBox_->currentIndex()).toInt();
645  int currUnit = unitBox_->itemData(unitBox_->currentIndex()).toInt();
646  channelBox_->clear();
647  unitBox_->clear();
648 
649  channelBox_->addItem("All Channels", 0);
650  foreach(int channel, ChannelsUnits_.keys())
651  {
652  channelBox_->addItem(QString("Channel %1").arg(channel), channel);
653  }
654  //Reset channel to what it was before
655  for (int i = 0; i<channelBox_->count(); i++)
656  {
657  if (channelBox_->itemData(i).toInt() == currCh)
658  {
659  channelBox_->setCurrentIndex(i);
660  break;
661  }
662  }
663 
664  QList<int> unitList = ChannelsUnits_.value(selectedChannel_);
665  unitBox_->addItem("All Units", 0);
666  foreach(int unit, unitList)
667  {
668  unitBox_->addItem(QString("Unit %1").arg(unit), unit);
669  }
