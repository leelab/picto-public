36 bool PlaybackStateUpdater::updateState()
37 {
38  if(!designRoot_)
39  return false;
40  double runToTime = timerOffset_ + getPlaybackOffsetTime();
41  if( playbackSuspended() && (runToTime != sessionPlayer_->getTime()) )
42  {
43  //We are stepping the session player even though we're not playing back.
44  //Rendering should be disabled.
45  enableRendering(false);
46  }
47  if(sessionPlayer_->stepToTime(runToTime))
48  {
49  //Once we have caught up to runToTime rendering should always be enabled.
50  enableRendering(true);
51 
52  if(!paused_ && playbackSuspended())
53  {
54  resumePlayback();
55  }
56  }
57  return true;
58 }

181 bool PlaybackStateUpdater::loadRun(int index)
182 {
183  if(!fileSessionLoader_)
184  return false;
185  if(index < 0)
186  return false;
187  if(index >= fileSessionLoader_->getRunNames().length())
188  return false;
189  paused_ = false;
190  timerOffset_ = 0.0;
191  suspendPlayback();
192  runLoaded_ = false;
193  currRunIndex_ = index;
194  currRunLength_ = fileSessionLoader_->runDuration(currRunIndex_);
195  //Make sure the lfp enabling is set up before loading the run
196  if(sessionState_->lfpEnabled() != enableLfp_)
197  sessionState_->enableLfp(enableLfp_);
198  //Load the run
199  return fileSessionLoader_->loadRun(index);
