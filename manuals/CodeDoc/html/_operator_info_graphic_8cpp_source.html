166 void OperatorInfoGraphic::propValueChanged(Property*,QVariant)

169  if(settingPropsFromMap_)

175  QStringList values = propertyContainer_->getPropertyValue("Values").toString().split(",");
176  orderedFields_.clear();

178  QVariant value;

180  {
181  if(i >= values.length())
182  value.clear();
183  else
184  value = values[i];
