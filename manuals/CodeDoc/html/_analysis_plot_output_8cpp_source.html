77 void AnalysisPlotOutput::setupFile()
78 {
79  if (valid_)
80  return;
81 
82  QString outputDir = getTempOutputDir();
83  if (!QFile::exists(outputDir + "/" + getRunName()))
84  { //The directory doesn't exist yet. Make it.
85  QDir dir;
86  dir.mkpath(outputDir + "/" + getRunName());
87  }
88 
89  QString fileName = operatorPlot_->getTitle();
90  fileName.replace(' ', '_');
91 
92  switch (getExportType())
93  {
94  case ExportType::EXPORT_PDF:
95  fileName += ".pdf";
96  break;
97  case ExportType::EXPORT_PNG:
98  fileName += ".png";
99  break;
100  case ExportType::EXPORT_BMP:
101  fileName += ".bmp";
102  break;
103  case ExportType::EXPORT_POSTSCRIPT:
104  fileName += ".ps";
105  break;
106  default:
107  qDebug() << "Unexpected Export Type";
108  return;
109  }
110 
111  fileName_ = outputDir + "/" + getRunName() + "/" + getRunName() + "." + fileName;
112 
113  if (filenameMap_.contains(fileName_))
114  {
115  int version = filenameMap_[fileName_]++;
116 
117  fileName_ = outputDir + "/" + getRunName() + "/" + getRunName() + "." + QString("%1").arg(version).rightJustified(2,'0') + "." + fileName;
118  }
119  else
120  {
121  filenameMap_[fileName_] = 1;
122  }
123 
124  valid_ = true;
125 }

171  QString targetName = propertyContainer_->getPropertyValue("OperatorPlot").toString();
172  foreach(QString childTag, analysis->getValidChildTags())

179  {
