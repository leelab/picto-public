60 void CurrentSessionData::writeData(int, QVariantList data)
61 {
62  Q_ASSERT(data.size() == 4);
63  int dataId = data[0].toInt();
64  int variableType = data[1].toInt();
65  int dataBin = variableType;
66  double timestamp = data[2].toDouble();
67  int oldDataId = 0;
68  double oldTimestamp = 0;
69  bool bAppend = false;
70 
71  //For StateTransition Data, we use a Ringbuffer of size MAX_RETAIN_TRANSITION_INFO, but we need to append the data
72  // when we submit data of the same timestamp, or the data will be reordered across the boundary.
73  if (variableType == TRANSITION_STATE_VAR_ID)
74  {
75  dataBin = variableType + lastOffset_[variableType];
76  if (dataMap_.contains(dataBin) && dataMap_[dataBin][2].toDouble() == timestamp)
77  {
78  bAppend = true;
79  }
80  else
81  {
82  lastOffset_[variableType] = (lastOffset_[variableType] + 1) % MAX_RETAIN_TRANSITION_INFO;
83  dataBin = variableType + lastOffset_[variableType];
84  }
85  }
86  else if (variableType >= 0) //Asset Property
87  {
88  //If variableType >= 0, then it's an Asset Property, so we need to use the MAX_RETAIN_PROPERTY_DATA
89  dataBin = variableType + lastOffset_[variableType];
90  if (dataMap_.contains(dataBin) && dataMap_[dataBin][2].toDouble() == timestamp)
91  {
92  bAppend = true;
93  }
94  else
95  {
96  lastOffset_[variableType] = (lastOffset_[variableType] + 1) % MAX_RETAIN_PROPERTY_DATA;
97  dataBin = variableType + lastOffset_[variableType];
98  }
99  }
100 
101  if (dataMap_.contains(dataBin))
102  {
103  oldDataId = dataMap_[dataBin][0].toInt();
104  oldTimestamp = dataMap_[dataBin][2].toDouble();
105  }
106 
107  if (oldDataId > dataId && !bAppend)
108  {
109  return;
110  }
111 
112  if(oldTimestamp > timestamp)
113  return;
114 
115  //If this has the same timestamp as the last chunk, we need to append the data to maintain the same ordering.
116  // If we don't, then the data will be reordered across the ring buffer's boundary
117  if (bAppend)
118  {
119  dataMap_[dataBin][3] = dataMap_[dataBin][3].toString() + data[3].toString();
120  }
121  else
122  {
123  dataMap_[dataBin] = data;
124  }
125 
126  if (!bAppend && dataByTime_.contains(oldTimestamp) && dataByTime_[oldTimestamp].contains(dataBin))
127  {
128  dataByTime_[oldTimestamp].remove(dataBin);
129  if(dataByTime_[oldTimestamp].isEmpty())
130  dataByTime_.remove(oldTimestamp);
131  }
132  dataByTime_[timestamp][dataBin] = variableType;
133 }
