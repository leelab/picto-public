209 bool Property::deserializeFromXml(QSharedPointer<QXmlStreamReader> xmlStreamReader)
210 {
211  Q_ASSERT_X(!runMode_,"Property::deserializeFromXml","deserializeFromXml should not be called when property is in run mode.");
212  serialSyntaxUpgraded_ = false;
213  //Create XMLStreamWriter to store this DataStore's tag text in a string
214  tagText_ = "";
215  QSharedPointer<QXmlStreamWriter> xmlWriter(new QXmlStreamWriter(&tagText_));
216  xmlWriter->writeCurrentToken(*xmlStreamReader);// Lets write the start tag to our tagText.
217 
218  //Get Start tag name (depending on who the parent is, they may have given us different names.
219  tagName_ = xmlStreamReader->name().toString();
220  bool emptyTag = true;
221  bool hadName = false;
222  if(xmlStreamReader->attributes().hasAttribute("name"))
223  {
224  setName(xmlStreamReader->attributes().value("name").toString());
225  hadName = true;
226  }
227  else
228  setName(tagName_);
229 
230  //In cases where a Asset Factory used a type attribute to choose between types, a type that we don't use but need to
231  // write out would be in the tag.
232  if(xmlStreamReader->attributes().hasAttribute("type"))
233  {
234  typeVal_ = xmlStreamReader->attributes().value("type").toString();
235  }
236  else
237  typeVal_ = "";
238 
239  //QXmlStreamAttributes attributeList = xmlStreamReader->attributes();
240  //foreach(QXmlStreamAttribute attribute,attributeList)
241  //{
242  // if(serializationAttributes_.contains(attribute.name().toString()))
243  // {
244  // serializationAttributes_[attribute.name().toString()] = attribute.value().toString();
245  // emptyTag = false;
246  // }
247  // else if((attribute.name() != "name") && (attribute.name() != "type") )
248  // {
249  // QString allowedAttributes = "name\ntype";
250  // foreach(QString attrName,serializationAttributes_.keys())
251  // {
252  // allowedAttributes.append("\n").append(attrName);
253  // }
254  // addError(getName().toLatin1(), QString("Invalid attribute: \n%1\nThe following attributes are defined for this property:\n%2\n-----").arg(attribute.name().toString()).arg(allowedAttributes).toLatin1(), xmlStreamReader);
255  // return false;
256  // }
257  //}
258  //Get the tag's attributes. Put the attributes that are not expected (ie. obsolete) in an unexpectedAttributeMap
259  QMap<QString,QVariant> unexpectedAttributeMap;
260  QXmlStreamAttributes attributeList = xmlStreamReader->attributes();
261  foreach(QXmlStreamAttribute attribute,attributeList)
262  {
263  if((attribute.name() != "name") && (attribute.name() != "type") )
264  {
265  emptyTag = false;
266  if(serializationAttributes_.contains(attribute.name().toString()))
267  {
268  serializationAttributes_[attribute.name().toString()] = attribute.value().toString();
269  continue;
270  }
271  //If we got here, there are unexpected attributes, tell the system that we encountered obsolete syntax.
272  hadObsoleteSerialSyntax_ = true;
273  serialSyntaxUpgraded_ = true;
274  emit edited();
275  unexpectedAttributeMap[attribute.name().toString()] = attribute.value().toString();
276  }
277  }
278 
279  //Loop until we're done with the tag or we reach the end of the XMLStream
280  if(xmlStreamReader->readNext() == QXmlStreamReader::Invalid)
281  {
282  addError("XML syntax has been violated.");
283  return false;
284  }
285  xmlWriter->writeCurrentToken(*xmlStreamReader);// Write everything left to tagText.
286  if(!xmlStreamReader->isCharacters() && !xmlStreamReader->isEndElement())
287  {
288  addError(QString("Unexpected value read.").toLatin1());
289  return false;
290  }
291  QString value = xmlStreamReader->text().toString();
292 
293 // QString value = xmlStreamReader->readElementText();
294  if(value != "")
295  emptyTag = false;
296 
297  // If its an empty tag, we should just go with the default value.
298  if(!emptyTag)
299  {
300  QString error = "";
301  QVariant result = stringToVariant(value,error);
302  if(!error.isEmpty())
303  { //Maybe this is an older version of a property that stored its data in attributes
304  //Try to get a variant value from the attributes
305  QString attributeError = "";
306  result = attributeMapToVariantValue(unexpectedAttributeMap, attributeError);
307  //If we couldn't get a value, report an error
308  if(!attributeError.isEmpty())
309  error = error + "\n" + attributeError; //Add error details
310  else
311  {
312  error = ""; //No more error
313  }
314  }
315  if(error.size())
316  {
317  addError(error.toLatin1());
318  return false;
319  }
320 
321  //Check if our serialization syntax will be upgraded on the next save out.
322  QString reserialized = variantToString(result);
323  if(value != reserialized)
324  {
325  //If we got here, syntax will be changed on the next save, tell the system that we encountered obsolete syntax.
326  hadObsoleteSerialSyntax_ = true;
327  serialSyntaxUpgraded_ = true;
328  emit edited();
329  }
330  setValue(result);
331  }
332 
333  //Loop until we're done with the tag or we reach the end of the XMLStream
334  while(!(xmlStreamReader->isEndElement() && (xmlStreamReader->name().toString() == tagName_)) && !xmlStreamReader->atEnd())
335  {
336  xmlStreamReader->readNext();
337  xmlWriter->writeCurrentToken(*xmlStreamReader);// Write everything left to tagText.
338  }
339  //Make sure we didn't finish the document.
340  if(xmlStreamReader->atEnd())
341  {
342  addError("Unexpected end of document");
343  return false;
344  }
345  return true;
