80 QString State::run(QSharedPointer<Engine::PictoEngine> engine)
81 {
82  resetScriptableValues();
83 
84  //Set up frame script.
85  bool runFrameScript = !propertyContainer_->getPropertyValue("FrameScript").toString().isEmpty();
86  QString frameScriptName = getName().simplified().remove(' ')+"Frame";
87 
88  //run the entry script
89  runEntryScript();
90 
91  //Start up all of the control elements
92  foreach(QSharedPointer<ResultContainer> control, elements_)
93  {
94  control.staticCast<ControlElement>()->start(engine);
95  }
96 
97  QString result = "";
98  QSharedPointer<Transition> resultTrans;
99  bool isDone = false;
100 
101  //This is the "rendering loop" It gets run for every frame
102  while(!isDone)
103  {
104  //--------- Check control elements------------
105  foreach(QSharedPointer<ResultContainer> control, elements_)
106  {
107  if(control.staticCast<ControlElement>()->isDone(engine))
108  {
109  //Get the result and call its entry script
110  QSharedPointer<Result> controlResult = control.staticCast<ControlElement>()->getResult();
111  control->setLatestResult(controlResult->getName());
112  controlResult->runResultScript();
113  QList<QSharedPointer<Transition>> elemLinks = transitions_.values(control->getName());
114  foreach(QSharedPointer<Transition> link,elemLinks)
115  {
116  if(link->getSourceResultAsset() == controlResult)
117  {
118  result = link->getDestination();
119  resultTrans = link;
120  break;
121  }
122  }
123  Q_ASSERT(resultTrans);
124  isDone = true;
125  }
126  }
127  if(elements_.isEmpty())
128  isDone = true;
129 
130  if(!isDone)
131  {
132  //-------------- If the state is not over, run the frame script ----------------
133  //
134  if(runFrameScript)
135  runScript(frameScriptName);
136 
137  //---------- Draw the scene. This will automatically send behavioral data to server --------------
138  scene_->render(engine,getAssetId());
139 
140  //---------- Run the analysis frame scripts ---------------------------------------
141  runAnalysisFrameScripts();
142  }
143 
144  //------ Check for engine stop commands ---------------
145  if(checkForEngineStop(engine))
146  {
147  isDone = true;
148  setLatestResult("EngineAbort");
149  result = "EngineAbort";
150  }
151  else if(isDone)
152  {
153  engine->addStateTransitionForServer(resultTrans); //Added in Picto Version 1.0.12. Before this transitions within a state weren't being recorded in the session file.
154  Q_ASSERT(results_.contains(result)); //Added in Picto Version 1.0.12.
155  setLatestResult(result);
156  //Run result script if there is one.
157  results_.value(result)->runResultScript();
158  }
159  //If we're not in exclusive mode, we should allocate time to process events
160  //This would occur if we were running a state machine somewhere other than
161  //Director (e.g. debugging it in Workstation)
162  if(!engine->getExclusiveMode())
163  {
164  QCoreApplication::processEvents();
165  }
166  }
167 
168  //Stop all of the control elements
169  foreach(QSharedPointer<ResultContainer> control, elements_)
170  {
171  control.staticCast<ControlElement>()->stop(engine);
172  }
173 
174  //run the exit script
175  runExitScript();
176 
177  return result;
178 }

253 void State::upgradeVersion(QString deserializedVersion)
254 {
255  MachineContainer::upgradeVersion(deserializedVersion);
256 
257  if(deserializedVersion < "0.0.1")
258  {
259  //Before 0.0.1, newClick on operatorClick info had the effect of reading out the current
260  //"clicked" value, and also resetting it. In 0.0.1, we changed it such that the user
261  //must reset it manually. Since there aren't too many experiments yet, we know how and
262  //where this script was used and we are changing operatorClick.newClick to
263  //(operatorClick.newClick && !(operatorClick.newClick = false)) in all places that it was
264  //used.
265  QString frameScript = propertyContainer_->getPropertyValue("FrameScript").toString();
266  frameScript.replace("operatorClick.newClick","(operatorClick.newClick && !(operatorClick.newClick = false))");
267  propertyContainer_->setPropertyValue("FrameScript",frameScript);
268  }

340 void State::rebuildScene()
341 {
342  //Since the scene needs access to visual elements stored above it in the tree, we get
343  //our output elements from the output element list.
344  scene_ = Scene::createScene();
345  hasCursor_ = false;
346  QList<QSharedPointer<OutputElement>> outputs = getOutputElementList();
347  QList<QUuid> activeAnalyses = getDesignConfig()->getActiveAnalysisIds();
348  QHash<QUuid,bool> activeAnalysisHash;
349  foreach(QUuid analysisId,activeAnalyses)
350  {
351  activeAnalysisHash[analysisId] = true;
352  }
353  foreach(QSharedPointer<OutputElement> output,outputs)
354  {
355  if(output.isNull())
356  continue;
357  if(output.dynamicCast<VisualElement>())
358  {
359  //If this is an analysis element and its analysis id isn't active, don't add it to the scene.
360  AssociateElement* analysisVisElem = dynamic_cast<AssociateElement*>(output.data());
361  if(analysisVisElem && !activeAnalysisHash.contains(analysisVisElem->getAssociateId()))
362  continue;
363  scene_->addVisualElement(output.staticCast<VisualElement>());
364  }
365  else if (output.dynamicCast<OutputSignal>())
366  {
367  scene_->addOutputSignal(output.staticCast<OutputSignal>());
368  }
369  else if(output.dynamicCast<AudioElement>())
370  {
371  scene_->addAudioElement(output.staticCast<AudioElement>());
372  }
373  else if(output.dynamicCast<InputSignal>())
374  {
375  scene_->addInputSignal(output.staticCast<InputSignal>());
376  }
377  }
378  QColor backgroundColor;
379  backgroundColor.setNamedColor(propertyContainer_->getPropertyValue("BackgroundColor").toString());
380  scene_->setBackgroundColor(QColor(propertyContainer_->getPropertyValue("BackgroundColor").toString()));
381 }
