104 void VisualElement::upgradeVersion(QString deserializedVersion)
105 {
106  OutputElement::upgradeVersion(deserializedVersion);
107  if(deserializedVersion < "0.0.1")
108  {
109  //As of version 0.0.1, we have removed the Dimensions properties from all VisualElements
110  //and replaced them with Size properties. We take advantage of the fact that the x,y
111  //properties of the Dimensions rectangle were never used so we can just throw them out.
112 
113  //Get the old obsolete Dimensions Rect property
114  QList<QSharedPointer<Asset>> oldRectProps = getGeneratedChildren("Dimensions");
115  //Get the new Size property
116  QList<QSharedPointer<Asset>> newSizeProps = getGeneratedChildren("Size");
117  if(oldRectProps.size() && newSizeProps.size())
118  {
119  Q_ASSERT((oldRectProps.size() == 1) && (newSizeProps.size() == 1));
120  QSharedPointer<ObsoleteAsset> oldRectProp = oldRectProps.first().staticCast<ObsoleteAsset>();
121  QSharedPointer<Property> newSizeProp = newSizeProps.first().staticCast<Property>();
122  //Try getting the old rect from the two methods of serializing it that were used, attributes and string
123  QStringList rectStringVals;
124  if(oldRectProp->getAttributeValue("x").isEmpty())
125  {
126  rectStringVals = oldRectProp->getValue().split(",");
127  }
128  else
129  {
130  rectStringVals << oldRectProp->getAttributeValue("x")
131  << oldRectProp->getAttributeValue("y")
132  << oldRectProp->getAttributeValue("width")
133  << oldRectProp->getAttributeValue("height");
134  }
135  Q_ASSERT(rectStringVals.size() == 4);
136  QRect oldRect( rectStringVals[0].toInt(),
137  rectStringVals[1].toInt(),
138  rectStringVals[2].toInt(),
139  rectStringVals[3].toInt());
140  newSizeProp->setValue(QSize(oldRect.width(),oldRect.height()));
141  }
142  }
143 }
