147  }
148

150 
151  //Find the behavTime_ - preflagwindow(converted to seconds)
152  double retainMin = behavTime_ - 0.001 * getPreFlagWindow();
153 
154  QLinkedList<double>::iterator it;
155  for (it = spikeData_[data.getChannel()][data.getUnit()].d.begin();
156  it != spikeData_[data.getChannel()][data.getUnit()].d.end();
157  it++)
158  {
159  if (*it >= retainMin)
160  {
161  break;
162  }
163  }
164  spikeData_[data.getChannel()][data.getUnit()].d.erase(spikeData_[data.getChannel()][data.getUnit()].d.begin(), it);

167  {
