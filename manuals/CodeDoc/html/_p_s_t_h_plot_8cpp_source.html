120 void PSTHPlot::addSpikeData(const NeuralDataUnit &data)
121 {
122  double behavTime = 0.0;
123  if (getDesignConfig()->getFrameReader())
124  {
125  behavTime = getDesignConfig()->getFrameReader()->getLatestTime();
126  }
127 
128  if (behavTime <= 0.0)
129  {
130  return;
131  }
132 
133  //If there aren't enough curves slots yet, make the slots
134  while (spikeData_.size() <= data.getChannel())
135  {
136  spikeData_.push_back(QVector<DataList>());
137  }
138  while (spikeData_[data.getChannel()].size() <= data.getUnit())
139  {
140  spikeData_[data.getChannel()].push_back(DataList());
141  }
142  if (!spikeData_[data.getChannel()][data.getUnit()].exists)
143  {
144  spikeData_[data.getChannel()][data.getUnit()].exists = true;
145  }
146 
147  spikeData_[data.getChannel()][data.getUnit()].d.push_back(data.getFittedtime());
148 
149  //Find the behavTime_ - preflagwindow(converted to seconds)
150  double retainMin = behavTime - 0.001 * getPreFlagWindow();
151 
152  QLinkedList<double>::iterator it;
153  for (it = spikeData_[data.getChannel()][data.getUnit()].d.begin();
154  it != spikeData_[data.getChannel()][data.getUnit()].d.end();
155  it++)
156  {
157  if (*it >= retainMin)
158  {
159  break;
160  }
161  }
162  spikeData_[data.getChannel()][data.getUnit()].d.erase(spikeData_[data.getChannel()][data.getUnit()].d.begin(), it);
163 
164  for (QList<ActiveScan>::iterator i = activeScans_.begin(); i != activeScans_.end();)
165  {
166  if (i->scanEndTime_ < data.getFittedtime())
167  {
168  submitScan(*i);
169 
170  i = activeScans_.erase(i);
171  }
172  else
173  {
174  i->addSpike(data.getChannel(), data.getUnit(), data.getFittedtime());
175  ++i;
176  }
177  }
178 }
