372 bool DirectorInterface::sendCommandGetResponse(Picto::ProtocolCommand command,QString* reply)
373 {
374  Q_ASSERT(reply);
375  QTcpSocket *commSock = panelInfo->getCommandSocket();
376  if(!commSock)
377  return false;
378  if(command.getContent().size())
379  command.setFieldValue("Content-Length", QString("%1").arg(command.getContent().size()));
380  QString commandId = QString::number(++lastCommandId_);
381  command.setFieldValue("Command-ID",commandId);
382  command.write(commSock);
383  //qDebug("SentID: " + commandId.toLatin1());
384  Picto::ProtocolResponse response;
385  int r = response.read(commSock,RESPONSEDELAYMS);
386  //If we got a reponse (r>=0) but it wasn't for our command, throw it out and read a new one.
387  while((r >= 0) && (response.getFieldValue("Command-ID") != commandId))
388  {
389  //qDebug("ReceivedID: " + response.getFieldValue("Command-ID").toLatin1());
390  r = response.read(commSock,RESPONSEDELAYMS);
391  }
392  (*reply) = "";
393  if(r < 0) //A response to our command didn't arrive in time.
394  return false;
395  //qDebug("ReceivedID: " + response.getFieldValue("Command-ID").toLatin1());
396 
397  (*reply) = QString(response.getDecodedContent());
398  if(response.getResponseCode() != 200)
399  return false;
400  return true;
401 }
