319  stateTrigger_ = Connected;
320  break;

322  activeExpName_->setText(myDesignRoot_->getDesignName()); //In case operator loads new file
323 
324  if (awaitingRejoin_ == RejoinStatus::AwaitingRejoin)
325  {
326  stateTrigger_ = JoinSessionRequest;
327  }
328  else if (connectAction_->isChecked())
329  {
330  if(remoteStatus == Idle)
331  stateTrigger_ = StartSessionRequest;
332  if(remoteStatus > Ending)
333  stateTrigger_ = JoinSessionRequest;
334  }
335  break;
336  case JoiningSession:
337  activeExpName_->setText(myDesignRoot_->getDesignName()); //In case operator loads new file
338  if(joinSession())
339  {
340  if (remoteStatus == Running)
341  {
342  if (slaveExpDriver_) slaveExpDriver_->pause(false);
343  stateTrigger_ = SessionRunning;
344  }
345  else if (remoteStatus == Paused)
346  {
347  if (slaveExpDriver_) slaveExpDriver_->pause(true);
348  stateTrigger_ = SessionPaused;
349  }
350  else
351  {
352  stateTrigger_ = SessionStopped;
353  }
354  }
355 
356  awaitingRejoin_ = RejoinStatus::None;
357 
358  if(remoteStatus < Stopped)
359  {
360  disjoinSession();
361  stateTrigger_ = DisjoinSessionRequest;
362  }
363  break;
364  case CreatingSession:
365  activeExpName_->setText(myDesignRoot_->getDesignName()); //In case operator loads new file
366  if(startSession())
367  stateTrigger_ = SessionStarted;
368  else
369  stateTrigger_ = SessionDidntStart;
370  break;
371  case StoppedSession:
372  if(zoomChanged_)
373  {
374  zoomChanged_ = false;
375  pixmapVisualTarget_->setZoom(zoomValue_);
376  renderingTarget_->showSplash();
377  }
378  if (remoteStatus == Paused)
379  {
380  if (slaveExpDriver_) slaveExpDriver_->pause(true);
381  stateTrigger_ = SessionPaused;
382  }
383  if (remoteStatus == Running)
384  {
385  if (slaveExpDriver_) slaveExpDriver_->pause(false);
386  stateTrigger_ = SessionRunning;
387  }
388  if(!connectAction_->isChecked())
389  {
390  if(shouldEndSession())
391  stateTrigger_ = EndSessionRequest;
392  else
393  {
394  disjoinSession();
395  stateTrigger_ = DisjoinSessionRequest;
396  }
397  }
398  if (awaitingRejoin_ == RejoinStatus::AwaitingDisconnect)
399  {
400  awaitingRejoin_ = RejoinStatus::AwaitingRejoin;
401  disjoinSession();
402  stateTrigger_ = RejoinSessionRequest;
403  }
404  if((remoteStatus < Stopped) && (remoteStatus > NotFound))
405  {
406  disjoinSession();
407  stateTrigger_ = DisjoinSessionRequest;
408  }
409  updateSessionDataPackage();
410  if(proxyListBox_->currentIndex() > 0)
411  {
412  updateNeuralData();
413  neuralDataViewer_->replot();
414  }
415  break;
416  case PausedSession:
417  if(zoomChanged_)
418  {
419  zoomChanged_ = false;
420  Scene::setZoom(zoomValue_);
421  }
422  if (remoteStatus == Running)
423  {
424  if (slaveExpDriver_) slaveExpDriver_->pause(false);
425  stateTrigger_ = SessionRunning;
426  }
427  //If the engine isn't running for some reason, or the director stopped, go to the stopped state. In the case
428  // where the director is still running but our engine stopped, the stop state will just restart the engine and
429  // send us back here.
430  if (remoteStatus == Stopped)
431  {
432  stateTrigger_ = SessionStopped;
433  }
434  if(!connectAction_->isChecked() || ((remoteStatus < Stopped) && (remoteStatus > NotFound)))
435  {
436  disjoinSession();
437  stateTrigger_ = DisjoinSessionRequest;
438  }
439  if (awaitingRejoin_ == RejoinStatus::AwaitingDisconnect)
440  {
441  awaitingRejoin_ = RejoinStatus::AwaitingRejoin;
442  disjoinSession();
443  stateTrigger_ = RejoinSessionRequest;
444  }
445  if(stateTrigger_ != NoViewerTrigger)
446  stopExperiment();
447  updateSessionDataPackage();
448  if(proxyListBox_->currentIndex() > 0)
449  {
450  updateNeuralData();
451  neuralDataViewer_->replot();
452  }
453  break;
454  case RunningSession:
455  if(zoomChanged_)
456  {
457  zoomChanged_ = false;
458  Scene::setZoom(zoomValue_);
459  }
460  if (remoteStatus == Paused)
461  {
462  if (slaveExpDriver_) slaveExpDriver_->pause(true);
463  stateTrigger_ = SessionPaused;
464  }
465  //If the engine isn't running for some reason, or the director stopped, go to the stopped state. In the case
466  // where the director is still running but our engine stopped, the stop state will just restart the engine and
467  // send us back here.
468  if(remoteStatus == Stopped)
469  stateTrigger_ = SessionStopped;
470  if(!connectAction_->isChecked() || ((remoteStatus < Stopped) && (remoteStatus > NotFound)))
471  {
472  disjoinSession();
473  stateTrigger_ = DisjoinSessionRequest;
474  }
475  if (awaitingRejoin_ == RejoinStatus::AwaitingDisconnect)
476  {
477  awaitingRejoin_ = RejoinStatus::AwaitingRejoin;
478  disjoinSession();
479  stateTrigger_ = RejoinSessionRequest;
480  }
481  if(stateTrigger_ != NoViewerTrigger)
482  stopExperiment();
483  updateSessionDataPackage();
484  if(proxyListBox_->currentIndex() > 0)
485  {
486  updateNeuralData();
487  neuralDataViewer_->replot();
488  }
489  break;
490  case EndingSession:
491  if(endSession() || remoteStatus < Stopped)
492  disjoinSession();
493  stateTrigger_ = SessionEnded;
494  break;
495  }
496  if(!assureChannelConnections())
497  {
498  stateTrigger_ = Disconnected;
499  }

884 {
885  //Its not so pretty to have to work outside the state machine like this, but if we're about to quit, we don't know

2064 QSharedPointer<Picto::ProtocolResponse> RemoteViewer::sendCommandGetReply(QSharedPointer<Picto::ProtocolCommand> cmd, int nonDefaultTimeoutMs)
2065 {
2066  QTime timer;
2067  timer.start();
2068  serverChannel_->sendRegisteredCommand(cmd);
2069  //qDebug("Sent " + cmd->getMethod().toLatin1() + " Command.");
2070  bool useAdaptiveTimeout = (nonDefaultTimeoutMs == -1);
2071  QString method = cmd->getMethod();
2072  int* responseDelay = &nonDefaultTimeoutMs;
2073  if(useAdaptiveTimeout)
2074  {
2075  if(!adaptiveResponseDelays_.contains(method))
2076  adaptiveResponseDelays_[method] = MIN_ADAPTIVE_TIMEOUT;
2077  responseDelay = &adaptiveResponseDelays_[method];
2078  }
2079  QString commandID = cmd->getFieldValue("Command-ID");
2080  QSharedPointer<Picto::ProtocolResponse> cmdResponse;
2081  pendingResponses_[commandID] = cmdResponse;
2082  //Get the response to this command
2083  do
2084  {
2085  //Update the delay according to the time since sending the command
2086  int delay = (*responseDelay)-timer.elapsed();
2087  if(delay < 0)
2088  delay = 0;
2089  //Wait up to the maximum delay for a response
2090  serverChannel_->waitForResponse(delay);
2091 
2092  //Insert all responses received into pendingResponses_ hash.
2093  while(serverChannel_->incomingResponsesWaiting())
2094  {
2095  QSharedPointer<Picto::ProtocolResponse> response = serverChannel_->getResponse();
2096  if(pendingResponses_.contains(response->getFieldValue("Command-ID")))
2097  {
2098  pendingResponses_[response->getFieldValue("Command-ID")] = response;
2099  }
2100  }
2101 
2102  //Check if our reponse is in the pending responses hash.
2103  if(pendingResponses_.contains(commandID))
2104  { //We got our response. Break out of the loop
2105  break;
2106  }
2107  //Keep looping until the responseDelay time has passed or we break out (after getting our response), whichever
2108  // comes first.
2109  }while(timer.elapsed() < (*responseDelay));
2110 
2111  //Get our response from the pendingResponses_ hash and remove it.
2112  cmdResponse = pendingResponses_[commandID];
2113  pendingResponses_.remove(commandID);
2114 
2115  //If there was no response. Print a status message indicating this.
2116  if(cmdResponse.isNull())
2117  {
2118  QString statusMessage = method + " command timed out.";
2119  if(useAdaptiveTimeout)
2120  {
2121  int lastAdaptDelay = *responseDelay;
2122  (*responseDelay) *= 2;
2123  if((*responseDelay) > MAX_ADAPTIVE_TIMEOUT)
2124  (*responseDelay) = MAX_ADAPTIVE_TIMEOUT;
2125  if(lastAdaptDelay != (*responseDelay))
2126  statusMessage += " Compensating... Timeout: " + QString::number((*responseDelay)) + " ms.";
2127  //qDebug(cmd->getMethod().toLatin1() + " command timed out.");
2128  }
2129  if(currState_ != WaitForConnect)
2130  { //If we're not even connected, there will obviously be timeouts here. No need to keep on whining about them.
2131  setStatus(statusMessage,true);
2132  }
2133  //Return an empty reply, since we didn't get the one we wanted.
2134  return QSharedPointer<Picto::ProtocolResponse>();
2135  }
2136 
2137  //If we're here, we got a response. Update the adaptive timeout for this command type according to the time it took
2138  // to receive the reply.
2139  //qDebug(cmd->getMethod().toLatin1() + " command reply received.");
2140  if(useAdaptiveTimeout)
2141  {
2142  (*responseDelay) = 2*timer.elapsed();
2143  if((*responseDelay)<MIN_ADAPTIVE_TIMEOUT)
2144  (*responseDelay) = MIN_ADAPTIVE_TIMEOUT;
2145  if((*responseDelay) > MAX_ADAPTIVE_TIMEOUT)
2146  (*responseDelay) = MAX_ADAPTIVE_TIMEOUT;
2147  }
2148 
2149  //Return the response.
2150  return cmdResponse;
2151 }
