96 QSharedPointer<Property> PropertyContainer::addProperty(int _type, QString _identifier, QVariant _value)
97 {
98  Q_ASSERT_X(!properties_.contains(_identifier),
99  "PropertyContainer::addProperty",
100  QString("Attempted to add multiple properties to a tag (%1) for which this operation is forbidden").arg(_identifier).toLatin1());
101 
102  QSharedPointer<Property> newProperty;
103  if(_type == enumTypeId())
104  newProperty = QSharedPointer<Property>( new EnumProperty(_identifier,_value) );
105  else if (_type == viewPropertyTypeId())
106  {
107  newProperty = QSharedPointer<Property>(new ViewProperty(_identifier, _value));
108  }
109  else
110  {
111  switch(_type)
112  {
113  case QVariant::Size:
114  newProperty = QSharedPointer<Property>( new SizeProperty(_identifier,_value) );
115  break;
116  case QVariant::Point:
117  newProperty = QSharedPointer<Property>( new PointProperty(_identifier,_value) );
118  break;
119  case QVariant::Color:
120  newProperty = QSharedPointer<Property>( new ColorProperty(_identifier,_value) );
121  break;
122  case QVariant::List:
123  newProperty = QSharedPointer<Property>( new ListProperty(_identifier,_value) );
124  break;
125  case QVariant::Map:
126  newProperty = QSharedPointer<Property>( new MapProperty(_identifier,_value) );
127  break;
128  default:
129  newProperty = QSharedPointer<Property>( new Property(_type,_identifier,_value) );
130  break;
131  }
132  }
133  connect(newProperty.data(),SIGNAL(valueChanged(Property*,QVariant)),this,SIGNAL(propertyValueChanged(Property*,QVariant)));
134  //containerGroupItem_->addSubProperty(newProperty);
135  properties_[_identifier].push_back(newProperty);
136  return newProperty;
137 }
