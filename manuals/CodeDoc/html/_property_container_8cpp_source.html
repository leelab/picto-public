85 QSharedPointer<Property> PropertyContainer::addProperty(int _type, QString _identifier, QVariant _value)
86 {
87  Q_ASSERT_X(!properties_.contains(_identifier),
88  "PropertyContainer::addProperty",
89  QString("Attempted to add multiple properties to a tag (%1) for which this operation is forbidden").arg(_identifier).toLatin1());
90 
91  QSharedPointer<Property> newProperty;
92  if(_type == enumTypeId())
93  newProperty = QSharedPointer<Property>( new EnumProperty(_identifier,_value) );
94  else
95  {
96  switch(_type)
97  {
98  case QVariant::Size:
99  newProperty = QSharedPointer<Property>( new SizeProperty(_identifier,_value) );
100  break;
101  case QVariant::Point:
102  newProperty = QSharedPointer<Property>( new PointProperty(_identifier,_value) );
103  break;
104  case QVariant::Color:
105  newProperty = QSharedPointer<Property>( new ColorProperty(_identifier,_value) );
106  break;
107  case QVariant::List:
108  newProperty = QSharedPointer<Property>( new ListProperty(_identifier,_value) );
109  break;
110  case QVariant::Map:
111  newProperty = QSharedPointer<Property>( new MapProperty(_identifier,_value) );
112  break;
113  default:
114  newProperty = QSharedPointer<Property>( new Property(_type,_identifier,_value) );
115  break;
116  }
117  }
118  connect(newProperty.data(),SIGNAL(valueChanged(Property*,QVariant)),this,SIGNAL(propertyValueChanged(Property*,QVariant)));
119  //containerGroupItem_->addSubProperty(newProperty);
120  properties_[_identifier].push_back(newProperty);
121  return newProperty;
122 }
