62 void RandomlyFilledGridGraphic::buildColorList()
63 {
64  int numColor2 = propertyContainer_->getPropertyValue("Color2Squares").toInt();
65  int numHorizSquares = propertyContainer_->getPropertyValue("HorizontalSquares").toInt();
66  int numVertSquares = propertyContainer_->getPropertyValue("VerticalSquares").toInt();
67 
68  //If none of the relevant parameters changed, we're done
69  if((numColor2 == prevColor2Squares_)
70  && (numHorizSquares == prevHorizSquares_)
71  && (numVertSquares) == prevVertSquares_)
72  return;
73  prevColor2Squares_ = numColor2;
74  prevHorizSquares_ = numHorizSquares;
75  prevVertSquares_ = numVertSquares;
76 
77  colorList_.clear();
78  colorList_.resize(numHorizSquares*numVertSquares);
79  for(int i=0; i<numColor2; i++)
80  colorList_[i] = 0;
81 
82  for(int i=numColor2; i<numHorizSquares*numVertSquares; i++)
83  colorList_[i] = 1;
84 }

89 void RandomlyFilledGridGraphic::draw()
90 {
91  int foregroundIndex;
92 
93  buildColorList();
94  //int numColor2 = propertyContainer_->getPropertyValue("Color2Squares").toInt();
95  int numHorizSquares = propertyContainer_->getPropertyValue("HorizontalSquares").toInt();
96  int numVertSquares = propertyContainer_->getPropertyValue("VerticalSquares").toInt();
97 
98  QColor backgroundColor = propertyContainer_->getPropertyValue("Color").value<QColor>();
99  QColor foregroundColor = propertyContainer_->getPropertyValue("Color2").value<QColor>();
100  foregroundIndex = 1;
101 
102  std::random_shuffle(colorList_.begin(), colorList_.end());
103 
104  QRect dimensions = QRect(QPoint(),propertyContainer_->getPropertyValue("Size").toSize());
105 
106  QImage image(dimensions.width(),dimensions.height(),QImage::Format_ARGB32);
107  posOffset_ = QPoint(dimensions.width()/2.0,dimensions.height()/2.0);
108  image.fill(backgroundColor.rgba());
109  QPainter p(&image);
110  p.setRenderHint(QPainter::Antialiasing, true);
111  p.setBrush(foregroundColor);
112  p.setPen(foregroundColor);
113 
114  float gridWidth = 0;
115  float gridHeight = 0;
116  if(numHorizSquares && numVertSquares)
117  {
118  gridWidth = float(image.width())/float(numHorizSquares);
119  gridHeight = float(image.height())/float(numVertSquares);
120  }
121 
122 
123  for(int x=0; x<numHorizSquares; x++)
124  {
125  for(int y=0; y<numVertSquares; y++)
126  {
127  if(colorList_[y*numHorizSquares + x] != foregroundIndex)
128  continue;
129  p.drawRect(x*gridWidth,y*gridHeight,gridWidth,gridHeight);
130  }
131  }
132  p.end();
133  image_ = image;
134 
135  updateCompositingSurfaces();
136 }
