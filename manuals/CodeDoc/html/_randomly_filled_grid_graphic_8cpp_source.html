96 void RandomlyFilledGridGraphic::draw()
97 {
98  int foregroundIndex;
99 
100  buildColorList();
101  int numColor2 = propertyContainer_->getPropertyValue("Color2Squares").toInt();
102  int numHorizSquares = propertyContainer_->getPropertyValue("HorizontalSquares").toInt();
103  int numVertSquares = propertyContainer_->getPropertyValue("VerticalSquares").toInt();
104 
105  QColor backgroundColor = propertyContainer_->getPropertyValue("Color").value<QColor>();
106  QColor foregroundColor = propertyContainer_->getPropertyValue("Color2").value<QColor>();
107  foregroundIndex = 1;
108 
109  std::random_shuffle(colorList_.begin(), colorList_.end());
110 
111  QRect dimensions = QRect(QPoint(),propertyContainer_->getPropertyValue("Size").toSize());
112 
113  QImage image(dimensions.width(),dimensions.height(),QImage::Format_ARGB32);
114  posOffset_ = QPoint(dimensions.width()/2.0,dimensions.height()/2.0);
115  image.fill(backgroundColor.rgba());
116  QPainter p(&image);
117  p.setRenderHint(QPainter::Antialiasing, true);
118  p.setBrush(foregroundColor);
119  p.setPen(foregroundColor);
120 
121  float gridWidth = 0;
122  float gridHeight = 0;
123  if(numHorizSquares && numVertSquares)
124  {
125  gridWidth = float(image.width())/float(numHorizSquares);
126  gridHeight = float(image.height())/float(numVertSquares);
127  }
128 
129 
130  for(int x=0; x<numHorizSquares; x++)
131  {
132  for(int y=0; y<numVertSquares; y++)
133  {
134  if(colorList_[y*numHorizSquares + x] != foregroundIndex)
135  continue;
136  p.drawRect(x*gridWidth,y*gridHeight,gridWidth,gridHeight);
137  }
138  }
139  p.end();
140  image_ = image;
141 
142  updateCompositingSurfaces();
143 }
