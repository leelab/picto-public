12 //After a whole bunch of debugging, we came to the following conclusion:
13 
14  //When rendering on PictoBox (EPIA-M700) with VIA drivers, the rendering

17 //In exclusive mode, the present function isn't actually flipping buffers, so
18 //all rendering gets done on the front surface. This causes serious display issues
19

22 //remaining frames render 10 times slower (~5 ms). This is not due to the buffer
23 //filling up, since waiting for vsync doesn't solve the problem
24

26 //will be done in Direct3D (under Windows XP). I have attempted to leave the code

424 void D3DMVisualTarget::draw(QPoint location, QSharedPointer<CompositingSurface> compositingSurface)
425 {
426  if(compositingSurface->getTypeName() == "Direct3DMobile")
427  {
428  //retreive a pointer to the texture
429  LPDIRECT3DMOBILETEXTURE tex = compositingSurface.staticCast<D3DMCompositingSurface>()->getTexture();
430 
431  //add texture pointer to the rendering list
432  textureList_.push_back(tex);
433 
434  D3DMSURFACE_DESC texDesc;
435  tex->GetLevelDesc(0, &texDesc);
436 
437  unsigned int imgHeight = compositingSurface.staticCast<D3DMCompositingSurface>()->getImageHeight();
438 
439  //figure out the position for the sprite, and add it to the list
440  //NOTE: Picto uses the standard screen coodinates with (0,0) in the top left
441  //corner. However, the D3DM setup we are using has (0,0) in the bottom left corner, so
442  //we need to find the "complement" of the y-coordinate.
443  //The value stored in the positionList_ is the bottom left corner (inclusive)
444  //of the sprite.
445  QPointF quadPoint;
446  quadPoint.setX((float)location.x());
447  quadPoint.setY((float)(height_-location.y()-imgHeight));
448  positionList_.push_back(quadPoint);
449  }
450  else
451  {
452  d3dFail("D3DMVisualTarget: draw called with a compositing surface not of type \"Direct3DMobile\"");
453  }
454 }

460 void D3DMVisualTarget::present()
461 {
462  LPDIRECT3DMOBILETEXTURE tex;
463  QPointF quadPoint;
464  D3DMXMATRIX matScale, matTrans;
465  D3DMSURFACE_DESC texDesc;
466 
467  HRESULT hr;
468 
469  //clear the back buffer
470  pD3dmDevice_->Clear( 0, NULL, D3DMCLEAR_TARGET, 0, 0.0f, 0 );
471 
472 
473  //begin the scene
474  hr = pD3dmDevice_->BeginScene();
475  if(hr)
476  d3dFail("D3DMVisualTarget: BeginScene FAILED");
477 
478  //render each textured quad
479  while(!textureList_.isEmpty())
480  {
481  tex = textureList_.takeFirst();
482  quadPoint = positionList_.takeFirst();
483 
484  tex->GetLevelDesc(0, &texDesc);
485 
486  //The 0.5 offset on the translation matrix is done to ensure perfect texture mapping
487  D3DMXMatrixScaling(&matScale, (float)texDesc.Width, (float)texDesc.Height, 0.0f);
488  D3DMXMatrixTranslation(&matTrans, quadPoint.x()-0.5f, quadPoint.y()-0.5f, 0.0f);
489  pD3dmDevice_->SetTransform( D3DMTS_WORLD, (D3DMMATRIX*)&(matScale*matTrans), D3DMFMT_D3DMVALUE_FLOAT );
490 
491  pD3dmDevice_->SetTexture( 0, tex );
492  pD3dmDevice_->DrawPrimitive(D3DMPT_TRIANGLELIST,
493  0, //Start Vertex
494  2); //Primitive Count
495 
496 
497  }
498 
499 
500  //end the scene
501  hr = pD3dmDevice_->EndScene();
502  if(hr)
503  d3dFail("D3DMVisualTarget: EndScene FAILED");
504 
505  //Don't call present until we are in vertical blanking
506  BYTE outBuf;
507  DWORD bytesReturned;
508  DeviceIoControl(hCRT,IOCTRL_WAIT_VBLANK_START,NULL,0,&outBuf,1,&bytesReturned,NULL);
509 
510  static int test = 0;
511 
512  //present the back buffer
513  hr = pD3dmDevice_->Present( NULL, NULL, NULL, NULL );
514  if(hr)
515  d3dFail("D3DMVisualTarget: Present FAILED");
516 }
