129 void TaskConfig::managerConnectionEstablished(bool connected)
130 {
131  bool justConnected = false;
132  switch (managerConnectionStatus_)
133  {
134  case ConnectionStatus::Unconnected:
135  if (connected)
136  {
137  managerConnectionStatus_ = ConnectionStatus::Connected;
138  justConnected = true;
139  }
140  else
141  {
142  qDebug() << "Tried to break connection when No Connection enabled.";
143  }
144  break;
145  case ConnectionStatus::Connected:
146  if (connected)
147  {
148  qDebug() << "Tried to open connection when already enabled.";
149  }
150  else
151  {
152  managerConnectionStatus_ = ConnectionStatus::ConnectionEnded;
153  }
154  break;
155  case ConnectionStatus::ConnectionEnded:
156  if (connected)
157  {
158  //Reconnected
159  managerConnectionStatus_ = ConnectionStatus::Connected;
160  }
161  else
162  {
163  qDebug() << "Tried to modify ended connection with connected =" << connected;
164  }
165  break;
166  default:
167  qDebug() << "Unrecognized ConnectionStatus (managerConnectionEstablished)";
168  }
169 
170  if (justConnected)
171  {
172  QStringList plotPaths = waitingOperatorPlots_.keys();
173  foreach(QString plotPath, plotPaths)
174  {
175  if (!waitingOperatorPlots_[plotPath].isNull())
176  {
177  emit plotHandlerRequested(plotPath);
178  }
179  else
180  {
181  waitingOperatorPlots_.remove(plotPath);
182  }
183  }
184  }
