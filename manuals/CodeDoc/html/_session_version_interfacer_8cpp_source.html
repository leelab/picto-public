220 QString SessionVersionInterfacer::recurseResetAssetIds(QSharedPointer<Asset> asset,QSharedPointer<SVIAssetNode> node)
221 {
222  asset->setAssetId(node->assetId);
223  if(!asset->inherits("Picto::DataStore"))
224  return "";
225  QSharedPointer<DataStore> dataStore = asset.staticCast<DataStore>();
226  QStringList childTags = dataStore->getDefinedChildTags();
227  QList<QSharedPointer<Asset>> childAssets;
228  foreach(QString childTag,childTags)
229  {
230  childAssets << dataStore->getGeneratedChildren(childTag);
231  }
232  foreach(int childId,node->children)
233  {
234  QSharedPointer<SVIAssetNode> childNode = nodesById_[childId];
235  QSharedPointer<Asset> matchingNode;
236  foreach(QSharedPointer<Asset> child,childAssets)
237  {
238  if(childNode->isTransition != child->inherits("Picto::Transition"))
239  continue;
240  if(childNode->isTransition)
241  {
242  QSharedPointer<Transition> transChild = child.staticCast<Transition>();
243  if( (childNode->transSource == transChild->getSource())
244  && (childNode->transResult == transChild->getSourceResult())
245  )
246  {
247  matchingNode = child;
248  break;
249  }
250  }
251  else if(childNode->name == child->getName())
252  {
253  matchingNode = child;
254  break;
255  }
256  }
257  if(matchingNode)
258  {
259  QString r = recurseResetAssetIds(matchingNode,childNode);
260  if(r != "")
261  return r;
262  }
263  else
264  {
265  //If no matching node was found, the asset must either be one of the types not
266  //recorded in the session (UI stuff) or come from an updated version of an asset
267  //used in the experiment. Either way add it to an obsolete assets list that can
268  //be used to filter out changes.
269  recurseAddObsoleteAssets(childNode->assetId);
270  }
271  }
272  return "";
273 }
