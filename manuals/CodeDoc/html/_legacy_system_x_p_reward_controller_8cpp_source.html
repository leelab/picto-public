76 void LegacySystemXPRewardController::startReward(unsigned int,int quantity)

78  if(!hasDevice_)

80  //We don't really care about the channel here because there's currenlty

85  //turn on the reward controller (remember, we're using active high logic)
86  DAQmxErrChk(DAQmxCreateTask("RewardTask",(TaskHandle*)&daqTaskHandle_));
87  taskExists_ = true;
88  DAQmxErrChk(DAQmxCreateAOVoltageChan(daqTaskHandle_,"Dev2/ao0","",0.0,5.0,DAQmx_Val_Volts,NULL));
89  DAQmxErrChk (DAQmxCfgSampClkTiming(daqTaskHandle_,"",1000.0/float64(quantity),DAQmx_Val_Rising,DAQmx_Val_FiniteSamps,2));
90  int32 sampsPerChanWritten;
91  DAQmxErrChk(DAQmxWriteAnalogF64(daqTaskHandle_,2,0,-1,DAQmx_Val_GroupByChannel,outputData,&sampsPerChanWritten,NULL));
92  DAQmxErrChk(DAQmxStartTask(daqTaskHandle_));
93  //Since this code creates a task that raises the voltage for quantity ms and then lowers it for quantity ms, as it is it
94  //doesn't support inter reward intervals of less than quantity. Since we couldn't find a better way to do this, we just
95  //set our own timer and stop/clear the task anytime that it still exists and more than quantity ms have passed.
96  stopwatch_.startWatch();
97  latestOnTime_ = quantity;
98 }
