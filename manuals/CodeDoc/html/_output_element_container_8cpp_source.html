112 void OutputElementContainer::postDeserialize()
113 {
114  StateMachineElement::postDeserialize();
115  QStringList childTags = getDefinedChildTags();
116  foreach(QString childTag,childTags)
117  {
118  QList<QSharedPointer<Asset>> tagChildren = getGeneratedChildren(childTag);
119  foreach(QSharedPointer<Asset> tagChild,tagChildren)
120  {
121  addChildToElementLists(tagChild);
122  }
123  }
124  connect(this,SIGNAL(childAddedAfterDeserialize(QSharedPointer<Asset>)),this,SLOT(addChildToElementLists(QSharedPointer<Asset>)));
125 }
