48 void RewardController::triggerRewards(bool appendToList)
49 {
50  QHash<int,RewardChannel>::iterator it;
51  for(it = rewardChannels_.begin();it!=rewardChannels_.end();it++)
52  {
53  //Check if the inter reward/flush interval has been met
54  if( it.value().stopwatch.elapsedMs() < it.value().lastRewardPeriod)
55  {
56  if(discardOverlap_)
57  it.value().pendingRewards.clear();
58  continue;
59  }
60  //If it was met once, set it to zero to avoid stopwatch wrapping issues.
61  it.value().lastRewardPeriod = 0;
62 
63  //If there is a flush in progress stop it
64  if(it.value().inFlush)
65  {
66  stopFlush(it.key());
67  it.value().inFlush = false;
68  }
69 
70  //Check if this channel has any new rewards
71  if(!it.value().pendingRewards.size())
72  continue;
73 
74  //Check if the last reward is still being supplied
75  if(!rewardWasSupplied(it.key()))
76  {
77  if(discardOverlap_)
78  it.value().pendingRewards.takeFirst();
79  continue;
80  }
81 
82  //Get the new reward for this channel
83  RewardUnit reward = it.value().pendingRewards.takeFirst();
84  if(!reward.isFlush)
85  { //Start reward
86  startReward(it.key(),reward.quantity);
87  emit rewarded(reward.quantity);
88  }
89  else
90  { //Start flush
91  startFlush(it.key());
92  it.value().inFlush = true;
93  }
94  double rewardTime = stamper_.stampSec();
95  it.value().stopwatch.startWatch();
96  it.value().lastRewardPeriod = reward.minRewardPeriod;
97  if(!appendToList)
98  continue;
99  if(!reward.isFlush)
100  appendDeliveredRewards(QSharedPointer<RewardDataUnit>(new RewardDataUnit(reward.quantity,it.key(),rewardTime)));
101  }
102 }
