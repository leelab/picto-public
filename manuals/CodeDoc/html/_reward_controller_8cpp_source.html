49 void RewardController::triggerRewards(bool appendToList)
50 {
51  QHash<int,RewardChannel>::iterator it;
52  for(it = rewardChannels_.begin();it!=rewardChannels_.end();it++)
53  {
54  //Check if the inter reward/flush interval has been met
55  if( it.value().stopwatch.elapsedMs() < it.value().lastRewardPeriod)
56  {
57  if(discardOverlap_)
58  it.value().pendingRewards.clear();
59  continue;
60  }
61  //If it was met once, set it to zero to avoid stopwatch wrapping issues.
62  it.value().lastRewardPeriod = 0;
63 
64  //If there is a flush in progress stop it
65  if(it.value().inFlush)
66  {
67  stopFlush(it.key());
68  it.value().inFlush = false;
69  }
70 
71  //Check if this channel has any new rewards
72  if(!it.value().pendingRewards.size())
73  continue;
74 
75  //Check if the last reward is still being supplied
76  if(!rewardWasSupplied(it.key()))
77  {
78  if(discardOverlap_)
79  it.value().pendingRewards.takeFirst();
80  continue;
81  }
82 
83  //Get the new reward for this channel
84  RewardUnit reward = it.value().pendingRewards.takeFirst();
85  if(!reward.isFlush)
86  { //Start reward
87  startReward(it.key(),reward.quantity);
88  }
89  else
90  { //Start flush
91  startFlush(it.key());
92  it.value().inFlush = true;
93  }
94  double rewardTime = stamper_.stampSec();
95  it.value().stopwatch.startWatch();
96  it.value().lastRewardPeriod = reward.minRewardPeriod;
97  if(!appendToList)
98  continue;
99  if(!reward.isFlush)
100  appendDeliveredRewards(QSharedPointer<RewardDataUnit>(new RewardDataUnit(reward.quantity,it.key(),rewardTime)));
101  }
