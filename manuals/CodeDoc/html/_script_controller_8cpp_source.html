68 void ScriptController::upgradeVersion(QString deserializedVersion)
69 {
70  ControlElement::upgradeVersion(deserializedVersion);
71  if(deserializedVersion < "0.0.1")
72  { //Before 0.0.1, Script controllers had a single required result, "Done" that was
73  //triggered if the script returned a boolean true value. As of Syntax version 0.0.1,
74  //the user can set as many results as they want with the result chosen by matching the
75  //script's return string to the result name. To upgrade, we add a "Done" result. This
76  //will cause this object's parent to hook up any saved "Done" result transitions to that
77  //result. We then change the name of the "Done" result to "true" which will cause
78  //those transitions to update automatically. This way when true is returned from these
79  //script controllers, it will be cast to the string "true" and that result will be triggered,
80  //and the correct transition traversed.
81  QList<QSharedPointer<Asset>> resultKids = getGeneratedChildren("Result");
82  Q_ASSERT(resultKids.size() <= 1);
83  if(resultKids.isEmpty())
84  { //If no "Done" result was deserialized, add it.
85  addRequiredResult("Done");
86  resultKids = getGeneratedChildren("Result");
87  }
88  Q_ASSERT(resultKids.size() == 1);
89  //Change the name of the "Done" result to "true"
90  resultKids.first().staticCast<UIEnabled>()->setName("true");
91 
92 
93  //Before 0.0.1, newClick on operatorClick info had the effect of reading out the current
94  //"clicked" value, and also resetting it. In 0.0.1, we changed it such that the user
95  //must reset it manually. Since there aren't too many experiments yet, we know how and
96  //where this script was used and we are changing operatorClick.newClick to
97  //(operatorClick.newClick && !(operatorClick.newClick = false)) in all places that it was
98  //used.
99  QString frameScript = propertyContainer_->getPropertyValue("Script").toString();
100  frameScript.replace("operatorClick.newClick","(operatorClick.newClick && !(operatorClick.newClick = false))");
101  propertyContainer_->setPropertyValue("Script",frameScript);
102 
103 
104  }
105 }
