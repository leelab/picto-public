71 void TargetController::upgradeVersion(QString deserializedVersion)
72 {
73  ControlElement::upgradeVersion(deserializedVersion);
74  if(deserializedVersion < "0.0.1")
75  { // In design syntax version "0.0.1", we added userOnTarget, userEnteredTarget, userExitedTarget functions.
76  // In an effort to keep the allowed script locations uniform, we are making TargetEntry
77  // and TargetExit scripts obsolete.
78  // We upgrade these older experiments by automatically adding their entry and exit scripts
79  // to the frame script and exit scripts in if(userEnteredTarget()) and if(userExitedTarget) blocks.
80 
81  //WARNING!!! userOnTarget(), userEnteredTarget() and userExitedTarget()
82  //functions cannot be used in AnalysisScripts of experiments that were saved with version below 0.0.1.
83  //This is because experiments with those versions used TargetEntry and TargetExit scripts and did not
84  //save OnTarget or OffTarget data in their session files. Attempting to use these functions in Analysis
85  //Scripts would cause return of invalid data.
86  QSharedPointer<Property> parentFrameScriptProp = getParentAsset().staticCast<DataStore>()->getGeneratedChildren("FrameScript").first().staticCast<Property>();
87  QSharedPointer<Property> parentExitScriptProp = getParentAsset().staticCast<DataStore>()->getGeneratedChildren("ExitScript").first().staticCast<Property>();
88  Q_ASSERT(parentFrameScriptProp);
89  QString parentFrameScript = parentFrameScriptProp->value().toString();
90  QString parentExitScript = parentExitScriptProp->value().toString();
91  QSharedPointer<ObsoleteAsset> obsScript;
92  if(getGeneratedChildren("TargetExitScript").size())
93  obsScript = getGeneratedChildren("TargetExitScript").first().staticCast<ObsoleteAsset>();
94  if(obsScript && obsScript->getValue().size())
95  {
96  //Put exit script into parent's frame script.
97  parentFrameScript.prepend(QString("if(%1.userExitedTarget()){\n%2\n}\n").arg(getName()).arg(obsScript->getValue()));
98  //Put exit script into parent's exit script.
99  parentExitScript.prepend(QString("if(%1.userExitedTarget()){\n%2\n}\n").arg(getName()).arg(obsScript->getValue()));
100  }
101  obsScript.clear();
102  if(getGeneratedChildren("TargetEntryScript").size())
103  obsScript = getGeneratedChildren("TargetEntryScript").first().staticCast<ObsoleteAsset>();
104  if(obsScript && obsScript->getValue().size())
105  {
106  //Put entry script into parent's frame script.
107  parentFrameScript.prepend(QString("if(%1.userEnteredTarget()){\n%2\n}\n").arg(getName()).arg(obsScript->getValue()));
108  //Put entry script into parent's exit script.
109  parentExitScript.prepend(QString("if(%1.userEnteredTarget()){\n%2\n}\n").arg(getName()).arg(obsScript->getValue()));
110  }
111  parentFrameScriptProp->setValue(parentFrameScript);
112  parentExitScriptProp->setValue(parentExitScript);
113  }
114 }

207 bool TargetController::isDonePrivate(QSharedPointer<Engine::PictoEngine> engine)
208 {
209  Controller::TimerUnits::TimerUnits timeUnits;
210  if(unitList_.value(propertyContainer_->getPropertyValue("TimeUnits").toInt(),"") == "Sec")
211  timeUnits = Controller::TimerUnits::sec;
212  else if(unitList_.value(propertyContainer_->getPropertyValue("TimeUnits").toInt(),"") == "Ms")
213  timeUnits = Controller::TimerUnits::ms;
214  else if(unitList_.value(propertyContainer_->getPropertyValue("TimeUnits").toInt(),"") == "Us")
215  timeUnits = Controller::TimerUnits::us;
216  else
217  {
218  timeUnits = Controller::TimerUnits::ms;
219  Q_ASSERT(false);
220  }
221 
222  int totalTime = getTotalTime();
223  int fixTime = getFixationTime();
224 
225  int currTotalTime = cumulativeTimer_->elapsedTime(timeUnits);
226  int currAcqTime = acquisitionTimer_->elapsedTime(timeUnits);
227  int currReAcqTime = reacquisitionTimer_->elapsedTime(timeUnits);
228 
229  int remainingFixTime = targetAcquired_ ? fixTime - currAcqTime : fixTime;
230 
231  //check to see if we've met or exceeded the total time
232  if ( currTotalTime + remainingFixTime > totalTime)
233  {
234  isDone_ = true;
235  result_ = "Total Time Excceeded";
236  return true;
237  }
238 
239  bool isInsideTarget = insideTarget(engine);
240 
241  //just entered target
242  if(!targetAcquired_ && isInsideTarget)
243  {
244  //if we're past the Min Initial Acquisition Time mark the target as acquired
245  //and start running the timer
246  if (currTotalTime >= getMinAcqTime())
247  {
248  acquisitionTimer_->start();
249  targetAcquired_ = true;
250  initialAcquisitionOccurred_ = true;
251 
252  if(!engine->slaveMode())
253  {
254  propertyContainer_->getProperty("OnTarget")->setValue(true);
255  propertyContainer_->getProperty("OnTargetChanged")->setValue(true);
256  }
257  //If fixation time is zero, we're done
258  if(fixTime <= 0)
259  {
260  isDone_ = true;
261  result_ = "Success";
262  return true;
263  }
264  }
265  }
266  //just left target
267  else if(targetAcquired_ && !isInsideTarget)
268  {
269  targetAcquired_ = false;
270 
271  if(!engine->slaveMode())
272  {
273  propertyContainer_->getProperty("OnTarget")->setValue(false);
274  propertyContainer_->getProperty("OnTargetChanged")->setValue(true);
275  }
276 
277  //If reacquisition isn't allowed, then we're done with a failure value.
278  if (!getReacqAllowed())
279  {
280  isDone_ = true;
281  result_ = "Broke Fixation";
282  return true;
283  }
284  else
285  {
286  if (currTotalTime + fixTime > totalTime)
287  { //There's not enough time left for a complete fixation.
288  isDone_ = true;
289  result_ = "Total Time Excceeded";
290  return true;
291  }
292 
293  waitingForReacquisition_ = true;
294  reacquisitionTimer_->start();
295  }
296  }
297  //staying inside target
298  else if(targetAcquired_ && isInsideTarget)
299  {
300  //check to see if we've met or exceeded the fixation time
301  if (currAcqTime >= fixTime)
302  {
303  isDone_ = true;
304  result_ = "Success";
305  return true;
306  }
307 
308  }
309  //staying outside target
310  else if(!targetAcquired_ && !isInsideTarget)
311  {
312  //check to see if we've met or exceeded the max acquisition time (if relevant)
313  if(!initialAcquisitionOccurred_)
314  {
315  if (currTotalTime >= getMaxAcqTime())
316  {
317  isDone_ = true;
318  result_ = "Initial Aquistion Time Exceeded";
319  return true;
320  }
321  }
322 
323  //check to see if we've met or exceeded the reacquisition time
324  if(waitingForReacquisition_)
325  {
326  if (currReAcqTime >= getMaxReacqTime())
327  {
328  isDone_ = true;
329  result_ = "Reaquistion Time Exceeded";
330  return true;
331  }
332  }
333 
334  }
335 
336  return false;
337 }

341 bool TargetController::insideTarget(QSharedPointer<Engine::PictoEngine> engine)
342 {
343  if(engine.isNull())
344  return false;
345  if(signal_.isNull())
346  {
347  //grab the signal channel
348  signal_ = engine->getSignalChannel(propertyContainer_->getPropertyValue("SignalChannel").toString());
349  if(signal_.isNull())
350  return false;
351 
352  //confirm that the signal channel contains "x" and "y" subchannels
353  Q_ASSERT(signal_->getSubchannels().contains("x"));
354  Q_ASSERT(signal_->getSubchannels().contains("y"));
355  }
356 
357  int x = signal_->peekValue("x");
358  int y = signal_->peekValue("y");
359  if(controlTarget_ && controlTarget_.toStrongRef()->contains(x,y))
360  return true;
361  return false;
