26 bool DesignRoot::resetDesignRoot(QString DesignRootText)
27 {
28  //Deserialize Design from DesignRootText
29  QString errors = "";
30  QSharedPointer<QXmlStreamReader> xmlReader(new QXmlStreamReader(DesignRootText));
31  pictoData_ = QSharedPointer<PictoData>(PictoData::Create().staticCast<PictoData>());
32  while(!xmlReader->isStartElement() && (xmlReader->name() != pictoData_->identifier()) && !xmlReader->atEnd())
33  xmlReader->readNext();
34  if(xmlReader->atEnd())
35  {
36  lastError_.details = "Could not find " + pictoData_->identifier() + " in DesignRoot Text.";
37  lastError_.name = "XML Parsing Error ";
38  return false;
39  }
40 
41  compiled_ = false;
42  bool res = pictoData_->fromXml(xmlReader);
43  if(!res)
44  {
45  errors = "DesignRoot Text Error\n"+Serializable::getErrors();
46  }
47  if(!errors.isEmpty())
48  {
49  lastError_.details = errors;
50  lastError_.name = "XML Parsing Error ";
51  pictoData_.clear();
52  return false;
53  }
54  connect(pictoData_.data(),SIGNAL(edited()),this,SLOT(designEdited()));
55  setDesignName(pictoData_->getName());
56  if(OldVersionAsset::encounteredOldVersionAsset() || Property::encounteredObsoleteSerialSyntax())
57  {
58  lastWarning_.name="Obsolete Syntax Encountered ";
59  lastWarning_.details="File contents have been automatically upgraded to "
60  "function with the latest version of Picto. If you save this design, "
61  "it will be incompatible with older versions of Picto.";
62  OldVersionAsset::clearOldVersionAssetFlag();
63  Property::clearObsoleteSerialSyntax();
64  //Move to the upgraded version
65  return resetDesignRoot(pictoData_->toXml());
66  }
67 
68  setUndoPoint(); //Has the effect of serializing the root object tree into designRootText_
69  designRootText_.clearUndoRedoStacks();
70  designRootText_.setModified(false);
71 
72  emit refreshedFromXml();
73  return true;
74 }

140 void DesignRoot::setUndoPoint()

143  return;
144  QString newDesignRootText = pictoData_->toXml();
