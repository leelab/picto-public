28 bool DesignRoot::resetDesignRoot(QString DesignRootText)
29 {
30  //Deserialize Design from DesignRootText
31  QString errors = "";
32  QSharedPointer<QXmlStreamReader> xmlReader(new QXmlStreamReader(DesignRootText));
33  pictoData_ = QSharedPointer<PictoData>(PictoData::Create().staticCast<PictoData>());
34  while(!xmlReader->isStartElement() && (xmlReader->name() != pictoData_->identifier()) && !xmlReader->atEnd())
35  xmlReader->readNext();
36  if(xmlReader->atEnd())
37  {
38  lastError_.details = "Could not find " + pictoData_->identifier() + " in DesignRoot Text.";
39  lastError_.name = "XML Parsing Error ";
40  return false;
41  }
42 
43  compiled_ = false;
44  bool res = pictoData_->fromXml(xmlReader);
45  if(!res)
46  {
47  errors = "DesignRoot Text Error\n"+Serializable::getErrors();
48  }
49  if(!errors.isEmpty())
50  {
51  lastError_.details = errors;
52  lastError_.name = "XML Parsing Error ";
53  pictoData_.clear();
54  return false;
55  }
56  connect(pictoData_.data(),SIGNAL(edited()),this,SLOT(designEdited()));
57  setDesignName(pictoData_->getName());
58  if(OldVersionAsset::encounteredOldVersionAsset() || Property::encounteredObsoleteSerialSyntax())
59  {
60  lastWarning_.name="Obsolete Syntax Encountered ";
61  lastWarning_.details="File contents have been automatically upgraded to "
62  "function with the latest version of Picto. If you save this design, "
63  "it will be incompatible with older versions of Picto.";
64  OldVersionAsset::clearOldVersionAssetFlag();
65  Property::clearObsoleteSerialSyntax();
66  //Move to the upgraded version
67  return resetDesignRoot(pictoData_->toXml());
68  }
69 
70  //Has the effect of serializing the root object tree into undoStack
71  setUndoPoint();
72  setUnmodified();
73 
74  emit refreshedFromXml();
75  return true;
76 }

181 void DesignRoot::setUndoPoint()
182 {
183  if(!pictoData_)
184  return;
185  QString newDesignRootText = pictoData_->toXml();
186  if (designUndoStack_.empty() || newDesignRootText != designUndoStack_[undoStateOffset_])
187  {
188  while (undoStateOffset_ > 0)
189  {
190  undoStateOffset_--;
191  designUndoStack_.pop_front();
192  }
193 
194  //Sometimes with giant experiment designs this fails to allocate memory
195  try
196  {
197  designUndoStack_.push_front(newDesignRootText);
198  }
199  catch(...)
200  {
201  //Retain at least the last 3 changes
202  while (designUndoStack_.size() > 3)
203  {
204  designUndoStack_.pop_back();
205  }
206  }
207 
208  emit redoAvailable(hasRedo());
209  emit undoAvailable(hasUndo());
210 
211 }
212 }
