129 void BarBasePlotHandler::updateColumns(const QString &setName, const QColor &color, ColumnType::ColumnType eType)
130 {
131  if (!m_qhpHistoPlotItems.contains(setName))

133  addDataSet(setName);

136  m_qhpHistoPlotItems[setName]->setBrush(QBrush(color));
137  m_qhcHistoPlotColor[setName] = color;

139  if (m_pDataSetBox && m_pDataSetBox->findText(setName) != -1)
140  {

143 
144 
145  QPen pen(Qt::black, 0);
146  QwtColumnSymbol *symbol = nullptr;

148  switch (eType)

152  m_qhpHistoPlotItems[setName]->setSymbol(nullptr);
153  m_qhpHistoPlotItems[setName]->setPen(pen);
154  break;
155  case ColumnType::COLUMN_FLAT_COLUMN:
156  m_qhpHistoPlotItems[setName]->setStyle(QwtPlotHistogram::Columns);
157  m_qhpHistoPlotItems[setName]->setSymbol(nullptr);
158  m_qhpHistoPlotItems[setName]->setPen(pen);
159  break;
160  case ColumnType::COLUMN_RAISED_COLUMN:
161  m_qhpHistoPlotItems[setName]->setStyle(QwtPlotHistogram::Columns);
162 
163  symbol = new QwtColumnSymbol(QwtColumnSymbol::Box);
164  symbol->setFrameStyle(QwtColumnSymbol::Raised);
165  symbol->setLineWidth(2);
166  symbol->setPalette(QPalette(m_qhpHistoPlotItems[setName]->brush().color()));
167 
168  m_qhpHistoPlotItems[setName]->setSymbol(symbol);
169  break;
170  case ColumnType::COLUMN_LINES:
171  m_qhpHistoPlotItems[setName]->setStyle(QwtPlotHistogram::Lines);
172  m_qhpHistoPlotItems[setName]->setSymbol(nullptr);
173  pen.setBrush(m_qhpHistoPlotItems[setName]->brush());
174  pen.setWidth(2);
175  m_qhpHistoPlotItems[setName]->setPen(pen);
176  break;
177  default:
178  break;
179  }
