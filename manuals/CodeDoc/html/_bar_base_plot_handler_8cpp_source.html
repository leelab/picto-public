405  createErrorBars(setName);
406

442 void BarBasePlotHandler::exportClicked()

475  {
476  exportSize = DataViewSize::VIEW_SIZE_4x4;
477  }
478 
479  if (fileName.endsWith(".png", Qt::CaseInsensitive))
480  {
481  exportType = ExportType::EXPORT_PNG;
482  }
483  else if (fileName.endsWith(".bmp", Qt::CaseInsensitive))
484  {
