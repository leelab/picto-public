8 void SpikeState::setDatabase(QSqlDatabase session)
9 {
10  runStart_ = runEnd_ = curr_ = -1;
11  session_ = session;
12  query_ = QSharedPointer<QSqlQuery>(new QSqlQuery(session_));
13 
14  //Get timestamp alignment constants
15  query_->prepare("SELECT value FROM sessioninfo WHERE key=\"AlignmentInfo\"");
16  if(!query_->exec() || !query_->next())
17  {
18  return;
19  }
20  AlignmentInfo inf;
21  inf.fromXml(query_->value(0).toString());
22  double offsetTime = inf.getOffsetTime();
23  double temporalFactor = inf.getTemporalFactor();
24 
25  //Get spike resolution
26  sampPeriod_ = 0;
27  query_->prepare("SELECT value FROM sessioninfo WHERE key='DataSource'");
28  if(!query_->exec())
29  {
30  return;
31  }
32  while(query_->next())
33  {
34  DataSourceInfo info;
35  info.fromXml(query_->value(0).toString());
36  if(info.getName() == "spikes")
37  {
38  sampPeriod_ = temporalFactor*info.getResolution();
39  break;
40  }
41  }
42  if(sampPeriod_ <= 0)
43  return;
44 
45 
46  //Preallocate Spike data array
47  query_->exec("SELECT COUNT(*) FROM spikes");
48  if(!query_->exec() || !query_->next())
49  {
50  Q_ASSERT(false);
51  return;
52  }
53  data_.clear();
54  data_.reserve(query_->value(0).toInt());
55  for(int i=0;i<query_->value(0).toInt();i++)
56  {
57  data_.append(PlaybackSpikeData());
58  }
59 
60  query_->exec("SELECT s.timestamp,s.channel,s.unit,s.waveform FROM spikes s "
61  "ORDER BY s.timestamp");
62  if(!query_->exec())
63  {
64  Q_ASSERT(false);
65  return;
66  }
67 
68  int arrayIndex = 0;
69  QVector<float> waveform;
70  QByteArray waveformBA;
71  int channel;
72  int unit;
73  QMap<int,QMap<int,bool>> channelUnitsLookup;
74  while(query_->next())
75  {
76  waveformBA = query_->value(3).toByteArray();
77  waveform.clear();
78  const float* pots = reinterpret_cast<const float*>(waveformBA.constData());
79  for(unsigned int i=0;i<waveformBA.size()/sizeof(float);i++)
80  {
81  waveform.append(pots[i]);
82  }
83  channel = query_->value(1).toInt();
84  unit = query_->value(2).toInt();
85  data_[arrayIndex++] = PlaybackSpikeData(offsetTime + (temporalFactor*query_->value(0).toDouble()),channel,unit,waveform);
86  channelUnitsLookup[channel][unit] = true;
87  }
88  foreach(int chan,channelUnitsLookup.keys())
89  {
90  channels_.append(chan);
91  foreach(int un,channelUnitsLookup[chan].keys())
92  {
93  unitsLookup_[chan].append(un);
94  }
95  }
96  Q_ASSERT(data_.size() == arrayIndex);
97 }
