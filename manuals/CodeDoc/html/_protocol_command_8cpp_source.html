24 {

28  QStringList tokens = firstLine.split(QRegExp("[ ][ ]*"));
29  if(tokens.count() != 3)
30  {
31  parseError = "Method Target Protocol/Version header malformed";
32  }
33  else
34  {
35  method_ = tokens[0];
36  target_ = tokens[1];
37

41  parseError = "Method Target Protocol/Version header malformed";
42  }
43  else
44  {
45  protocolName_ = tokens[2].left(protocolVersionPosition);
46  protocolVersion_ = tokens[2].mid(protocolVersionPosition+1);
47  }
48  }

220 {
221  QString currentLine;
222  QStringList tokens;
223 
224  if(socket->bytesAvailable() <= 0)
225  {
226  if(!socket->waitForReadyRead(timeoutMs))
227  {
228  return -1;
229  }
230  }
231 
232  //read the first line(e.g. "GET /sometarget PICTO/1.0")
233  currentLine = socket->readLine();
234  currentLine.remove("\r\n");
235 
236  tokens = currentLine.split(" ", QString::SkipEmptyParts);
237  if(tokens.size() != 3)
238  return -2;
239 
240  method_ = tokens[0];
241  target_ = tokens[1];
242 
243  tokens = tokens[2].split("/");
244  if(tokens.size() != 2)
245  return -2;
246  protocolName_ = tokens[0];
247  protocolVersion_ = tokens[1];
248 
249  //next, clear then populate the fields map
250  fields_.clear();
251 
252  currentLine = socket->readLine();
253  while(currentLine != "\r\n")
254  {
255  currentLine.remove("\r\n");
256  QString fieldKey,fieldValue;
257 
258  int keyEndPosition = currentLine.indexOf(':');
259  if(keyEndPosition < 0)
260  return -4;
261 
262  fieldKey = currentLine.left(keyEndPosition);
263  fieldValue = currentLine.mid(keyEndPosition+2);
264  fields_[fieldKey] = fieldValue;
265 
266  currentLine = socket->readLine();
267  }
268 
269  //finally, if there's content, read it
270  int contentLength = fields_.value("Content-Length", "0").toInt();
271  if(contentLength == 0)
272  return 0;
273 
274  content_ = socket->read(contentLength);
275  while(content_.size() < contentLength &&
276  (socket->bytesAvailable() > 0 || socket->waitForReadyRead(500)))
277  {

279  }
280
