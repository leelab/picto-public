99 void ScriptFunction::upgradeVersion(QString deserializedVersion)
100 {
101  ScriptableContainer::upgradeVersion(deserializedVersion);
102  if(deserializedVersion < "0.0.1")
103  {
104  //Before 0.0.1, newClick on operatorClick info had the effect of reading out the current
105  //"clicked" value, and also resetting it. In 0.0.1, we changed it such that the user
106  //must reset it manually. Since there aren't too many experiments yet, we know how and

108  //(operatorClick.newClick && !(operatorClick.newClick = false)) in all places that it was
109  //used.
110  QString frameScript = propertyContainer_->getPropertyValue("Script").toString();
