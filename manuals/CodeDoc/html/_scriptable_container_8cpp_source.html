43 void ScriptableContainer::addScriptable(QWeakPointer<Scriptable> scriptable)
44 {
45  //If the scriptable is empty (ie. Whatever the weak pointer was pointing to has been deleted) return
46  if(scriptable.isNull())
47  return;
48  QSharedPointer<Scriptable> strScriptable = scriptable.toStrongRef();
49  QSharedPointer<AssociateElement> assocElement = strScriptable.dynamicCast<AssociateElement>();
50  if(assocElement)
51  {
52  //The input scriptable is part of an Associate. Only add it if its part of the same Associate
53  //as me
54  if(!isAssociateElement_)
55  unboundScriptables_.push_back(scriptable);
56  else
57  {
58  QUuid scriptableAssocId = assocElement->getAssociateId();
59  AssociateElement* thisAsAssocElem = dynamic_cast<AssociateElement*>(this);
60  if(thisAsAssocElem->getAssociateId() != scriptableAssocId)
61  {
62  unboundScriptables_.push_back(scriptable);
63  }
64  else
65  scriptables_.push_back(scriptable);
66  }
67  }
68  else
69  scriptables_.push_back(scriptable);
70 
71  //If we added a new scriptable, scripting is no longer properly initialized.
72  scriptingInitialized_ = false;
73  //If the new scriptable's name was edited, we'll need to reinitialize scripting again.
74  connect(scriptable.toStrongRef().data(),SIGNAL(nameEdited()),this,SLOT(deinitScripting()));
75  foreach(QSharedPointer<ScriptableContainer> child,scriptableContainers_)
76  {
77  child->addScriptable(scriptable);
78  }

134 bool ScriptableContainer::initScripting(bool enableDebugging)
135 {
136  if( scriptingInitialized_ )
137  return true;
138 
139  debuggingEnabled_ = enableDebugging;
140  //Initialize scripting on this ScriptableContainer
141  if(canHaveScripts())
142  {
143  if(hasScripts())
144  {
145  //Looks like we're going to need a script engine
146  //create the engine
147  qsEngine_ = QSharedPointer<QScriptEngine>(new QScriptEngine());
148 
149  if(debuggingEnabled_)
150  {
151  //Attach the debugger to the script engine
152  qsEngineDebugger_ = QSharedPointer<QScriptEngineDebugger>(new QScriptEngineDebugger());
153  qsEngineDebugger_->attachTo(qsEngine_.data());
154 
155  //Add result check function to qsEngine
156  qsEngine_->globalObject().setProperty("checkScriptResults",qsEngine_->newFunction(checkScriptResults));
157  }
158 
159 
160  //Bind all scriptable to the script engine (parameters and ui elements)
161  if(!bindScriptablesToScriptEngine(*qsEngine_))
162  return false;
163 
164  QMap<QString,QString> scriptMap = getScripts();
165  scriptPrograms_.clear();
166  for(QMap<QString,QString>::iterator it = scriptMap.begin();it!=scriptMap.end();it++)
167  {
168  //if(qsEngine_->globalObject().property(it.key()).isValid())
169  //{
170  // // There is already a scriptable component with this name.
171  // Q_ASSERT(false);
172  // return false;
173  //}
174 
175  //Make a Qt Script Function out of the script and its name
176  QString scriptTag = it.value();
177  QString script = propertyContainer_->getPropertyValue(scriptTag).toString();
178  Q_ASSERT(script.length());
179  Q_ASSERT(!scriptPrograms_.contains(it.key()));
180  QString programText;
181  if(!debuggingEnabled_)
182  {
183  //Add a version of the script with return value checking.
184  programText = QString("function script(){\n\n//YOUR CODE STARTS HERE-------\n%1\n//YOUR CODE ENDS HERE----------\n\n}\nscript()").arg(script);
185  }
186  else
187  {
188  programText = QString("function script(){\n\n//YOUR CODE STARTS HERE-------\n%1\n//YOUR CODE ENDS HERE----------\n\n}\nvar returnVal = script();\n").arg(script) + QString("checkScriptResults(\"%1\",returnVal);\nreturnVal;").arg(it.key());
189  }
190  scriptPrograms_[it.key()] = QSharedPointer<QScriptProgram>(new QScriptProgram(programText));
191  }
192  }
193  //Connect this object's edited slot to deinitScripting, because if one of its scripts changes, we'll need to reinsert
194  //it into the scripting environment
195  //Calling disconnect first assures us that we don't connect the signal more than once and end up calling deinitScripting
196  //more than once when edited() is emitted.
197  disconnect(this,SIGNAL(edited()),this,SLOT(deinitScripting()));
198  connect(this,SIGNAL(edited()),this,SLOT(deinitScripting()));
199  }
200 
201  //Inform child classes that the scriptable container has been reinitialized.
202  scriptableContainerWasReinitialized();
203 
204  //Intialize Scripting on all child ScriptableContainers
205  foreach(QSharedPointer<ScriptableContainer> child,scriptableContainers_)
206  {
207  child->initScripting(debuggingEnabled_);
208  }
209  scriptingInitialized_ = true;
210  return true;
211 }

428 {

431 
432  //Check Script Syntax

463 bool ScriptableContainer::executeSearchAlgorithm(SearchRequest searchRequest)
464 {
465  if(Scriptable::executeSearchAlgorithm(searchRequest))
466  return true;
467  switch(searchRequest.type)
468  {
469  case SearchRequest::SCRIPT:
470  {
471  //Check to see if this object contains any non-empty scripts
472  QMap<QString,QString> scriptMap = getScripts();
473  QSharedPointer<Property> currScriptProp;
474  QString currScript;
475  foreach(QString key,scriptMap.keys())
476  {
477  QString propName = scriptMap.value(key);
478  currScriptProp = propertyContainer_->getProperty(propName);
479  if(currScriptProp.isNull())
480  continue;
481  if(!currScriptProp->value().toString().isEmpty())
482  return true;
483  }
484  }
485  break;
486  case SearchRequest::STRING:
487  {
488  //Search through my scripts to see if any contain the query string
489  QMap<QString,QString> scriptMap = getScripts();
490  QSharedPointer<Property> currScriptProp;
491  QString currScript;
492  foreach(QString key,scriptMap.keys())
493  {
494  QString propName = scriptMap.value(key);
495  //Search script contents
496  currScriptProp = propertyContainer_->getProperty(propName);
497  if(currScriptProp.isNull())
498  continue;
499  if(currScriptProp->value().toString().contains(searchRequest.query,searchRequest.caseSensitive?Qt::CaseSensitive:Qt::CaseInsensitive))
500  return true;
501  }
502  }
503  break;
504  default:
505  return false;
506  };
507  return false;
508 }

516 bool ScriptableContainer::bindScriptablesToScriptEngine(QScriptEngine &engine)
517 {
518  //If any scriptables have the same name, the one with the shorter path
519  //should not be bound. Create a list with all bindable scriptables.
520  QHash<QString,QSharedPointer<Scriptable>> scriptableHash;
521  QString currName;
522  foreach(QSharedPointer<Scriptable> scriptable, scriptables_)
523  {
524  currName = scriptable->getName();
525  if(!scriptableHash.contains(currName))
526  {
527  scriptableHash[currName] = scriptable;
528  }
529  else
530  {
531  if(scriptableHash.value(currName)->getPath().size() < scriptable->getPath().size())
532  scriptableHash[currName] = scriptable;
533  }
534  }
535  QList<QSharedPointer<Scriptable>> bindableScriptables = scriptableHash.values();
536  foreach(QSharedPointer<Scriptable> scriptable, bindableScriptables)
537  {
538  if(!scriptable->bindToScriptEngine(engine))
539  return false;
540  }
541  return true;
