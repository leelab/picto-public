44 void ScriptableContainer::addScriptable(QWeakPointer<Scriptable> scriptable)
45 {
46  //If the scriptable is empty (ie. Whatever the weak pointer was pointing to has been deleted) return

52  {
53  //The input scriptable is part of an Associate. Only add it if its part of the same Associate
54  //as me
55  if (!isAssociateElement_)
56  unboundScriptables_.push_back(scriptable);
57  else
58  {

61  if (thisAsAssocElem->getAssociateId() != scriptableAssocId)
62  {
63  unboundScriptables_.push_back(scriptable);
64  }
65  else
66  {
67  scriptables_.push_back(scriptable);
68  }
69  }
70  }
71  else

73  scriptables_.push_back(scriptable);
74  }

77  scriptingInitialized_ = false;

80  foreach(QSharedPointer<ScriptableContainer> child, scriptableContainers_)

139 bool ScriptableContainer::initScripting(bool enableDebugging)
140 {
141  if (scriptingInitialized_)
142  return true;
143 
144  debuggingEnabled_ = enableDebugging;
145  //Initialize scripting on this ScriptableContainer
146  if (canHaveScripts())
147  {
148  if (hasScripts())
149  {
150  //Looks like we're going to need a script engine
151  //create the engine
152  qsEngine_ = QSharedPointer<QScriptEngine>(new QScriptEngine());
153 
154  if (debuggingEnabled_)
155  {
156  //Attach the debugger to the script engine
157  qsEngineDebugger_ = QSharedPointer<QScriptEngineDebugger>(new QScriptEngineDebugger());
158  qsEngineDebugger_->attachTo(qsEngine_.data());
159 
160  //Add result check function to qsEngine
161  qsEngine_->globalObject().setProperty("checkScriptResults", qsEngine_->newFunction(checkScriptResults));
162  }
163 
164 
165  //Bind all scriptable to the script engine (parameters and ui elements)
166  if (!bindScriptablesToScriptEngine(*qsEngine_))
167  return false;
168 
169  QMap<QString, QString> scriptMap = getScripts();
170  scriptPrograms_.clear();
171  for (QMap<QString, QString>::iterator it = scriptMap.begin(); it != scriptMap.end(); it++)
172  {
173  //if(qsEngine_->globalObject().property(it.key()).isValid())
174  //{
175  // // There is already a scriptable component with this name.
176  // Q_ASSERT(false);
177  // return false;
178  //}
179 
180  //Make a Qt Script Function out of the script and its name
181  QString scriptTag = it.value();
182  QString script = propertyContainer_->getPropertyValue(scriptTag).toString();
183  Q_ASSERT(script.length());
184  Q_ASSERT(!scriptPrograms_.contains(it.key()));
185  QString programText;
186  if (!debuggingEnabled_)
187  {
188  //Add a version of the script with return value checking.
189  programText = QString("function script(){\n\n//YOUR CODE STARTS HERE-------\n%1\n//YOUR CODE ENDS HERE----------\n\n}\nscript()").arg(script);
190  }
191  else
192  {
193  programText = QString("function script(){\n\n//YOUR CODE STARTS HERE-------\n%1\n//YOUR CODE ENDS HERE----------\n\n}\nvar returnVal = script();\n").arg(script) + QString("checkScriptResults(\"%1\",returnVal);\nreturnVal;").arg(it.key());
194  }
195  scriptPrograms_[it.key()] = QSharedPointer<QScriptProgram>(new QScriptProgram(programText));
196  }
197  }
198  //Connect this object's edited slot to deinitScripting, because if one of its scripts changes, we'll need to reinsert
199  //it into the scripting environment
200  //Calling disconnect first assures us that we don't connect the signal more than once and end up calling deinitScripting
201  //more than once when edited() is emitted.
202  disconnect(this, SIGNAL(edited()), this, SLOT(deinitScripting()));
203  connect(this, SIGNAL(edited()), this, SLOT(deinitScripting()));
204  }
205 
206  //Inform child classes that the scriptable container has been reinitialized.
207  scriptableContainerWasReinitialized();
208 
209  //Intialize Scripting on all child ScriptableContainers
210  foreach(QSharedPointer<ScriptableContainer> child, scriptableContainers_)
211  {
212  child->initScripting(debuggingEnabled_);
213  }
214  scriptingInitialized_ = true;
215 
216  scriptableContainerCustomInitialization();
217 
218  return true;
219 }

435 bool ScriptableContainer::validateObject(QSharedPointer<QXmlStreamReader> xmlStreamReader)
436 {
437  if (!Scriptable::validateObject(xmlStreamReader))
438  return false;
439 
440  //Check Script Syntax
441  QMap<QString, QString> scriptMap = getScripts();
442  bool scriptsAreOkay = true;
443  for (QMap<QString, QString>::iterator it = scriptMap.begin(); it != scriptMap.end(); it++)
444  {
445  QString scriptTag = it.value();
446  QString script = propertyContainer_->getPropertyValue(scriptTag).toString();
447  QString function = "function " + it.key() + "() { " + script + "\n}";
448  QScriptSyntaxCheckResult res = QScriptEngine::checkSyntax(function);
449  if (res.state() != QScriptSyntaxCheckResult::Valid)
450  {
451  QString errorMessage = QString("Syntax Error in \"%1\"(Line:%2)")

454  .arg(res.errorMessage());
455  if (res.errorMessage().size())
456  errorMessage.append(" : " + res.errorMessage());
457  addError(errorMessage);
458  scriptsAreOkay = false;
459  }
460  if (!scriptsAreOkay)
461  return false;
462  }
463  //We don't check if multiple scriptables have the same name, because our bindScriptablesToScriptEngine()
464  //function makes sure that only the one with the longest path is added to the scriptEngine
465  return true;

471 bool ScriptableContainer::executeSearchAlgorithm(SearchRequest searchRequest)
472 {
473  if (Scriptable::executeSearchAlgorithm(searchRequest))
474  return true;
475  switch (searchRequest.type)
476  {
477  case SearchRequest::SCRIPT:
478  {
479  //Check to see if this object contains any non-empty scripts
480  QMap<QString, QString> scriptMap = getScripts();
481  QSharedPointer<Property> currScriptProp;
482  QString currScript;
483  foreach(QString key, scriptMap.keys())
484  {
485  QString propName = scriptMap.value(key);
486  currScriptProp = propertyContainer_->getProperty(propName);
487  if (currScriptProp.isNull())
488  continue;
489  if (!currScriptProp->value().toString().isEmpty())
490  return true;
491  }
492  }
493  break;
494  case SearchRequest::STRING:
495  {
496  //Search through my scripts to see if any contain the query string
497  QMap<QString, QString> scriptMap = getScripts();
498  QSharedPointer<Property> currScriptProp;
499  QString currScript;
500  foreach(QString key, scriptMap.keys())
501  {
502  QString propName = scriptMap.value(key);
503  //Search script contents
504  currScriptProp = propertyContainer_->getProperty(propName);
505  if (currScriptProp.isNull())
506  continue;
507  if (currScriptProp->value().toString().contains(searchRequest.query, searchRequest.caseSensitive ? Qt::CaseSensitive : Qt::CaseInsensitive))
508  return true;
509  }
510  }
511  break;
512  default:
513  return false;
514  };
515  return false;
516 }

524 bool ScriptableContainer::bindScriptablesToScriptEngine(QScriptEngine &engine)
525 {
526  //If any scriptables have the same name, the one with the shorter path
527  //should not be bound. Create a list with all bindable scriptables.
528  QHash<QString, QSharedPointer<Scriptable>> scriptableHash;
529  QString currName;
530  foreach(QSharedPointer<Scriptable> scriptable, scriptables_)
531  {
532  currName = scriptable->getName();
533  if (!scriptableHash.contains(currName))
534  {
535  scriptableHash[currName] = scriptable;
536  }
537  else
538  {
539  if (scriptableHash.value(currName)->getPath().size() < scriptable->getPath().size())
540  scriptableHash[currName] = scriptable;
541  }
542  }
543  QList<QSharedPointer<Scriptable>> bindableScriptables = scriptableHash.values();
544  foreach(QSharedPointer<Scriptable> scriptable, bindableScriptables)
545  {
546  if (!scriptable->bindToScriptEngine(engine))
547  return false;
548  }
549  return true;
