313 void AnalysisOptionWidget::deleteSelectedAnalysis()
314 {
315  if(!isEnabled())
316  return;
317  Q_ASSERT(editorState_ && editorState_->getTopLevelAsset());
318  QSharedPointer<Analysis> selectedAnalysis = getSelectedAnalysis();
319  if(!selectedAnalysis)
320  return;
321 
322  int ret = QMessageBox::warning(this, tr("Picto"),
323  QString("Are you sure you want to delete the %1 Analysis?").arg(selectedAnalysis->getName()),
324  QMessageBox::Yes | QMessageBox::Cancel,
325  QMessageBox::Cancel);
326  if(ret != QMessageBox::Yes)
327  return;
328 
329  //Prevent the changes in this range from calling EditorState::SetWindowAsset
330  updateLocker_++;
331 
332  //Detach analysis from experiment
333  QUuid currAnalysisId = selectedAnalysis->getAssociateId();
334  editorState_->getTopLevelAsset().staticCast<DataStore>()->ClearAssociateDescendants(currAnalysisId);
335 
336  //Remove analysis from design
337  int analysisIndexToRemove = selectAnalysisBox_->currentIndex() - 1;
338  Q_ASSERT(analysisIndexToRemove < designRoot_->getNumAnalyses());
339  if(!designRoot_->removeAnalysis(analysisIndexToRemove))
340  Q_ASSERT(false);
341 
342  //Reset experiment so that the removal of the analysis (setDeleted() is used) takes effect.
343  //This will result in updateAnalysisList() getting called.
344  editorState_->triggerExperimentReset();
345 
346  //Unlock calls to EditorState::SetWindowAsset
347  updateLocker_--;
348 
349  //Set current analysis to none.
350  setSelectedAnalysis(QUuid());
351 }
