180 void AnalysisOptionWidget::updateAnalysisList()
181 {
182  if(!isEnabled())
183  return;
184  if(updatingList_)
185  return;
186  updatingList_ = true;
187  QUuid lastAnalysisId;
188  if(selectedAnalysis_)
189  lastAnalysisId = selectedAnalysis_->getAssociateId();
190  ignoreSelectBoxEvents_ = true;
191  int newSelectionIndex = 0;
192  selectAnalysisBox_->clear();
193  selectAnalysisBox_->addItem("None");
194  QUuid currAnalysisId;
195  for(int i=0;i<designRoot_->getNumAnalyses();i++)
196  {
197  currAnalysisId = designRoot_->getAnalysis(i).staticCast<Analysis>()->getAssociateId();
198  selectAnalysisBox_->addItem(designRoot_->getAnalysis(i)->getName(), currAnalysisId);
199  if(!lastAnalysisId.isNull() && currAnalysisId == lastAnalysisId)
200  newSelectionIndex = i+1;
201  }
202 
203  //We set the current index to -1 so that it will cause selectedIndexChanged to get triggered when set.
204  selectAnalysisBox_->setCurrentIndex(-1);
205  ignoreSelectBoxEvents_ = false;
206  selectAnalysisBox_->setCurrentIndex(newSelectionIndex);
207  updatingList_ = false;
208 }

241 void AnalysisOptionWidget::selectedIndexChanged(int selectedIndex)
242 {
243  if(!isEnabled())
244  return;
245  if(ignoreSelectBoxEvents_)
246  return;
247 
248  Q_ASSERT(designRoot_);
249  Q_ASSERT(selectedIndex <= designRoot_->getNumAnalyses());
250 
251  QSharedPointer<Analysis> lastCurrAnalysis = editorState_->getCurrentAnalysis();
252 
253  //If no analysis was selected
254  if(selectedIndex <= 0)
255  {
256  selectAnalysisBox_->setEditable(false);
257  deleteAnalysisAction_->setEnabled(false);
258 
259  //Erase the current selected analysis pointer
260  selectedAnalysis_.clear();
261  }
262  else
263  {
264  selectAnalysisBox_->setEditable(true);
265  deleteAnalysisAction_->setEnabled(true);
266 
267  //Update the current selected analysis
268  selectedAnalysis_ = designRoot_->getAnalysis(selectedIndex-1).staticCast<Analysis>();
269  }
270  if(selectedAnalysis_ != lastCurrAnalysis)
271  {
272  //Tell the editorState about the new current analysis
273  editorState_->setCurrentAnalysis(selectedAnalysis_);
274  }

302 void AnalysisOptionWidget::deleteSelectedAnalysis()
303 {
304  if(!isEnabled())
305  return;
306  Q_ASSERT(editorState_ && editorState_->getTopLevelAsset());
307  QSharedPointer<Analysis> selectedAnalysis = getSelectedAnalysis();
308  if(!selectedAnalysis)
309  return;
310 
311  int ret = QMessageBox::warning(this, tr("Picto"),
312  QString("Are you sure you want to delete the %1 Analysis?").arg(selectedAnalysis->getName()),
313  QMessageBox::Yes | QMessageBox::Cancel,
314  QMessageBox::Cancel);
315  if(ret != QMessageBox::Yes)
316  return;
317 
318  //Detach analysis from experiment
319  QUuid currAnalysisId = selectedAnalysis->getAssociateId();
320  editorState_->getTopLevelAsset().staticCast<DataStore>()->ClearAssociateDescendants(currAnalysisId);
321 
322  //Remove analysis from design
323  int analysisIndexToRemove = selectAnalysisBox_->currentIndex() - 1;
324  Q_ASSERT(analysisIndexToRemove < designRoot_->getNumAnalyses());
325  if(!designRoot_->removeAnalysis(analysisIndexToRemove))
326  Q_ASSERT(false);
327 
328  //Reset experiment so that the removal of the analysis (setDeleted() is used) takes affect.
329  //This will result in updateAnalysisList() getting called.
330  editorState_->triggerExperimentReset();
331 
332  //Set current analysis to none.
333  setSelectedAnalysis(QUuid());
