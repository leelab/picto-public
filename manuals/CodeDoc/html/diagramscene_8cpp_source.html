138 QRectF DiagramScene::getDefaultZoomRect()
139 {
140  if(!startBar_)
141  return itemsBoundingRect();
142  QRectF returnVal;
143  //Iterate through all items in scene
144  QGraphicsItem* itemBuffer;
145  foreach(QGraphicsItem* item,items())
146  {
147  //If item isn't the start bar or in the startbar or a transition, add it to the returnVal rectangle
148  itemBuffer = item;
149  bool inStartBar = false;
150  while(itemBuffer)
151  {
152  if(itemBuffer == startBar_)
153  {
154  inStartBar = true;
155  break;
156  }
157  itemBuffer = itemBuffer->parentItem();
158  }
159  if(inStartBar)
160  continue;
161  if(dynamic_cast<Arrow*>(item))
162  continue;
163 
164  QRectF itemBoundRect = item->boundingRect();

166  if((itemPos.x() < 0) || itemPos.y() < 0)
167  continue;

169  //point, but removing items with pos less than zero and seting top left of bounding rect to pos
170  //seemed to fix the problem.

172  if(returnVal.isNull())
173  returnVal = itemBoundRect;
174  else
175  returnVal = returnVal.united(itemBoundRect);
176  }
177  if(returnVal.x() < 0)
178  returnVal.setX(0);
179  if(returnVal.y() < 0)

193 void DiagramScene::setSceneAsset(QSharedPointer<Asset> asset)
194 {
195  QSharedPointer<DataStore> dataStore(asset.staticCast<DataStore>());
196 
197  clear();
198  //Add a start bar
199  startBar_ = NULL;
200  newItemIndex_ = 1;
201  QStringList childTypes = dataStore->getDefinedChildTags();
202  //Set the default item placement location to the top left of the scene
203  QPointF childAssetLoc(sceneRect().topLeft().x()+DEFAULT_ITEM_BORDER_BUFFER,sceneRect().topLeft().y()+DEFAULT_ITEM_BORDER_BUFFER);
204  QList<QSharedPointer<Transition>> transitions;
205  QList<DiagramItem*> diagItems;
206  if(dataStore->inherits("Picto::StateMachine") || dataStore->inherits("Picto::State"))
207  {
208  startBar_ = new StartBarItem("",editorState_,NULL,dataStore);
209  diagItems.push_back(startBar_);
210  addItem(startBar_);
211  //Put the default placement location at the top right of the start bar
212  childAssetLoc = startBar_->boundingRect().topRight()+QPointF(DEFAULT_ITEM_BORDER_BUFFER,DEFAULT_ITEM_BORDER_BUFFER);
213  }
214  //Add all experiment assets
215  foreach(QString childType, childTypes)
216  {
217  QList<QSharedPointer<Asset>> assets = dataStore->getGeneratedChildren(childType);
218  foreach(QSharedPointer<Asset> childAsset,assets)
219  {
220  if( childAsset->inherits("Picto::Transition") )
221  {
222  transitions.push_back(childAsset.staticCast<Transition>());
223  continue;
224  }
225  DiagramItem* diagItem = insertDiagramItem(childAsset,childAssetLoc);
226  if(diagItem)
227  {
228  //If something here has a default name, we should start after its index.
229  if(diagItem->getName().contains(QRegExp("NewItem[1-9]")))
230  {
231  int itemIndex = diagItem->getName().remove("NewItem").toInt();
232  if(itemIndex >= newItemIndex_)
233  newItemIndex_ = itemIndex+1;
234  }
235 
236  //Now add it to the list and update the default placement location to the top
237  //right of the currently added items
238  diagItems.push_back(diagItem);
239  childAssetLoc = QPointF(getDefaultZoomRect().right(),diagItem->pos().y());
240  }
241  }
242  }
243 
244  //Add all transitions
245  foreach(QSharedPointer<Transition> transition,transitions)
246  {
247  DiagramItem* start = NULL;
248  DiagramItem* end = NULL;
249  int sourceId = transition->getSourceId();
250  int sourceResultId = transition->getSourceResultId();
251  int destId = transition->getDestinationId();
252  if(!sourceId && !sourceResultId && startBar_)
253  start = startBar_;
254  WireableItem* wireItem;
255  QSharedPointer<Asset> asset;
256  QSharedPointer<Asset> resultAsset;
257  foreach(DiagramItem* diagItem,diagItems)
258  {
259  wireItem = dynamic_cast<WireableItem*>(diagItem);
260  if(!wireItem)
261  continue;
262  asset = wireItem->getAsset();
263  if(asset.isNull())
264  continue;
265  if(!start && (sourceId == asset->getAssetId()))
266  {
267  QList<DiagramItem*> results = wireItem->getArrowSources();
268  foreach(DiagramItem* result,results)
269  {
270  resultAsset = static_cast<ArrowSourceItem*>(result)->getAsset();
271  if(resultAsset && resultAsset->getAssetId() == sourceResultId)
272  {
273  start = result;
274  break;
275  }
276  }
277  }
278  if(!end && (destId == asset->getAssetId()))
279  {
280  end = wireItem->getArrowDest();
281  }
282  if(start && end)
283  {
284  break;
285  }
286  }
287  if(start && end)
288  {
289  insertTransition(start,end,transition);
290  }
291  }
292 
293  //Add Analysis assets for the current analysis
294  QSharedPointer<Analysis> currAnalysis = editorState_->getCurrentAnalysis();
295  if(!currAnalysis.isNull())
296  {
297  childTypes = dataStore->getAssociateChildTags(currAnalysis->getAssociateId());
298  foreach(QString childType, childTypes)
299  {
300  QList<QSharedPointer<Asset>> assets = dataStore->getAssociateChildren(currAnalysis->getAssociateId(),childType);
301  foreach(QSharedPointer<Asset> childAsset,assets)
302  {
303  DiagramItem* diagItem = insertDiagramItem(childAsset,childAssetLoc);
304  if(diagItem)
305  {
306  //If something here has a default name, we should start after its index.
307  if(diagItem->getName().contains(QRegExp("NewItem[1-9]")))
308  {
309  int itemIndex = diagItem->getName().remove("NewItem").toInt();
310  if(itemIndex >= newItemIndex_)
311  newItemIndex_ = itemIndex+1;
312  }
313 
314  //Now add it to the list and update the default placement location to the top
315  //right of the currently added items
316  diagItems.push_back(diagItem);
317  childAssetLoc = QPointF(getDefaultZoomRect().right(),diagItem->pos().y());
318  }
319  }
320  }
321  }
322  editorState_->setWindowItemsLoaded();
323 }

418  }
419  //delete qgraphicsitem_cast<DiagramItem *>(item);

560 void DiagramScene::mouseReleaseEvent(QGraphicsSceneMouseEvent *mouseEvent)
561 {
562  //Only respond to left buttons
563  if (mouseEvent->button() != Qt::LeftButton)
564  {
565  QGraphicsScene::mouseReleaseEvent(mouseEvent);
566  return;
567  }
568 
569  //If we're in Select mode, check if a new asset needs to be generated.
570  //If it does, generate it.
571  if(editorState_->getEditMode() == EditorState::PlaceItem)
572  {
573  QSharedPointer<Asset> newAsset = createNewAsset();
574  if(newAsset)
575  {
576  DiagramItem *newItem = insertDiagramItem(newAsset,mouseEvent->scenePos());
577  if(newItem)
578  {
579  if(newAsset.dynamicCast<UIEnabled>())
580  newAsset.staticCast<UIEnabled>()->setName(QString("NewItem%1").arg(newItemIndex_++));
581  editorState_->setLastActionUndoable();
582  editorState_->triggerItemInserted();
583  clearSelection();
584  newItem->setSelected(true);
585  }
586  }
587  }
588  else
589  //If a line is being drawn, check if we should add it or delete it.
590  //If we should add it, create the transition and add it.
591  if (line != 0 && editorState_->getEditMode() == EditorState::DrawLine)
592  {
593 
594  QList<QGraphicsItem *> startItems = items(line->line().p1());
595  for(QList<QGraphicsItem *>::iterator iter = startItems.begin();iter != startItems.end();)
596  {
597  if((*iter) == line)
598  iter = startItems.erase(iter);
599  else
600  iter++;
601  }
602 
603  QList<QGraphicsItem *> endItems = items(line->line().p2());
604  for(QList<QGraphicsItem *>::iterator iter = endItems.begin();iter != endItems.end();)
605  {
606  if((*iter) == line)
607  iter = endItems.erase(iter);
608  else
609  iter++;
610  }
611 
612  removeItem(line);
613  delete line;
614  line = NULL;
615 
616  //Ignore text graphics.
617  while(startItems.count() && (!dynamic_cast<ArrowPortItem*>(startItems.first())))
618  {
619  startItems.removeFirst();
620  }
621  while(endItems.count() && (!dynamic_cast<ArrowPortItem*>(endItems.first())))
622  {
623  endItems.removeFirst();
624  }
625 
626  if (startItems.count() > 0 && endItems.count() > 0 &&
627  startItems.first()->type() == DiagramItem::Type &&
628  endItems.first()->type() == DiagramItem::Type &&
629  startItems.first() != endItems.first()) {
630  DiagramItem *startItem =
631  qgraphicsitem_cast<DiagramItem *>(startItems.first());
632  DiagramItem *endItem =
633  qgraphicsitem_cast<DiagramItem *>(endItems.first());
634  insertTransition(startItem,endItem);
635  }
636  editorState_->setEditMode(EditorState::Select);
637  }
638 
639  line = 0;
640  QGraphicsScene::mouseReleaseEvent(mouseEvent);
641 }
