24 void PictoData::postDeserialize()
25 {
26  //If this PictoData has any StateMachineEditorData child assets, this object should be set as edited.
27  // ideally, the ObsoleteAsset child would just set itself deleted and that would
28  // propogate its edited state to here, the ObsoleteAsset can't though because
29  // the latest it can do that is in postDeserialize, and this PictoData has its
30  // edited flag set back to false after that point. This should be fixed at some point.
31  QList<QSharedPointer<Asset>> tagChildren = getGeneratedChildren("StateMachineEditorData");
32  if(tagChildren.size())
33  emit edited();
34 
35  //Add all associate roots to their hosts (this has the affect of adding all analyses to their tasks and all
36  // UIData to their experiments and analyses.
37  QList<QSharedPointer<Asset>> allAssociateRoots = getGeneratedChildren("Analysis");
38  allAssociateRoots.append(getGeneratedChildren("UIData"));
39  QSharedPointer<Asset> linkAsset;
40  QString feedback;
41  foreach(QSharedPointer<Asset> associateRoot,allAssociateRoots)
42  {
43  linkAsset = associateRoot.staticCast<AssociateRoot>()->getLinkableAsset();
44  //If the Associate Root has nothing to link to, delete it.
45  if(!linkAsset)
46  {
47  associateRoot->setDeleted();
48  continue;
49  }
50  associateRoot.staticCast<AssociateRoot>()->LinkToAsset(linkAsset,feedback);
51  }
52 
53  //Now that the entire tree is set up, and the saved associate roots are attached to their linked elements based on
54  // saved asset ids, make sure that all assets have unique IDs and the DesignConfig object is tracking them. This way,
55  // if an assetId needs to be changed for some reason, the changed id will propegate to any linked associates, and
56  // resulting hostId changes will propogate to linked AssociateRoots as well.
57  //Since Associates to AssociateHost linking does not use the design config, it is okay that we disallowIdDuplication
58  // after connecting Associates.
59  designConfig_->disallowIdDuplication();
60 
61  //Now call inherited postDeserialize which will actually call fixDuplicatedAssetIds on the designConfig
62  DataStore::postDeserialize();
63 
64  QList<QSharedPointer<Asset>> assocRootHosts = getGeneratedChildren("Experiment");
65  assocRootHosts.append(getGeneratedChildren("Analysis"));
66 
69  //assocRootHosts.append(getGeneratedChildren("AnalysisContainer"));
70  AssociateRootHost* assocRootHost;
71  foreach(QSharedPointer<Asset> associateRootHostAsset,assocRootHosts)
72  {
73  //If the host has no UIData AssociateRoot attached
74  assocRootHost = dynamic_cast<AssociateRootHost*>(associateRootHostAsset.data());
75  Q_ASSERT(assocRootHost);
76  if(!assocRootHost->getAssociateRoot("UIData"))
77  {
78  //Create a new UIData object and attach it to this host.
79  QUuid hostId = assocRootHost->getHostId();
80  QSharedPointer<Asset> newUIData = createChildAsset("UIData",QString(),QString());
81  QString feedback;
82  newUIData.staticCast<AssociateRoot>()->LinkToAsset(associateRootHostAsset,feedback);
83  }
84  }
85 
86  //We handle version upgrading after attaching AssociateRoots so that those Associate roots
87  // can be modified, if necessary, as part of the upgrade process and so that changes to their
88  // hosts will not cause them to get disconnected (hostId change will propegate to associateRoot).
89  QString syntaxVer = propertyContainer_->getPropertyValue("SyntaxVersion").toString();
90  if(syntaxVer != DESIGNSYNTAXVERSION)
91  {
92  designConfig_->setDeserializedVersion(syntaxVer);
93  upgradeVersion(syntaxVer);
94  propertyContainer_->setPropertyValue("SyntaxVersion",DESIGNSYNTAXVERSION);
95  }
96  propertyContainer_->getProperty("SyntaxVersion")->setVisible(false);
97 }
