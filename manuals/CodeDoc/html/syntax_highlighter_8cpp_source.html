131  for(it = myData->begin();it!=myData->end();it++)
132  {
133  it->start_ = pos;
134  }
135  while (pos < len)
136  { // Loop through text block. In the context of the construct on top of the stack there are permissable
137  // start strings that start new constructs and stop strings that stop this construct. Search for these
138  // strings and format accordingly.
139 
140  ConstructFlag flag;
141  flag.start_ = -2; // -2 indicates no match
142  foreach(int constructID,constructs_[myData->top().id_].subConstructs_)
143  { // Loop through all constructs that can be within the top construct, looking for their start string. Keep the longest one that is allowed.
144  matchStr = constructs_[constructID].opener_;
145  if(text.mid(pos,matchStr.length()) == matchStr)
146  { // We found this match string. Check if this works as a new construct
147  if((flag.start_<0) || (matchStr.length() > constructs_[flag.id_].opener_.length()))
148  {
149  flag.id_ = constructID;
150  flag.start_ = pos;
151  }
152  }
153  }
154  if(myData->size() > 1)
155  { // If the top construct in the stack isn't the BASE construct, look for its end string
156  matchStr = constructs_[myData->top().id_].closer_;
157  for(it = myData->end()-1;it != myData->begin();it--)
158  { // Move down stack until we find construct who closer can be used
159  matchStr = constructs_[it->id_].closer_;
160  if(matchStr != USE_NEXT_CONSTRUCT)
161  break;
162  }

166  {

170  }
171  }

180  myData->push(flag);

182  }

192  }
