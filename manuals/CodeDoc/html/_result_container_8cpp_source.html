40 bool ResultContainer::hasEditableDescendants()
41 {
42  int maxResults = 0;
43  int currMax;
44  //Add up maxAssets of all types of all resultFactories and put it in maxResults.
45  //If any of them have a -1 max, it means that there is no limit and therefore
46  //There are user addable results and we should return true.
47  foreach(QSharedPointer<AssetFactory> assetFactory,resultFactoryByType_)
48  {
49  foreach(QString type,assetFactory->getTypes())
50  {
51  currMax = assetFactory->getMaxAssets(type);
52  if(currMax < 0)
53  return true;
54  maxResults += currMax;
55  }
56  }
57 
58  //Required assets are not editable, all others are. If the number of required results equals
59  //maxResults, there are no editable descendants; otherwise, there are.
60  int numRequiredResults = requiredResults_.size();
61  Q_ASSERT(numRequiredResults <= maxResults);
62  if(numRequiredResults == maxResults)
63  return false;
64  return true;
65 }

89 void ResultContainer::addRequiredResult(QString resultName)
90 {
91  if(maxOptionalResults_[""] != -1)
92  {
93  //This will increase the max results value in the factory according to
94  //the new number of required results of this type. If we didn't do
95  //this, we couldn't get a new result from the factory below.
96  //Note: if we didn't add the new result to requiredResults_ below, this
97  //would make the factories max Assets inconsistant with requireResults_.
98  resultFactoryByType_[""]->setMaxAssets(resultFactoryByType_[""]->getMaxAssets()+1);
99  resultFactoryByType_[""]->setMinAssets(resultFactoryByType_[""]->getMinAssets()+1);
100  }
101  QSharedPointer<Result> requiredResult = createChildAsset("Result","",QString()).staticCast<Result>();
102  Q_ASSERT(requiredResult);
103  requiredResult->setName(resultName);
104  requiredResult->getPropertyContainer()->getProperty("Name")->setGuiEditable(false);
105  requiredResults_.insert("",requiredResult);
106 }

137 void ResultContainer::postDeserialize()
138 {
139  ScriptableContainer::postDeserialize();
140  //Check child list to see if any of the requiredResults were serialized in, if not, add them to the child
141  //list.
142  QList<QSharedPointer<Asset>> deserializedResults = getGeneratedChildren("Result");
143  for(QMultiMap<QString,QSharedPointer<Result>>::iterator it = requiredResults_.begin();it!=requiredResults_.end();it++)
144  {
145  bool found = false;
146  foreach(QSharedPointer<Asset> deserializedResult,deserializedResults)
147  {
148  if(deserializedResult->getName() == it.value()->getName())
149  {
150  found = true;
151  break;
152  }
153  }
154  if(!found)
155  {
156  AddChild("Result",it.value());
157  //Since the requiredResults weren't generated by the factory, we need to increment the number
158  //of generated results so that the factory will no if it can generate any more of the given type.
159  resultFactoryByType_[it.key()]->setGeneratedAssets(resultFactoryByType_[it.key()]->getGeneratedAssets()+1);
160  }
161  }
162 
163  //Add all child results to results_ map.
164  results_.clear();
165  QList<QSharedPointer<Asset>> results = getGeneratedChildren("Result");
166  foreach(QSharedPointer<Asset> result,results)
167  {
168  results_[result->getName()] = result.staticCast<Result>();
169  }
170 }
