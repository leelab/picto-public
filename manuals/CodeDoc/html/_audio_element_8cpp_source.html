170 void AudioElement::updateFileObject(QString fileElementName)
171 {
172  QList<QWeakPointer<Scriptable>> scriptables = getScriptableList();
173  if((!fileObj_.isNull()) && fileObj_.toStrongRef()->getName() == fileElementName)
174  return;
175  foreach(QWeakPointer<Scriptable> scriptable,scriptables)
176  {
177  if(scriptable.isNull())
178  continue;
179  if(scriptable.toStrongRef()->getName() == fileElementName && scriptable.toStrongRef()->inherits("Picto::AudioFileParameter"))
180  {
181  QSharedPointer<PreloadedSound> sound = (scriptable.toStrongRef().staticCast<AudioFileParameter>())->createSound();
182  cachedSounds_[fileElementName] = sound;
183  foreach(QSharedPointer<MixingSample> mixSamp,mixingSamples_.values())
184  {
185  mixSamp->setSoundEffect(sound);
186  }
187  return;
188  }
189  }
190  QScriptContext* ctx = context();
191  if(ctx)
192  ctx->throwError("Couldn't find file object: " + fileElementName);
193 }
