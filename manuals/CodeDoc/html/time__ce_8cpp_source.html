263 // time_t represents seconds since midnight January 1, 1970 UTC
264 // (coordinated universal time) in 32-bits Win32 FILETIME structure is 64-bit,
265 // which represents the number of 100-nanosecond (hns) intervals since
266 // January 1, 1601 UTC (coordinate universal time) the time difference
267 // between midnight January 1, 1970 and midnight January 1, 1601 is 11644473600 seconds
268 //
269 time_t_ce mktime_ce(struct tm *tptr)
270 {
271  SYSTEMTIME st;
272  TmToSystemTime(tptr, &st);
273  SetTz(&st);
274 
275  DWORD day = 0;
276  DWORD year = 0;
277  DWORD seconds = 0;
278  DWORD overflow;
279  DWORD tm_year;
280  DWORD yday, month;
281 
282  // see if seconds are < 0
283  while(tptr->tm_sec < 0)
284  {
285  // steal 60 seconds from the minutes
286  tptr->tm_sec += 60;
287  tptr->tm_min--;
288  }
289  // roll any seconds > 60 into the minutes
290  tptr->tm_min += tptr->tm_sec / 60;
291  // then crop them off
292  tptr->tm_sec %= 60;
293 
294  // see if minutes are < 0
295  while(tptr->tm_min < 0)
296  {
297  // steal 60 minutes from the hours
298  tptr->tm_min += 60;
299  tptr->tm_hour--;
300  }
301  // roll any minutes > 60 into the hours
302  tptr->tm_hour += tptr->tm_min / 60;
303  // then crop them off
304  tptr->tm_min %= 60;
305 
306  // see if hours are < 0
307  while(tptr->tm_hour < 0)
308  {
309  // steal 24 hours from the days
310  tptr->tm_hour += 24;
311  day--;
312  }
313  // keep any "excess" days (tm doesn't have a convenient place for this)
314  day += tptr->tm_hour / 24;
315  // crop
316  tptr->tm_hour %= 24;
317 
318  // roll any months > 12 into the years
319  tptr->tm_year += tptr->tm_mon / 12;
320  // then crop the off
321  tptr->tm_mon %= 12;
322 
323  // see if months are < 0
324  if (tptr->tm_mon < 0)
325  {
326  // steal 12 months from the years
327  tptr->tm_mon += 12;
328  tptr->tm_year--;
329  }
330 
331  // add number of days into the month to total day
332  day += (tptr->tm_mday - 1);
333 
334  // if days are < 0 then calculate the number of days
335  // checking to see if the month is a leap year month
336  while (day < 0)
337  {
338  // If months are < 0 then steal 12 months from number of years
339  // for the day calculation
340  if(--tptr->tm_mon < 0)
341  {
342  tptr->tm_year--;
343  tptr->tm_mon = 11;
344  }
345  day += _ytab[LEAPYEAR(YEAR0 + tptr->tm_year)][tptr->tm_mon];
346  }
347  // if day is greater then the number of days in the month
348  // subtract the number of days in the month and adjust the
349  // month
350  while (day >= _ytab[LEAPYEAR(YEAR0 + tptr->tm_year)][tptr->tm_mon])
351  {
352  day -= _ytab[LEAPYEAR(YEAR0 + tptr->tm_year)][tptr->tm_mon];
353  if (++(tptr->tm_mon) == 12)
354  {
355  tptr->tm_mon = 0;
356  tptr->tm_year++;
357  }
358  }
359  tptr->tm_mday = day + 1;
360  year = EPOCH_YR;
361 
362  // if year is less then 1970 then return error
363  if (tptr->tm_year < year - YEAR0) return (time_t) -1;
364 
365  seconds = 0;
366  day = 0; // Means days since day 0 now
367  overflow = 0;
368 
369  // Assume that when day becomes negative, there will certainly
370  // be overflow on seconds.
371  // The check for overflow needs not to be done for leapyears
372  // divisible by 400.
373  // The code only works when year (1970) is not a leapyear.
374  tm_year = tptr->tm_year + YEAR0;
375 
376  // make sure we are not past the max year for 32-bit number
377  if (TIME_MAX / 365 < tm_year - year) overflow++;
378 
379  // calculate number of days since EPOCH
380  day = (tm_year - year) * 365;
381 
382  if (TIME_MAX - day < (tm_year - year) / 4 + 1) overflow++;
383 
384  day += (tm_year - year) / 4 + ((tm_year % 4) && tm_year % 4 < year % 4);
385  day -= (tm_year - year) / 100 + ((tm_year % 100) && tm_year % 100 < year % 100);
386  day += (tm_year - year) / 400 + ((tm_year % 400) && tm_year % 400 < year % 400);
387 
388  // setup for calculation of the yday or Julian day since Jan 1
389  yday = month = 0;
390 
391  // add up the number of days for the preceding months
392  while (month < tptr->tm_mon)
393  {
394  yday += _ytab[LEAPYEAR(tm_year)][month];
395  month++;
396  }
397  // add the number of days in the current month
398  yday += (tptr->tm_mday - 1);
399 
400  // make sure the didn't overflow
401  if (day + yday < 0) overflow++;
402 
403  day += yday;
404 
405  // set the year day in the structure
406  tptr->tm_yday = yday;
407 
408  // calculate the weekday
409  tptr->tm_wday = (day + 4) % 7; // Day 0 was thursday (4)
410 
411  // start the seconds calculation by totaling the hours, min, seconds
412  seconds = ((tptr->tm_hour * 60L) + tptr->tm_min) * 60L + tptr->tm_sec;
413 
414  // make sure we are not going to overflow
415  if ((TIME_MAX - seconds) / SECS_DAY < day) overflow++;
416 
417  // calculate the number of seconds for the number of days
418  seconds += day * SECS_DAY;
419 
420  // Now adjust according to timezone and daylight saving time
421  if (((_localtime > 0) && (TIME_MAX - _localtime < seconds))
422  || ((_localtime < 0) && (seconds < -_localtime)))
423  overflow++;
424 
425  // Adjust for local time zone
426  seconds += _localtime;
427 
428  // return error if we are going to blow the max values
429  if (overflow) return (time_t_ce) -1;
430 
431  if ((time_t_ce) seconds != seconds) return (time_t_ce) -1;
432 
433  // return the number of seconds since EPOCH
434  return (time_t_ce) seconds;
435 }

438 // Get the current system time and convert to seconds since
439 // 1/1/1970. Store the seconds value in tloc if not a NULL pointer then
440 // return the seconds value.
441 //
442 time_t_ce time_ce(time_t_ce *tloc)
443 {

446 
447  // Get current system time
448  GetSystemTime(&st);
449 
450  // Set time zone information
451  //
452  SetTz(&st);
453 
454  // convert system time to number of seconds since 1970
455  //
456  secs = SystemTimeToSecondsSince1970(&st);
457

459  //

461  {
462  return TIME_FAILURE;
463  }

472  }
473  memcpy(tloc, &secs, sizeof(time_t_ce));
474  }
475 
476  return secs;
477 }

479 // The strftime function is a modified version created by the following:
480 // written 6 september 1989 by jim nutt
481 // released into the public domain by jim nutt
482 //
483 // modified 21-Oct-89 by Rob Duff
484 //
485 //
486 // size_t strftime(char *str,
487 // size_t maxs,
488 // const char *fmt,
489 // const struct tm *t)
490 //
491 // this functions acts much like a sprintf for time/date output.
492 // given a pointer to an output buffer, a format string and a
493 // time, it copies the time to the output buffer formatted in
494 // accordance with the format string. the parameters are used
495 // as follows:
496 //
497 // str is a pointer to the output buffer, there should
498 // be at least maxs characters available at the address
499 // pointed to by str.
500 //
501 // maxs is the maximum number of characters to be copied
502 // into the output buffer, included the '\0' terminator
503 //
504 // fmt is the format string. a percent sign (%) is used
505 // to indicate that the following character is a special
506 // format character. the following are valid format
507 // characters:
508 //
509 // %A full weekday name (Monday)
510 // %a abbreviated weekday name (Mon)
511 // %B full month name (January)
512 // %b abbreviated month name (Jan)
513 // %c standard date and time representation
514 // %d day-of-month (01-31)
515 // %H hour (24 hour clock) (00-23)
516 // %I hour (12 hour clock) (01-12)
517 // %j day-of-year (001-366)
518 // %M minute (00-59)
519 // %m month (01-12)
520 // %p local equivalent of AM or PM
521 // %S second (00-59)
522 // %U week-of-year, first day sunday (00-53)
523 // %W week-of-year, first day monday (00-53)
524 // %w weekday (0-6, sunday is 0)
525 // %X standard time representation
526 // %x standard date representation
527 // %Y year with century
528 // %y year without century (00-99)
529 // %Z timezone name
530 // %% percent sign
531 //
532 // the standard date string is equivalent to:
533 //
534 // %a %b %d %Y
535 //
536 // the standard time string is equivalent to:
537 //
538 // %H:%M:%S
539 //
540 // the standard date and time string is equivalent to:
541 //
542 // %a %b %d %H:%M:%S %Y
543 //
544 // strftime returns the number of characters placed in the
545 // buffer, not including the terminating \0, or zero if more
546 // than maxs characters were produced.
547 //
548 size_t strftime_ce(char *s, size_t maxs, const char *f, const struct tm *t)
549 {
550  int w;
551  char *p, *q, *r;
552  static char buf[26];
553 
554  p = s;
555  q = s + maxs - 1;
556  while ((*f != '\0'))
557  {
558  if (*f++ == '%')
559  {
560  r = buf;
561  switch (*f++)
562  {
563  case '%' :
564  r = "%";
565  break;
566 
567  case 'a' :
568  r = aday[t->tm_wday];
569  break;
570 
571  case 'A' :
572  r = day[t->tm_wday];
573  break;
574 
575  case 'b' :
576  r = amonth[t->tm_mon];
577  break;
578 
579  case 'B' :
580  r = month[t->tm_mon];
581  break;
582 
583  case 'c' :
584  strfmt(r, "%0 %0 %2 %2:%2:%2 %4",
585  aday[t->tm_wday], amonth[t->tm_mon],
586  t->tm_mday,t->tm_hour, t->tm_min,
587  t->tm_sec, t->tm_year+1900);
588  break;
589 
590  case 'd' :
591  strfmt(r,"%2",t->tm_mday);
592  break;
593 
594  case 'H' :
595  strfmt(r,"%2",t->tm_hour);
596  break;
597 
598  case 'I' :
599  strfmt(r,"%2",(t->tm_hour%12)?t->tm_hour%12:12);
600  break;
601 
602  case 'j' :
603  strfmt(r,"%3",t->tm_yday+1);
604  break;
605 
606  case 'm' :
607  strfmt(r,"%2",t->tm_mon+1);
608  break;
609 
610  case 'M' :
611  strfmt(r,"%2",t->tm_min);
612  break;
613 
614  case 'p' :
615  r = (t->tm_hour>11)?"PM":"AM";
616  break;
617 
618  case 'S' :
619  strfmt(r,"%2",t->tm_sec);
620  break;
621 
622  case 'U' :
623  w = t->tm_yday/7;
624  if (t->tm_yday%7 > t->tm_wday)
625  w++;
626  strfmt(r, "%2", w);
627  break;
628 
629  case 'W' :
630  w = (t->tm_yday + DAYSPERWEEK -
631  (t->tm_wday ?
632  (t->tm_wday - 1) :
633  (DAYSPERWEEK - 1))) / DAYSPERWEEK;
634  strfmt(r, "%2", w);
635  break;
636 
637  case 'w' :
638  strfmt(r,"%1",t->tm_wday);
639  break;
640 
641  case 'x' :
642  strfmt(r, "%2/%2/%2", t->tm_mon + 1,
643  t->tm_mday, t->tm_year+1900);
644  break;
645 
646  case 'X' :
647  strfmt(r, "%2:%2:%2", t->tm_hour,
648  t->tm_min, t->tm_sec);
649  break;
650 
651  case 'y' :
652  strfmt(r,"%2",t->tm_year%100);
653  break;
654 
655  case 'Y' :
656  strfmt(r,"%4",t->tm_year+1900);
657  break;
658 
659  case 'Z' :
660  r = (t->tm_isdst && tz_name[1][0])?tz_name[1]:tz_name[0];
661  break;
662 
663  default:
664  buf[0] = '%'; // reconstruct the format
665  buf[1] = f[-1];
666  buf[2] = '\0';
667  if (buf[1] == 0)
668  f--; // back up if at end of string
669  }
670  while (*r)
671  {
672  if (p == q)
673  {
674  *q = '\0';
675  return 0;
676  }
677  *p++ = *r++;
678  }
679  }
680  else
681  {
682  if (p == q)
683  {
684  *q = '\0';
685  return 0;
686  }
687  *p++ = f[-1];
688  }
689  }
690  *p = '\0';
691  return p - s;
692 }

694 //
695 // static void strfmt(char *str, char *fmt);
696 //
697 // simple sprintf for strftime
698 //
699 // each format descriptor is of the form %n
700 // where n goes from zero to four
701 //
702 // 0 -- string %s
703 // 1..4 -- int %?.?d
704 //
705 static void strfmt(char *str, const char *fmt, ...)
706 {
707  int ival, ilen;
708  char *sval;
709  static int pow[5] = { 1, 10, 100, 1000, 10000 };
710  va_list vp;
711 
712  va_start(vp, fmt);
713  while (*fmt)
714  {
715  if (*fmt++ == '%')
716  {
717  ilen = *fmt++ - '0';
718  if (ilen == 0) // zero means string arg
719  {
720  sval = va_arg(vp, char*);
721  while (*sval)
722  *str++ = *sval++;
723  }
724  else // always leading zeros
725  {
726  ival = va_arg(vp, int);
727  while (ilen)
728  {
729  ival %= pow[ilen--];
730  *str++ = (char)('0' + ival / pow[ilen]);
731  }

928 // Get the JulianDay from a Gregorian Date for number of days into the current
929 // year
930 // Algorithm from: http://www.vsg.cape.com/~pbaum/date/jdalg2.htm
931 //
932 DWORD JulianDays(SYSTEMTIME * st)
933 {
934  int m = 0;
935  int y = 0;
936  double jdd = 0;
937  double jddYearStart = 0;
938 
939  // Calculate the Julian day for the beginning of the year
940  //
941  m = 13;
942  y = st->wYear - 1;
943  jddYearStart = 1 + (153 * m - 457) / 5 + 365 * y + (y / 4) - (y / 100) + (y / 400) + 1721118.5;
944 
945  // Calculate Julian Day for Current Date
946  //
947  if (st->wMonth >= 3)
948  {
949  m = st->wMonth;
950  y = st->wYear;
951  }
952  jdd = st->wDay + (153 * m - 457) / 5 + 365 * y + (y / 4) - (y / 100) + (y / 400) + 1721118.5;
953 
954  // Subract the year start Julian date from the Current Julian date to get
955  // the number of Julian days from the year start
956  //
957  return (DWORD)(jdd - jddYearStart);
958 }
