167 {
168  timer->stop();
169 
170  emit activity();
171 
172  while(tcpSocket->canReadLine())
173  {
174  QString currentLine = tcpSocket->readLine();
175  QRegExp newLineRegExp("[\r\n]+");
176 
177  if(newLineRegExp.exactMatch(currentLine))
178  {
179  QSharedPointer<Picto::ProtocolCommand> command(new Picto::ProtocolCommand(pendingCommand));
180  if(command->isPendingContent())
181  {
182  receiveContent(command);
183  }
184  QSharedPointer<Picto::ProtocolResponse> response;
185  do
186  {
187  response = processCommand(command);
188  QString acceptedEncodings = command->getFieldValue("Accept-Encoding");
189  QStringList acceptedEncodingList = acceptedEncodings.split(",");
190  if(response->getMultiPart() < Picto::MultiPartResponseType::MultiPartInitial)
191  {
192  if(acceptedEncodingList.contains("gzip", Qt::CaseInsensitive))
193  {
194  response->setContentEncoding(Picto::ContentEncodingType::gzip);
195  }
196  else if(acceptedEncodingList.contains("deflate", Qt::CaseInsensitive))
197  {
198  response->setContentEncoding(Picto::ContentEncodingType::deflate);
199  }
200  }
201  deliverResponse(response);
202  if(response->shouldStream())
203  {
204  command->setFieldValue("X-PictoStreamState",response->getFieldValue("X-PictoStreamState"));
205  }
206  } while(tcpSocket->state() == QTcpSocket::ConnectedState &&
207  tcpSocket->waitForBytesWritten() &&
208  response->shouldStream());
209 
210  pendingCommand.clear();
211  }
212  else
213  {
214  pendingCommand += currentLine;
215  }
216  }
217 
218  timer->start();
219 }
