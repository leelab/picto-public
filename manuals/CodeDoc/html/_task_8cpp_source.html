39 QString Task::run(QSharedPointer<Engine::PictoEngine> engine)
40 {
41  QString result = "";
42  if(stateMachine_.isNull())
43  {
44  return result;
45  }
46  else
47  {
48  //Before running the statemachine, we need to append the task name to its path
49  stateMachine_->setPath(QStringList()<<getName());
50 
51  if(engine->slaveMode())
52  {
53  result = stateMachine_->slaveRun(engine);
54  }
55  else
56  {
57  sendInitialStateDataToServer(engine);
58  resetScriptableValues();
59  result = stateMachine_->run(engine);
60 
61  //After the task has finished running, we need to report the final result.
62  // This can't be done within the state machine, because the state machine has no idea if it's the top level.
63  sendFinalStateDataToServer(result, engine);
64  result = "";
65  }
66 
67  return result;
68  }
69 }

112 void Task::sendInitialStateDataToServer(QSharedPointer<Engine::PictoEngine> engine)
113 {
114  // JOEY 12/24/10 If there's no data channel, we're running in test mode.
115  QSharedPointer<CommandChannel> dataChannel = engine->getDataCommandChannel();
116  if(dataChannel.isNull())
117  {
118  //If there's no data channel, this is a Test run. Restart the FrameResolutionTimer every time.
119  QDateTime dateTime = QDateTime::currentDateTime();
120  QString taskRunName = getName()+"_"+dateTime.toString("yyyy_MM_dd__hh_mm_ss");
121  engine->markTaskRunStart(taskRunName);
122  return;
123  }
124 
125  QSharedPointer<Transition> tran(new Transition(QSharedPointer<Asset>(),QSharedPointer<Asset>(),stateMachine_));
126  tran->setAssetId(-taskNumber_);
127  engine->addStateTransitionForServer(tran);
128  QDateTime dateTime = QDateTime::currentDateTime();
129  QString taskRunName = getName()+"_"+dateTime.toString("yyyy_MM_dd__hh_mm_ss");
130  engine->markTaskRunStart(taskRunName);
131  //At the beginning of a new TaskRun, all of the initial property values are sent to
132  //the server. This means that the director gets a huge number of "received" responses
133  //at an unknown time near the beginning of the experiment that can cause frame skipping.
134  //To work around this, we report a virtual initial frame that will automatically include
135  //all of the initial property data. We then wait until we get "received" responses for
136  //that data before continuing.
137  Timestamper stamper;
138  double frameTime = stamper.stampSec();
139  engine->reportNewFrame(frameTime,getAssetId());
140  while(dataChannel->pendingResponses())
141  {
142  dataChannel->processResponses(100);
143  }
144 }

153 void Task::sendFinalStateDataToServer(QString result, QSharedPointer<Engine::PictoEngine> engine)
154 {
155  // JOEY 12/24/10 If there's no data channel, we're running in test mode.
156  QSharedPointer<CommandChannel> dataChannel = engine->getDataCommandChannel();
157  if(dataChannel.isNull())
158  {
159  engine->markTaskRunStop();
160  return;
161  }
162 
163  if(result == "EngineAbort")
164  {
165 
166  QSharedPointer<Transition> tran(new Transition(result));
167  engine->addStateTransitionForServer(tran);
168  }
169  else
170  {
171  QSharedPointer<Transition> tran(new Transition(stateMachine_,result));
172  engine->addStateTransitionForServer(tran);
173  }
174  //Even though there's no actual frame here, calling reportNewFrame has the effect of
175  //marking all of the latest data with a new frame id and sending this data to the server.
176  //Since there's no frame after the end of the experiment, this "fake last frame" takes
177  //its place.
178  Timestamper stamper;
179  double frameTime = stamper.stampSec();
180  engine->markTaskRunStop();
181  engine->reportNewFrame(frameTime,getAssetId());
182 }
