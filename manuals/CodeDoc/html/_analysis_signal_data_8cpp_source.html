225 QVariantList AnalysisSignalData::getPrevValues(double secsPreceding)
226 {
227  QStringList subChannels = getComponentNames();
228  QList<QVariantList> resultsByName;
229  foreach(QString subChan,subChannels)
230  {
231  resultsByName.append(getPrevValues(subChan,secsPreceding));
232  }
233  //Make sure all lists are the same size
234  for(int i=1;i<resultsByName.size();i++)
235  {
236  if(resultsByName[i].size() != resultsByName[i-1].size())
237  {
238  Q_ASSERT(resultsByName[i].size() == resultsByName[i-1].size());
239  return QVariantList();
240  }
241  }
242  //Put together result list
243  QVariantList result;
244  QVariantList singleRow;
245  if(resultsByName.size())
246  {
247  result.reserve(resultsByName[0].size());
248  for(int i=0;i<resultsByName[0].size();i++)
249  {
250  singleRow.clear();
251  for(int j=0;j<resultsByName.size();j++)
252  {
253  singleRow.append(resultsByName[j][i]);
254  }
255  result.append(QVariant(singleRow));
256  }
257  }
258  return result;
