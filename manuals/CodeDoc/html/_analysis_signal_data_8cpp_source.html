240 QVariantList AnalysisSignalData::getPrevValues(double secsPreceding)
241 {
242  QStringList subChannels = getComponentNames();
243  QList<QVariantList> resultsByName;
244  foreach(QString subChan,subChannels)
245  {
246  resultsByName.append(getPrevValues(subChan,secsPreceding));
247  }
248  //Make sure all lists are the same size
249  for(int i=1;i<resultsByName.size();i++)
250  {
251  if(resultsByName[i].size() != resultsByName[i-1].size())
252  {
253  Q_ASSERT(resultsByName[i].size() == resultsByName[i-1].size());
254  return QVariantList();
255  }
256  }
257  //Put together result list
258  QVariantList result;
259  QVariantList singleRow;
260  if(resultsByName.size())
261  {
262  result.reserve(resultsByName[0].size());
263  for(int i=0;i<resultsByName[0].size();i++)
264  {
265  singleRow.clear();
266  for(int j=0;j<resultsByName.size();j++)
267  {
268  singleRow.append(resultsByName[j][i]);
269  }
270  result.append(QVariant(singleRow));
271  }
272  }
273  return result;
274 }
