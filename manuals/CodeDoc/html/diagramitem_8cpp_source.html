92 void DiagramItem::updateLabel()
93 {
94  if(!nameText_)
95  {
96  nameText_ = new QGraphicsTextItem(this);
97  nameText_->setDefaultTextColor(QColor(Qt::white));
98  nameText_->setZValue(1000.0);
99  }
100  QString text = getName();
101 
102  //If there is currently a search string, highlight the appropriate characters in the name.
103  if(!searchString_.isEmpty())
104  {
105  int foundPos = 0;
106  QString colorStartTag = QString("<span style=\"background-color: %1\">")
107  .arg(highlightColors_[searchHighlightIndex_].name());
108  QString colorEndTag("</span>");
109  while((foundPos = text.indexOf(searchString_,foundPos,searchCaseSensitive_?Qt::CaseSensitive:Qt::CaseInsensitive)) >= 0)
110  {
111  text = text.insert(foundPos+searchString_.length(),colorEndTag);
112  text = text.insert(foundPos,colorStartTag);
113  foundPos += searchString_.length()+colorStartTag.length()+colorEndTag.length();
114  }
115  }
116 
117  //Add the type info
118  if((text != "") && (getType() != ""))
119  {
120  text.append("<br>");
121  }
122  if(getType() != "")
123  {
124  text.append("(").append(getType()).append(")");
125  }
126  //Write the text string to the label
127  nameText_->setHtml(text);
128 
129  //Find out how much room the label takes up
130  //If the graphic needs to make room for this label, change its size accordingly
131  QRectF textbound = nameText_->boundingRect();
132  //I found that an empty name nameText_ still has a finite boundingRect for some reason. This corrects that.
133  if(text.isEmpty())
134  {
135  textbound.setWidth(0);
136  textbound.setHeight(0);
137  }
138  QRectF polybound = polygon().boundingRect();
139  bool needsToStretch = false;
140  if(polybound.width() < textbound.width())
141  {
142  needsToStretch = true;
143  polybound.setWidth(textbound.width());
144  }
145  if(polybound.height() < textbound.height())
146  {
147  needsToStretch = true;
148  polybound.setHeight(textbound.height());
149  }
150  if(needsToStretch)
151  {
152  //Set the new size to the diagramItem so that the text fits.
153  setRect(polybound);
154  }
155  nameText_->setPos(rect_.topLeft());
156 }

176 void DiagramItem::paint( QPainter * painter, const QStyleOptionGraphicsItem * option, QWidget * widget)
177 {
178  QGraphicsPolygonItem::paint(painter,option,widget);
179  QRectF rect = getRect();
180  QPainter::CompositionMode currCompMode = painter->compositionMode();
181  int paintWidth = 2;
182  Q_ASSERT(paintWidth > 0 && paintWidth%2 == 0);
183  painter->translate(-paintWidth/2,-paintWidth/2);
184  rect.setWidth(rect.width()+paintWidth);

186  foreach(int highlightIndex,outlines_.keys())

190  rect.setWidth(rect.width()-(2*paintWidth));

192  painter->setPen(QPen(QBrush(highlightColors_[highlightIndex]),paintWidth));
193  painter->setBrush(Qt::NoBrush);
194  painter->drawRect(rect);
195  }
196  painter->setCompositionMode(currCompMode);
197 }

214 QVariant DiagramItem::itemChange(GraphicsItemChange change,
215  const QVariant &value)
216 {
217  QVariant inputVal = value;
218  //See if the change is a position change.
219  if (change == QGraphicsItem::ItemPositionChange)
220  {
221  //See how big the change is. We don't accept small changes
222  //because they can get triggered by double clicking.
223  QPointF translation = inputVal.toPointF() - pos();
224  QRectF bounds = boundingRect();
225  if (
226  stickInPlace_
227  && isSelected()
228  && ( (abs(translation.x()) > (bounds.width()/3))
229  || (abs(translation.y()) > (bounds.height()/3))
230  )
231  )
232  //This is a large change, stop sticking the item in place
233  stickInPlace_ = false;
234 
235  //We only allow movement in MoveItem mode
236  if(editorState_->getEditMode() != EditorState::MoveItem)
237  stickInPlace_ = true;
238 
239  if(!stickInPlace_)
240  {
241  if(catchUpFrames_ < 60)
242  {
243  //We haven't yet caught our position up from sticking in place.
244  //This will move the item smoothly to the desired location after sticking.
245  catchUpFrames_++;
246  inputVal.setValue(pos()+((catchUpFrames_/60.0)*translation));
247  }
248  }
249  if(!stickInPlace_ || !isSelected())
250  { //If we're not sticking, update graphics underneath this one and report
251  //to child classes that the position change.
252  //Note that we also do this if the item isn't selected. This occurs when it
253  //is being placed in the scene when it is first created.
254  updateDependantGraphics();
255  positionChanged(inputVal.toPoint());
256  }
257  else
258  { //The item is sticking right now. Don't allow a position change.
259  return pos();
260  }
261  }
262  else if (change == QGraphicsItem::ItemSelectedChange)
263  {
265  //if(editorState_->getEditMode() != EditorState::Select)
266  // return !inputVal.toBool();
267  }
268  else if (change == QGraphicsItem::ItemSelectedHasChanged)
269  {
270  if(inputVal.toBool())
271  {
272  editorState_->setSelectedItem(this);
273  }
274  else
275  {
276  //When an item is deselected, it should be reinitialized to stick in place.
277  stickInPlace_ = true;
278  catchUpFrames_ = 0;
279 
280  //If this was the last selected item, set the selected item to NULL. This
281  //has the effect of switching the selected asset to the window asset.
282  if(editorState_->getSelectedItem() == this)
283  editorState_->setSelectedItem(NULL);
284  }
285  }
286 
287 
288  return inputVal;
289 }

330 void DiagramItem::enableOutline(int highlightIndex,bool enabled)
331 {
332  //Loop through all enabled groups, and enable/disable outlines for each one according to the input
333  unsigned int outlineIndex;
334  unsigned short upperNybble = 0xF0 & highlightIndex;
335  for(unsigned short lowerNybble=0x01;lowerNybble<0x10;lowerNybble = lowerNybble<<1)
336  {
337  if(!(highlightIndex & lowerNybble))
338  continue;
339  outlineIndex = upperNybble | lowerNybble;
340  if(!enabled)
341  {
342  if(outlines_.contains(outlineIndex))
343  {
344  outlines_.remove(outlineIndex);
345  setRect(getRect());
346  update();
347  }
348  return;
349  }
350  outlines_[outlineIndex] = true;
351  }
352  update(getRect());
353 }
