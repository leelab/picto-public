124 void OperatorPlot::replot()
125 {
126  QVector<QwtIntervalSample> qvSamples;
127  QVector<QwtIntervalSample> qvError;
128 
129  QList<long> keyList = m_qhlCumulNum.keys();
130  std::sort(keyList.begin(), keyList.end());
131 
132  bool bNormalized = propertyContainer_->getPropertyValue("NormalizedDisplay").toBool();
133 
134  double total = 0.0;
135 
136  foreach(long key, keyList)
137  {
138  double dMin = key - 0.5;
139  double dMax = key + 0.5;
140  double dAvg = m_qhdCumulValue[key] / double(m_qhlCumulNum[key]);
141  qvSamples.push_back(std::move(QwtIntervalSample(dAvg, dMin, dMax)));
142 
143  if (bNormalized)
144  {
145  total += dAvg * (dMax - dMin);
146  }
147 
148  if (m_bLastErrorBarState)
149  {
150  double dValue = key;
151  double stdDev = sqrt((m_qhdCumulValSq[key] / m_qhlCumulNum[key]) - dAvg*dAvg);
152  dMin = dAvg - stdDev;
153  dMax = dAvg + stdDev;
154  qvError.push_back(std::move(QwtIntervalSample(dValue, dMin, dMax)));
155  }
156  }
157 
158  m_pHistoPlotItem->setSamples(new QwtIntervalSeriesData(qvSamples));
159 
160  if (bNormalized)
161  {
162  //Change the y-axis labeling to give the appearance of normalization.
163  QList<double> majorTicks;
164  QList<double> mediumTicks;
165 
166  int numOfDivs = 2;
167  for (int i = 0; i < numOfDivs+1; i++)
168  {
169  majorTicks += (i/(1.0*numOfDivs))*total;
170  }
171 
172  numOfDivs = 5;
173  for (int i = 0; i < numOfDivs + 1; i++)
174  {
175  mediumTicks += (i / (1.0*numOfDivs))*total;
176  }
177 
178  QList<double> minorTicks;
179 
180  QwtScaleDiv scaleDiv(0, total + 1, minorTicks, mediumTicks, majorTicks);
181 
182  QwtScaleDraw *scaleDraw = new NormalizedScaleDraw(total);
183 
184  m_pPlot->setAxisScaleDiv(QwtPlot::yLeft, scaleDiv);
185  m_pPlot->setAxisScaleDraw(QwtPlot::yLeft, scaleDraw);
186  }
187 
188  m_pErrorBars->setSamples(new QwtIntervalSeriesData(qvError));
189 
190  m_pPlot->replot();
191 
192  //QPixmap pixmap(m_pPlot->width(), m_pPlot->height());
193  //pixmap.grabWidget(m_pPlot);
194  //image_ = QImage(pixmap);
195 }
