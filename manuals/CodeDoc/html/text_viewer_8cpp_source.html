113 void TextViewer::goToTask(QString taskname)
114 {
115  xmlEdit_->setFocus();
116  if(taskname == "Experiment top")
117  {
118  xmlEdit_->moveCursor(QTextCursor::Start);
119  xmlEdit_->moveCursor(QTextCursor::StartOfLine);
120  xmlEdit_->moveCursor(QTextCursor::EndOfLine,QTextCursor::KeepAnchor);
121  return;
122  }
123 
124  //Find the line number of the task with the matching name
125  QXmlStreamReader xmlReader(pictoDataText_->toPlainText());
126 
127  int taskOffset;
128 
129  while(!xmlReader.atEnd())
130  {
131  if(xmlReader.name() == "Task" && xmlReader.isStartElement())
132  {
133  taskOffset = xmlReader.characterOffset();
134 
135  while(xmlReader.name() != "Name" && !xmlReader.atEnd())
136  xmlReader.readNext();
137  if(!xmlReader.atEnd())
138  {
139  if(taskname == xmlReader.readElementText())
140  {
141  //We do some fancy cursor moving in an attempt to make the
142  //selected task obvious
143  xmlEdit_->moveCursor(QTextCursor::End);
144  QTextCursor cursor = xmlEdit_->textCursor();
145  cursor.setPosition(taskOffset);
146  cursor = xmlEdit_->document()->find("<Task>",cursor, QTextDocument::FindBackward);
147  xmlEdit_->setTextCursor(cursor);
148  break;
149  }
150  }
151  }
152  xmlReader.readNext();
153  }
