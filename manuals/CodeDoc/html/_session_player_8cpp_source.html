186  processing_ = false;

191  PlaybackIndex frameIndexForTime = nextFrameId;

193  {
194  PlaybackIndex prevSpikeInd = sessionState_->getSpikeState()->getCurrentIndex();

269 {

272  QList<QSharedPointer<DataState>> idIndexedStates = sessionState_->getStatesIndexedById();
273  int nBuffer;

277 
278  //Intialize the nBuffer and indexBuffer with the first

280  int i = 0;
281  do

283  nBuffer = i;
284  indexBuffer = idIndexedStates[i++]->getNextIndex(boundaryTime);
285  } while (!indexBuffer.isValid() && i < idIndexedStates.size());
286 
287  //Get the state with the lowest index
288  for (; i<idIndexedStates.size(); i++)
289  {

291  if (iterationIndex.isValid()
292  && (iterationIndex < indexBuffer)
293  )
294  {
295  indexBuffer = iterationIndex;
296  nBuffer = i;
297  }
298 
299  }
300  if (!indexBuffer.isValid())
