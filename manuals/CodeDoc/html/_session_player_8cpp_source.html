176 bool SessionPlayer::step(double lookForward)
177 {
178  processing_ = true;
179  PlaybackIndex currFrameId = sessionState_->getFrameState()->getCurrentIndex();
180  PlaybackIndex nextFrameId = sessionState_->getFrameState()->getNextIndex(lookForward);
181  QSharedPointer<DataState> stateToTrigger = getNextTriggerState(lookForward);
182  if(stateToTrigger.isNull())
183  {
184  processing_ = false;
185  return false;
186  }
187  if (stateToTrigger->getNextIndex(nextFrameId.time()) == nextFrameId)
188  { //If the next stateToTrigger is a frame, step to all time based data states first (ie. reward, signals, spike, lfp).
189  PlaybackIndex frameIndexForTime = nextFrameId;
190  if(frameIndexForTime.isValid())
191  {
192  PlaybackIndex timeIndex = PlaybackIndex::maxForTime(frameIndexForTime.time());
193  QList<QSharedPointer<DataState>> timeIndexedStates = sessionState_->getStatesIndexedByTime();
194  foreach(QSharedPointer<DataState> dataState,timeIndexedStates)
195  {
196  dataState->moveToIndex(timeIndex);
197  }
198  }
199  }
200  stateToTrigger->moveToIndex(stateToTrigger->getNextIndex(nextFrameId.time()));
201  lastIndex_ = stateToTrigger->getCurrentIndex();
202  processing_ = false;
203  return true;
204 }
