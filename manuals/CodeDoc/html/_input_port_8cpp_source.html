133 void InputPort::updateDataBuffer(double currTimestamp)
134 {
135  //If the task isn't running yet, mark this time as the last syncTimstamp and start it.
136  if(!taskRunning_)
137  {
138  lastSyncTimestamp_ = currTimestamp;
139 
140  bool started = start();
141  Q_ASSERT_X(started,"InputPort::updateDataBuffer","Port failed to start");
142  return;
143  }
144  //Only allow update data buffer once all signal channels have looked at their data from the last update.
145  if(!needsUpdate_)
146  return;
147  needsUpdate_ = false;
148 
149  double dataReadTime = updateDataBuffer();
150 
151  //If any data was left over last time this function was called (ie. it corresponded
152  //to times that were less than dataReadTime but greater than currTimestamp, it is
153  //added to the head of the list here.
154  if(channelLeftOvers[channelNums_[0]].size())
155  {
156  int lastTimesLeftovers = channelLeftOvers[channelNums_[0]].size();
157  QHash<int, QVector<double>>::iterator it;
158  for(it = channelBuffers_.begin();it != channelBuffers_.end();it++)
159  {
160  (*it) = channelLeftOvers[it.key()] << (*it);
161  channelLeftOvers[it.key()].clear();
162  }
163  }
164 
165  int totalSampsRead = channelBuffers_[channelNums_[0]].size();
166 
167  //Throw out or duplicate readings to bring sample times into alignment
168  //Since we are using at least 4 times the sample rate of the highest
169  //resolution channel, and the clock is good enough that we should never
170  //need to duplicate more than 1 sample to regain alignment, this is
171  //within any set resolution boundaries.
172  double calculatedFinalReadTime = (sampsCollected_+totalSampsRead)/deviceSampRate_;
173  double jitter = calculatedFinalReadTime-dataReadTime;
174  int sampleListSizeChange = -jitter*deviceSampRate_;
175  if(sampleListSizeChange)
176  {
177  QHash<int, QVector<double>>::iterator chanIter;
178  int origSize;
179  for(chanIter = channelBuffers_.begin();chanIter !=channelBuffers_.end();chanIter++)
180  {
181  origSize = chanIter.value().size();
182  if(sampleListSizeChange + origSize < 0)
183  sampleListSizeChange = origSize;
184  chanIter.value().resize(origSize+sampleListSizeChange);
185  if((sampleListSizeChange > 0) && origSize)
186  {
187  //If we increased the size of the vector, we should duplicate the last entry
188  for(int i=origSize;i<origSize+sampleListSizeChange;i++)
189  {
190  chanIter.value()[i] = chanIter.value()[origSize-1];
191  }
192  }
193  }
194  }
195 
196  //We used the actual data read time reported from updateDataBuffer() to calculate
197  //jitter and synchronize accordingly. There may have been some delay between the
198  //passed in currTimestamp and the actual read time though. The caller wants data
199  //that was read before currTimestamp. For this reason, we remove the entries off
200  //the back of the channelBuffers_ vectors and save them for the next data read.
201  int leftoverData = 0;
202  calculatedFinalReadTime = (sampsCollected_+totalSampsRead+sampleListSizeChange)/deviceSampRate_;
203  if(calculatedFinalReadTime > currTimestamp)
204  {
205  QHash<int, QVector<double>>::iterator it;
206  leftoverData = 1+(calculatedFinalReadTime-currTimestamp)*deviceSampRate_;
207  for(it = channelBuffers_.begin();it != channelBuffers_.end();it++)
208  {
209  channelLeftOvers[it.key()] << it->mid(it->size()-leftoverData,-1);
210  it->resize(it->size()-leftoverData);
211  }
212  }
213 
214  //Update frame offset time values
215  int sampsPerRead;
216  int currPhase;
217  foreach(int chanNum,sampsPerReadMap_.keys())
218  {
219  sampsPerRead = sampsPerReadMap_[chanNum];
220  currPhase = channelPhaseMap_[chanNum];
221  latestFrameOffsetTimes_[chanNum] = ((sampsCollected_+(sampsPerRead-currPhase))/deviceSampRate_)-lastSyncTimestamp_;
222  }
223 
224  //Update number of samples collected
225  sampsCollected_ += (totalSampsRead+sampleListSizeChange-leftoverData);
226 
227  //Update last Sync Timestamp
228  lastSyncTimestamp_ = currTimestamp;
229 
230  //Mark all channels as unpolled
231  for(QHash<int, bool>::iterator it = polledDataMap_.begin();it != polledDataMap_.end();it++)
232  {
233  (*it) = false;
234  }
235 }

252  QVector<double>* currList = &channelBuffers_[channelNum];
253  //Create the return array with the correct size

255  QVector<double> returnVal((currList->size()+currPhase)/sampsPerRead);
256

261  {

263  if(currPhase == sampsPerRead)
264  {
265  double currVal = (*currList)[i];
266  returnVal[j++] = (*currList)[i];
267  currPhase = 0;
268  }
269  }

271 
272  //Mark this channel's data as polled.
273  polledDataMap_[channelNum] = true;
274

276  channelBuffers_[channelNum].clear();
277 
278  //If any data channel hasn't been polled yet, we're done.
279  foreach(bool polled,polledDataMap_)
280  {
281  if(!polled)
282  return returnVal;
283  }
284  //All data channels have been polled, we're ready for the next nidaq card update.
285  needsUpdate_ = true;
