236 void ProgressWidget::updateHighlights(bool forceUpdate)
237 {
238  //To update the highlight values we just change the widget stylesheet. This takes a lot of processor
239  //time but is by far the easiest way to do this. In order to avoid causing UI lag when this function
240  //is called a lot, we just make sure that we only reset the stylesheet one per REFRESHINTERVALMS
241  //which should be on the order of 30ms.
242  if(!forceUpdate && (frameTimer_.elapsed() < REFRESHINTERVALMS))
243  return;
244  frameTimer_.restart();
245  //Use styles to create highlight bar in widget
246  QString styleUpdate = QString("QSlider::groove:horizontal {background: #fff, stop:1 #fff;}");
247  if(progressToSlider(max_))
248  {
249  //Build list of necessary color stops to include in style string
250  QList<HighlightData> highlights = highlights_.values();
251  qStableSort<QList<HighlightData>>(highlights);
252  QVector<QString> colors;
253  QVector<double> pos;
254  double lastStopPt = 0.0;
255  foreach(HighlightData data,highlights)
256  {
257  double min = data.getMin()/max_;
258  double max = data.getMax()/max_;
259  if(max < lastStopPt + 2*HIGHLIGHTBORDERWIDTH)
260  continue;
261  if(min < lastStopPt+HIGHLIGHTBORDERWIDTH)
262  min = lastStopPt+HIGHLIGHTBORDERWIDTH;
263  if(min > 1-2*HIGHLIGHTBORDERWIDTH)
264  break;
265  if(min > lastStopPt+2*HIGHLIGHTBORDERWIDTH)
266  {
267  colors.append("#fff");
268  pos.append(lastStopPt+HIGHLIGHTBORDERWIDTH);
269  colors.append("#fff");
270  pos.append(min-HIGHLIGHTBORDERWIDTH);
271  }
272  colors.append(data.getMinColorStr());
273  pos.append(min);
274 
275  if(max < min + HIGHLIGHTBORDERWIDTH)
276  max = min + HIGHLIGHTBORDERWIDTH;
277  if(max > 1-HIGHLIGHTBORDERWIDTH)
278  max = 1-HIGHLIGHTBORDERWIDTH;
279  colors.append(data.getMaxColorStr());
280  pos.append(max);
281  lastStopPt = max;
282  }
283  //Add white color stop if last stop isn't near the end of the bar
284  if(pos.size() && pos.last() < 1-HIGHLIGHTBORDERWIDTH)
285  {
286  pos.append(pos.last()+HIGHLIGHTBORDERWIDTH);
287  colors.append("#fff");
288  }
289 
290  //Add color stops to string
291  QString stopStr = "";
292  for(int i=0;i<pos.size();i++)
293  {
294  stopStr.append(QString("stop:%1 %2,")
295  .arg(pos[i])
296  .arg(colors[i]));
297  }
298  styleUpdate = QString("QSlider::groove:horizontal{background:qlineargradient(x1:0,y1:0.5,x2:1,y2:.5,stop:0 #fff,%1stop:1 #fff);}").arg(stopStr);
299  }

302  progSlider_->setStyleSheet(styleUpdate);
