432 void TokenTrayGraphic::draw()
433 {
434  //return;
435  //if(updatingParameterLists_)
436  // return; //This is being called recursively. Leave.
437  int numTokens = getNumTokens();
438  QRect trayDimensions = QRect(QPoint(),propertyContainer_->getPropertyValue("Size").toSize());
439  QRect tokenDimensions = QRect(QPoint(),propertyContainer_->getPropertyValue("TokenSize").toSize());
440  //Make sure size of sizes,colors,shapes lists matches numTokens before drawing
441  QStringList sizes = propertyContainer_->getPropertyValue("TokenSizes").toString().split(",",QString::SkipEmptyParts);
442  QStringList colors = propertyContainer_->getPropertyValue("TokenColors").toString().split(",",QString::SkipEmptyParts);
443  QStringList shapes = propertyContainer_->getPropertyValue("TokenShapes").toString().split(",",QString::SkipEmptyParts);
444  QStringList outlines = propertyContainer_->getPropertyValue("TokenOutlines").toString().split(",",QString::SkipEmptyParts);
445  QStringList outlineWidths = propertyContainer_->getPropertyValue("TokenOutlineWidths").toString().split(",",QString::SkipEmptyParts);
446  if ( (sizes.size() != getNumTokens())
447  || (colors.size() != getNumTokens())
448  || (shapes.size() != getNumTokens())
449  || (outlines.size() != getNumTokens())
450  || (outlineWidths.size() != getNumTokens())
451  )
452  {
453  //Stuff is in the process of being updated.
454  return;
455  }
456  //Whenever a list's value is -1 it should juist used the default value
457  for(int i=0;i<numTokens;i++)
458  {
459  if(sizes[i] == "-1")
460  sizes[i] = "1.0";
461  if(colors[i] == "-1")
462  colors[i] = getColor().name();
463  if(shapes[i] == "-1")
464  shapes[i] = getTokenShape();
465  if(outlines[i] == "-1")
466  outlines[i] = QString::number(int(getOutline()));
467  if(outlineWidths[i] == "-1")
468  outlineWidths[i] = QString::number(getOutlineWidth());
469  }
470  int maxTokenDiam= getMaxTokenDiam();
471  QRect imageDimensions(0,0,trayDimensions.width()+maxTokenDiam,trayDimensions.height()+maxTokenDiam);
472  QImage image(imageDimensions.width()+maxTokenDiam,imageDimensions.height()+maxTokenDiam,QImage::Format_ARGB32);
473  image.fill(0);
474  posOffset_ = QPoint(imageDimensions.width()/2.0,imageDimensions.height()/2.0);
475 
476  for(int i=0;i<numTokens;i++)
477  {
478  QColor color(colors[i]);
479  double size = sizes[i].toDouble();
480  QRect dimensions(tokenDimensions.x(),tokenDimensions.y(),tokenDimensions.width()*size,tokenDimensions.height()*size);
481  QPoint tokenPosOffset = QPoint(dimensions.width()/2.0,dimensions.height()/2.0);
482  QPoint offset = posOffset_+getTokenOffset(i)-tokenPosOffset;
483  QPainter p(&image);
484  QPen pen(color);
485  p.setBrush(color);
486  p.translate(offset);
487  if(outlines[i].toInt())
488  {
489  p.setBrush(QColor(0,0,0,0));
490  pen.setWidth(outlineWidths[i].toInt());
491  }
492  p.setPen(pen);
493  p.setRenderHint(QPainter::Antialiasing, true);
494 
495  if(shapes[i] == "Ellipse")
496  {
497  p.drawEllipse(dimensions);
498  }
499  else if(shapes[i] == "Rectangle")
500  {
501  p.drawRect(dimensions);
502  }
503  else if(shapes[i] == "Diamond")
504  {
505  QPolygon diamond;
506  int maxWPoint = dimensions.width()-1;
507  int maxHPoint = dimensions.height()-1;
508  diamond << QPoint(maxWPoint/2,0)
509  << QPoint(maxWPoint,maxHPoint/2)
510  << QPoint(maxWPoint/2,maxHPoint)
511  << QPoint(0,maxHPoint/2)
512  << QPoint(maxWPoint/2,0);
513  p.drawPolygon(diamond);
514  }
515 
516  p.end();
517  }
518  image_ = image;
519  //updateCompositingSurfaces();
520 
521  shouldUpdateCompositingSurfaces_ = true;
522 }

603 QPoint TokenTrayGraphic::getTokenOffset(int index)
604 {
605  QRect dims = getTrayDimensions();
606  int numTokens = getNumTokens();
607  double angle = double(index)*2.0*PI/double(numTokens);
608  angle += propertyContainer_->getPropertyValue("Phase").toInt()*PI/180.0;
609  //Keep angle in 0->2PI
610  int fullLoops = angle/(2*PI);
611  angle = angle-double(fullLoops*2*PI);
612  double h = dims.height();
613  double w = dims.width();
614  if((w <= 0) || (h <= 0))//Ideally, these cases would be allowed and just signify a straight line. I'm not going to worry about this right now.
615  return QPoint(0,0);
616  double x=0;
617  double y=0;
618 
619  QString trayShape = shapeList_.value(propertyContainer_->getPropertyValue("Shape").toInt(),"");
620  if(trayShape == "Ellipse")
621  {
622  //Lots of algebra leads to the following x value where a line at angle (angle) intersects our ellipse
623  x = sqrt( (h*h*w*w) / (4.0* ((w*w*tan(angle)*tan(angle))+(h*h)) ) );
624  if((angle > PI/2.0) && (angle <= 3.0*PI/2.0))
625  x = -x;
626  //More algebra leads to the value of y below (where the line at angle (angle) intersects our ellipse).
627  double buff = (w*w*.25)-(x*x);
628  if(buff < 0) buff = 0;
629  y=-sqrt( h*h*(buff)/(w*w) ); //negative sine is because y is down in screen coordinates but up in standard coordinates
630  if(angle > PI)
631  y = -y;
632  }
633  else if(trayShape == "Rectangle")
634  {
635 
636  }
637  else if(trayShape == "Diamond")
638  {
639 
640  }
641  return QPoint(x,y);
642 }

647 int TokenTrayGraphic::getMaxTokenDiam()
648 {
649  int defaultOutlineWidth = getOutlineWidth();
650  int defaultWidth = getTokenWidth()+getOutlineWidth();
651  int defaultHeight = getTokenHeight()+getOutlineWidth();
652  int defaultTokenDiam = defaultWidth;
653  if(defaultHeight > defaultWidth) defaultTokenDiam = defaultHeight;
654  int maxDiam = defaultTokenDiam;
655  if(defaultHeight > defaultWidth) maxDiam = defaultHeight;
656  QStringList sizes = propertyContainer_->getPropertyValue("TokenSizes").toString().split(",",QString::SkipEmptyParts);
657  QStringList outlineWidths = propertyContainer_->getPropertyValue("TokenOutlineWidths").toString().split(",",QString::SkipEmptyParts);
658  if ( (sizes.size() < getNumTokens())
659  || (outlineWidths.size() < getNumTokens())
660  )
661  return defaultTokenDiam + defaultOutlineWidth + 1; //If this is the case, somethings being updated and calling draw and the system won't really need this data yet.
662  for(int i=0;i<getNumTokens();i++)
663  {
664  int oWidth = defaultOutlineWidth;
665  int diam = defaultTokenDiam;
666  if(sizes[i] != "-1")
667  diam *= sizes[i].toDouble();
668  if(outlineWidths[i] != "-1")
669  oWidth = outlineWidths[i].toInt();
670  int fullDiam = diam+oWidth;
671  if(fullDiam > maxDiam)
672  maxDiam = fullDiam;
673  }
674  return maxDiam+1;
675 }
