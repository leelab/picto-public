164 void AnalysisSelectorWidget::updateLocalAnalysisList()
165 {
166  //Remove all old buttons from layout. We do this in a roundabout but simple way. Moving a widget's
167  //layout to another widget and then destroying that widget destroys the layout and all the children that
168  //were attached to it.
169  if(analysesBox_->layout())
170  QWidget().setLayout(analysesBox_->layout());
171  analysesBox_->setLayout(new QVBoxLayout());
172 
173  //Add new buttons to layout
174  if(localAnalysisLookup_.contains(currFilePath_))
175  {
176  for(int i=0;i<localAnalysisLookup_[currFilePath_].size();i++)
177  {
178  QCheckBox* analysisCheckbox(new QCheckBox(localAnalysisLookup_[currFilePath_][i].name_));
179  connect(analysisCheckbox,SIGNAL(clicked(bool)),this,SLOT(checkboxChanged(bool)));
180  qobject_cast<QVBoxLayout*>(analysesBox_->layout())->addWidget(analysisCheckbox,0,Qt::AlignTop);
181  qobject_cast<QVBoxLayout*>(analysesBox_->layout())->setStretch(i,0);
182  }
183  }
184  qobject_cast<QVBoxLayout*>(analysesBox_->layout())->addStretch(1);
185  //Enable this tab if it contains any analyses
186  tabWidget_->setTabEnabled(0,localAnalysisLookup_.contains(currFilePath_) && localAnalysisLookup_[currFilePath_].size());
187  emit analysisWidgetChanged();
188 }
