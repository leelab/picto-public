165 void AnalysisSelectorWidget::updateLocalAnalysisList()
166 {
167  //Remove all old buttons from layout. We do this in a roundabout but simple way. Moving a widget's
168  //layout to another widget and then destroying that widget destroys the layout and all the children that
169  //were attached to it.
170  if (analysesBox_->layout())
171  QWidget().setLayout(analysesBox_->layout());
172  analysesBox_->setLayout(new QVBoxLayout());
173 
174  //Add new buttons to layout
175  if (localAnalysisLookup_.contains(currFilePath_))
176  {
177  for (int i = 0; i<localAnalysisLookup_[currFilePath_].size(); i++)
178  {
179  QCheckBox* analysisCheckbox(new QCheckBox(localAnalysisLookup_[currFilePath_][i].name_));
180  checkboxInfoHash[analysisCheckbox] = &localAnalysisLookup_[currFilePath_][i];
181  connect(analysisCheckbox,SIGNAL(clicked(bool)),this,SLOT(checkboxChanged(bool)));
182  qobject_cast<QVBoxLayout*>(analysesBox_->layout())->addWidget(analysisCheckbox,0,Qt::AlignTop);
183  qobject_cast<QVBoxLayout*>(analysesBox_->layout())->setStretch(i,0);
184  }
185  }
186  qobject_cast<QVBoxLayout*>(analysesBox_->layout())->addStretch(1);
187  //Enable this tab if it contains any analyses
188  tabWidget_->setTabEnabled(0,localAnalysisLookup_.contains(currFilePath_) && localAnalysisLookup_[currFilePath_].size());
189  emit analysisWidgetChanged();
190 }
