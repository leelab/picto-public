420 bool ReplayViewer::popRunQueueFront()
421 {
422  bool returnValue = true;
423  runQueue_.pop_front();
424  if(runQueue_.isEmpty())
425  {
426  //Playback is complete.
427  QList<PlayRunInfo> selectedRuns = getSelectedPlayRunInfo();
428  //Check if any of the run playbacks failed for the currently selected runs
429  QList<PlayRunInfo> failedRuns;
430  foreach(PlayRunInfo runInfo,selectedRuns)
431  {
432  if(runs_->getRunStatus(runInfo.filePath_,runInfo.runIndex_) == RunSelectorWidget::ERROROCCURED)
433  {
434  failedRuns.append(runInfo);
435  }
436  }
437  //If a playback failed
438  if(failedRuns.size())
439  {
440  //Tell the user we're done and offer to re-run failed playbacks.
441  QMessageBox::StandardButton result = QMessageBox::question(this,"Playback Complete with Errors","Playback is complete; however, some runs could not successfully be played. This is most likely due to limited system memory. Try closing other applications and then select \"Retry\" to retry playback on the failed runs only. If playback still doesn't work, try restarting Picto.",QMessageBox::StandardButton::Retry | QMessageBox::StandardButton::Cancel,QMessageBox::StandardButton::Cancel);
442  //If the user wants to run failed playbacks again
443  if(result == QMessageBox::Retry)
444  {
445  //Update the runs_ dialog and selectedRuns list, then set return value to true (for continue).
446  runs_->deselectAll();
447  foreach(PlayRunInfo runInfo,failedRuns)
448  {
449  runs_->selectRun(runInfo.filePath_,runInfo.runIndex_);
450  }
451  selectedRuns = getSelectedPlayRunInfo();
452  returnValue = true;
453  }
454  else
455  {
456  //Don't continue
457  returnValue = false;
458  }
459  }
460  else
461  {
462  //All playbacks were succesfull. Tell the user
463  QMessageBox::information(this,"Playback Complete","All selected runs completed successfully");
464  returnValue = false;
465  }
466  //Update the runQueue_ to contain all the selected runs.
467  setRunQueue(selectedRuns);
468  }
469  return returnValue;
470 }

621  bool analysisIsSelected = analysisSelector_->hasSelectedIds();
622  switch(status)
623  {
624  case PlaybackControllerData::Idle:
625  progress_->setHighlightRange(0,0,0);
626  progress_->setHighlightRange(1,0,0,true);
627  progress_->setSliderProgress(0);

640  {
641  static_cast<OutputSignalWidget*>(outSigWidg)->enable(false);
642  }
643  playbackController_->getRenderingTarget()->showSplash();
644  break;
645  case PlaybackControllerData::PreLoading:
646  progress_->setHighlightRange(0,0,0);
647  progress_->setHighlightRange(1,0,0,true);
648  progress_->setSliderProgress(0);
649  progress_->setEnabled(false);

656  toggleRecord_->setEnabled(false);
657  loadProgress_->setMaximum(0);
658  loadSessionAction_->setEnabled(false);
659  lfpRequirements_->setEnabled(false);
660  analysisSelector_->setEnabled(true);
661  analysisSelector_->enableCheckboxes(true);
662  foreach(QWidget * outSigWidg, outputSignalsWidgets_)
663  {
664  static_cast<OutputSignalWidget*>(outSigWidg)->enable(false);
665  }
666  playbackController_->getRenderingTarget()->showSplash();
667  break;
668  case PlaybackControllerData::Stopped:
669  runs_->enableCheckBoxes(true);
