320 bool ReplayViewer::popRunQueueFront()
321 {
322  bool returnValue = true;
323  runQueue_.pop_front();
324  if(runQueue_.isEmpty())
325  {
326  //Playback is complete.
327  QList<PlayRunInfo> selectedRuns = getSelectedPlayRunInfo();
328  //Check if any of the run playbacks failed for the currently selected runs
329  QList<PlayRunInfo> failedRuns;
330  foreach(PlayRunInfo runInfo,selectedRuns)
331  {
332  if(runs_->getRunStatus(runInfo.filePath_,runInfo.runIndex_) == RunSelectorWidget::ERROROCCURED)
333  {
334  failedRuns.append(runInfo);
335  }
336  }
337  //If a playback failed
338  if(failedRuns.size())
339  {
340  //Tell the user we're done and offer to re-run failed playbacks.
341  QMessageBox::StandardButton result = QMessageBox::question(this,"Playback Complete with Errors","Playback is complete; however, some runs could not successfully be played. This is most likely due to limited system memory. Try closing other applications and then select \"Retry\" to retry playback on the failed runs only. If playback still doesn't work, try restarting Picto.",QMessageBox::StandardButton::Retry | QMessageBox::StandardButton::Cancel,QMessageBox::StandardButton::Cancel);
342  //If the user wants to run failed playbacks again
343  if(result == QMessageBox::Retry)
344  {
345  //Update the runs_ dialog and selectedRuns list, then set return value to true (for continue).
346  runs_->deselectAll();
347  foreach(PlayRunInfo runInfo,failedRuns)
348  {
349  runs_->selectRun(runInfo.filePath_,runInfo.runIndex_);
350  }
351  selectedRuns = getSelectedPlayRunInfo();
352  returnValue = true;
353  }
354  else
355  {
356  returnValue = false; //Don't continue
357  }
358  }
359  else
360  { //All playbacks were succesfull. Tell the user
361  QMessageBox::information(this,"Playback Complete","All selected runs completed successfully");
362  returnValue = false;
363  }
364  //Update the runQueue_ to contain all the selected runs.
365  setRunQueue(selectedRuns);
366  }
367  return returnValue;
368 }

744 void ReplayViewer::runSelectionChanged()
745 {
746  QList<PlayRunInfo> newRunQueue = getSelectedPlayRunInfo();
747 
748  //If the first run in the new run queue isn't part of the
749  //same file that we last loaded, set the load progress to zero
750  if ( runQueue_.size()
751  && newRunQueue.size()
752  && (newRunQueue.first().filePath_ != runQueue_.first().filePath_)
753  )
754  {
755  loadProgress_->setValue(0);
756  }
757 
758  //If we're not currently running
759  if(latestStatus_ == PlaybackControllerData::Stopped)
760  {
761  //If nothing is selected, disable the commands that let you
762  //start playback
763  if(newRunQueue.isEmpty())
764  {
765  pauseAction_->setEnabled(false);
766  playAction_->setEnabled(false);
767  runToEndAction_->setEnabled(false);
768  toggleRecord_->setEnabled(false);
769  }
770  else
771  { //Otherwise enable them
772  pauseAction_->setEnabled(true);
773  playAction_->setEnabled(true);
774  runToEndAction_->setEnabled(analysisSelector_->hasSelectedIds());
775  toggleRecord_->setEnabled(true);
776  }
777  }
778  runQueue_ = newRunQueue;
779  //Set up local analysis selector based on currently selected runs
780  if(runQueue_.isEmpty())
781  analysisSelector_->setCurrentFile("");
782  else
783  {
784  //See if there are runs from more than one file selected. If so, disable
785  //local analyses
786  QString filePath = runQueue_.first().filePath_;
787  bool enableLocalAnalyses = true;
788  foreach(PlayRunInfo runInfo,runQueue_)
789  {
790  if(runInfo.filePath_ != filePath)
791  {
792  enableLocalAnalyses = false;
793  break;
794  }
795  }
796  if(enableLocalAnalyses)
797  analysisSelector_->setCurrentFile(filePath);
798  else
799  analysisSelector_->setCurrentFile("");
800  }
801  jumpDownRequested_ = false;
802  stopAction_->trigger();
803  progress_->setSliderProgress(0);
804  progress_->setHighlightRange(0,0,0);
805  progress_->setHighlightRange(1,0,0);
806 }
