229  return;

454 void PictoEngine::reportNewFrame(double frameTime,int runningStateId)
455 {
456  //Create a new frame data store
457  FrameDataUnitPackage frameData;
458 
459  //Behavioral data occured during the course of the previous frame and so its offsettime value is with respect to the
460  // first phosphor time of the previous frame. The frame is stamped when we call getBehavioralDataPackages, so we
461  // need to call it before we call setLastFrame with the new frame id.
462  QList<QSharedPointer<BehavioralDataUnitPackage>> behavPackList = getBehavioralDataPackages();
463  QList<QSharedPointer<InputDataUnitPackage>> inputPackList = getInputDataPackages();
464 
465  frameData.addFrame(frameTime,runningStateId);
466  setLastFrame(frameData.getLatestFrameId());
467 
468  QSharedPointer<CommandChannel> dataChannel = getDataCommandChannel();
469 
470  //If this is an experimental run or this is a test run (not playback)
471  if(frameTimerFactory_)
472  {
473  //Tell all timers that work on single frame resolution what the latest frame time is
474  frameTimerFactory_->setLastFrameTime(frameTime);
475  }
476 
477  //If there's an active lfpReader_ (ie. We're using the TestViewer with Analysis)
478  if (lfpReader_)
479  {
480  //Set the latest frame time to the frameReader_
481  frameReader_->setLatestFrameTime(frameTime);
482  //Set the latest frame time to the lfpReader_
483  lfpReader_->createVirtualLfpData(frameTime);
484  //Set the latest reward data to the rewardReader_
485  QList<QSharedPointer<RewardDataUnit>> rewards = getDeliveredRewards();
486  foreach(QSharedPointer<RewardDataUnit> reward,rewards)
487  {
488  rewardReader_->setLatestRewardData(reward->getTime().toDouble(),reward->getDuration());
489  }
490  //Add a spike to the spikeReader
491  spikeReader_->createVirtualSpike(frameTime);
492 
493  //Set the latest signal data to the signal readers
494  QStringList subChannels;
495  QVector<float> chanData;
496  //Loop through channels
497  foreach(QSharedPointer<Picto::SignalChannel> chan,signalChannels_)
498  {
499  //Loop through this channel's subchannels and build a list of float data
500  subChannels = chan->getSubchannels();
501  chanData.clear();
502  foreach(QString subChan,subChannels)
503  {
504  chanData.append(chan->peekValue(subChan));
505  }
506  //Add data to the appropriate signal reader
507  Q_ASSERT(signalReaders_.contains(chan->getName().toLower()));
508  QSharedPointer<LiveSignalReader> signalReader = signalReaders_.value(chan->getName().toLower());
509  //Write the input signal the correct number of times according to the signal reader's set
510  //sample period
511  double samplePeriod = signalReader->getSamplePeriod();
512  double nextSignalTime = signalReader->getLatestTime() + samplePeriod;
513  while(nextSignalTime <= frameTime)
514  {
515  signalReader->setLatestSignalData(nextSignalTime,chanData);
516  nextSignalTime += samplePeriod;
517  }
518  }
519  }
520 
521  if(dataChannel.isNull())
522  return;
523 
524  QSharedPointer<PropertyDataUnitPackage> propPack = getChangedPropertyPackage();
525  QSharedPointer<StateDataUnitPackage> statePack = getStateDataPackage();
526 
527  if(taskRunStarting_)
528  taskRunUnit_->endFrame_ = 0;
529  else if(taskRunEnding_)
530  taskRunUnit_->endFrame_ = getLastFrameId();
531 
532  //send a PUTDATA command to the server with the most recent behavioral data
533  QSharedPointer<Picto::ProtocolResponse> dataResponse;
534  QString status = "running";
535  int engCmd = getEngineCommand();
536  switch(engCmd)
537  {
538  case PlayEngine:
539  status = "running";
540  break;
541  case PauseEngine:
542  status = "paused";
543  break;
544  case StopEngine:
545  status = "stopped";
546  break;
547  }
548  QString dataCommandStr = "PUTDATA " + getName() + ":" + status + " PICTO/1.0";
549  QSharedPointer<Picto::ProtocolCommand> dataCommand(new Picto::ProtocolCommand(dataCommandStr));
550 
551  QByteArray dataXml;
552  QSharedPointer<QXmlStreamWriter> xmlWriter(new QXmlStreamWriter(&dataXml));
553 
554  xmlWriter->writeStartElement("Data");
555 
556  foreach(QSharedPointer<BehavioralDataUnitPackage> behavPack,behavPackList)
557  {
558  if(behavPack && behavPack->length())
559  behavPack->toXml(xmlWriter);
560  }
561 
562  foreach(QSharedPointer<InputDataUnitPackage> inputPack, inputPackList)
563  {
564  if (inputPack && inputPack->length())
565  inputPack->toXml(xmlWriter);
566  }
567 
568  if(propPack && propPack->length())
569  propPack->toXml(xmlWriter);
570  if(statePack && statePack->length())
571  statePack->toXml(xmlWriter);
572  QList<QSharedPointer<RewardDataUnit>> rewards = getDeliveredRewards();
573  foreach(QSharedPointer<RewardDataUnit> reward,rewards)
574  {
575  reward->toXml(xmlWriter);
576  }
577  if((taskRunStarting_ || taskRunEnding_) && taskRunUnit_)
578  {
579  taskRunUnit_->toXml(xmlWriter);
580  taskRunStarting_ = false;
581  taskRunEnding_ = false;
582  }
583  frameData.toXml(xmlWriter); //Frame data must go last so that server knows when it reads
584  //in a frame, that the data it has defines the state that was
585  //in place at that frame.
586  xmlWriter->writeEndElement();
587 
588  dataCommand->setContent(dataXml);
589  dataCommand->setFieldValue("Content-Length",QString::number(dataXml.length()));
590  QUuid commandUuid = QUuid::createUuid();
591 
592  dataChannel->sendRegisteredCommand(dataCommand);
593 
594  dataChannel->processResponses(0);
595  //The line below is very fast if the connection isn't broken and takes up to 5 ms reconnecting if it is.
596  //This shouldn't be a problem since there should be some time to spare in the state's run loop before the next frame.
597  dataChannel->assureConnection(5);
598
